218  void create_thread_object(threads::thread_id_type& thrd,
219  threads::thread_init_data& data, thread_state_enum state, Lock& lk)
220  {
221  HPX_ASSERT(lk.owns_lock());
222  HPX_ASSERT(data.stacksize != 0);
223 
224  std::ptrdiff_t stacksize = data.stacksize;
225 
226  std::list<thread_id_type>* heap = nullptr;
227 
228  if (stacksize == get_stack_size(thread_stacksize_small))
229  {
230  heap = &thread_heap_small_;
231  }
232  else if (stacksize == get_stack_size(thread_stacksize_medium))
233  {
234  heap = &thread_heap_medium_;
235  }
236  else if (stacksize == get_stack_size(thread_stacksize_large))
237  {
238  heap = &thread_heap_large_;
239  }
240  else if (stacksize == get_stack_size(thread_stacksize_huge))
241  {
242  heap = &thread_heap_huge_;
243  }
244  else {
245  switch(stacksize) {
246  case thread_stacksize_small:
247  heap = &thread_heap_small_;
248  break;
249 
250  case thread_stacksize_medium:
251  heap = &thread_heap_medium_;
252  break;
253 
254  case thread_stacksize_large:
255  heap = &thread_heap_large_;
256  break;
257 
258  case thread_stacksize_huge:
259  heap = &thread_heap_huge_;
260  break;
261 
262  default:
263  break;
264  }
265  }
266  HPX_ASSERT(heap);
267 
268  if (state == pending_do_not_schedule || state == pending_boost)
269  {
270  state = pending;
271  }
272 
273  // Check for an unused thread object.
274  if (!heap->empty())
275  {
276  // Take ownership of the thread object and rebind it.
277  thrd = heap->front();
278  heap->pop_front();
279  thrd->rebind(data, state);
280  }
281 
282  else
283  {
284  hpx::util::unlock_guard<Lock> ull(lk);
285 
286  // Allocate a new thread object.
287  thrd = threads::thread_data::create(data, memory_pool_, state);
288  }

292  // add new threads if there is some amount of work available
293  std::size_t add_new(std::int64_t add_count, thread_queue* addfrom,
294  std::unique_lock<mutex_type> &lk, bool steal = false)
295  {
296  HPX_ASSERT(lk.owns_lock());
297 
298  if (HPX_UNLIKELY(0 == add_count))
299  return 0;
300 
301  std::size_t added = 0;
302  task_description* task = nullptr;
303  while (add_count-- && addfrom->new_tasks_.pop(task, steal))
304  {
305 #ifdef HPX_HAVE_THREAD_QUEUE_WAITTIME
306  if (maintain_queue_wait_times) {
307  addfrom->new_tasks_wait_ +=
308  util::high_resolution_clock::now() - util::get<2>(*task);
309  ++addfrom->new_tasks_wait_count_;
310  }
311 #endif
312  --addfrom->new_tasks_count_;
313 
314  // measure thread creation time
315  util::block_profiler_wrapper<add_new_tag> bp(add_new_logger_);
316 
317  // create the new thread
318  threads::thread_init_data& data = util::get<0>(*task);
319  thread_state_enum state = util::get<1>(*task);
320  threads::thread_id_type thrd;
321 
322  create_thread_object(thrd, data, state, lk);
323 
324  delete task;
325 
326  // add the new entry to the map of all threads
327  std::pair<thread_map_type::iterator, bool> p =
328  thread_map_.insert(thrd);
329 
330  if (HPX_UNLIKELY(!p.second)) {
331  lk.unlock();
332  HPX_THROW_EXCEPTION(hpx::out_of_memory,
333  "threadmanager::add_new",
334  "Couldn't add new thread to the thread map");
335  return 0;
336  }
337  ++thread_map_count_;
338 
339  // only insert the thread into the work-items queue if it is in
340  // pending state
341  if (state == pending) {
342  // pushing the new thread into the pending queue of the
343  // specified thread_queue
344  ++added;
345  schedule_thread(thrd.get());
346  }
347 
348  // this thread has to be in the map now
349  HPX_ASSERT(thread_map_.find(thrd.get()) != thread_map_.end());
350  HPX_ASSERT(thrd->get_pool() == &memory_pool_);
351  }
352 
353  if (added) {
354  LTM_(debug) << "add_new: added " << added << " tasks to queues"; //-V128
355  }
356  return added;
357  }

360  bool add_new_always(std::size_t& added, thread_queue* addfrom,
361  std::unique_lock<mutex_type> &lk, bool steal = false)
362  {
363  HPX_ASSERT(lk.owns_lock());
364 
365 #ifdef HPX_HAVE_THREAD_CREATION_AND_CLEANUP_RATES
366  util::tick_counter tc(add_new_time_);
367 #endif
368 
369  // create new threads from pending tasks (if appropriate)
370  std::int64_t add_count = -1; // default is no constraint
371 
372  // if we are desperate (no work in the queues), add some even if the
373  // map holds more than max_count
374  if (HPX_LIKELY(max_count_)) {
375  std::size_t count = thread_map_.size();
376  if (max_count_ >= count + min_add_new_count) { //-V104
377  HPX_ASSERT(max_count_ - count <
378  static_cast<std::size_t>(
379  (std::numeric_limits<std::int64_t>::max)()
380  ));
381  add_count = static_cast<std::int64_t>(max_count_ - count);
382  if (add_count < min_add_new_count)
383  add_count = min_add_new_count;
384  if (add_count > max_add_new_count)
385  add_count = max_add_new_count;
386  }
387  else if (work_items_.empty()) {
388  add_count = min_add_new_count; // add this number of threads
389  max_count_ += min_add_new_count; // increase max_count //-V101
390  }
391  else {
392  return false;
393  }
394  }
395 
396  std::size_t addednew = add_new(add_count, addfrom, lk, steal);
397  added += addednew;
398  return addednew != 0;
399  }

401  void recycle_thread(thread_id_type thrd)
402  {
403  std::ptrdiff_t stacksize = thrd->get_stack_size();
404 
405  if (stacksize == get_stack_size(thread_stacksize_small))
406  {
407  thread_heap_small_.push_front(thrd);
408  }
409  else if (stacksize == get_stack_size(thread_stacksize_medium))
410  {
411  thread_heap_medium_.push_front(thrd);
412  }
413  else if (stacksize == get_stack_size(thread_stacksize_large))
414  {
415  thread_heap_large_.push_front(thrd);
416  }
417  else if (stacksize == get_stack_size(thread_stacksize_huge))
418  {
419  thread_heap_huge_.push_front(thrd);
420  }
421  else
422  {
423  switch(stacksize) {
424  case thread_stacksize_small:
425  thread_heap_small_.push_front(thrd);
426  break;
427 
428  case thread_stacksize_medium:
429  thread_heap_medium_.push_front(thrd);
430  break;
431 
432  case thread_stacksize_large:
433  thread_heap_large_.push_front(thrd);
434  break;
435 
436  case thread_stacksize_huge:
437  thread_heap_huge_.push_front(thrd);

453  bool cleanup_terminated_locked_helper(bool delete_all = false)
454  {
455 #ifdef HPX_HAVE_THREAD_CREATION_AND_CLEANUP_RATES
456  util::tick_counter tc(cleanup_terminated_time_);
457 #endif
458 
459  if (terminated_items_count_ == 0 && thread_map_.empty())
460  return true;
461 
462  if (delete_all) {
463  // delete all threads
464  thread_data* todelete;
465  while (terminated_items_.pop(todelete))
466  {
467  --terminated_items_count_;
468 
469  // this thread has to be in this map
470  HPX_ASSERT(thread_map_.find(todelete) != thread_map_.end());
471 
472  bool deleted = thread_map_.erase(todelete) != 0;
473  HPX_ASSERT(deleted);
474  if (deleted) {
475  --thread_map_count_;
476  HPX_ASSERT(thread_map_count_ >= 0);
477  }
478  }
479  }
480  else {
481  // delete only this many threads
482  std::int64_t delete_count =
483  (std::max)(
484  static_cast<std::int64_t>(terminated_items_count_ / 10),
485  static_cast<std::int64_t>(max_delete_count));
486 
487  thread_data* todelete;
488  while (delete_count && terminated_items_.pop(todelete))
489  {
490  --terminated_items_count_;
491 
492  thread_map_type::iterator it = thread_map_.find(todelete);
493 
494  // this thread has to be in this map
495  HPX_ASSERT(it != thread_map_.end());
496 
497  recycle_thread(*it);
498 
499  thread_map_.erase(it);
500  --thread_map_count_;
501  HPX_ASSERT(thread_map_count_ >= 0);
502 
503  --delete_count;
504  }
505  }
506  return terminated_items_count_ == 0;
507  }

1007  inline bool wait_or_add_new(bool running,
1008  std::int64_t& idle_loop_count, std::size_t& added,
1009  thread_queue* addfrom = nullptr, bool steal = false) HPX_HOT
1010  {
1011  // try to generate new threads from task lists, but only if our
1012  // own list of threads is empty
1013  if (0 == work_items_count_.load(std::memory_order_relaxed))
1014  {
1015  // see if we can avoid grabbing the lock below
1016  if (addfrom)
1017  {
1018  // don't try to steal if there are only a few tasks left on
1019  // this queue
1020  if (running && min_tasks_to_steal_staged >
1021  addfrom->new_tasks_count_.load(std::memory_order_relaxed))
1022  {
1023  return false;
1024  }
1025  }
1026  else
1027  {
1028  if (running &&
1029  0 == new_tasks_count_.load(std::memory_order_relaxed))
1030  {
1031  return false;
1032  }
1033  addfrom = this;
1034  }
1035 
1036  // No obvious work has to be done, so a lock won't hurt too much.
1037  //
1038  // We prefer to exit this function (some kind of very short
1039  // busy waiting) to blocking on this lock. Locking fails either
1040  // when a thread is currently doing thread maintenance, which
1041  // means there might be new work, or the thread owning the lock
1042  // just falls through to the cleanup work below (no work is available)
1043  // in which case the current thread (which failed to acquire
1044  // the lock) will just retry to enter this loop.
1045  std::unique_lock<mutex_type> lk(mtx_, std::try_to_lock);
1046  if (!lk.owns_lock())
1047  return false; // avoid long wait on lock
1048 
1049  // stop running after all HPX threads have been terminated
1050  bool added_new = add_new_always(added, addfrom, lk, steal);
1051  if (!added_new) {
1052  // Before exiting each of the OS threads deletes the
1053  // remaining terminated HPX threads
1054  // REVIEW: Should we be doing this if we are stealing?
1055  bool canexit = cleanup_terminated_locked(true);
1056  if (!running && canexit) {
1057  // we don't have any registered work items anymore
1058  //do_some_work(); // notify possibly waiting threads
1059  return true; // terminate scheduling loop
1060  }
1061  return false;
1062  }
1063 
1064  cleanup_terminated_locked();
1065  }
1066  return false;
1067  }

79 hpx::future<std::uint64_t> fibonacci_future(std::uint64_t n)
80 {
81  // if we know the answer, we return a future encapsulating the final value
82  if (n < 2)
83  return hpx::make_ready_future(n);
84  if (n < threshold)
85  return hpx::make_ready_future(fibonacci_serial(n));
86 
87  fibonacci_future_action fib;
88  hpx::id_type loc1 = here;
89  hpx::id_type loc2 = here;
90 
91  if (n == distribute_at) {
92  loc2 = get_next_locality(++next_locality);
93  }
94  else if (n-1 == distribute_at) {
95  std::uint64_t next = next_locality += 2;
96  loc1 = get_next_locality(next-1);
97  loc2 = get_next_locality(next);
98  }
99 
100  hpx::future<std::uint64_t> f = hpx::async(fib, loc1, n-1);
101  hpx::future<std::uint64_t> r = fib(loc2, n-2);
102 
103  return hpx::when_all(f, r).then(when_all_wrapper());
104 }

107 int hpx_main(boost::program_options::variables_map& vm)
108 {
109  // extract command line argument, i.e. fib(N)
110  std::uint64_t n = vm["n-value"].as<std::uint64_t>();
111  std::string test = vm["test"].as<std::string>();
112  std::uint64_t max_runs = vm["n-runs"].as<std::uint64_t>();
113 
114  if (max_runs == 0) {
115  std::cerr << "fibonacci_futures_distributed: wrong command "
116  "line argument value for "
117  "option 'n-runs', should not be zero" << std::endl;
118  return hpx::finalize(); // Handles HPX shutdown
119  }
120 
121  bool executed_one = false;
122  std::uint64_t r = 0;
123 
124  if (test == "all" || test == "0")
125  {
126  // Keep track of the time required to execute.
127  std::uint64_t start = hpx::util::high_resolution_clock::now();
128 
129  // Synchronous execution, use as reference only.
130  r = fibonacci_serial(n);
131 
132 // double d = double(hpx::util::high_resolution_clock::now() - start) / 1.e9;
133  std::uint64_t d = hpx::util::high_resolution_clock::now() - start;
134  char const* fmt = "fibonacci_serial(%1%) == %2%,"
135  "elapsed time:,%3%,[s]\n";
136  std::cout << (boost::format(fmt) % n % r % d);
137 
138  executed_one = true;
139  }
140 
141  if (test == "all" || test == "1")
142  {
143  // Keep track of the time required to execute.
144  std::uint64_t start = hpx::util::high_resolution_clock::now();
145 
146  for (std::size_t i = 0; i != max_runs; ++i)
147  {
148  // Create a Future for the whole calculation and wait for it.
149  next_locality.store(0);
150  r = fibonacci_future(n).get();
151  }
152 
153 // double d = double(hpx::util::high_resolution_clock::now() - start) / 1.e9;
154  std::uint64_t d = hpx::util::high_resolution_clock::now() - start;
155  char const* fmt = "fibonacci_future(%1%) == %2%,elapsed time:,%3%,[s],%4%\n";
156  std::cout << (boost::format(fmt) % n % r % (d / max_runs)
157  % next_locality.load());
158 
159  get_serial_execution_count_action serial_count;
160  for (hpx::id_type const& loc : hpx::find_all_localities())
161  {
162  std::size_t count = serial_count(loc);
163  std::cout << (boost::format(" serial-count,%1%,%2%\n") %
164  loc % (count / max_runs));
165  }
166 
167  executed_one = true;
168  }
169 
170  if (!executed_one)
171  {
172  std::cerr << "fibonacci_futures_distributed: wrong command line argument "
173  "value for option 'tests', should be either 'all' or a number between "
174  "zero and 1, value specified: " << test << std::endl;
175  }
176 
177  return hpx::finalize(); // Handles HPX shutdown
178 }

206 void init_globals()
207 {
208  // Retrieve command line using the Boost.ProgramOptions library.
209  boost::program_options::variables_map vm;
210  if (!hpx::util::retrieve_commandline_arguments(get_commandline_options(), vm))
211  {
212  HPX_THROW_EXCEPTION(hpx::commandline_option_error,
213  "fibonacci_futures_distributed",

215  return;
216  }
217 
218  std::uint64_t n = vm["n-value"].as<std::uint64_t>();
219 
220  threshold = vm["threshold"].as<unsigned int>();
221  if (threshold < 2 || threshold > n) {
222  HPX_THROW_EXCEPTION(hpx::commandline_option_error,
223  "fibonacci_futures_distributed",
224  "wrong command line argument value for option 'threshold', "
225  "should be in between 2 and n-value, value specified: " +
226  std::to_string(threshold));
227  return;
228  }
229 
230  distribute_at = vm["distribute-at"].as<unsigned int>();
231  if (distribute_at < 2 || distribute_at > n) {
232  HPX_THROW_EXCEPTION(hpx::commandline_option_error,
233  "fibonacci_futures_distributed",
234  "wrong command line argument value for option 'distribute-at', "
235  "should be in between 2 and n-value, value specified: " +
236  std::to_string(distribute_at));
237  return;
238  }
239 
240  here = hpx::find_here();
241  next_locality.store(0);
242  serial_execution_count.store(0);
243 
244  // try to more evenly distribute the work over the participating localities
245  std::vector<hpx::id_type> locs = hpx::find_all_localities();
246  std::size_t num_repeats = vm["loc-repeat"].as<int>();
247 
248  localities.push_back(here); // add ourselves
249  for (std::size_t j = 0; j != num_repeats; ++j)
250  {
251  for (std::size_t i = 0; i != locs.size(); ++i)
252  {
253  if (here == locs[i])
254  continue;
255  localities.push_back(locs[i]);
256  }
257  }
258 }

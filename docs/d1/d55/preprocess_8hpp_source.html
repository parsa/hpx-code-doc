47  void trigger()
48  {
49  // hpx::lcos::local::promise<void>::set_value() might need to acquire
50  // a lock, as such, we check the our triggering condition inside a
51  // critical section and trigger the promise outside of it.
52  bool set_value = false;
53  {
54  std::lock_guard<mutex_type> l(mtx_);
55  ++triggered_futures_;
56  set_value = (done_ && num_futures_ == triggered_futures_);
57  }
58  if(set_value)
59  {
60  promise_.set_value();
61  }
62  }
63 
64  void await_future(hpx::lcos::detail::future_data_refcnt_base & future_data)
65  {
66  {
67  std::lock_guard<mutex_type> l(mtx_);
68  ++num_futures_;
69  }
70  future_data.set_on_completed(
71  [this]()
72  {
73  this->trigger();
74  }
75  );
76  }

78  void add_gid(
79  naming::gid_type const & gid,
80  naming::gid_type const & split_gid)
81  {
82  std::lock_guard<mutex_type> l(mtx_);
83  HPX_ASSERT(split_gids_[&gid] == naming::invalid_gid);
84  split_gids_[&gid] = split_gid;
85  }

87  bool has_gid(naming::gid_type const & gid)
88  {
89  std::lock_guard<mutex_type> l(mtx_);
90  return split_gids_.find(&gid) != split_gids_.end();
91  }
92 
93  void reset()
94  {
95  size_ = 0;
96  done_ = false;
97  num_futures_ = 0;
98  triggered_futures_ = 0;
99  promise_ = hpx::lcos::local::promise<void>();
100  }
101 
102  bool has_futures()
103  {
104  if(num_futures_ == 0)
105  {
106  promise_.set_value();
107  }
108  return num_futures_ > 0;
109  }
110 
111  template <typename F>
112  void operator()(F f)
113  {
114  bool set_promise = false;
115  {
116  std::lock_guard<mutex_type> l(mtx_);
117  done_ = true;
118  if (num_futures_ == triggered_futures_)
119  set_promise = true;
120  }
121 
122  if (set_promise)
123  promise_.set_value();
124 
125  hpx::future<void> fut = promise_.get_future();
126  // we don't call f directly to avoid possible stack overflow.
127  auto shared_state_ = hpx::traits::future_access<hpx::future<void> >::
128  get_shared_state(fut);
129  shared_state_->set_on_completed(std::move(f));
130  }

107 void test_receive_buffer_server::do_work()
108 {
109  send(0, 0); // send initial value
110 
111  std::vector<hpx::future<std::size_t> > steps;
112  steps.reserve(MAX_ITERATIONS);
113 
114  for (std::size_t i = 0; i != MAX_ITERATIONS; ++i)
115  {
116  hpx::future<std::size_t> f = receive(i);
117  steps.push_back(
118  f.then(
119  [this, i](hpx::future<std::size_t> && f) -> std::size_t
120  {
121  std::size_t val = f.get();
122  send(i + 1, val + 1);
123  return val;
124  })
125  );
126  }
127 
128  // receive final value

130 
131  // verify received values
132  hpx::wait_all(steps);
133  for (std::size_t i = 0; i != MAX_ITERATIONS; ++i)
134  {
135  HPX_TEST_EQ(steps[i].get(), i);
136  }

15 //
16 // This is a simple example which generates random numbers and returns
17 // pass or fail from a routine.
18 // When called by many threads returning a vector of futures - if the user wants to
19 // reduce the vector of pass/fails into a single pass fail based on a simple
20 // any fail = !pass rule, then this example shows how to do it.
21 // The user can experiment with the failure rate to see if the statistics match
22 // their expectations.
23 // Also. Routine can use either a lambda, or a function under control of USE_LAMBDA
24 
25 #define TEST_SUCCESS 1
26 #define TEST_FAIL 0
27 //
28 #define FAILURE_RATE_PERCENT 5
29 #define SAMPLES_PER_LOOP 10
30 #define TEST_LOOPS 1000
31 //
32 std::random_device rseed;
33 std::mt19937 gen(rseed());
34 std::uniform_int_distribution<int> dist(0,99); // interval [0,100)
35 
36 #define USE_LAMBDA
37 
38 //----------------------------------------------------------------------------
39 int reduce(hpx::future<std::vector<hpx::future<int> > > &&futvec)
40 {
41  int res = TEST_SUCCESS;
42  std::vector<hpx::future<int> > vfs = futvec.get();
43  for (hpx::future<int>& f: vfs) {
44  if (f.get() == TEST_FAIL) return TEST_FAIL;
45  }
46  return res;
47 }

49 //----------------------------------------------------------------------------
50 int generate_one()
51 {
52  // generate roughly x% fails
53  int result = TEST_SUCCESS;
54  if (dist(gen)>=(100-FAILURE_RATE_PERCENT)) {
55  result = TEST_FAIL;
56  }
57  return result;
58 }

60 //----------------------------------------------------------------------------
61 hpx::future<int> test_reduce()
62 {
63  std::vector<hpx::future<int> > req_futures;
64  //
65  for (int i=0; i<SAMPLES_PER_LOOP; i++) {
66  // generate random sequence of pass/fails using % fail rate per incident
67  hpx::future<int> result = hpx::async(generate_one);
68  req_futures.push_back(std::move(result));
69  }
70 
71  hpx::future<std::vector<hpx::future<int> > > all_ready = hpx::when_all(req_futures);
72 
73 #ifdef USE_LAMBDA
74  hpx::future<int> result = all_ready.then(
75  [](hpx::future<std::vector<hpx::future<int> > > &&futvec) -> int {
76  // futvec is ready or the lambda would not be called
77  std::vector<hpx::future<int> > vfs = futvec.get();
78  // all futures in v are ready as fut is ready
79  int res = TEST_SUCCESS;
80  for (hpx::future<int>& f: vfs) {
81  if (f.get() == TEST_FAIL) return TEST_FAIL;
82  }
83  return res;
84  });
85 #else
86  hpx::future<int> result = all_ready.then(reduce);
87 #endif
88  //
89  return result;
90 }

93 int hpx_main()
94 {
95  hpx::util::high_resolution_timer htimer;
96  // run N times and see if we get approximately the right amount of fails
97  int count = 0;
98  for (int i=0; i<TEST_LOOPS; i++) {
99  int result = test_reduce().get();
100  count += result;
101  }
102  double pr_pass = std::pow(1.0 - FAILURE_RATE_PERCENT/100.0, SAMPLES_PER_LOOP);
103  double exp_pass = TEST_LOOPS*pr_pass;
104  hpx::cout << "From " << TEST_LOOPS << " tests, we got "
105  << "\n " << count << " passes"
106  << "\n " << exp_pass << " expected \n"
107  << "\n " << htimer.elapsed() << " seconds \n" << hpx::flush;
108  // Initiate shutdown of the runtime system.
109  return hpx::finalize();
110 }

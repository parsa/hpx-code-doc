20 // A custom allocator which takes a pointer in its constructor and then returns
21 // this pointer in response to any allocate request. It is here to try to fool
22 // the hpx serialization into copying directly into a user provided buffer
23 // without copying from a result into another buffer.
24 template <typename T>
25 class pointer_allocator
26 {
27 public:
28  typedef T value_type;
29  typedef T* pointer;
30  typedef const T* const_pointer;
31  typedef T& reference;
32  typedef const T& const_reference;
33  typedef std::size_t size_type;
34  typedef std::ptrdiff_t difference_type;
35 
36  pointer_allocator() noexcept
37  : pointer_(nullptr), size_(0)
38  {
39  }
40 
41  pointer_allocator(pointer p, size_type size) noexcept
42  : pointer_(p), size_(size)
43  {
44  }
45 
46  pointer address(reference value) const { return &value; }
47  const_pointer address(const_reference value) const { return &value; }
48 
49  pointer allocate(size_type n, void const* hint = nullptr)
50  {
51  HPX_ASSERT(n == size_);
52  return static_cast<T*>(pointer_);
53  }
54 
55  void deallocate(pointer p, size_type n)
56  {
57  HPX_ASSERT(p == pointer_ && n == size_);
58  }
59 
60 private:
61  // serialization support
62  friend class hpx::serialization::access;
63 
64  template <typename Archive>
65  void load(Archive& ar, unsigned int const version)
66  {
67  std::size_t t = 0;
68  ar >> size_ >> t;
69  pointer_ = reinterpret_cast<pointer>(t);
70  }
71 
72  template <typename Archive>
73  void save(Archive& ar, unsigned int const version) const
74  {
75  std::size_t t = reinterpret_cast<std::size_t>(pointer_);
76  ar << size_ << t;
77  }
78 
79  HPX_SERIALIZATION_SPLIT_MEMBER()
80 
81 private:
82  pointer pointer_;
83  size_type size_;
84 };

113  // Retrieve an array of doubles to the given address
114  transfer_buffer_type get_here(std::size_t size, std::size_t remote_buffer)
115  {
116  pointer_allocator<double> allocator(
117  reinterpret_cast<double*>(remote_buffer), size);
118 
119  // lock the mutex, will be unlocked by the transfer buffer's deleter
120  mtx_.lock();
121 
122  // we use our data directly without copying
123  return transfer_buffer_type(data_.data(), size,
124  transfer_buffer_type::reference,
125  hpx::util::bind(&zerocopy_server::release_lock, this),
126  allocator);
127  }
128  HPX_DEFINE_COMPONENT_ACTION(zerocopy_server, get_here, get_here_action);
129 
131  // Retrieve an array of doubles
132  general_buffer_type get(std::size_t size)
133  {
134  // lock the mutex, will be unlocked by the transfer buffer's deleter
135  mtx_.lock();

139  general_buffer_type::reference,
140  hpx::util::bind(&zerocopy_server::release_lock, this));
141  }

161 struct zerocopy
162  : hpx::components::client_base<

164 {
165 private:
166  // Copy he data once into the destination buffer if the get() operation was
167  // entirely local (no data copies have been made so far).
168  static void transfer_data(general_buffer_type recv,
169  hpx::future<transfer_buffer_type> f)
170  {
171  transfer_buffer_type buffer(f.get());
172  if (buffer.data() != recv.data())
173  {
174  std::copy(buffer.data(), buffer.data()+buffer.size(), recv.data());
175  }
176  }

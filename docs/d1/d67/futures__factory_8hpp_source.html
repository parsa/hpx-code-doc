109  {
110  this->check_started();
111 
112  typedef typename Base::future_base_type future_base_type;
113  future_base_type this_(this);
114 
115  if (this->sched_) {

329  {}

331  futures_factory(futures_factory&& rhs)
332  : task_(std::move(rhs.task_)),
333  future_obtained_(rhs.future_obtained_)
334  {
335  rhs.task_.reset();
336  rhs.future_obtained_ = false;
337  }
338 
339  futures_factory& operator=(futures_factory&& rhs)
340  {
341  if (this != &rhs) {
342  task_ = std::move(rhs.task_);
343  future_obtained_ = rhs.future_obtained_;
344 
345  rhs.task_.reset();
346  rhs.future_obtained_ = false;
347  }
348  return *this;
349  }
350 
351  // synchronous execution
352  void operator()() const
353  {
354  if (!task_) {
355  HPX_THROW_EXCEPTION(task_moved,
356  "futures_factory<Result()>::operator()",
357  "futures_factory invalid (has it been moved?)");
358  return;
359  }
360  task_->run();
361  }
362 
363  // asynchronous execution
364  threads::thread_id_type apply(
365  launch policy = launch::async,
366  threads::thread_priority priority = threads::thread_priority_default,
367  threads::thread_stacksize stacksize = threads::thread_stacksize_default,
368  error_code& ec = throws) const
369  {
370  if (!task_) {
371  HPX_THROW_EXCEPTION(task_moved,
372  "futures_factory<Result()>::apply()",
373  "futures_factory invalid (has it been moved?)");
374  return threads::invalid_thread_id;
375  }
376  return task_->apply(policy, priority, stacksize, ec);
377  }
378 
379  // This is the same as get_future, except that it moves the
380  // shared state into the returned future.
381  lcos::future<Result> get_future(error_code& ec = throws)
382  {
383  if (!task_) {
384  HPX_THROWS_IF(ec, task_moved,
385  "futures_factory<Result()>::get_future",
386  "futures_factory invalid (has it been moved?)");
387  return lcos::future<Result>();
388  }
389  if (future_obtained_) {
390  HPX_THROWS_IF(ec, future_already_retrieved,
391  "futures_factory<Result()>::get_future",
392  "future already has been retrieved from this factory");
393  return lcos::future<Result>();
394  }
395 
396  future_obtained_ = true;
397 
398  using traits::future_access;
399  return future_access<future<Result> >::create(std::move(task_));
400  }
401 
402  bool valid() const noexcept
403  {
404  return !!task_;
405  }
406 
407  void set_exception(std::exception_ptr const& e)
408  {
409  if (!task_) {
410  HPX_THROW_EXCEPTION(task_moved,
411  "futures_factory<Result()>::set_exception",
412  "futures_factory invalid (has it been moved?)");
413  return;
414  }
415  task_->set_exception(e);
416  }
417 
418  protected:
419  boost::intrusive_ptr<task_impl_type> task_;
420  bool future_obtained_;
421  };

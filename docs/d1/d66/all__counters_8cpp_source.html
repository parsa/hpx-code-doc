85 void test_all_locality_thread_counters(char const* const* counter_names,
86  std::size_t locality_id, std::size_t pool, std::size_t core)
87 {
88  for (char const* const*p = counter_names; *p != nullptr; ++p)
89  {
90  // split counter type into counter path elements
91  hpx::performance_counters::counter_path_elements path;
92  HPX_TEST_EQ(
93  hpx::performance_counters::get_counter_path_elements(*p, path),
94  hpx::performance_counters::status_valid_data);
95 
96  // augment the counter path elements
97  path.parentinstancename_ = "locality";
98  path.parentinstanceindex_ = locality_id;
99  if (pool != std::size_t(-1))
100  {
101  path.instancename_ = "pool";
102  path.instanceindex_ = pool;
103 
104  if (core != std::size_t(-1))
105  {
106  path.subinstancename_ = "worker-thread";
107  path.subinstanceindex_ = core;
108  }
109  else
110  {
111  path.subinstancename_ = "total";
112  path.subinstanceindex_ = std::size_t(-1);
113  }
114  }
115  else if (core != std::size_t(-1))
116  {
117  path.instancename_ = "worker-thread";
118  path.instanceindex_ = core;
119  }
120  else
121  {
122  path.instancename_ = "total";
123  path.instanceindex_ = std::size_t(-1);
124  }
125 
126  std::string name;
127  HPX_TEST_EQ(
128  hpx::performance_counters::get_counter_name(path, name),
129  hpx::performance_counters::status_valid_data);
130 
131  std::cout << name << '\n';
132 
133  try
134  {
135  hpx::performance_counters::performance_counter counter(name);
136  HPX_TEST_EQ(counter.get_name(hpx::launch::sync), name);
137  counter.get_value<std::size_t>(hpx::launch::sync);
138  }
139  catch (...)
140  {
141  HPX_TEST(false); // should never happen
142  }
143  }
144 }

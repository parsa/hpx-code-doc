77  {

80  if (std::size_t(N) <= sort_limit_per_task)
81  {
82  return execution::async_execute(

89  }

92  // split
93  //----------------------------------------------------------------
94 
95  //------------------- check if sorted ----------------------------
96  if (detail::is_sorted_sequential(first, last, comp))
97  return hpx::make_ready_future(last);
98 
99  //---------------------- pivot select ----------------------------
100  std::size_t nx = std::size_t(N) >> 1;
101 
102  RandomIt it_a = first + 1;
103  RandomIt it_b = first + nx;
104  RandomIt it_c = last - 1;
105 
106  if (comp(*it_b, *it_a))
107  std::iter_swap(it_a, it_b);
108 
109  if (comp(*it_c, *it_b))
110  {
111  std::iter_swap(it_c, it_b);
112  if (comp(*it_b, *it_a))
113  std::iter_swap(it_a, it_b);
114  }
115 
116  std::iter_swap(first, it_b);
117 
118  typedef
119  typename std::iterator_traits<RandomIt>::reference
120  reference;
121 
122  reference val = *first;
123  RandomIt c_first = first + 2, c_last = last - 2;
124 
125  while (c_first != last && comp(*c_first, val))
126  ++c_first;
127  while (comp(val, *c_last))
128  --c_last;
129  while (!(c_first > c_last))
130  {
131  std::iter_swap(c_first++, c_last--);
132  while (comp(*c_first, val))
133  ++c_first;
134  while (comp(val, *c_last))
135  --c_last;
136  } // End while
137 
138  std::iter_swap(first, c_last);
139 
140  // spawn tasks for each sub section
141  hpx::future<RandomIt> left =
142  execution::async_execute(
143  policy.executor(),
144  &sort_thread<ExPolicy, RandomIt, Compare>,
145  std::ref(policy), first, c_last, comp);

181  template <typename ExPolicy, typename RandomIt, typename Compare>
182  hpx::future<RandomIt>
183  parallel_sort_async(ExPolicy && policy, RandomIt first, RandomIt last,
184  Compare comp)
185  {
186  hpx::future<RandomIt> result;
187  try {
188  std::ptrdiff_t N = last - first;
189  HPX_ASSERT(N >= 0);
190 
191  if (std::size_t(N) < sort_limit_per_task)
192  {
193  std::sort(first, last, comp);
194  return hpx::make_ready_future(last);
195  }
196 
197  // check if already sorted
198  if (detail::is_sorted_sequential(first, last, comp))
199  return hpx::make_ready_future(last);
200 
201  result = execution::async_execute(policy.executor(),
202  &sort_thread<ExPolicy, RandomIt, Compare>,
203  std::ref(policy), first, last, comp);
204  }
205  catch (...) {
206  return detail::handle_exception<ExPolicy, RandomIt>::call(
207  std::current_exception());
208  }
209 
210  if (result.has_exception())
211  {
212  return detail::handle_exception<ExPolicy, RandomIt>::call(
213  std::move(result));
214  }
215

77 sierpinski get_sierpinski(std::uint64_t n, double len)
78 {
79  if (n == 0)
80  return sierpinski(1, 0, sqrt(3)/4.0*len*len);
81 
82  hpx::naming::id_type const locality_id = hpx::find_here();
83 
84  // The problem for iteration n is broken down into three sub-problems,
85  // each of size n-1 (and side length gets halved). It is very inefficient
86  // as it does the same calculations
87  // three times but this is the method used for the sake of using HPX.
88 
89  get_sierpinski_action s;
90  hpx::future<sierpinski> n1 = hpx::async(s, locality_id, n - 1, len/2);
91  hpx::future<sierpinski> n2 = hpx::async(s, locality_id, n - 1, len/2);
92  hpx::future<sierpinski> n3 = hpx::async(s, locality_id, n - 1, len/2);
93 
94  sierpinski ans = n1.get() + n2.get() + n3.get();
95  ans.white_triangles++;
96  return ans;
97 }

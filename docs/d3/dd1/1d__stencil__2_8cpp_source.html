8 // This is the second in a series of examples demonstrating the development of
9 // a fully distributed solver for a simple 1D heat distribution problem.
10 //
11 // This example shows how futurization can be applied to the code from example
12 // one. While this nicely parallelizes the code (note: without changing the
13 // overall structure of the algorithm), the achieved performance is bad (a lot
14 // slower than example one). This is caused by the large amount of overheads
15 // introduced by wrapping each and every grid point into its own future object.
16 // The amount of work performed by each of the created HPX threads (one thread
17 // for every grid point and time step) is too small compared to the imposed
18 // overheads.

37 inline std::size_t idx(std::size_t i, int dir, std::size_t size)
38 {
39  if(i == 0 && dir == -1)
40  return size-1;
41  if(i == size-1 && dir == +1)
42  return 0;
43 
44  HPX_ASSERT((i + dir) < size);
45 
46  return i + dir;
47 }

50 //[stepper_2
51 struct stepper
52 {
53  // Our partition type
54  typedef hpx::shared_future<double> partition;
55 
56  // Our data for one time step
57  typedef std::vector<partition> space;
58 
59  // Our operator
60  static double heat(double left, double middle, double right)
61  {
62  return middle + (k*dt/(dx*dx)) * (left - 2*middle + right);
63  }
64 
65  // do all the work on 'nx' data points for 'nt' time steps
66  hpx::future<space> do_work(std::size_t nx, std::size_t nt)
67  {
68  using hpx::dataflow;
69  using hpx::util::unwrapping;
70 
71  // U[t][i] is the state of position i at time t.
72  std::vector<space> U(2);
73  for (space& s : U)
74  s.resize(nx);
75 
76  // Initial conditions: f(0, i) = i
77  for (std::size_t i = 0; i != nx; ++i)
78  U[0][i] = hpx::make_ready_future(double(i));
79 
80  auto Op = unwrapping(&stepper::heat);
81 
82  // Actual time step loop
83  for (std::size_t t = 0; t != nt; ++t)
84  {
85  space const& current = U[t % 2];
86  space& next = U[(t + 1) % 2];
87 
88  // WHEN U[t][i-1], U[t][i], and U[t][i+1] have been computed, THEN we
89  // can compute U[t+1][i]
90  for (std::size_t i = 0; i != nx; ++i)
91  {
92  next[i] = dataflow(
93  hpx::launch::async, Op,
94  current[idx(i, -1, nx)], current[i], current[idx(i, +1, nx)]
95  );
96  }
97  }
98 
99  // Now the asynchronous computation is running; the above for-loop does not
100  // wait on anything. There is no implicit waiting at the end of each timestep;
101  // the computation of each U[t][i] will begin when as soon as its dependencies
102  // are ready and hardware is available.
103 
104  // Return the solution at time-step 'nt'.
105  return hpx::when_all(U[nt % 2]);
106  }
107 };
108 //]
110 int hpx_main(boost::program_options::variables_map& vm)
111 {
112  std::uint64_t nx = vm["nx"].as<std::uint64_t>(); // Number of grid points.
113  std::uint64_t nt = vm["nt"].as<std::uint64_t>(); // Number of steps.
114 
115  if (vm.count("no-header"))
116  header = false;
117 
118  // Create the stepper object
119  stepper step;
120 
121  // Measure execution time.
122  std::uint64_t t = hpx::util::high_resolution_clock::now();
123 
124  // Execute nt time steps on nx grid points.
125  hpx::future<stepper::space> result = step.do_work(nx, nt);
126 
127  stepper::space solution = result.get();
128  hpx::wait_all(solution);
129 
130  std::uint64_t elapsed = hpx::util::high_resolution_clock::now() - t;
131 
132  // Print the final solution
133  if (vm.count("results"))
134  {
135  for (std::size_t i = 0; i != nx; ++i)
136  std::cout << "U[" << i << "] = " << solution[i].get() << std::endl;
137  }
138 
139  std::uint64_t const os_thread_count = hpx::get_os_thread_count();
140  print_time_results(os_thread_count, elapsed, nx, nt, header);
141 
142  return hpx::finalize();
143 }

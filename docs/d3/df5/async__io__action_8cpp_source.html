6 // The purpose of this example is to demonstrate how it is possible to
7 // generically schedule an asynchronous IO task onto one of the IO-threads
8 // in HPX (which are OS-threads), how to synchronize the result of this IO
9 // task with a waiting HPX thread, and how to wrap all of this into a HPX
10 // action.

36  // This function will be executed by an HPX thread
37  hpx::lcos::future<int> async_io_worker(std::string const& string_to_write)
38  {
39  std::shared_ptr<hpx::lcos::local::promise<int> > p =
40  std::make_shared<hpx::lcos::local::promise<int> >();
41 
42  // Get a reference to one of the IO specific HPX io_service objects ...
43  hpx::threads::executors::io_pool_executor scheduler;
44 
45  // ... and schedule the handler to run on one of its OS-threads.
46  scheduler.add(hpx::util::bind(&do_async_io, string_to_write, p));
47 
48  return p->get_future();
49  }
50 }

52 // This function will be called whenever the action async_io_action is
53 // invoked. This allows to remotely execute the async_io.
54 int async_io(std::string const& string_to_write)
55 {
56  hpx::lcos::future<int> f = detail::async_io_worker(string_to_write);
57  return f.get(); // simply wait for the IO to finish
58 }

63 int hpx_main()
64 {
65  {
66  // Determine on what locality to run the IO operation. If this
67  // example is run on one locality, we perform the IO operation
68  // locally, otherwise we choose one of the remote localities to
69  // invoke the async_io_action.
70  hpx::naming::id_type id = hpx::find_here(); // default: local
71 
72  std::vector<hpx::naming::id_type> localities =
73  hpx::find_remote_localities();
74  if (!localities.empty())
75  id = localities[0]; // choose the first remote locality
76 
77  // Create an action instance.
78  async_io_action io;
79 
80  // Initiate an asynchronous (possibly remote) IO operation and
81  // wait for it to complete without blocking any of the HPX
82  // thread-manager threads. This will suspend the current HPX
83  // thread until the IO operation is finished. Other work could
84  // be executed in the meantime.
85  int result = io(id, "Write this string to std::cout");
86 
87  // Print the returned result.
88  hpx::cout << "HPX-thread: The asynchronous IO operation returned: "
89  << result << "\n" << hpx::flush;
90  }
91 
92  return hpx::finalize(); // Initiate shutdown of the runtime system.
93 }

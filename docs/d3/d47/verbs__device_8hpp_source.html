38  // Search for the specified InfiniBand device.
39  device_ = nullptr;
40  for (int index = 0; index < numDevices; ++index) {
41  LOG_DEBUG_MSG("checking device " << ibv_get_device_name(devices_[index])
42  << " against " << device.c_str());
43  if (strcmp(ibv_get_device_name(devices_[index]), device.c_str()) == 0) {
44  device_ = devices_[index];
45  break;
46  }
47  }
48 
49  // See if the requested InfiniBand device was found.
50  if (device_ == nullptr) {
51  LOG_ERROR_MSG("InfiniBand device " << device << " was not found");
52  std::ostringstream what;
53  what << "InfiniBand device " << device << " not found";
54  rdma_error e(ENODEV, what.str());
55  throw e;
56  }
57  LOG_DEBUG_MSG("found InfiniBand device " << get_device_name());
58 
59  interface_ = nullptr;
60  interfaces_ = nullptr;
61 
62  if (!interface.empty()) {
63  // Get the list of network interfaces.
64  if (getifaddrs(&interfaces_) != 0) {
65  rdma_error e(errno, "getifaddrs() failed");
66  LOG_ERROR_MSG("error getting list of network interfaces: "
67  << rdma_error::error_string(e.error_code()));
68  throw e;
69  }
70 
71  // Search for the specified network interface.
72  for (interface_ = interfaces_; interface_ != nullptr;
73  interface_ = interface_->ifa_next)
74  {
75  // std::cout << "checking interface "
76  // << interface_->ifa_name << " against "
77  // << interface.c_str() << std::endl;
78  if ((strcmp(interface_->ifa_name, interface.c_str()) == 0)
79  && (interface_->ifa_addr->sa_family == AF_INET))
80  {
81  break;
82  }
83  }
84 
85  // See if the specified network interface was found.
86  if (interface_ == nullptr) {
87  LOG_ERROR_MSG("network interface "
88  << interface << " was not found on the node");
89  std::ostringstream what;
90  what << "network interface " << interface << " not found";
91  rdma_error e(ENOENT, what.str());
92  throw e;
93  }
94  LOG_DEBUG_MSG("found network interface " << get_interface_name());
95  }
96  }

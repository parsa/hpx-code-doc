427  // template <class... UTypes>
428  // explicit constexpr tuple(UTypes&&... u);
429  // Initializes the elements in the tuple with the corresponding value
430  // in std::forward<UTypes>(u).
431  // This constructor shall not participate in overload resolution
432  // unless each type in UTypes is implicitly convertible to its
433  // corresponding type in Types.
434  template <typename U, typename ...Us, typename Enable =
435  typename std::enable_if<
436  detail::pack<U, Us...>::size == detail::pack<Ts...>::size
437  && std::conditional<
438  detail::pack<Us...>::size == 0
439  , typename std::enable_if<
440  !std::is_same<tuple, typename std::decay<U>::type>::value
441  && !detail::are_tuples_compatible_not_same<tuple, U&&>::value
442  , detail::are_tuples_compatible<tuple, tuple<U>&&>
443  >::type
444  , detail::are_tuples_compatible<tuple, tuple<U, Us...>&&>
445  >::type::value
446  >::type>
447  explicit HPX_CONSTEXPR HPX_HOST_DEVICE tuple(U&& v, Us&&... vs)
448  : _impl(std::forward<U>(v), std::forward<Us>(vs)...)
449  {}

458  // For all i, initializes the ith element of *this with
459  // std::forward<Ti>(get<i>(u)).
460  tuple(tuple&&) = default;
461 #else

463  // Initializes each element of *this with the corresponding element

68  void console_logging_locked(naming::id_type const& prefix,
69  messages_type const& msgs)
70  {
71  // If we're not in an HPX thread, we cannot call apply as it may access
72  // the AGAS components. We just throw an exception here - there's no
73  // thread-manager, so the exception will probably be unhandled. This is
74  // desirable in this situation, as we can't trust the logging system to
75  // report this error.
76  if (HPX_UNLIKELY(!threads::get_self_ptr()))
77  {
78  // write the message to a local file in any case
79  fallback_console_logging_locked(msgs);
80 
81  // raise error as this should get called from outside a HPX-thread
82  HPX_THROW_EXCEPTION(null_thread_id,
83  "components::console_logging_locked",
84  "console_logging_locked was not called from a HPX-thread");
85  }
86 
87  try {
88  hpx::apply<server::console_logging_action<> >(prefix, msgs);
89  }
90  catch (hpx::exception const& e) {
91  // if this is not the console locality (or any other error occurs)
92  // we might be too late for any logging, write to local file
93  fallback_console_logging_locked(msgs, e.what());
94  }
95  }

104  void pending_logs::add(message_type const& msg)
105  {
106  if (nullptr == hpx::get_runtime_ptr()) {
107  // This branch will be taken if it's too early or too late in the
108  // game. We do local logging only. Any queued messages which may be
109  // still left in the queue are logged locally as well.
110 
111  // queue up the new message and log it with the rest of it
112  messages_type msgs;
113  {
114  std::lock_guard<queue_mutex_type> l(queue_mtx_);
115  queue_.push_back(msg);
116  queue_.swap(msgs);
117  }
118 
119  fallback_console_logging_locked(msgs);
120  }
121  else if (is_active()) {
122  // This branch will be taken under normal circumstances. We queue
123  // up the message and either log it immediately (if we are on the
124  // console locality) or defer logging until 'max_pending' messages
125  // have been queued. Note: we can invoke send only from within a
126  // HPX-thread.
127  std::size_t size = 0;
128 
129  {
130  std::lock_guard<queue_mutex_type> l(queue_mtx_);
131  queue_.push_back(msg);
132  size = queue_.size();
133  }
134 
135  // Invoke actual logging immediately if we're on the console or
136  // if the number of waiting log messages is too large.
137  if (naming::get_agas_client().is_console() || size > max_pending)
138  send();
139  }
140  else {
141  // This branch will be taken if the runtime is up and running, but
142  // either the thread manager is not active or this is not invoked
143  // on a HPX-thread.
144 
145  // Note: is_console can be called outside of a HPX-thread
146  if (!naming::get_agas_client().is_console())
147  {
148  // queue it for delivery to the console
149  std::lock_guard<queue_mutex_type> l(queue_mtx_);
150  queue_.push_back(msg);
151  }
152  else
153  {
154  // log it locally on the console
155  messages_type msgs;
156  msgs.push_back(msg);
157  fallback_console_logging_locked(msgs);
158  }
159  }
160  }

182  bool pending_logs::ensure_prefix()
183  {
184  // Resolve the console prefix if it's still invalid.
185  if (HPX_UNLIKELY(naming::invalid_id == prefix_))
186  {
187  std::unique_lock<prefix_mutex_type> l(prefix_mtx_, std::try_to_lock);
188 
189  if (l.owns_lock() && (naming::invalid_id == prefix_))
190  {
191  naming::gid_type raw_prefix;
192  {
193  util::unlock_guard<std::unique_lock<prefix_mutex_type> > ul(l);
194  naming::get_agas_client().get_console_locality(raw_prefix);
195  }
196 
197  HPX_ASSERT(naming::invalid_gid != raw_prefix);
198  if (!prefix_) {
199  prefix_ = naming::id_type(raw_prefix, naming::id_type::unmanaged);
200  }
201  else {
202  HPX_ASSERT(prefix_.get_gid() == raw_prefix);
203  }
204  }
205 
206  // Someone else started getting the console prefix.
207  else {
208  return false;
209  }
210  }
211  return true;

214  void pending_logs::send()
215  {
216  // WARNING: Never, ever call this outside of a HPX-thread.
217  HPX_ASSERT(threads::get_self_ptr());
218 
219  bool expected = false;
220  if (!is_sending_.compare_exchange_strong(expected, true))
221  return;
222 
223  try {
224  {
225  std::lock_guard<queue_mutex_type> l(queue_mtx_);
226  if (queue_.empty())
227  return; // some other thread did the deed
228  }
229 
230  if (!ensure_prefix())
231  return; // some other thread tries to do logging
232 
233  messages_type msgs;
234  {
235  std::lock_guard<queue_mutex_type> l(queue_mtx_);
236  if (queue_.empty())
237  return; // some other thread did the deed
238  queue_.swap(msgs);
239  }
240 
241  console_logging_locked(prefix_, msgs);
242  }
243  catch(...) {
244  is_sending_ = false;
245  throw;
246  }
247 
248  is_sending_ = false;
249  }

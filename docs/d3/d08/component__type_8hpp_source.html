108  inline bool types_are_compatible(component_type lhs, component_type rhs)
109  {
110  // don't compare types if one of them is unknown
111  if (component_invalid == rhs || component_invalid == lhs)
112  return true; // no way of telling, so we assume the best :-P
113 
114  // don't compare types if one of them is component_runtime_support
115  if (component_runtime_support == rhs || component_runtime_support == lhs)
116  return true;
117 
118  component_type lhs_base = get_base_type(lhs);
119  component_type rhs_base = get_base_type(rhs);
120 
121  if (lhs_base == rhs_base)
122  return true;
123 
124  // special case for lco's
125  if (lhs_base == component_base_lco &&
126  (rhs_base == component_base_lco_with_value_unmanaged ||
127  rhs_base == component_base_lco_with_value))
128  {
129  return true;
130  }
131 
132  if (rhs_base == component_base_lco &&
133  (lhs_base == component_base_lco_with_value_unmanaged ||
134  lhs_base == component_base_lco_with_value))
135  {
136  return true;
137  }
138 
139  if (lhs_base == component_base_lco_with_value_unmanaged &&
140  rhs_base == component_base_lco_with_value)
141  {
142  return true;
143  }
144 
145  if (lhs_base == component_base_lco_with_value &&
146  rhs_base == component_base_lco_with_value_unmanaged)
147  {
148  return true;
149  }
150 
151  return false;
152  }

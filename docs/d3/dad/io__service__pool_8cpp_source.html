103  {

105  if (startup != nullptr)

169  bool io_service_pool::run_locked(std::size_t num_threads, bool join_threads,
170  compat::barrier* startup)
171  {
172  if (io_services_.empty())
173  {
174  pool_size_ = num_threads;
175 
176  for (std::size_t i = 0; i < num_threads; ++i)
177  {
178  io_services_.emplace_back(new boost::asio::io_service);
179  work_.emplace_back(initialize_work(*io_services_[i]));
180  }
181  }
182 
183  for (std::size_t i = 0; i < num_threads; ++i)
184  {
185  compat::thread t(util::bind(
186  &io_service_pool::thread_run, this, i, startup));
187  threads_.emplace_back(std::move(t));
188  }
189 
190  next_io_service_ = 0;
191  stopped_ = false;
192 
193  HPX_ASSERT(pool_size_ == io_services_.size());
194  HPX_ASSERT(threads_.size() == io_services_.size());
195  HPX_ASSERT(work_.size() == io_services_.size());
196 
197  if (join_threads)
198  join_locked();
199 
200  return true;
201  }

259  boost::asio::io_service& io_service_pool::get_io_service(int index)
260  {
261  // use this function for single group io_service pools only
262  std::lock_guard<compat::mutex> l(mtx_);
263 
264  if (index == -1) {
265  if (++next_io_service_ == pool_size_)
266  next_io_service_ = 0;

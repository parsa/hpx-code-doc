365  template <typename InIter, typename OutIter, typename Pred, typename Proj>
366  inline std::pair<InIter, OutIter>
367  sequential_copy_if(InIter first, InIter last, OutIter dest,
368  Pred && pred, Proj && proj)
369  {
370  while (first != last)
371  {
372  if (hpx::util::invoke(pred, hpx::util::invoke(proj, *first)))
373  *dest++ = *first;
374  first++;
375  }
376  return std::make_pair(first, dest);

396  template <typename ExPolicy, typename FwdIter1, typename FwdIter2,
397  typename Pred, typename Proj = util::projection_identity>
398  static typename util::detail::algorithm_result<
399  ExPolicy, std::pair<FwdIter1, FwdIter2>
400  >::type
401  parallel(ExPolicy && policy, FwdIter1 first, FwdIter1 last,
402  FwdIter2 dest, Pred && pred, Proj && proj/* = Proj()*/)
403  {
404  typedef hpx::util::zip_iterator<FwdIter1, bool*> zip_iterator;
405  typedef util::detail::algorithm_result<
406  ExPolicy, std::pair<FwdIter1, FwdIter2>
407  > result;
408  typedef typename std::iterator_traits<FwdIter1>::difference_type
409  difference_type;
410 
411  if (first == last)
412  return result::get(std::make_pair(last, dest));
413 
414  difference_type count = std::distance(first, last);
415 
416  boost::shared_array<bool> flags(new bool[count]);
417  std::size_t init = 0;
418 
419  using hpx::util::get;
420  using hpx::util::make_zip_iterator;
421  typedef util::scan_partitioner<
422  ExPolicy, std::pair<FwdIter1, FwdIter2>, std::size_t
423  > scan_partitioner_type;
424 
425  auto f1 =
426  [pred, proj, flags, policy]
427  (
428  zip_iterator part_begin, std::size_t part_size
429  ) -> std::size_t
430  {
431  HPX_UNUSED(flags);
432  HPX_UNUSED(policy);
433 
434  std::size_t curr = 0;
435 
436  // MSVC complains if proj is captured by ref below
437  util::loop_n<ExPolicy>(
438  part_begin, part_size,
439  [&pred, proj, &curr](zip_iterator it) mutable
440  {
441  using hpx::util::invoke;
442  bool f = invoke(pred, invoke(proj, get<0>(*it)));
443 
444  if ((get<1>(*it) = f))
445  ++curr;
446  });
447 
448  return curr;
449  };
450  auto f3 =
451  [dest, flags, policy](
452  zip_iterator part_begin, std::size_t part_size,
453  hpx::shared_future<std::size_t> curr,
454  hpx::shared_future<std::size_t> next
455  ) mutable
456  {
457  HPX_UNUSED(flags);
458  HPX_UNUSED(policy);
459 
460  next.get(); // rethrow exceptions
461 
462  std::advance(dest, curr.get());
463  util::loop_n<ExPolicy>(
464  part_begin, part_size,
465  [&dest](zip_iterator it) mutable
466  {
467  if(get<1>(*it))
468  *dest++ = get<0>(*it);
469  });
470  };
471 
472  return scan_partitioner_type::call(
473  std::forward<ExPolicy>(policy),
474  make_zip_iterator(first, flags.get()), count, init,
475  // step 1 performs first part of scan algorithm
476  std::move(f1),
477  // step 2 propagates the partition results from left
478  // to right
479  hpx::util::unwrapping(std::plus<std::size_t>()),
480  // step 3 runs final accumulation on each partition
481  std::move(f3),
482  // step 4 use this return value
483  [last, dest, flags](
484  std::vector<hpx::shared_future<std::size_t> > && items,
485  std::vector<hpx::future<void> > &&) mutable
486  -> std::pair<FwdIter1, FwdIter2>
487  {
488  HPX_UNUSED(flags);
489 
490  std::advance(dest, items.back().get());
491  return std::make_pair(last, dest);
492  });
493  }
494  };

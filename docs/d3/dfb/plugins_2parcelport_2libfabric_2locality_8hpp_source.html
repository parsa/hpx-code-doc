39 // --------------------------------------------------------------------
40 // Locality, in this structure we store the informartion required by
41 // libfabric to make a connection to another node.
42 // With libfabric 1.4.x the array contains the fabric ip address stored
43 // as the second uint32_t in the array. For this reason we use an
44 // array of uint32_t rather than uint8_t/char so we can easily access
45 // the ip for debug/validation purposes
46 // --------------------------------------------------------------------
47 struct locality {
48 
49  // the number of 32bit ints stored in our array
50  static const uint32_t array_length = HPX_PARCELPORT_LIBFABRIC_LOCALITY_SIZE/4;
51  static const uint32_t array_size = HPX_PARCELPORT_LIBFABRIC_LOCALITY_SIZE;
52 
53  // array type of our locality data
54  typedef std::array<uint32_t, array_length> locality_data;
55 
56  static const char *type() {
57  return "libfabric";
58  }

60  explicit locality(const locality_data &in_data)
61  {
62  std::memcpy(&data_[0], &in_data[0], array_size);
63  fi_address_ = 0;
64  LOG_DEBUG_MSG("explicit constructing locality from "
65  << ipaddress(ip_address()) << ":" << decnumber(port()));
66  }
67 
68  locality() {
69  std::memset(&data_[0], 0x00, array_size);
70  fi_address_ = 0;

72  << ipaddress(ip_address()) << ":" << decnumber(port()));
73  }

96  const uint32_t & ip_address() const {
97 #if defined (HPX_PARCELPORT_LIBFABRIC_LOCALITY_SOCKADDR)

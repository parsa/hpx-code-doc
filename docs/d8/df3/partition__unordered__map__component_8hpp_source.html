217  T get_value(Key const& key, bool erase)
218  {
219  typename data_type::iterator it = partition_unordered_map_.find(key);
220  if (it == partition_unordered_map_.end())
221  {
222  HPX_THROW_EXCEPTION(bad_parameter,
223  "partition_unordered_map::get_value",
224  "unable to find requested key in this partition of the "
225  "unordered_map");
226  }
227 
228  if (!erase)
229  return it->second;
230 
231  erase_on_exit t(partition_unordered_map_, it);
232  return it->second;
233  }

243  std::vector<T> get_values(std::vector<Key> const& keys)
244  {
245  std::vector<T> result;
246  result.reserve(keys.size());
247 
248  for (std::size_t i = 0; i != keys.size(); ++i)
249  {
250  typename data_type::iterator it =
251  partition_unordered_map_.find(keys[i]);
252  if (it == partition_unordered_map_.end())
253  {
254  HPX_THROW_EXCEPTION(bad_parameter,
255  "partition_unordered_map::get_values",
256  "unable to find requested key in this partition of the "
257  "unordered_map");
258  break;
259  }
260  result.push_back(it->second);
261  }
262  return result;
263  }

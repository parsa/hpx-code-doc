31 void test_multiple_readers()
32 {
33  typedef hpx::lcos::local::shared_mutex shared_mutex_type;
34  typedef hpx::lcos::local::mutex mutex_type;
35 
36  unsigned const number_of_threads = 10;
37 
38  test::thread_group pool;
39 
40  hpx::lcos::local::shared_mutex rw_mutex;
41  unsigned unblocked_count = 0;
42  unsigned simultaneous_running_count = 0;
43  unsigned max_simultaneous_running = 0;
44  mutex_type unblocked_count_mutex;
45  hpx::lcos::local::condition_variable unblocked_condition;
46  mutex_type finish_mutex;
47  std::unique_lock<mutex_type> finish_lock(finish_mutex);
48 
49  try
50  {
51  for (unsigned i = 0; i != number_of_threads; ++i)
52  {
53  pool.create_thread(
54  test::locking_thread<boost::shared_lock<shared_mutex_type> >(
55  rw_mutex, unblocked_count, unblocked_count_mutex,
56  unblocked_condition, finish_mutex,
57  simultaneous_running_count, max_simultaneous_running
58  )
59  );
60  }
61 
62  {
63  std::unique_lock<mutex_type> lk(unblocked_count_mutex);
64  while(unblocked_count < number_of_threads)
65  {
66  unblocked_condition.wait(lk);
67  }
68  }
69 
70  CHECK_LOCKED_VALUE_EQUAL(unblocked_count_mutex,
71  unblocked_count, number_of_threads);
72 
73  finish_lock.unlock();
74  pool.join_all();
75  }
76  catch(...)
77  {
78  pool.interrupt_all();
79  pool.join_all();
80  HPX_TEST(false);
81  }
82 
83  CHECK_LOCKED_VALUE_EQUAL(unblocked_count_mutex,
84  max_simultaneous_running, number_of_threads);

51  void done() {
52  double elapsed = this->_timer.elapsed();
53  if (this->_parent) {
54  this->_parent->addProfile(this->_title, std::make_tuple(elapsed,0,1));
55  std::for_each(this->_profiles.begin(), this->_profiles.end(),
56  [=](std::map<const char *, valtype>::value_type &p) {
57  this->_parent->addProfile(p.first, p.second);
58  }
59  );
60  }
61  else {
62  // get the max depth of the profile tree so we can prepare string lengths
63  int maxlevel = 0;
64  std::for_each(this->_profiles.begin(), this->_profiles.end(),
65  [&](std::map<const char *, valtype>::value_type &p) {
66  maxlevel = (std::max)(maxlevel, std::get<1>(p.second));
67  }
68  );
69  // prepare format string for output
70  char const* fmt1 = "Profile %20s : %2i %5i %9.3f %s %7.3f";
71  std::string fmt2 = "Total " + std::string(41,' ') + " %s %7.3f";
72  // add this level to top of list
73  this->_profiles[this->_title] = std::make_tuple(elapsed,0,1);
74  // print each of the sub nodes
75  std::vector<double> level_totals(5,0);
76  int last_level = 0;
77  if (_output) hpx::cout << std::string(58+maxlevel*9,'-') << "\n";
78  for (auto p=this->_profiles.begin(); p!=this->_profiles.end(); ) {
79  int &level = std::get<1>(p->second);
80  level_totals[level] += std::get<0>(p->second);
81  if (level<last_level) {
82  if (_output) hpx::cout << std::string(52,' ')
83  << std::string (last_level*9, ' ')
84  << "------\n";
85  if (_output) hpx::cout << (boost::format(fmt2)
86  % std::string (last_level*9, ' ')
87  % (100.0*level_totals[last_level]/elapsed)) << "\n";
88  last_level = level;
89  }
90  else if (level>last_level) {
91  last_level = level;
92  }
93  if (_output) hpx::cout << (boost::format(fmt1)
94  % p->first
95  % level
96  % std::get<2>(p->second)
97  % std::get<0>(p->second)
98  % std::string (level*9, ' ')
99  % (100.0*std::get<0>(p->second)/elapsed)) << "\n";
100  if ((++p)==this->_profiles.end()) {
101  if (_output) hpx::cout << std::string(52,' ')
102  << std::string (last_level*9, ' ')
103  << "------\n";
104  if (_output) hpx::cout << (boost::format(fmt2)
105  % std::string (last_level*9, ' ')
106  % (100.0*level_totals[last_level]/elapsed)) << "\n";
107  last_level = level;
108  }
109  }
110  if (_output) hpx::cout << std::string(58+maxlevel*9,'-') << "\n";
111  }
112  this->_done = true;
113  }

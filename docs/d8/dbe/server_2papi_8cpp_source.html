98  bool thread_counters::remove_event(papi_counter *cnt)
99  {
100  if (!stop_all() || PAPI_cleanup_eventset(evset_) != PAPI_OK)
101  return false;
102 
103  // store the most recent value
104  int rmix = cnt->get_counter_index();
105  cnt->update_state(timestamp_, counts_[rmix], PAPI_COUNTER_STOPPED);
106  // erase entries corresponding to removed event
107  counts_.erase(counts_.begin()+rmix);
108  cnttab_.erase(cnttab_.begin()+rmix);
109  // For the lack of better strategy the remaining events are added in the
110  // same order as before. This avoids reordering of remaining counter
111  // values and at least some surprises on architectures with asymmetric
112  // functionality of counting registers.
113  unsigned i = 0;
114  while (i < counts_.size())
115  {
116  papi_counter *c = cnttab_[i];
117  if (PAPI_add_event(evset_, c->get_event()) != PAPI_OK)
118  { // failed to add event
119  c->update_state(timestamp_, counts_[c->get_counter_index()],
120  PAPI_COUNTER_SUSPENDED);
121  counts_.erase(counts_.begin()+i);
122  cnttab_.erase(cnttab_.begin()+i);
123  continue;
124  }
125  // update indices of remaining counters
126  c->update_index(i++);
127  }
128  return start_all();
129  }

156  bool thread_counters::stop_all()
157  {
158  int stat;
159  if (PAPI_state(evset_, &stat) != PAPI_OK) return false;
160 
161  if ((stat & PAPI_RUNNING) != 0)
162  {
163  std::vector<long long> tmp(counts_.size());
164  if (PAPI_stop(evset_, &tmp[0]) != PAPI_OK) return false;
165  // accumulate existing counts before modifying event set
166  if (PAPI_accum(evset_, &counts_[0]) != PAPI_OK) return false;
167  timestamp_ = static_cast<std::int64_t>(hpx::get_system_uptime());
168  }
169  return true;
170  }

172  bool thread_counters::finalize()
173  {
174  if (evset_ != PAPI_NULL)
175  {
176  if (stop_all())
177  {
178  for (std::uint32_t i = 0; i < cnttab_.size(); ++i)
179  { // retrieve the most recent values for the still active counters
180  papi_counter *c = cnttab_[i];
181  if (c->get_status() == PAPI_COUNTER_ACTIVE)
182  c->update_state(timestamp_, counts_[i], PAPI_COUNTER_TERMINATED);
183  else c->update_status(PAPI_COUNTER_TERMINATED);
184  }
185  }
186  // release event set resources
187  return PAPI_destroy_eventset(&evset_) == PAPI_OK;
188  }
189  return true;

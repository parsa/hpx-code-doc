98  // BulkTwoWayExecutor interface
99  template <typename F, typename S, typename ... Ts>
100  std::vector<hpx::future<
101  typename detail::bulk_function_result<F, S, Ts...>::type
102  > >
103  bulk_async_execute(F && f, S const& shape, Ts &&... ts) const
104  {
105  // lazily initialize once
106  static std::size_t global_num_tasks =
107  (std::min)(std::size_t(128), hpx::get_os_thread_count());
108 
109  std::size_t num_tasks =
110  (num_tasks_ == std::size_t(-1)) ? global_num_tasks : num_tasks_;
111 
112  typedef std::vector<hpx::future<
113  typename detail::bulk_function_result<
114  F, S, Ts...
115  >::type
116  > > result_type;
117 
118  result_type results;
119  std::size_t size = hpx::util::size(shape);
120  results.resize(size);
121 
122  spawn(results, 0, size, num_tasks, f, hpx::util::begin(shape), ts...).get();
123  return results;
124  }
126 
127  protected:
129  template <typename Result, typename F, typename Iter, typename ... Ts>
130  hpx::future<void> spawn(std::vector<hpx::future<Result> >& results,
131  std::size_t base, std::size_t size, std::size_t num_tasks,
132  F const& func, Iter it, Ts const&... ts) const
133  {
134  if (size > num_tasks)
135  {
136  // spawn hierarchical tasks
137  std::size_t chunk_size = (size + num_spread_) / num_spread_ - 1;
138  chunk_size = (std::max)(chunk_size, num_tasks);
139 
140  std::vector<hpx::future<void> > tasks;
141  tasks.reserve(num_spread_);
142 
143  hpx::future<void> (parallel_executor::*spawn_func)(
144  std::vector<hpx::future<Result> >&, std::size_t,
145  std::size_t, std::size_t, F const&, Iter, Ts const&...
146  ) const = &parallel_executor::spawn;
147 
148  while (size != 0)
149  {
150  std::size_t curr_chunk_size = (std::min)(chunk_size, size);
151 
152  hpx::future<void> f = hpx::async(
153  spawn_func, this, std::ref(results), base,
154  curr_chunk_size, num_tasks, std::ref(func), it,
155  std::ref(ts)...);
156  tasks.push_back(std::move(f));
157 
158  base += curr_chunk_size;
159  it = hpx::parallel::v1::detail::next(it, curr_chunk_size);
160  size -= curr_chunk_size;
161  }
162 
163  HPX_ASSERT(size == 0);
164 
165  return hpx::when_all(tasks);
166  }
167 
168  // spawn all tasks sequentially
169  HPX_ASSERT(base + size <= results.size());
170 
171  for (std::size_t i = 0; i != size; ++i, ++it)
172  {
173  results[base + i] = hpx::async(l_, func, *it, ts...);
174  }
175 
176  return hpx::make_ready_future();
177  }
179 
180  private:

183

186  {

188  }

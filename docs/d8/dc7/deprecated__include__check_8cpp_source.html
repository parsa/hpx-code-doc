88  void deprecated_include_check::inspect(
89  const string & library_name,
90  const path & full_path, // example: c:/foo/boost/filesystem/path.hpp
91  const string & contents) // contents of file to be inspected
92  {
93  std::string::size_type p = contents.find( "hpxinspect:" "nodeprecatedinclude" );
94  if (p != string::npos)
95  {
96  // ignore this directive here (it is handled below) if it is followed
97  // by a ':'
98  if (p == contents.size() - 30 ||
99  (contents.size() > p + 30 && contents[p + 30] != ':'))
100  {
101  return;
102  }
103  }
104 
105  std::set<std::string> found_includes;
106 
107  // check for all given includes
108  for (deprecated_includes_regex_data const& d : regex_data)
109  {
110  boost::sregex_iterator cur(contents.begin(), contents.end(), d.pattern), end;
111  for(; cur != end; ++cur)
112  {
113  auto m = *cur;
114  if (m[1].matched || m[2].matched)
115  {
116  int idx = (m[1].matched ? 1 : 2);
117 
118  // avoid errors to be reported twice
119  std::string found_include(m[1].first, m[1].second);
120  if (found_includes.find(found_include) == found_includes.end())
121  {
122  std::string tag("hpxinspect:" "nodeprecatedinclude:" + found_include);
123  if (contents.find(tag) != string::npos)
124  continue;
125 
126  // name was found
127  found_includes.insert(found_include);
128 
129  // include is missing
130  auto it = contents.begin();
131  auto match_it = m[idx].first;
132  auto line_start = it;
133 
134  string::size_type line_number = 1;
135  for (; it != match_it; ++it)
136  {
137  if (string::traits_type::eq(*it, '\n'))
138  {
139  ++line_number;
140  line_start = it + 1; // could be end()
141  }
142  }
143 
144  ++m_errors;
145  error(library_name, full_path, string(name())
146  + " deprecated #include ("
147  + found_include
148  + ") on line "
149  + linelink(full_path, boost::lexical_cast<string>(line_number))
150  + " use " + m.format(d.data->use_instead) + " instead");
151  }
152  }
153  }
154  }
155  }

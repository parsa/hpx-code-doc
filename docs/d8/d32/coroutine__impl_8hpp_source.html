90  static inline coroutine_impl* create(
91  functor_type&& f, thread_id_repr_type id = nullptr,
92  std::ptrdiff_t stack_size = default_stack_size)
93  {
94  coroutine_impl* p = allocate(id, stack_size);
95 
96  if (!p)
97  {
98  std::size_t const heap_num = std::size_t(id) / 32; //-V112
99 
100  // allocate a new coroutine object, if non is available (or all
101  // heaps are locked)
102  context_base::increment_allocation_count(heap_num);
103  p = new coroutine_impl(std::move(f),
104  id, stack_size);
105  } else {
106  // we reuse an existing object, we need to rebind its function
107  p->rebind(std::move(f), id);
108  }
109  return p;
110  }

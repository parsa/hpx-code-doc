62  for(std::uint64_t b = 0; b < num_blocks; ++b)
63  {
64  for(std::uint64_t i = 0; i < order; ++i)

167 void transpose(sub_block A, sub_block B, std::uint64_t block_order,
168  std::uint64_t tile_size)
169 {
170  if(tile_size < block_order)
171  {
172  for(std::uint64_t i = 0; i < block_order; i += tile_size)
173  {
174  for(std::uint64_t j = 0; j < block_order; j += tile_size)
175  {
176  std::uint64_t i_max = (std::min)(block_order, i + tile_size);
177  for(std::uint64_t it = i; it < i_max; ++it)
178  {
179  std::uint64_t j_max = (std::min)(block_order, j + tile_size);
180  for(std::uint64_t jt = j; jt < j_max; ++jt)
181  {
182  B[it + block_order * jt] = A[jt + block_order * it];
183  }
184  }
185  }
186  }
187  }
188  else
189  {
190  for(std::uint64_t i = 0; i < block_order; ++i)
191  {
192  for(std::uint64_t j = 0; j < block_order; ++j)
193  {
194  B[i + block_order * j] = A[j + block_order * i];
195  }
196  }
197  }
198 }

200 double test_results(std::uint64_t order, std::uint64_t block_order,
201  std::vector<block> const & trans)
202 {
203  double errsq = 0.0;
204 
205  for(std::uint64_t b = 0; b < trans.size(); ++b)
206  {
207  for(std::uint64_t i = 0; i < order; ++i)
208  {
209  double col_val = COL_SHIFT * i;
210  for(std::uint64_t j = 0; j < block_order; ++j)
211  {
212  double diff = trans[b][i * block_order + j] -
213  (col_val + ROW_SHIFT * (b * block_order + j));
214  errsq += diff * diff;
215  }
216  }
217  }
218 
219  if(verbose)
220  std::cout << " Squared sum of differences: " << errsq << "\n";
221 
222  return errsq;
223 }

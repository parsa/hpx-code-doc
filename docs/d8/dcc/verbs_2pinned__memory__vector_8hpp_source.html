24  // this class looks like a vector, but can be initialized from a pointer and size,
25  // it is used by the verbs parcelport to pass an rdma memory chunk with received
26  // data into the decode parcel buffer routines.
27  // it cannot be resized or changed once created and does not delete wrapped memory

167  inline void resize(std::size_t s) {
168  LOG_TRACE_MSG("pinned_memory_vector "
169  << "size " << hexuint32(m_size_)
170  << "array " << hexpointer(m_array_)
171  << "region " << hexpointer(m_region_)
172  << "alloc " << hexpointer(m_alloc_)
173  << "resizing from " << m_size_ << " to " << s);
174 
175  if (m_region_) {
176  if (s > m_region_->get_size()) {
177  LOG_ERROR_MSG(
178  "Resizing from " << m_region_->get_size() << " to " << s);
179  throw std::runtime_error(
180  "pinned_memory_vector should never be resized once an "
181  "allocation has been assigned");
182  }
183  m_size_ = s;
184  }
185  else {
186  m_region_ = m_alloc_->allocate_region(s);
187  m_array_ = static_cast<T*>(m_region_->get_address());
188  m_size_ = s;
189  }
190  }

197  << "alloc " << hexpointer(m_alloc_)

199  if (m_array_ || m_region_) {
200  throw std::runtime_error(
201  "pinned_memory_vector should never be resized once an "
202  "allocation has been assigned");
203  }
204  m_region_ = m_alloc_->allocate_region(s);
205  m_array_ = static_cast<T*>(m_region_->get_address());
206  }

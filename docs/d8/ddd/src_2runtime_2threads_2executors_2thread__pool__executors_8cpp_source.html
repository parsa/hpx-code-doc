89  template <typename Scheduler>
90  thread_pool_executor<Scheduler>::~thread_pool_executor()
91  {
92  // if we're still starting up, give this executor a chance of executing
93  // its tasks
94  while (!scheduler_.has_reached_state(state_running))
95  this_thread::suspend();
96 
97  // Inform the resource manager that this executor is about to be
98  // destroyed. This will cause it to invoke remove_processing_unit below
99  // for each of the currently allocated virtual cores.
100  resource_manager& rm = resource_manager::get();
101  rm.stop_executor(cookie_);
102 
103  // wait for executor to finish executing
104  shutdown_sem_.wait(max_current_concurrency_.load());
105 
106  // detach this executor from resource manager
107  rm.detach(cookie_); // this releases proxy (manage_thread_executor)
108 
109 #if defined(HPX_DEBUG)
110  // all resources should have been stopped at this point (or have never
111  // been initialized)
112  for (std::size_t i = 0; i != max_punits_; ++i)
113  {
114  hpx::state s = scheduler_.get_state(i).load();
115  HPX_ASSERT(s == state_initialized || s == state_stopped);
116  }
117 
118  // all scheduled tasks should have completed executing
119  HPX_ASSERT(tasks_completed_ == tasks_scheduled_);
120 
121  // all driver threads should have stopped executing
122  HPX_ASSERT(current_concurrency_ == 0);
123 #endif
124  }

175  // Schedule given function for execution in this executor no sooner
176  // than time abs_time. This call never blocks, and may violate
177  // bounds on the executor's queue size.
178  template <typename Scheduler>
179  void thread_pool_executor<Scheduler>::add_at(
180  util::steady_clock::time_point const& abs_time,
181  closure_type && f, util::thread_description const& desc,
182  threads::thread_stacksize stacksize, error_code& ec)
183  {
184  // create a new suspended thread
185  thread_init_data data(util::bind(
186  util::one_shot(&thread_pool_executor::thread_function_nullary),
187  this, std::move(f)), desc);
188  data.stacksize = threads::get_stack_size(stacksize);
189 
190  threads::thread_id_type id = threads::invalid_thread_id;
191  threads::detail::create_thread( //-V601
192  &scheduler_, data, id, suspended, true, ec);
193  if (ec) return;
194  HPX_ASSERT(invalid_thread_id != id); // would throw otherwise
195 
196  // update statistics
197  ++tasks_scheduled_;
198 
199  // now schedule new thread for execution
200  threads::detail::set_thread_state_timed(scheduler_, abs_time, id, ec);
201  if (ec) {
202  --tasks_scheduled_;
203  return;
204  }
205 
206  if (&ec != &throws)
207  ec = make_success_code();

254  template <typename Scheduler>
255  void thread_pool_executor<Scheduler>::suspend_back_into_calling_context(
256  std::size_t virt_core)
257  {
258  // give invoking context a chance to catch up with its tasks, but only
259  // if this scheduler is currently in running state
260  std::atomic<hpx::state>& state = scheduler_.get_state(virt_core);
261  hpx::state expected = state_running;
262  if (state.compare_exchange_strong(expected, state_suspended))
263  {
264  {
265  on_self_reset on_exit(self_[virt_core]);
266  this_thread::suspend();
267  }
268 
269  // reset state to running if current state is still suspended
270  expected = state_suspended;
271  state.compare_exchange_strong(expected, state_running);
272  }
273  else
274  {
275  HPX_ASSERT(expected != state_suspended);
276  }
277  }

306  template <typename Scheduler>
307  void thread_pool_executor<Scheduler>::run(std::size_t virt_core,
308  std::size_t thread_num)
309  {
310  // Set the state to 'state_running' only if it's still in 'state_starting'
311  // state, otherwise our destructor is currently being executed, which
312  // means we need to still execute all threads.
313  std::atomic<hpx::state>& state = scheduler_.get_state(virt_core);
314  hpx::state expected = state_starting;
315  if (state.compare_exchange_strong(expected, state_running))
316  {
317  ++max_current_concurrency_;
318 
319 // {
320 // std::lock_guard<mutex_type> l(mtx_);
321 // resource::get_partitioner().get_affinity_data().add_punit(
322 // virt_core, thread_num);
323 // scheduler_.on_start_thread(virt_core);
324 // }
325 
326  self_[virt_core] = threads::get_self_ptr();
327 
328  on_run_exit on_exit(current_concurrency_, shutdown_sem_,
329  self_[virt_core]);
330 
331  // FIXME: turn these values into performance counters
332  std::int64_t executed_threads = 0, executed_thread_phases = 0;
333  std::uint64_t overall_times = 0, thread_times = 0;
334  std::int64_t idle_loop_count = 0, busy_loop_count = 0;
335  std::uint8_t task_active = 0;
336 
337  threads::detail::scheduling_counters counters(
338  executed_threads, executed_thread_phases,
339  overall_times, thread_times, idle_loop_count, busy_loop_count,
340  task_active);
341 
342  threads::detail::scheduling_callbacks callbacks(
343  threads::detail::scheduling_callbacks::callback_type(),
344  util::bind( //-V107
345  &thread_pool_executor::suspend_back_into_calling_context,
346  this, virt_core));
347 
348  scheduler_.set_scheduler_mode(policies::fast_idle_mode);
349  threads::detail::scheduling_loop(virt_core, scheduler_,
350  counters, callbacks);
351 
352  // the scheduling_loop is allowed to exit only if no more HPX
353  // threads exist
354  HPX_ASSERT(!scheduler_.get_thread_count(
355  unknown, thread_priority_default, virt_core) ||
356  state >= state_terminating);
357  }

425  // Remove the given processing unit from the scheduler.
426  template <typename Scheduler>
427  void thread_pool_executor<Scheduler>::remove_processing_unit(
428  std::size_t virt_core, error_code& ec)
429  {
430  // inform the scheduler to stop the virtual core
431  std::atomic<hpx::state>& state = scheduler_.get_state(virt_core);
432  hpx::state oldstate = state.exchange(state_stopped);
433  HPX_ASSERT(oldstate == state_starting ||
434  oldstate == state_running || oldstate == state_suspended ||
435  oldstate == state_stopped);
436  --curr_punits_;
437  }

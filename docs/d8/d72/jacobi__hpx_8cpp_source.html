52

58  hpx::util::high_resolution_timer t;
59  for(std::size_t i = 0; i < iterations; ++i)
60  {
61  for(std::size_t y = 1, j = 0; y < n -1; y += block_size, ++j)
62  {
63  std::size_t y_end = (std::min)(y + block_size, n - 1);
64  std::vector<hpx::shared_future<void> > trigger;
65  trigger.reserve(3);
66  trigger.push_back((*deps_old)[j]);
67  if(j > 0) trigger.push_back((*deps_old)[j-1]);
68  if(j + 1 < n_block) trigger.push_back((*deps_old)[j+1]);

70  /*

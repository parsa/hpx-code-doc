51  // --------------------------------------------------------------------
52  void rma_receiver::read_message(region_type* region,
53  fi_addr_t const& src_addr)
54  {
55  start_time_ = util::high_resolution_clock::now();
56  HPX_ASSERT(rma_count_ == 0);
57  HPX_ASSERT(header_ == nullptr);
58  HPX_ASSERT(header_region_ == nullptr);
59  HPX_ASSERT(chunk_region_ == nullptr);
60  HPX_ASSERT(message_region_ == nullptr);
61  HPX_ASSERT(rma_regions_.size() == 0);
62  HPX_ASSERT(chunk_fetch_ == false);
63 
64  // where this message came from
65  src_addr_ = src_addr;
66 
67  // the region posted as a receive contains the received header
68  header_region_ = region;
69  header_ = reinterpret_cast<header_type*>(header_region_->get_address());
70 
71  HPX_ASSERT(header_);
72  HPX_ASSERT(header_region_->get_address());
73 
74  LOG_DEBUG_MSG("receiver " << hexpointer(this) << "Header : " << *header_);
75 
76  LOG_TRACE_MSG(
77  CRC32_MEM(header_, header_->header_length(), "Header region (recv)"));
78 
79  if (header_->chunk_ptr()==nullptr) {
80  // the header does not have piggybacked chunks, we must rma-get them before
81  // we can decode the message, they may need further rma-get operations
82  handle_message_no_chunk_data();
83  return;
84  }
85 
86  // how many RMA operations are needed
87  rma_count_ = header_->num_zero_copy_chunks();
88 
89  LOG_DEBUG_MSG("receiver " << hexpointer(this)
90  << "is expecting " << decnumber(rma_count_) << "read completions");
91 
92  // If we have no zero copy chunks and piggy backed data, we can
93  // process the message immediately, otherwise, dispatch to receiver
94  // If we have neither piggy back, nor zero copy chunks, rma_count is 0
95  if (rma_count_ == 0)
96  {
97  handle_message_no_rma();
98  ++msg_plain_;
99  }
100  else {
101  handle_message_with_zerocopy_rma();
102  ++msg_rma_;
103  }
104  }

107  void rma_receiver::handle_message_no_rma()
108  {
109  HPX_ASSERT(header_);
110  LOG_DEBUG_MSG("receiver " << hexpointer(this)
111  << "handle piggy backed send without zero copy regions");
112 
113  char *piggy_back = header_->message_data();
114  HPX_ASSERT(piggy_back);
115 
116  LOG_TRACE_MSG(
117  CRC32_MEM(piggy_back, header_->message_size(),
118  "(Message region recv piggybacked - no rdma)"));
119 
120  typedef pinned_memory_vector<char, header_size, region_type, memory_pool_type>
121  rcv_data_type;
122  typedef parcel_buffer<rcv_data_type, std::vector<char>> rcv_buffer_type;
123 
124  // when parcel decoding from the wrapped pointer buffer has completed,
125  // the lambda function will be called
126  rcv_data_type wrapped_pointer(
127  piggy_back, header_->message_size(), [](){}, nullptr, nullptr);
128 
129  rcv_buffer_type buffer(std::move(wrapped_pointer), nullptr);
130 
131  int zc_chunks =
132  std::count_if(chunks_.begin(), chunks_.end(), [](chunk_struct &c) {
133  return c.type_ == serialization::chunk_type_pointer;
134  });
135  int oo_chunks = chunks_.size() - zc_chunks;
136 
137  buffer.num_chunks_ = std::make_pair(zc_chunks, oo_chunks);
138  buffer.data_size_ = header_->message_size();
139  performance_counters::parcels::data_point& data = buffer.data_point_;
140  data.bytes_ = static_cast<std::size_t>(header_->message_size());
141  data.time_ = util::high_resolution_clock::now() - start_time_;
142  LOG_DEBUG_MSG("receiver " << hexpointer(this)
143  << "calling parcel decode for complete NORMAL parcel");
144  std::size_t num_thread = hpx::get_worker_thread_num();
145  decode_message_with_chunks(*pp_, std::move(buffer), 0, chunks_, num_thread);
146  LOG_DEBUG_MSG("receiver " << hexpointer(this)
147  << "parcel decode called for complete NORMAL (small) parcel");
148 
149  cleanup_receive();
150  }

154  {
155  chunks_.resize(header_->num_chunks());
156  char *chunk_data = header_->chunk_data();
157  HPX_ASSERT(chunk_data);
158 
159  size_t chunkbytes =
160  chunks_.size() * sizeof(chunk_struct);
161 
162  std::memcpy(chunks_.data(), chunk_data, chunkbytes);
163  LOG_DEBUG_MSG("receiver " << hexpointer(this)
164  << "Copied chunk data from header : size "
165  << decnumber(chunkbytes));
166 
167  LOG_EXCLUSIVE(
168  for (const chunk_struct &c : chunks_)
169  {
170  LOG_DEBUG_MSG("receiver " << hexpointer(this)
171  << "recv : chunk : size " << hexnumber(c.size_)
172  << " type " << decnumber((uint64_t)c.type_)
173  << " rkey " << hexpointer(c.rkey_)
174  << " cpos " << hexpointer(c.data_.cpos_)
175  << " index " << decnumber(c.data_.index_));
176  });
177 
178  rma_regions_.reserve(header_->num_zero_copy_chunks());
179 
180  // for each zerocopy chunk, schedule a read operation
181  read_chunk_list();
182  }

184  // --------------------------------------------------------------------
185  void rma_receiver::handle_message_no_chunk_data()
186  {
187  chunk_fetch_ = true;
188  // get the remote chunk block memory region details
189  auto &cb = header_->chunk_header_ptr()->chunk_rma;
190  LOG_DEBUG_MSG("receiver " << hexpointer(this)
191  << "Fetching RMA chunk chunk data with "
192  << "size " << decnumber(cb.size_)
193  << "rkey " << hexpointer(cb.rkey_)
194  << "addr " << hexpointer(cb.data_.cpos_));
195 
196  // we need a local memory region to read the chunks into
197  chunk_region_ = memory_pool_->allocate_region(cb.size_);
198  chunk_region_->set_message_length(cb.size_);
199  uint64_t rkey1 = cb.rkey_;
200  const void *remoteAddr1 = cb.data_.cpos_;
201  // add it to the list of rma regions to fetch
202  rma_regions_.push_back(chunk_region_);
203  LOG_DEBUG_MSG("receiver " << hexpointer(this)
204  << "Fetching chunk region with size " << decnumber(cb.size_));
205  rma_count_ = 1;
206 
207  // if the message region requires rma-get, we can get it now as well
208  if (!header_->message_piggy_back())
209  {
210  auto &mc = header_->message_chunk_ptr()->message_rma;
211  message_region_ = memory_pool_->allocate_region(mc.size_);
212  message_region_->set_message_length(mc.size_);
213  uint64_t rkey2 = mc.rkey_;
214  const void *remoteAddr2 = mc.data_.cpos_;
215  // add it to the list of rma regions to fetch
216  rma_regions_.push_back(message_region_);
217  LOG_DEBUG_MSG("receiver " << hexpointer(this)
218  << "Fetching message region with size " << decnumber(mc.size_));
219  ++rma_count_;
220  // call the rma read function for the message chunk
221  read_one_chunk(src_addr_, message_region_, remoteAddr2, rkey2);
222  }
223 
224  // call the rma read function for the chunk-info chunk
225  // we do this last so that a completion does not come before the message
226  // chunk read has been triggered - this would cause the rma receive to decrement
227  // the counter and might hit zero before we had sent the second request
228  read_one_chunk(src_addr_, chunk_region_, remoteAddr1, rkey1);
229  }

231  // --------------------------------------------------------------------
232  int rma_receiver::handle_chunks_read_message()
233  {
234  char *chunk_data = chunk_region_->get_address();
235  HPX_ASSERT(chunk_data);
236  //
237  uint64_t chunkbytes = chunk_region_->get_message_length();
238  uint64_t num_chunks = chunkbytes/sizeof(chunk_struct);
239  chunks_.resize(num_chunks);
240  std::memcpy(chunks_.data(), chunk_data, chunkbytes);
241  LOG_DEBUG_MSG("receiver " << hexpointer(this)
242  << "Copied chunk data from chunk_region: size " << decnumber(chunkbytes)
243  << "with num chunks " << decnumber(num_chunks));
244  //
245  rma_regions_.clear();
246  chunk_fetch_ = false;
247  // for each zerocopy chunk, schedule a read operation
248  uint64_t zc_count =
249  std::count_if(chunks_.begin(), chunks_.end(), [](chunk_struct &c) {
250  return c.type_ == serialization::chunk_type_pointer;
251  });
252  LOG_DEBUG_MSG("receiver " << hexpointer(this)
253  << "Restarting RMA reads with " << decnumber(rma_count_) << "chunks");
254  rma_count_ = zc_count;
255  // perform an rma read for each zero copy chunk
256  read_chunk_list();
257  // do not return rma_count_ as it might already have decremented! (racey)
258  HPX_ASSERT(rma_regions_.size() == zc_count );
259  return rma_regions_.size();
260  }

262  // --------------------------------------------------------------------
263  void rma_receiver::read_chunk_list()
264  {
265  for (chunk_struct &c : chunks_)
266  {
267  if (c.type_ == serialization::chunk_type_pointer)
268  {
269  region_type *get_region =
270  memory_pool_->allocate_region(c.size_);
271  LOG_TRACE_MSG(
272  CRC32_MEM(get_region->get_address(), c.size_,
273  "(RDMA GET region (new))"));
274 
275  rma_regions_.push_back(get_region);
276  get_region->set_message_length(c.size_);
277 
278  // overwrite the serialization chunk data pointer because the chunk
279  // info sent contains the pointer to the remote data and when we
280  // deocde the parcel we want the chunk to point to the local copy of it
281  const void *remoteAddr = c.data_.cpos_;
282  c.data_.cpos_ = get_region->get_address();
283  // call the rma read function for the chunk
284  read_one_chunk(src_addr_, get_region, remoteAddr, c.rkey_);
285  }
286  }

290  void rma_receiver::read_one_chunk(
291  fi_addr_t src_addr, region_type *get_region,
292  const void *remoteAddr, uint64_t rkey)
293  {
294  // post the rdma read/get
295  LOG_DEBUG_MSG("rma_receiver " << hexpointer(this)
296  << "RDMA Get fi_read :"
297  << "client " << hexpointer(endpoint_)
298  << "fi_addr " << hexpointer(src_addr_)
299  << "tag " << hexuint64(header_->tag())
300  << "local addr " << hexpointer(get_region->get_address())
301  << "local desc " << hexpointer(get_region->get_desc())
302  << "size " << hexlength(get_region->get_message_length())
303  << "rkey " << hexpointer(rkey)
304  << "remote cpos " << hexpointer(remoteAddr));
305 
306  // count reads
307  ++rma_reads_;
308 
309  ssize_t ret = 0;
310  for (std::size_t k = 0; true; ++k)
311  {
312  LOG_EXCLUSIVE(
313  // write a pattern and dump out data for debugging purposes
314  uint32_t *buffer =
315  reinterpret_cast<uint32_t*>(get_region->get_address());
316  std::fill(buffer, buffer + get_region->get_size()/4, 0xDEADC0DE);
317  LOG_TRACE_MSG(
318  CRC32_MEM(get_region->get_address(), c.size_,
319  "(RDMA GET region (pre-fi_read))"));
320  );
321 
322  ret = fi_read(endpoint_,
323  get_region->get_address(),
324  get_region->get_message_length(),
325  get_region->get_desc(),
326  src_addr_,
327  (uint64_t)(remoteAddr), rkey, this);
328  if (ret == -FI_EAGAIN)
329  {
330  LOG_ERROR_MSG("receiver " << hexpointer(this)
331  << "reposting fi_read...\n");
332  hpx::util::detail::yield_k(k, "libfabric::receiver::async_read");
333  continue;
334  }
335  if (ret) throw fabric_error(ret, "fi_read");
336  break;
337  }
338  }

340  // --------------------------------------------------------------------
341  // whenever and rma read completion event occurs, this function is entered.
342  // an atomic counter, counts down with each completion, until we hit zero,
343  // when all expected read results are available.
344  void rma_receiver::handle_rma_read_completion()
345  {
346  FUNC_START_DEBUG_MSG;
347  HPX_ASSERT(rma_count_ > 0);
348  // If we haven't read all chunks, we can return and wait
349  // for the other incoming read completions
350  if (--rma_count_ > 0)
351  {
352  LOG_DEBUG_MSG("receiver " << hexpointer(this)
353  << "Not yet read all RMA regions " << hexpointer(this));
354  FUNC_START_DEBUG_MSG;
355  return;
356  }
357 
358  HPX_ASSERT(rma_count_ == 0);
359 
360  // when the chunk structure could not be piggybacked, the chunk_fetch_
361  // flag is set prior to reading the chunks.
362  if (chunk_fetch_) {
363  LOG_DEBUG_MSG("receiver " << hexpointer(this)
364  << "rma read chunk list complete");
365  if (handle_chunks_read_message()>0) {
366  // more rma reads have been started, so exit and wait for them
367  return;
368  }
369  }
370  else {
371  LOG_DEBUG_MSG("receiver " << hexpointer(this)
372  << "all RMA regions now read ");
373  }
374 
375  // If the main message was not piggy backed, then the message region
376  // is either the final chunk of the rma list (if chunks were piggybacked)
377  // or read via rma during the chunk fetch (chunks not piggybacked)
378  if (!header_->message_piggy_back())
379  {
380  if (header_->chunk_ptr()) {
381  message_region_ = rma_regions_.back();
382  //
383  rma_regions_.resize(rma_regions_.size()-1);
384  chunks_.resize(chunks_.size()-1);
385  }
386  else {
387  LOG_DEBUG_MSG("No piggy back message or chunks");
388  // message region should have been read by handle_message_no_chunk_data
389  HPX_ASSERT(message_region_);
390  }
391  }
392 
393  std::size_t message_length = header_->message_size();
394  char *message = nullptr;
395  if (message_region_)
396  {
397  message = static_cast<char *>(message_region_->get_address());
398  HPX_ASSERT(message);
399  LOG_DEBUG_MSG("receiver " << hexpointer(this)
400  << "No piggy_back RDMA message "
401  << "region " << hexpointer(message_region_)
402  << "address " << hexpointer(message_region_->get_address())
403  << "length " << hexuint32(message_length));
404  LOG_TRACE_MSG(
405  CRC32_MEM(message, message_length, "Message region (recv rdma)"));
406 
407  // do this after dumping out data as otherwise we lose some debug info
408  HPX_ASSERT(message_region_->get_message_length() == header_->message_size());
409  }
410  else
411  {
412  HPX_ASSERT(header_->message_data());
413  message = header_->message_data();
414  LOG_TRACE_MSG(CRC32_MEM(message, message_length,
415  "Message region (recv piggyback with rdma)"));
416  }
417 
418  for (auto &r : rma_regions_)
419  {
420  LOG_TRACE_MSG(CRC32_MEM(r->get_address(), r->get_message_length(),
421  "rdma region (recv) "));
422  }
423 
424  // wrap the message and chunks into a pinned vector so that they
425  // can be passed into the parcel decode functions and when released have
426  // the pinned buffers returned to the memory pool
427  typedef pinned_memory_vector<char, header_size, region_type, memory_pool_type>
428  rcv_data_type;
429  typedef parcel_buffer<rcv_data_type, std::vector<char>> rcv_buffer_type;
430 
431  rcv_data_type wrapped_pointer(message, message_length,
432  [this, message, message_length]()
433  {
434  // deleted cleanup until all problems resolved, now after parcel decode
435  if (message_region_) {
436  LOG_TRACE_MSG(CRC32_MEM(message, message_length,
437  "Message region (receiver delete)"));
438  }
439  }, nullptr, nullptr);
440 
441  rcv_buffer_type buffer(std::move(wrapped_pointer), nullptr);
442 
443  LOG_EXCLUSIVE(
444  for (chunk_struct &c : chunks_) {
445  LOG_DEBUG_MSG("get : chunk : size " << hexnumber(c.size_)
446  << " type " << decnumber((uint64_t)c.type_)
447  << " rkey " << hexpointer(c.rkey_)
448  << " cpos " << hexpointer(c.data_.cpos_)
449  << " index " << decnumber(c.data_.index_));
450  });
451 
452  int zc_chunks =
453  std::count_if(chunks_.begin(), chunks_.end(), [](chunk_struct &c) {
454  return c.type_ == serialization::chunk_type_pointer;
455  });
456  int oo_chunks = chunks_.size() - zc_chunks;
457 
458  buffer.num_chunks_ = std::make_pair(zc_chunks, oo_chunks);
459  buffer.data_size_ = header_->message_size();
460  performance_counters::parcels::data_point& data = buffer.data_point_;
461  data.bytes_ = static_cast<std::size_t>(header_->message_size());
462  data.time_ = util::high_resolution_clock::now() - start_time_;
463 
464  LOG_DEBUG_MSG("receiver " << hexpointer(this)
465  << "calling parcel decode for ZEROCOPY complete parcel");
466  std::size_t num_thread = hpx::get_worker_thread_num();
467  decode_message_with_chunks(*pp_, std::move(buffer), 0, chunks_, num_thread);
468  LOG_DEBUG_MSG("receiver " << hexpointer(this)
469  << "parcel decode called for ZEROCOPY complete parcel");
470 
471  LOG_DEBUG_MSG("receiver " << hexpointer(this) << "Sending ack");
472  send_rdma_complete_ack();
473 
474  cleanup_receive();
475  FUNC_END_DEBUG_MSG;
476  }

478  // --------------------------------------------------------------------
479  void rma_receiver::send_rdma_complete_ack()
480  {
481  LOG_DEBUG_MSG("receiver " << hexpointer(this)
482  << "RDMA Get tag " << hexuint64(header_->tag())
483  << " has completed : posting 8 byte ack to origin");
484 
485  ++sent_ack_;
486 
487  int ret = 0;
488  for (std::size_t k = 0; true; ++k)
489  {
490  // when we received the incoming message, the tag was already set
491  // with the sender context so that we can signal it directly
492  // that we have completed RMA and the sender my now cleanup.
493  std::uint64_t tag = header_->tag();
494  ret = fi_inject(endpoint_, &tag, sizeof(std::uint64_t), src_addr_);
495  if (ret == -FI_EAGAIN)
496  {
497  LOG_ERROR_MSG("receiver " << hexpointer(this)
498  << "reposting fi_inject...\n");
499  hpx::util::detail::yield_k(k,
500  "libfabric::receiver::send_rdma_complete_ack");
501  continue;
502  }
503  if (ret) throw fabric_error(ret, "fi_inject ack notification error");
504  break;
505  }

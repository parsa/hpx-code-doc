86 int omp_main(
87  variables_map&
88  )
89 {
90  // Validate command line.
91  if (0 == tasks)
92  throw std::invalid_argument("count of 0 tasks specified\n");
93 
94  // Start the clock.
95  high_resolution_timer t;
96 
97  #pragma omp parallel
98  #pragma omp single
99  {
100  for (std::uint64_t i = 0; i < tasks; ++i)
101  #if _OPENMP>=200805
102  #pragma omp task untied
103  #endif
104  worker_timed(delay * 1000);
105 
106  // Yield until all work is done.
107  #if _OPENMP>=200805
108  #pragma omp taskwait
109  #endif
110 
111  print_results(omp_get_num_threads(), t.elapsed());
112  }
113 
114  return 0;
115 }

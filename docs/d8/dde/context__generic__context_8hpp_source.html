163  // Generic implementation for the context_impl_base class based on
164  // Boost.Context.
165 #if BOOST_VERSION < 106100
166  template <typename T>
167  HPX_FORCEINLINE void trampoline(intptr_t pv)
168  {
169  T* fun = reinterpret_cast<T*>(pv);
170  HPX_ASSERT(fun);
171  (*fun)();
172  std::abort();
173  }
174 #else
175  template <typename T>
176  HPX_FORCEINLINE void trampoline(boost::context::detail::transfer_t tr)
177  {
178  auto arg = reinterpret_cast<std::pair<void*,
179  boost::context::detail::fcontext_t*>*>(tr.data);
180 
181  HPX_ASSERT(arg->second);
182  *arg->second = tr.fctx;
183 
184  T* fun = reinterpret_cast<T*>(arg->first);
185  HPX_ASSERT(fun);
186  (*fun)();
187 
188  std::abort();
189  }

215  // Create a context that on restore invokes Functor on
216  // a new stack. The stack size can be optionally specified.
217  template <typename Functor>
218  fcontext_context_impl(Functor& cb, std::ptrdiff_t stack_size)
219 #if BOOST_VERSION < 106100
220  : cb_(reinterpret_cast<intptr_t>(&cb))
221 #else
222  : cb_(std::make_pair(reinterpret_cast<void*>(&cb), nullptr))
223 #endif
224  , funp_(&trampoline<Functor>)
225 #if BOOST_VERSION > 105500
226  , ctx_(0)
227 #endif
228  , alloc_()
229  , stack_size_(
230  (stack_size == -1) ?
231  alloc_.minimum_stacksize() : std::size_t(stack_size)
232  )
233  , stack_pointer_(alloc_.allocate(stack_size_))
234  {
235 #if BOOST_VERSION < 105600
236  boost::context::fcontext_t* ctx =
237  boost::context::make_fcontext(stack_pointer_, stack_size_, funp_);
238 
239  std::swap(*ctx, ctx_);
240 #elif BOOST_VERSION < 106100
241  ctx_ =
242  boost::context::make_fcontext(stack_pointer_, stack_size_, funp_);
243 #else
244  ctx_ =
245  boost::context::detail::make_fcontext(
246  stack_pointer_, stack_size_, funp_);
247 #endif
248  }
249 
250  ~fcontext_context_impl()
251  {
252 #if BOOST_VERSION < 105600
253  if (ctx_.fc_stack.sp && stack_pointer_)
254 #else
255  if (ctx_ && stack_pointer_)
256 #endif
257  {
258  alloc_.deallocate(stack_pointer_, stack_size_);
259 #if BOOST_VERSION < 105600
260  ctx_.fc_stack.size = 0;
261  ctx_.fc_stack.sp = 0;
262 #endif
263  }

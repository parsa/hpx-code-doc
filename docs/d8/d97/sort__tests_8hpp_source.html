41 // Fill a vector with random numbers in the range [lower, upper]
42 template <typename T>
43 void rnd_fill(std::vector<T> &V, const T lower, const T upper, const T seed)
44 {
45  // use the default random engine and an uniform distribution
46  std::mt19937 eng(static_cast<unsigned int>(seed));
47  std::uniform_real_distribution<double> distr(lower, upper);
48 
49  for (auto &elem : V) {
50  elem = static_cast<T>(distr(eng));
51  }
52 }

55 // generate a random string of a given length
56 std::string random_string( size_t length )
57 {
58  auto randchar = []() -> char
59  {
60  const char charset[] =
61  "0123456789"
62  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
63  "abcdefghijklmnopqrstuvwxyz";
64  const size_t max_index = (sizeof(charset) - 1);
65  return charset[ rand() % max_index ];
66  };
67  std::string str(length,0);
68  std::generate_n( str.begin(), length, randchar );
69  return str;
70 }

87 // check that the array is sorted correctly
88 template <class IA, typename Compare>
89 int verify(const std::vector <IA> &A, Compare comp, std::uint64_t elapsed,
90  bool print)
91 {
92  if (A.size()<2) {
93  // skip checks as we must be sorted
94  }
95  else {
96  IA temp = *(A.begin());
97  for (typename std::vector<IA>::const_iterator it=A.begin();
98  it != A.end(); ++it)
99  {
100  if (comp((*it), temp)) {
101  if (print) std::cout << "fail "
102  << boost::format("%8.6f") % (elapsed / 1e9)
103  << A.size() << std::endl;
104  return 0;
105  }
106  temp = (*it);
107  }
108  }
109  if (print) std::cout
110  << "OK "
111  << boost::format("%8.6f") % (elapsed / 1e9)
112  << A.size() << std::endl;
113  return 1;
114 }

37 std::uint64_t fibonacci(std::uint64_t n)
38 {
39  ++count;
40 
41  if (n < 2)
42  return n;
43 
44  // We restrict ourselves to execute the Fibonacci function locally.
45  hpx::naming::id_type const locality_id = hpx::find_here();
46 
47  // Invoking the Fibonacci algorithm twice is inefficient.
48  // However, we intentionally demonstrate it this way to create some
49  // heavy workload.
50 
51  fibonacci_action fib;
52  hpx::future<std::uint64_t> n1 =
53  hpx::async(fib, locality_id, n - 1);
54  hpx::future<std::uint64_t> n2 =
55  hpx::async(fib, locality_id, n - 2);
56 
57  return n1.get() + n2.get(); // wait for the Futures to return their values
58 }

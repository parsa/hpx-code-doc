120  void runtime_support::free_component_sync(agas::gva const& g,
121  naming::gid_type const& gid, std::uint64_t count)
122  {
123  typedef server::runtime_support::free_component_action action_type;
124 
125  // Determine whether the gid of the component to delete is local or
126  // remote
127  std::pair<bool, components::pinned_ptr> r;
128 
129  naming::address addr;
130  if (g.prefix == hpx::get_locality() ||
131  agas::is_local_address_cached(gid, addr))
132  {
133  typedef action_type::component_type component_type;
134  if (traits::component_supports_migration<component_type>::call())
135  {
136  r = traits::action_was_object_migrated<action_type>::call(
137  hpx::id_type(gid, hpx::id_type::unmanaged), addr.address_);
138  if (!r.first)
139  {
140  // apply locally
141  components::server::runtime_support* p =
142  reinterpret_cast<components::server::runtime_support*>(
143  hpx::get_runtime().get_runtime_support_lva());
144  p->free_component(g, gid, count);
145  return;
146  }
147  }
148  else
149  {
150  // apply locally
151  components::server::runtime_support* p =
152  reinterpret_cast<components::server::runtime_support*>(
153  hpx::get_runtime().get_runtime_support_lva());
154  p->free_component(g, gid, count);
155  return;
156  }
157  }
158 
159  // apply remotely (only if runtime is not stopping)
160  naming::id_type id = get_colocation_id(
161  launch::sync, naming::id_type(gid, naming::id_type::unmanaged));
162 
163  lcos::packaged_action<action_type, void> p;
164  lcos::future<void> f = p.get_future();
165  p.apply(id, g, gid, count);
166  f.get();
167  }

48 template <typename Executor>
49 void test_executors(std::size_t processing_units, std::size_t num_pus)
50 {
51  std::atomic<std::size_t> count_invocations(0);
52  std::size_t const num_tasks = 100;
53 
54  typedef hpx::parallel::executor_information_traits<Executor> infotraits;
55 
56  std::size_t num_execs = processing_units / num_pus;
57 
58  {
59  // create as many executors as we have cores available using num_pus
60  // processing unit each
61  std::vector<Executor> execs;
62  for (std::size_t i = 0; i != num_execs; ++i)
63  execs.push_back(Executor(num_pus));
64 
65  verify_resource_allocation(num_execs, num_pus);
66 
67  // give executors a chance to get started
68  hpx::this_thread::yield();
69 
70  for(Executor& exec : execs)
71  {
72  HPX_TEST_EQ(infotraits::processing_units_count(exec, test::dummy),
73  num_pus);
74  }
75 
76  // schedule a couple of tasks on each of the executors
77  for(Executor& exec : execs)
78  {
79  for (int i = 0; i != num_tasks; ++i)
80  {
81  hpx::parallel::execution::post(exec,
82  [&count_invocations]()
83  {
84  ++count_invocations;
85  });
86  }
87  }
88 
89  // test again
90  for(Executor& exec : execs)
91  {
92  HPX_TEST_EQ(infotraits::processing_units_count(exec, test::dummy),
93  num_pus);
94  }
95  }
96 
97  HPX_TEST_EQ(count_invocations, num_execs * num_tasks);
98 }

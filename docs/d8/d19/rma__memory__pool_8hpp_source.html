52 // Description of memory pool objects:
53 //
54 // memory_region_allocator:
55 // An allocator that returns memory of the requested size. The memory is pinned
56 // and ready to be used for RDMA operations. A memory_region object is
57 // used, it contains the memory registration information needed by the libfabric API.
58 //
59 // rdma_chunk_pool :
60 // Allocate N chunks of memory in one go, a single memory registration is
61 // used for the whole block, and this is divided into smaller blocks that
62 // are then used by the pool_container.
63 // @TODO: Note that originally a boost::pool was used but this was replaced by
64 // a simple block allocation and needs to be cleaned as it puts blocks onto
65 // a stack that is duplicated in pool_container.
66 //
67 // pool_container:
68 // The pool container wraps an rdma_chunk_pool and provides a stack. When a user
69 // requests a small block, one is popped off the stack. At startup, the pool_container
70 // requests a large number of blocks from the rdma_chunk_pool and sets the correct
71 // address offset within each larger chunk for each small block and pushes the mini
72 // Region onto the stack. Thus N small rdma_regions are created from a
73 // single larger one and memory blocks come from contiguous memory.
74 //
75 // rma_memory_pool:
76 // The rma_memory_pool maintains 4 pool_container (stacks) of different sized blocks
77 // so that most user requests can be fulfilled.
78 // If a request cannot be filled, the pool can generate temporary blocks with
79 // new allocations and on-the-fly registration of the memory.
80 // Additionally, it also provides a simple API so users may pass pre-allocated
81 // memory to the pool for on-the-fly registration (rdma transfer of user memory chunks)
82 // and later de-registration.

127  // ---------------------------------------------------------------------------
128  // pool_container, collect some routines for reuse with
129  // small, medium, large chunks etc
130  // ---------------------------------------------------------------------------
131  template <typename RegionProvider,
132  typename Allocator,
133  typename PoolType,
134  std::size_t ChunkSize,
135  std::size_t MaxChunks>
136  struct pool_container
137  {
138  typedef typename RegionProvider::provider_domain domain_type;
139  typedef rma_memory_region<RegionProvider> region_type;
140  typedef std::shared_ptr<region_type> region_ptr;
141 
142  // ------------------------------------------------------------------------
143  pool_container(domain_type *pd) :
144  accesses_(0), in_use_(0), pd_(pd)
145  {
146  }
147 
148  // ------------------------------------------------------------------------
149  bool allocate_pool()
150  {
151  LOG_DEBUG_MSG(PoolType::desc() << "Allocating "
152  << "ChunkSize " << hexuint32(ChunkSize)
153  << "num_chunks " << decnumber(MaxChunks)
154  << "total " << hexuint32(ChunkSize*MaxChunks));
155 
156  // Allocate one very large registered block for N small blocks
157  region_ptr block =
158  Allocator().malloc(pd_, ChunkSize*MaxChunks);
159  // store a copy of this to make sure it is 'alive'
160  block_list_[block->get_address()] = block;
161 
162  // break the large region into N small regions
163  uint64_t offset = 0;
164  for (std::size_t i=0; i<MaxChunks; ++i) {
165  // we must keep a copy of the sub-region since we only pass
166  // pointers to regions around the code.
167  region_list_[i] = region_type(
168  block->get_region(),
169  static_cast<char*>(block->get_base_address()) + offset,
170  static_cast<char*>(block->get_base_address()),
171  ChunkSize,
172  region_type::BLOCK_PARTIAL
173  );
174  LOG_TRACE_MSG(PoolType::desc() << "Allocate Block "
175  << decnumber(i)
176  << region_list_[i]);
177  // push the pointer onto our stack
178  push(&region_list_[i]);
179  offset += ChunkSize;
180  }
181  in_use_ = 0;
182  return true;
183  }
184 
185  // ------------------------------------------------------------------------
186  void DeallocatePool()
187  {
188  if (in_use_!=0) {
189  LOG_ERROR_MSG(PoolType::desc()
190  << "Deallocating free_list : Not all blocks were returned "
191  << " refcounts " << decnumber(in_use_));
192  }
193  region_type* region = nullptr;
194  while (!free_list_.pop(region)) {
195  // clear our stack
196  delete region;
197  }
198  // wipe our copies of sub-regions (no clear function for std::array)
199  std::fill(region_list_.begin(), region_list_.end(), region_type());
200  // release references to shared arrays
201  block_list_.clear();
202  }
203 
204  // ------------------------------------------------------------------------
205  inline void push(region_type *region)
206  {
207  LOG_TRACE_MSG(PoolType::desc() << "Push block " << *region
208  << "Used " << decnumber(in_use_-1)
209  << "Accesses " << decnumber(accesses_));
210 
211  LOG_EXCLUSIVE(
212  uintptr_t val = uintptr_t(region->get_address());
213  LOG_TRACE_MSG(PoolType::desc()
214  << "Writing 0xdeadbeef to region address "
215  << hexpointer(val));
216  if (region->get_address()!=nullptr) {
217  // get use the pointer to the region
218  uintptr_t *ptr = reinterpret_cast<uintptr_t*>(val);
219  for (unsigned int c=0; c<ChunkSize/8; ++c) {
220  ptr[c] = 0xdeadbeef;
221  }
222  }
223  );
224 
225  if (!free_list_.push(region)) {
226  LOG_ERROR_MSG(PoolType::desc() << "Error in memory pool push");
227  }
228  // decrement one reference
229  --in_use_;
230  }
231 
232  // ------------------------------------------------------------------------
233  inline region_type *pop()
234  {
235  // get a block
236  region_type *region = nullptr;
237  if (!free_list_.pop(region)) {
238  LOG_DEBUG_MSG(PoolType::desc() << "Error in memory pool pop");
239  return nullptr;
240  }
241  ++in_use_;
242  ++accesses_;
243  LOG_TRACE_MSG(PoolType::desc() << "Pop block "
244  << *region
245  << "Used " << decnumber(in_use_)
246  << "Accesses " << decnumber(accesses_));
247  return region;
248  }
249 
250  // ------------------------------------------------------------------------
251  // at shutdown we can disregrad any bocks still prepoted as we can't
252  // unpost them
253  void decrement_used_count(uint32_t N) {
254  in_use_ -= N;
255  }
256 
257  // ------------------------------------------------------------------------
258  // for debug log messages
259  std::string status() {
260  std::stringstream temp;
261  temp << "| " << PoolType::desc()
262  << "ChunkSize " << hexlength(ChunkSize)
263  << "Free " << decnumber(MaxChunks-in_use_)
264  << "Used " << decnumber(in_use_)
265  << "Accesses " << decnumber(accesses_);
266  return temp.str();
267  }

270  constexpr std::size_t chunk_size() const { return ChunkSize; }
271  //
272  performance_counter<unsigned int> accesses_;
273  performance_counter<unsigned int> in_use_;
274  //
275  domain_type *pd_;
276  std::unordered_map<const char *, region_ptr> block_list_;
277  std::array<region_type, MaxChunks> region_list_;
278  bl::stack<region_type*, bl::capacity<MaxChunks>> free_list_;
279  };

285  template <typename RegionProvider>

296  // constructor

329  // -------------------------
330  // User allocation interface
331  // -------------------------
332  // The Region* versions of allocate/deallocate
333  // should be used in preference to the std:: compatible
334  // versions using char* for efficiency
335 
336  //----------------------------------------------------------------------------
337  // query the pool for a chunk of a given size to see if one is available
338  // this function is 'unsafe' because it is not thread safe and another
339  // thread may push/pop a block after/during this call and invalidate the result.
340  inline bool can_allocate_unsafe(size_t length) const
341  {
342  if (length<=tiny_.chunk_size()) {
343  return !tiny_.free_list_.empty();
344  }
345  else if (length<=small_.chunk_size()) {
346  return !small_.free_list_.empty();
347  }
348  else if (length<=medium_.chunk_size()) {
349  return !medium_.free_list_.empty();
350  }
351  else if (length<=large_.chunk_size()) {
352  return !large_.free_list_.empty();
353  }
354  return true;
355  }
356 
357  //----------------------------------------------------------------------------
358  // allocate a region, if size=0 a tiny region is returned
359  inline region_type *allocate_region(size_t length)
360  {
361  region_type *region = nullptr;
362  //
363  if (length<=tiny_.chunk_size()) {
364  region = tiny_.pop();
365  }
366  else if (length<=small_.chunk_size()) {
367  region = small_.pop();
368  }
369  else if (length<=medium_.chunk_size()) {
370  region = medium_.pop();
371  }
372  else if (length<=large_.chunk_size()) {
373  region = large_.pop();
374  }
375  // if we didn't get a block from the cache, create one on the fly
376  if (region==nullptr) {
377  region = allocate_temporary_region(length);
378  }
379 
380  LOG_TRACE_MSG("Popping Block "
381  << *region
382  << tiny_.status()
383  << small_.status()
384  << medium_.status()
385  << large_.status()
386  << large_.status()
387  << "temp regions " << decnumber(temp_regions));
388  //
389  return region;
390  }
391 
392  //----------------------------------------------------------------------------
393  // release a region back to the pool
394  inline void deallocate(region_type *region)
395  {
396  // if this region was registered on the fly, then don't return it to the pool
397  if (region->get_temp_region() || region->get_user_region()) {
398  if (region->get_temp_region()) {
399  --temp_regions;
400  LOG_TRACE_MSG("Deallocating temp region "
401  << *region
402  << "temp regions " << decnumber(temp_regions));
403  }
404  else if (region->get_user_region()) {
405  --user_regions;
406  LOG_TRACE_MSG("Deleting (user region) "
407  << *region
408  << "user regions " << decnumber(user_regions));
409  }
410  delete region;
411  return;
412  }
413 
414  // put the block back on the free list
415  if (region->get_size()<=tiny_.chunk_size()) {
416  tiny_.push(region);
417  }
418  else if (region->get_size()<=small_.chunk_size()) {
419  small_.push(region);
420  }
421  else if (region->get_size()<=medium_.chunk_size()) {
422  medium_.push(region);
423  }
424  else if (region->get_size()<=large_.chunk_size()) {
425  large_.push(region);
426  }
427 
428  LOG_TRACE_MSG("Pushing Block "
429  << *region
430  << tiny_.status()
431  << small_.status()
432  << medium_.status()
433  << large_.status()
434  << "temp regions " << decnumber(temp_regions));
435  }
436 
437  //----------------------------------------------------------------------------
438  // allocates a region from the heap and registers it, it bypasses the pool
439  // when deallocted, it will be unregistered and deleted, not returned to the pool
440  inline region_type* allocate_temporary_region(std::size_t length)
441  {
442  region_type *region = new region_type();
443  region->set_temp_region();
444  region->allocate(protection_domain_, length);
445  ++temp_regions;
446  LOG_TRACE_MSG("Allocating temp region "
447  << *region
448  << "temp regions " << decnumber(temp_regions));
449  return region;
450  }

453  // protection domain that memory is registered with
454  domain_type *protection_domain_;
455 
456  // maintain 4 pools of thread safe pre-allocated regions of fixed size.
457  pool_container<RegionProvider, allocator_type, pool_tiny,
458  RDMA_POOL_1K_CHUNK_SIZE, RDMA_POOL_MAX_1K_CHUNKS> tiny_;
459  pool_container<RegionProvider, allocator_type, pool_small,

463  pool_container<RegionProvider, allocator_type, pool_large,
464  RDMA_POOL_LARGE_CHUNK_SIZE, RDMA_POOL_MAX_LARGE_CHUNKS> large_;
465 
466  // counters
467  hpx::util::atomic_count temp_regions;
468  hpx::util::atomic_count user_regions;
469  };

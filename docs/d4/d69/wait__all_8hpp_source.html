233  // Current element is a range (vector or array) of futures
234  template <std::size_t I, typename Iter>
235  void await_range(Iter next, Iter end)
236  {
237  typedef typename std::iterator_traits<Iter>::value_type
238  future_type;
239  typedef typename detail::future_or_shared_state_result<
240  future_type
241  >::type future_result_type;
242 
243  void (wait_all_frame::*f)(Iter, Iter) =
244  &wait_all_frame::await_range<I>;
245 
246  for (; next != end; ++next)
247  {
248  typename traits::detail::shared_state_ptr<
249  future_result_type
250  >::type next_future_data =
251  traits::detail::get_shared_state(*next);
252 
253  if (next_future_data.get() != nullptr &&
254  !next_future_data->is_ready())
255  {
256  next_future_data->execute_deferred();
257 
258  // execute_deferred might have made the future ready
259  if (!next_future_data->is_ready())
260  {
261  // Attach a continuation to this future which will
262  // re-evaluate it and continue to the next element
263  // in the sequence (if any).
264  boost::intrusive_ptr<wait_all_frame> this_(this);
265  next_future_data->set_on_completed(
266  util::deferred_call(f, std::move(this_),
267  std::move(next), std::move(end)));
268  return;
269  }
270  }
271  }
272 
273  // All elements of the sequence are ready now, proceed to the
274  // next argument.
275  do_await<I + 1>(is_end<I + 1>());
276  }

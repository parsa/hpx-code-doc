34 
36  // Compute the minimum, maximum and delta values for each dimension.

38  shen.get_dimension(sheneos::dimension::ye, min_ye, max_ye);
39  double const delta_ye = (max_ye - min_ye) / num_ye_points;
40 
41  double min_temp = 0, max_temp = 0;

44

46  shen.get_dimension(sheneos::dimension::rho, min_rho, max_rho);
47  double const delta_rho = (max_rho - min_rho) / num_rho_points;
48 
50  // Generate the data points, spacing them out equally.
51  std::vector<double> values_ye(num_ye_points);
52  std::vector<std::size_t> sequence_ye(num_ye_points);
53  double ye = min_ye;
54  for (std::size_t i = 0; i < num_ye_points; ++i) {

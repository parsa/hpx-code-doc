53 double ireceive(hpx::naming::id_type dest, std::size_t loop,
54  std::size_t size, std::size_t window_size)
55 {
56  std::size_t skip = SKIP;
57 
58  if (size <= LARGE_MESSAGE_SIZE) {
59  loop *= LOOP_SMALL_MULTIPLIER;
60  skip *= LOOP_SMALL_MULTIPLIER;
61  }

64 
65  // align used buffers on page boundaries
66  unsigned long align_size = getpagesize();
67  (void)align_size;
68  HPX_ASSERT(align_size <= MAX_ALIGNMENT);
69 
70  boost::scoped_array<char> send_buffer(new char[size]);
71  std::memset(send_buffer.get(), 'a', size);
72 
73  hpx::util::high_resolution_timer t;
74 
75  isend_action send;
76  for (std::size_t i = 0; i != loop + skip; ++i) {
77  // do not measure warm up phase
78  if (i == skip)
79  t.restart();
80

82  using hpx::parallel::execution::par;
83 
84  std::size_t const start = 0;
85 
86  auto range = boost::irange(start, window_size);
87  for_each(par, std::begin(range), std::end(range),
88  [&](std::uint64_t j)
89  {
90  send(dest, buffer_type(
91  send_buffer.get(), size, buffer_type::reference));
92  }

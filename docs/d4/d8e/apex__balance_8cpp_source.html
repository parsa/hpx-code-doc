39 size_t next_locality( const std::vector<double> & probs) {
40  static std::default_random_engine generator;
41  static std::uniform_real_distribution<double> distribution(0.0, 1.0);
42  const double eps = 1e-9;
43  double r = distribution(generator);
44 
45  size_t i = 0;
46  for(const double p : probs) {
47  r -= p;
48  if(r < eps) {
49  return i;
50  }
51  ++i;
52  }
53  return 0;
54 }

56 int hpx_main(boost::program_options::variables_map& vm)
57 {
58  // extract command line argument, i.e. fib(N)
59  std::uint64_t n = vm["n-value"].as<std::uint64_t>();
60  std::uint64_t units = vm["units"].as<std::uint64_t>();
61  std::uint64_t blocks = vm["blocks"].as<std::uint64_t>();
62 
63  // Keep track of the time required to execute.
64  hpx::util::high_resolution_timer t;
65 
66  std::vector<hpx::naming::id_type> localities = hpx::find_all_localities();
67  std::vector<double> probabilities(localities.size());
68  probabilities[0] = 1.0;
69 
70  std::cout << "There are " << localities.size() << " localities." << std::endl;
71  std::cout << "Units: " << units << " n: " << n << std::endl;
72 
73  for(std::uint64_t block = 0; block < blocks; ++block) {
74  std::cout << "Block " << block << std::endl;
75  std::list<std::uint64_t> work(units, n);
76  std::list<hpx::lcos::future<double> > futures;
77  for(std::uint64_t & item : work) {
78  do_work_action act;
79  size_t next = next_locality(probabilities);
80  std::cout << "Will issue work to loc " << next << std::endl;
81  futures.push_back(hpx::async(act, localities[next], item));
82  }
83  std::cout << "Issued work for block " << block << std::endl;
84  hpx::lcos::wait_all(futures.begin(), futures.end());
85  std::cout << "Work done for block " << block << std::endl;
86  }
87 
88 
89  char const* fmt = "elapsed time: %1% [s]\n";
90  std::cout << (boost::format(fmt) % t.elapsed());
91 
92  return hpx::finalize(); // Handles HPX shutdown
93 }

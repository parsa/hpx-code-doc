46  tokenizer<char_separator<char>> tokens(contents, sep);
47  for (const auto& t : tokens) {
48  size_t rend = t.find_first_of("\r"), size = t.size();
49  if (rend == size - 1)
50  {
51  someline.push_back(t);
52  }
53  else
54  {
55  char_separator<char> sep2("\r", "", boost::keep_empty_tokens);
56  tokenizer<char_separator<char>> tokens2(t, sep2);
57  for (const auto& u : tokens2) {
58  someline.push_back(u);
59  }
60  }
61  }

63  {
64  currline++;
65  if (someline[p].find('\t') != string::npos)
66  {

69  lineorder.push_back(linenum);
70  }
71  p++;
72  }

20 // Create a distributing factory locally. The distributing factory can
21 // be used to create blocks of components that are distributed across
22 // all localities that support that component type.
23 void test_allgather(std::size_t np)
24 {
25  // Get the component type for our point component.
26  hpx::components::component_type block_type =
27  ag::server::allgather::get_component_type();
28 
29  std::vector<hpx::id_type> localities =
30  hpx::find_all_localities(block_type);
31  std::cout << " Number of localities: " << localities.size()
32  << std::endl;
33 
34  // Create np allgather components. These components will be evenly
35  // distributed among all available localities supporting the
36  // component type.
37  std::vector<hpx::id_type> components =
38  hpx::new_<ag::server::allgather[]>(
39  hpx::default_layout(localities), np).get();
40 
41  hpx::util::high_resolution_timer kernel1time;
42  {
43  std::vector<hpx::lcos::future<void> > init_phase;
44  ag::server::allgather::init_action init;
45  for (std::size_t i=0;i<np;i++) {
46  init_phase.push_back(hpx::async(init, components[i], i, np));
47  }
48  hpx::wait_all(init_phase);
49  }
50  double k1time = kernel1time.elapsed();
51 
52  hpx::util::high_resolution_timer computetime;
53  {
54  std::vector<hpx::lcos::future<void> > compute_phase;
55  ag::server::allgather::compute_action compute;
56  for (std::size_t i=0;i<np;i++) {
57  compute_phase.push_back(hpx::async(compute, components[i], components));
58  }
59  hpx::wait_all(compute_phase);
60  }
61  double ctime = computetime.elapsed();
62 
63  // print out the sums
64  ag::server::allgather::print_action print;
65  for (std::size_t i=0;i<np;i++) {
66  print(components[i]);
67  }
68 
69  std::cout << " init time: " << k1time << std::endl;
70  std::cout << " compute time: " << ctime << std::endl;
71 }

135 // component which allows for it to be created and accessed remotely through

137 struct partition_server
138  : hpx::components::component_base<partition_server>
139 {
140  enum partition_type
141  {
142  left_partition, middle_partition, right_partition
143  };
144 
145  // construct new instances
146  partition_server() {}
147 
148  partition_server(partition_data const& data)
149  : data_(data)
150  {}
151 
152  partition_server(std::size_t size, double initial_value)
153  : data_(size, initial_value)
154  {}
155 
156  // Access data. The parameter specifies what part of the data should be
157  // accessed. As long as the result is used locally, no data is copied,
158  // however as soon as the result is requested from another locality only
159  // the minimally required amount of data will go over the wire.
160  partition_data get_data(partition_type t) const
161  {
162  switch (t)
163  {
164  case left_partition:
165  return partition_data(data_, data_.size()-1);
166 
167  case middle_partition:
168  break;
169 
170  case right_partition:
171  return partition_data(data_, 0);
172 
173  default:
174  HPX_ASSERT(false);
175  break;
176  }
177  return data_;
178  }
179 
180  // Every member function which has to be invoked remotely needs to be
181  // wrapped into a component action. The macro below defines a new type
182  // 'get_data_action' which represents the (possibly remote) member function
183  // partition::get_data().
184  HPX_DEFINE_COMPONENT_DIRECT_ACTION(partition_server, get_data, get_data_action);
185 
186 private:
187  partition_data data_;
188 };

245 struct stepper
246 {
247  // Our data for one time step
248  typedef std::vector<partition> space;
249 
250  // Our operator
251  static double heat(double left, double middle, double right)
252  {
253  return middle + (k*dt/(dx*dx)) * (left - 2*middle + right);
254  }
255 
256  //[stepper_7
257  // The partitioned operator, it invokes the heat operator above on all elements
258  // of a partition.
259  static partition heat_part(partition const& left,
260  partition const& middle, partition const& right)
261  {
262  using hpx::dataflow;
263  using hpx::util::unwrapping;
264 
265  hpx::shared_future<partition_data> middle_data =
266  middle.get_data(partition_server::middle_partition);
267 
268  hpx::future<partition_data> next_middle = middle_data.then(
269  unwrapping(
270  [middle](partition_data const& m) -> partition_data
271  {
272  HPX_UNUSED(middle);
273 
274  // All local operations are performed once the middle data of
275  // the previous time step becomes available.
276  std::size_t size = m.size();
277  partition_data next(size);
278  for (std::size_t i = 1; i != size-1; ++i)
279  next[i] = heat(m[i-1], m[i], m[i+1]);
280  return next;
281  }
282  )
283  );
284 
285  return dataflow(
286  hpx::launch::async,
287  unwrapping(
288  [left, middle, right](partition_data next, partition_data const& l,
289  partition_data const& m, partition_data const& r) -> partition
290  {
291  HPX_UNUSED(left);
292  HPX_UNUSED(right);
293 
294  // Calculate the missing boundary elements once the
295  // corresponding data has become available.
296  std::size_t size = m.size();
297  next[0] = heat(l[size-1], m[0], m[1]);
298  next[size-1] = heat(m[size-2], m[size-1], r[0]);
299 
300  // The new partition_data will be allocated on the same locality
301  // as 'middle'.
302  return partition(middle.get_id(), next);
303  }
304  ),
305  std::move(next_middle),
306  left.get_data(partition_server::left_partition),
307  middle_data,
308  right.get_data(partition_server::right_partition)
309  );
310  }
311  //]
312 
313  // do all the work on 'np' partitions, 'nx' data points each, for 'nt'
314  // time steps
315  space do_work(std::size_t np, std::size_t nx, std::size_t nt);
316 };

320 // 'heat_part_action' which wraps the global function heat_part(). It can be

325 // do all the work on 'np' partitions, 'nx' data points each, for 'nt'
326 // time steps
327 stepper::space stepper::do_work(std::size_t np, std::size_t nx, std::size_t nt)
328 {
329  using hpx::dataflow;
330 
331  std::vector<hpx::id_type> localities = hpx::find_all_localities();
332  std::size_t nl = localities.size(); // Number of localities
333 
334  // U[t][i] is the state of position i at time t.
335  std::vector<space> U(2);
336  for (space& s: U)
337  s.resize(np);
338 
339  // Initial conditions: f(0, i) = i
340  for (std::size_t i = 0; i != np; ++i)
341  U[0][i] = partition(localities[locidx(i, np, nl)], nx, double(i));
342 
343  heat_part_action act;
344  for (std::size_t t = 0; t != nt; ++t)
345  {
346  space const& current = U[t % 2];
347  space& next = U[(t + 1) % 2];
348 
349  for (std::size_t i = 0; i != np; ++i)
350  {
351  // we execute the action on the locality of the middle partition
352  using hpx::util::placeholders::_1;
353  using hpx::util::placeholders::_2;
354  using hpx::util::placeholders::_3;
355  auto Op = hpx::util::bind(act, localities[locidx(i, np, nl)], _1, _2, _3);
356  next[i] = dataflow(
357  hpx::launch::async, Op,
358  current[idx(i, -1, np)], current[i], current[idx(i, +1, np)]
359  );
360  }
361  }
362 
363  // Return the solution at time-step 'nt'.
364  return U[nt % 2];
365 }

368 int hpx_main(boost::program_options::variables_map& vm)
369 {
370  std::uint64_t np = vm["np"].as<std::uint64_t>(); // Number of partitions.
371  std::uint64_t nx = vm["nx"].as<std::uint64_t>(); // Number of grid points.
372  std::uint64_t nt = vm["nt"].as<std::uint64_t>(); // Number of steps.
373 
374  std::vector<hpx::id_type> localities = hpx::find_all_localities();
375  std::size_t nl = localities.size(); // Number of localities
376 
377  if (vm.count("no-header"))
378  header = false;
379 
380  if (np < nl)
381  {
382  std::cout << "The number of partitions should not be smaller than "
383  "the number of localities" << std::endl;
384  return hpx::finalize();
385  }
386 
387  // Create the stepper object
388  stepper step;
389 
390  // Measure execution time.
391  std::uint64_t t = hpx::util::high_resolution_clock::now();
392 
393  // Execute nt time steps on nx grid points and print the final solution.
394  stepper::space solution = step.do_work(np, nx, nt);
395  for (std::size_t i = 0; i != np; ++i)
396  solution[i].get_data(partition_server::middle_partition).wait();
397 
398  std::uint64_t elapsed = hpx::util::high_resolution_clock::now() - t;
399 
400  // Print the final solution
401  if (vm.count("results"))
402  {
403  for (std::size_t i = 0; i != np; ++i)
404  {
405  std::cout << "U[" << i << "] = "
406  << solution[i].get_data(partition_server::middle_partition).get()
407  << std::endl;
408  }
409  }
410  std::uint64_t const num_worker_threads = hpx::get_num_worker_threads();
411  hpx::future<std::uint32_t> locs = hpx::get_num_localities();
412  print_time_results(locs.get(),num_worker_threads, elapsed, nx, np, nt, header);
413 
414  return hpx::finalize();
415 }

103 std::uint64_t average_out_task(std::size_t vector_size)
104 {
105  if (num_overlapping_loops <= 0)
106  {
107  std::uint64_t start = hpx::util::high_resolution_clock::now();
108 
109  for(auto i = 0; i < test_count; i++)
110  measure_task_foreach(vector_size).wait();
111 
112  return (hpx::util::high_resolution_clock::now() - start) / test_count;
113  }
114 
115  std::vector<hpx::shared_future<void> > tests;
116  tests.resize(num_overlapping_loops);
117 
118  std::uint64_t start = hpx::util::high_resolution_clock::now();
119 
120  for(auto i = 0; i < test_count; i++)
121  {
122  hpx::future<void> curr = measure_task_foreach(vector_size);
123  if (i >= num_overlapping_loops)
124  tests[(i-num_overlapping_loops) % tests.size()].wait();
125  tests[i % tests.size()] = curr.share();
126  }
127 
128  hpx::wait_all(tests);
129  return (hpx::util::high_resolution_clock::now() - start) / test_count;
130 }

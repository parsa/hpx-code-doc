83  inline void* alloc_stack(std::size_t size)
84  {
85  void* real_stack = ::mmap(nullptr,
86  size + EXEC_PAGESIZE,
87  PROT_EXEC | PROT_READ | PROT_WRITE,
88 #if defined(__APPLE__)
89  MAP_PRIVATE | MAP_ANON | MAP_NORESERVE,
90 #elif defined(__FreeBSD__)
91  MAP_PRIVATE | MAP_ANON,
92 #else
93  MAP_PRIVATE | MAP_ANONYMOUS | MAP_NORESERVE,
94 #endif
95  -1,
96  0
97  );
98 
99  if (real_stack == MAP_FAILED)
100  {
101  if (ENOMEM == errno)
102  throw std::runtime_error("mmap() failed to allocate thread stack due "
103  "to insufficient resources, "
104  "increase /proc/sys/vm/max_map_count or add "
105  "-Ihpx.stacks.use_guard_pages=0 to the command line");
106  else
107  throw std::runtime_error("mmap() failed to allocate thread stack");
108  }

111  if (use_guard_pages)
112  {
113  // Add a guard page.
114  ::mprotect(real_stack, EXEC_PAGESIZE, PROT_NONE);

116  void** stack = static_cast<void**>(real_stack) + (EXEC_PAGESIZE

126  inline void watermark_stack(void* stack, std::size_t size)
127  {
128  HPX_ASSERT(size > EXEC_PAGESIZE);
129 
130  // Fill the bottom 8 bytes of the first page with 1s.
131  void** watermark = static_cast<void**>(stack) + ((size - EXEC_PAGESIZE)
132  / sizeof(void*));
133  *watermark = reinterpret_cast<void*>(0xDEADBEEFDEADBEEFull);
134  }

136  inline bool reset_stack(void* stack, std::size_t size)
137  {
138  void** watermark = static_cast<void**>(stack) + ((size - EXEC_PAGESIZE)
139  / sizeof(void*));
140 
141  // If the watermark has been overwritten, then we've gone past the first
142  // page.
143  if ((reinterpret_cast<void*>(0xDEADBEEFDEADBEEFull)) != *watermark)
144  {
145  // We never free up the first page, as it's initialized only when the
146  // stack is created.
147  ::madvise(stack, size - EXEC_PAGESIZE, MADV_DONTNEED);
148  return true;
149  }
150 
151  return false;
152  }

182 bool symbol_namespace::bind(
183  std::string key
184  , naming::gid_type gid
185  )
186 { // {{{ bind implementation
187  // parameters
188  util::scoped_timer<std::atomic<std::int64_t> > update(
189  counter_data_.bind_.time_
190  );
191  counter_data_.increment_bind_count();
192 
193  std::unique_lock<mutex_type> l(mutex_);
194 
195  gid_table_type::iterator it = gids_.find(key);
196  gid_table_type::iterator end = gids_.end();
197 
198  if (it != end)
199  {
200  std::int64_t const credits = naming::detail::get_credit_from_gid(gid);
201  naming::gid_type raw_gid = *(it->second);
202 
203  naming::detail::strip_internal_bits_from_gid(raw_gid);
204  naming::detail::strip_internal_bits_from_gid(gid);
205 
206  // increase reference count
207  if (raw_gid == gid)
208  {
209  LAGAS_(info) << (boost::format(
210  "symbol_namespace::bind, key(%1%), gid(%2%), old_credit(%3%), "
211  "new_credit(%4%)")
212  % key % gid
213  % naming::detail::get_credit_from_gid(*(it->second))
214  % (naming::detail::get_credit_from_gid(*(it->second)) + credits));
215 
216  // REVIEW: do we need to add the credit of the argument to the table?
217  naming::detail::add_credit_to_gid(*(it->second), credits);
218 
219  return true;
220  }
221 
222  if (LAGAS_ENABLED(info))
223  {
224  naming::detail::add_credit_to_gid(gid, credits);
225  LAGAS_(info) << (boost::format(
226  "symbol_namespace::bind, key(%1%), gid(%2%), response(no_success)")
227  % key % gid);
228  }
229 
230  return false;
231  }
232 
233  if (HPX_UNLIKELY(!util::insert_checked(gids_.insert(
234  std::make_pair(key, std::make_shared<naming::gid_type>(gid))))))
235  {
236  l.unlock();
237 
238  HPX_THROW_EXCEPTION(lock_error
239  , "symbol_namespace::bind"
240  , "GID table insertion failed due to a locking error or "
241  "memory corruption");
242  }
243 
244  // handle registered events
245  typedef on_event_data_map_type::iterator iterator;
246  std::pair<iterator, iterator> p = on_event_data_.equal_range(key);
247 
248  std::vector<hpx::id_type> lcos;
249  if (p.first != p.second)
250  {
251  iterator it = p.first;
252  while (it != p.second)
253  {
254  lcos.push_back((*it).second);
255  ++it;
256  }
257 
258  on_event_data_.erase(p.first, p.second);
259 
260  // notify all LCOS which were registered with this name
261  for (hpx::id_type const& id : lcos)
262  {
263  // re-locate the entry in the GID table for each LCO anew, as we
264  // need to unlock the mutex protecting the table for each iteration
265  // below
266  gid_table_type::iterator gid_it = gids_.find(key);
267  if (gid_it == gids_.end())
268  {
269  l.unlock();
270 
271  HPX_THROW_EXCEPTION(invalid_status
272  , "symbol_namespace::bind"
273  , "unable to re-locate the entry in the GID table");
274  }
275 
276  // hold on to the gid while the map is unlocked
277  std::shared_ptr<naming::gid_type> current_gid = gid_it->second;
278 
279  {
280  util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
281 
282  // split the credit as the receiving end will expect to keep the
283  // object alive
284  naming::gid_type new_gid =
285  naming::detail::split_gid_if_needed(*current_gid).get();
286 
287  // trigger the lco
288  set_lco_value(id, std::move(new_gid));
289  }
290  }
291  }
292 
293  l.unlock();
294 
295  LAGAS_(info) << (boost::format(
296  "symbol_namespace::bind, key(%1%), gid(%2%)")
297  % key % gid);
298 
299  return true;
300 } // }}}

438  on_event_data_map_type::value_type(std::move(name), lco));
439 
440  l.unlock();
441 
442  if (it == on_event_data_.end())
443  {
444  LAGAS_(info) << (boost::format(
445  "symbol_namespace::on_event, name(%1%), response(no_success)")
446  % name);
447 
448  return false;
449  }
450  }
451  else
452  {
453  l.unlock();
454  }
455 
456  LAGAS_(info) << "symbol_namespace::on_event";
457

461 naming::gid_type symbol_namespace::statistics_counter(std::string const& name)
462 { // {{{ statistics_counter implementation
463  LAGAS_(info) << "symbol_namespace::statistics_counter";
464 
465  performance_counters::counter_path_elements p;
466  performance_counters::get_counter_path_elements(name, p);
467 
468  if (p.objectname_ != "agas")
469  {
470  HPX_THROW_EXCEPTION(bad_parameter,
471  "symbol_namespace::statistics_counter",
472  "unknown performance counter (unrelated to AGAS)");
473  }
474 
475  namespace_action_code code = invalid_request;
476  detail::counter_target target = detail::counter_target_invalid;
477  for (std::size_t i = 0;
478  i != detail::num_symbol_namespace_services;
479  ++i)
480  {
481  if (p.countername_ == detail::symbol_namespace_services[i].name_)
482  {
483  code = detail::symbol_namespace_services[i].code_;
484  target = detail::symbol_namespace_services[i].target_;
485  break;
486  }
487  }
488 
489  if (code == invalid_request || target == detail::counter_target_invalid)
490  {
491  HPX_THROW_EXCEPTION(bad_parameter,
492  "symbol_namespace::statistics_counter",
493  "unknown performance counter (unrelated to AGAS)");
494  }
495 
496  typedef symbol_namespace::counter_data cd;
497 
498  using util::placeholders::_1;
499  util::function_nonser<std::int64_t(bool)> get_data_func;
500  if (target == detail::counter_target_count)
501  {
502  switch (code) {
503  case symbol_ns_bind:
504  get_data_func = util::bind(&cd::get_bind_count, &counter_data_, _1);
505  break;
506  case symbol_ns_resolve:
507  get_data_func = util::bind(&cd::get_resolve_count, &counter_data_, _1);
508  break;
509  case symbol_ns_unbind:
510  get_data_func = util::bind(&cd::get_unbind_count, &counter_data_, _1);
511  break;
512  case symbol_ns_iterate_names:
513  get_data_func = util::bind(&cd::get_iterate_names_count,
514  &counter_data_, _1);
515  break;
516  case symbol_ns_on_event:
517  get_data_func = util::bind(&cd::get_on_event_count, &counter_data_, _1);
518  break;
519  case symbol_ns_statistics_counter:
520  get_data_func = util::bind(&cd::get_overall_count, &counter_data_, _1);
521  break;
522  default:
523  HPX_THROW_EXCEPTION(bad_parameter
524  , "symbol_namespace::statistics"
525  , "bad action code while querying statistics");
526  }
527  }
528  else {
529  HPX_ASSERT(detail::counter_target_time == target);
530  switch (code) {
531  case symbol_ns_bind:
532  get_data_func = util::bind(&cd::get_bind_time, &counter_data_, _1);
533  break;
534  case symbol_ns_resolve:
535  get_data_func = util::bind(&cd::get_resolve_time, &counter_data_, _1);
536  break;
537  case symbol_ns_unbind:
538  get_data_func = util::bind(&cd::get_unbind_time, &counter_data_, _1);
539  break;
540  case symbol_ns_iterate_names:
541  get_data_func = util::bind(&cd::get_iterate_names_time,
542  &counter_data_, _1);
543  break;
544  case symbol_ns_on_event:
545  get_data_func = util::bind(&cd::get_on_event_time, &counter_data_, _1);
546  break;
547  case symbol_ns_statistics_counter:
548  get_data_func = util::bind(&cd::get_overall_time, &counter_data_, _1);
549  break;
550  default:
551  HPX_THROW_EXCEPTION(bad_parameter
552  , "symbol_namespace::statistics"
553  , "bad action code while querying statistics");
554  }
555  }
556 
557  performance_counters::counter_info info;
558  performance_counters::get_counter_type(name, info);
559 
560  performance_counters::complement_counter_info(info);
561 
562  using performance_counters::detail::create_raw_counter;
563  naming::gid_type gid = create_raw_counter(info, get_data_func, hpx::throws);
564  return naming::detail::strip_credits_from_gid(gid);
565 } // }}}

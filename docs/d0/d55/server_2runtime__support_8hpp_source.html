754  template <typename Component>
755  naming::gid_type runtime_support::migrate_component_to_here(
756  std::shared_ptr<Component> const& p, naming::id_type to_migrate)
757  {
758  components::component_type const type =
759  components::get_component_type<
760  typename Component::wrapped_type>();
761 
762  std::shared_ptr<component_factory_base> factory;
763  naming::gid_type migrated_id;
764 
765  {
766  std::unique_lock<component_map_mutex_type> l(cm_mtx_);
767  component_map_type::const_iterator it = components_.find(type);
768  if (it == components_.end()) {
769  std::ostringstream strm;
770  strm << "attempt to migrate component instance of "
771  << "invalid/unknown type: "
772  << components::get_component_type_name(type)
773  << " (component type not found in map)";
774 
775  l.unlock();
776  HPX_THROW_EXCEPTION(hpx::bad_component_type,
777  "runtime_support::migrate_component_to_here",
778  strm.str());
779  return naming::invalid_gid;
780  }
781 
782  if (!(*it).second.first) {
783  std::ostringstream strm;
784  strm << "attempt to migrate component instance of "
785  << "invalid/unknown type: "
786  << components::get_component_type_name(type)
787  << " (map entry is nullptr)";
788 
789  l.unlock();
790  HPX_THROW_EXCEPTION(hpx::bad_component_type,
791  "runtime_support::migrate_component_to_here",
792  strm.str());
793  return naming::invalid_gid;
794  }
795 
796  // create a local instance by copying the bits and remapping the id in
797  // AGAS
798  migrated_id = to_migrate.get_gid();
799  factory = (*it).second.first;
800  }
801 
802  typedef typename Component::wrapping_type wrapping_type;
803  typename wrapping_type::derived_type* new_instance = nullptr;
804 
805  naming::gid_type id = factory->create_with_args(migrated_id,
806  detail::construct_function<wrapping_type>(std::move(*p)),
807  reinterpret_cast<void**>(&new_instance));
808 
809  // sanity checks
810  if (!id || new_instance == nullptr)
811  {
812  // we should not get here (id should not be invalid)
813  HPX_THROW_EXCEPTION(hpx::invalid_status,
814  "runtime_support::migrate_component_to_here",
815  "could not create copy of given component");
816  return naming::invalid_gid;
817  }
818  if (id != migrated_id)
819  {
820  // we should not get here either (the ids should be the same)
821  HPX_THROW_EXCEPTION(hpx::invalid_status,
822  "runtime_support::migrate_component_to_here",
823  "could not create copy of given component (the new id is "
824  "different from the original id)");
825  return naming::invalid_gid;
826  }
827 
828  LRT_(info) << "successfully migrated component " << id
829  << " of type: " << components::get_component_type_name(type)
830  << " to locality: " << find_here();
831 
832  // inform the newly created component that it has been migrated
833  new_instance->on_migrated();
834 
835  // At this point the object has been fully migrated. We now remove
836  // the object from the AGAS table of migrated objects. This is
837  // necessary as this object might have been migrated off this locality
838  // before it was migrated back.
839  agas::unmark_as_migrated(id);
840 
841  to_migrate.make_unmanaged();
842 
843  return id;
844  }

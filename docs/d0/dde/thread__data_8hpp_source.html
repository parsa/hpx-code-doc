151  thread_state set_state(thread_state_enum state,
152  thread_state_ex_enum state_ex = wait_unknown)
153  {
154  thread_state prev_state =
155  current_state_.load(std::memory_order_acquire);
156 
157  for (;;) {
158  thread_state tmp = prev_state;
159 
160  // ABA prevention for state only (not for state_ex)
161  std::int64_t tag = tmp.tag();
162  if (state != tmp.state())
163  ++tag;
164 
165  if (state_ex == wait_unknown)
166  state_ex = tmp.state_ex();
167 
168  if (HPX_LIKELY(current_state_.compare_exchange_strong(tmp,
169  thread_state(state, state_ex, tag))))
170  {
171  return prev_state;
172  }
173 
174  prev_state = tmp;
175  }
176  }

178  bool set_state_tagged(thread_state_enum newstate,
179  thread_state& prev_state, thread_state& new_tagged_state)
180  {
181  thread_state tmp = prev_state;
182  thread_state_ex_enum state_ex = tmp.state_ex();
183 
184  new_tagged_state = thread_state(newstate, state_ex,
185  prev_state.tag() + 1);
186 
187  if (!current_state_.compare_exchange_strong(tmp, new_tagged_state))
188  return false;
189 
190  prev_state = tmp;
191  return true;

215  bool restore_state(thread_state new_state, thread_state old_state)
216  {
217  // ABA prevention for state only (not for state_ex)
218  std::int64_t tag = old_state.tag();
219  if (new_state.state() != old_state.state())
220  ++tag;
221 
222  // ignore the state_ex while compare-exchanging
223  thread_state_ex_enum state_ex =
224  current_state_.load(std::memory_order_relaxed).state_ex();
225 
226  thread_state old_tmp(old_state.state(), state_ex, old_state.tag());
227  thread_state new_tmp(new_state.state(), state_ex, tag);
228 
229  return current_state_.compare_exchange_strong(old_tmp, new_tmp);
230  }

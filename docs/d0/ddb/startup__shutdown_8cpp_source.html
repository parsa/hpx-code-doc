7 // The purpose of this example is to show how you can access the command line
8 // arguments from inside a startup function of a shared library which is loaded
9 // for a HPX component.
10 //
11 // The startup function is called as a HPX-thread during runtime startup. It
12 // is guaranteed to be executed before hpx_main() is invoked.
13 //
14 // The HPX API function retrieve_commandline_arguments() demonstrated below
15 // expectes 2 arguments: a) an options_description object describing any
16 // special options to be recognized by this component and b) a variables_map
17 // object which on return will be initialized with all found command line
18 // options and arguments passed while invoking the application on the locality
19 // where this component got loaded.
20 
21 // The actual component of this module (see the subdirectory /server) does not
22 // expose any useful functionality. It has been added for the sole purpose of
23 // turning this module into a HPX component, as otherwise the startup function
24 // would not be executed by the runtime system.

39 namespace startup_shutdown
40 {
41  // This function will be registered as a startup function for HPX below.
42  //
43  // That means it will be executed in a HPX-thread before hpx_main, but
44  // after the runtime has been initialized and started.
45  void startup()
46  {
47  using boost::program_options::options_description;
48  using boost::program_options::variables_map;
49 
50  options_description desc_commandline("startup_shutdown_component");
51  desc_commandline.add_options()
52  ("additional",
53  "this is an additional option for the startup_shutdown_compnent")
54  ;
55 
56  // Retrieve command line using the Boost.ProgramOptions library.
57  variables_map vm;
58  if (!hpx::util::retrieve_commandline_arguments(desc_commandline, vm)) {
59  HPX_THROW_EXCEPTION(hpx::not_implemented,
60  "startup_shutdown::startup",
61  "Failed to handle command line options");
62  }
63 
64  if (vm.count("additional")) {
65  std::cout << "Found additional option on command line!"
66  << std::endl;
67  }
68  }

19 void run_with_large_stack()
20 {
21  int const array_size = 1000000;
22 
23  // Allocating a huge array on the stack would normally cause problems.
24  // For this reason, this function is scheduled on a thread using a large
25  // stack (see below).
26  char large_array[array_size]; // allocate 1 MByte of memory
27 
28  std::fill(large_array, &large_array[array_size], '\0');

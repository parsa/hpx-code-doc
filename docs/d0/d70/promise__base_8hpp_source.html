153  // Promise base contains the actual implementation for the remotely
154  // settable
155  // promise. It consists of two parts:
156  // 1) The shared state, which is used to signal the future
157  // 2) The LCO, which is used to set the promise remotely
158  //
159  // The LCO is a component, which lifetime is completely controlled by
160  // the shared state. That is, in the GID that we send along as the
161  // continuation can be unmanaged, AGAS doesn't participate in the
162  // promise lifetime management. The LCO is being reset once the shared
163  // state either contains a value or the data, which is set through the
164  // LCO interface, which can go out of scope safely when the shared
165  // state is set.
166  template <typename Result, typename RemoteResult, typename SharedState>
167  class promise_base
168  : public hpx::lcos::local::detail::promise_base<Result, SharedState>
169  {
170  typedef hpx::lcos::local::detail::promise_base<
171  Result, SharedState
172  > base_type;
173 
174  protected:
175  typedef Result result_type;
176  typedef SharedState shared_state_type;
177  typedef boost::intrusive_ptr<shared_state_type> shared_state_ptr;
178 
179  typedef promise_lco<Result, RemoteResult> wrapped_type;
180  typedef components::managed_component<wrapped_type> wrapping_type;
181 
182  public:
183  promise_base()
184  : base_type()
185  , id_retrieved_(false)
186  {
187  init_shared_state();
188  }
189 
190  template <typename Allocator>
191  promise_base(std::allocator_arg_t, Allocator const& a)
192  : base_type(std::allocator_arg, a)
193  , id_retrieved_(false)
194  {
195  init_shared_state();
196  }
197 
198  promise_base(promise_base&& other) noexcept
199  : base_type(std::move(other)),

229  naming::id_type get_id(bool mark_as_started = true,
230  error_code& ec = throws) const
231  {

233  {
234  HPX_THROWS_IF(ec, no_state,
235  "detail::promise_base<Result, RemoteResult>::get_id",
236  "this promise has no valid shared state");

239  if (!addr_ || !id_)

293  protected:
294  void init_shared_state()
295  {
296  // The lifetime of the LCO (component) part is completely
297  // handled by the shared state, we create the object to get our
298  // gid and then attach it to the completion handler of the
299  // shared state.
300  typedef std::unique_ptr<wrapping_type> wrapping_ptr;
301  wrapping_ptr lco_ptr(
302  new wrapping_type(new wrapped_type(this->shared_state_)));
303 
304  id_ = lco_ptr->get_unmanaged_id();
305  addr_ = naming::address(hpx::get_locality(),
306  lco_ptr->get_component_type(),
307  lco_ptr.get());
308 
309  // Pass id to shared state if it exposes the set_id() function
310  detail::call_set_id(this->shared_state_, id_, id_retrieved_);
311 
312  // This helper is used to keep the component alive until the
313  // completion handler has been called. We need to manually free
314  // the component here, since we don't rely on reference counting
315  // anymore
316  auto keep_alive = hpx::util::deferred_call(
317  [](wrapping_ptr ptr)
318  {
319  delete ptr->get(); // delete wrapped_type
320  },
321  std::move(lco_ptr));
322  this->shared_state_->set_on_completed(std::move(keep_alive));
323  }

325  void check_abandon_shared_state(const char* fun)
326  {
327  if (this->shared_state_ != nullptr &&
328  this->future_retrieved_ &&
329  !(this->shared_state_->is_ready() || id_retrieved_))
330  {
331  this->shared_state_->set_error(broken_promise, fun,
332  "abandoning not ready shared state");
333  return;
334  }
335  }
336  mutable bool id_retrieved_;
337 
338  naming::id_type id_;
339  naming::address addr_;
340  };
341  }

333  template <typename T, typename Data /*= std::vector<T> */>
334  template <typename DistPolicy, typename Create>
335  void partitioned_vector<T, Data>::create(
336  DistPolicy const& policy, Create&& creator)
337  {
338  std::size_t num_parts =
339  traits::num_container_partitions<DistPolicy>::call(policy);
340  std::size_t part_size = (size_ + num_parts - 1) / num_parts;
341 
342  // create as many partitions as required
343  hpx::future<std::vector<bulk_locality_result>> f =
344  creator(policy, num_parts, part_size);
345 
346  // now initialize our data structures
347  std::uint32_t this_locality = get_locality_id();
348  std::vector<future<void>> ptrs;
349 
350  std::size_t num_part = 0;
351  std::size_t allocated_size = 0;
352 
353  std::size_t l = 0;
354 
355  // Fixing the size of partitions to avoid race conditions between
356  // possible reallocations during push back and the continuation
357  // to set the local partition data
358  partitions_.resize(num_parts);
359  for (bulk_locality_result const& r : f.get())
360  {
361  using naming::get_locality_id_from_id;
362  std::uint32_t locality = get_locality_id_from_id(r.first);
363  for (hpx::id_type const& id : r.second)
364  {
365  std::size_t size = (std::min)(part_size, size_ - allocated_size);
366  partitions_[l] = partition_data(id, size, locality);
367 
368  if (locality == this_locality)
369  {
370  using util::placeholders::_1;
371  ptrs.push_back(
372  get_ptr<partitioned_vector_partition_server>(id).then(
373  util::bind(&partitioned_vector::get_ptr_helper, l,
374  std::ref(partitions_), _1)));
375  }
376  ++l;
377 
378  allocated_size += size;
379  if (++num_part == num_parts)
380  {
381  HPX_ASSERT(allocated_size == size_);
382 
383  // shrink last partition, if appropriate
384  if (size != part_size)
385  {
386  partitioned_vector_partition_client(
387  partitions_[l - 1].partition_)
388  .resize(size);
389  }
390  break;
391  }
392  else
393  {
394  HPX_ASSERT(size == part_size);
395  }
396  HPX_ASSERT(l < num_parts);
397  }
398  }
399  HPX_ASSERT(l == num_parts);
400 
401  when_all(ptrs).get();
402 
403  // cache our partition size
404  partition_size_ = get_partition_size();
405  }

436  HPX_PARTITIONED_VECTOR_SPECIALIZATION_EXPORT void
437  partitioned_vector<T, Data>::copy_from(partitioned_vector const& rhs)
438  {

440

442  const_iterator end = rhs.partitions_.end();

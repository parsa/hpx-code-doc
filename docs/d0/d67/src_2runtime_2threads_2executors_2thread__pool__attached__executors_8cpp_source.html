76  // Schedule given function for execution in this executor no sooner
77  // than time abs_time. This call never blocks, and may violate
78  // bounds on the executor's queue size.
79  template <typename Scheduler>
80  void thread_pool_attached_executor<Scheduler>::add_at(
81  util::steady_clock::time_point const& abs_time,
82  closure_type && f, util::thread_description const& description,
83  threads::thread_stacksize stacksize, error_code& ec)
84  {
85  if (stacksize == threads::thread_stacksize_default)
86  stacksize = stacksize_;
87 
88  // create new thread
89  thread_id_type id = register_thread_nullary(
90  std::move(f), description, suspended, false,
91  priority_, get_next_thread_num(), stacksize, ec);
92  if (ec) return;
93 
94  HPX_ASSERT(invalid_thread_id != id); // would throw otherwise
95 
96  // now schedule new thread for execution
97  set_thread_state(id, abs_time);
98  }

100  // Schedule given function for execution in this executor no sooner
101  // than time rel_time from now. This call never blocks, and may
102  // violate bounds on the executor's queue size.
103  template <typename Scheduler>
104  void thread_pool_attached_executor<Scheduler>::add_after(
105  util::steady_clock::duration const& rel_time,
106  closure_type && f, util::thread_description const& description,
107  threads::thread_stacksize stacksize, error_code& ec)
108  {

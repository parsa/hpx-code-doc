69  inline std::string
70  trim_whitespace (std::string const &s)
71  {
72  typedef std::string::size_type size_type;
73 
74  size_type first = s.find_first_not_of (" \t\r\n");
75  if (std::string::npos == first)
76  return (std::string ());
77 
78  size_type last = s.find_last_not_of (" \t\r\n");
79  return s.substr (first, last - first + 1);
80  }

157 {
158 #if defined(__AIX__) && defined(__GNUC__)
159  // NEVER ask why... seems to be some weird stdlib initialization problem
160  // If you don't call getline() here the while(getline...) loop below will
161  // crash with a bad_cast exception. Stupid AIX...
162  std::string l1;
163  std::ifstream i1;
164  i1.open(filename.c_str(), std::ios::in);
165  std::getline(i1, l1);
166  i1.close();

168 
169  // build ini - open file and parse each line
170  std::ifstream input(filename.c_str (), std::ios::in);
171  if (!input.is_open())
172  line_msg("Cannot open file: ", filename);
173 
174  // initialize
175  if (!regex_init())
176  line_msg ("Cannot init regex for ", filename);
177 
178  // read file
179  std::string line;
180  std::vector <std::string> lines;
181  while (std::getline(input, line))
182  lines.push_back (line);
183 
184  // parse file
185  parse(filename, lines, false);

199 // parse file
200 void section::parse (std::string const& sourcename,
201  std::vector<std::string> const& lines, bool verify_existing,
202  bool weed_out_comments)
203 {
204  int linenum = 0;
205  section* current = this;
206 
207  boost::regex regex_comment (pattern_comment, boost::regex::perl
208  | boost::regex::icase);
209  boost::regex regex_section (pattern_section, boost::regex::perl
210  | boost::regex::icase);
211  boost::regex regex_qualified_entry
212  (pattern_qualified_entry, boost::regex::perl | boost::regex::icase);
213  boost::regex regex_entry (pattern_entry, boost::regex::perl | boost::regex::icase);
214 
215  std::vector<std::string>::const_iterator end = lines.end();
216  for (std::vector<std::string>::const_iterator it = lines.begin();
217  it != end; ++it)
218  {
219  ++linenum;
220 
221  // remove trailing new lines and white spaces
222  std::string line(detail::trim_whitespace (*it));
223 
224  // skip if empty line
225  if (line.empty())
226  continue;
227 
228  // weed out comments
229  if (weed_out_comments)
230  {
231  boost::smatch what_comment;
232  if (boost::regex_match (line, what_comment, regex_comment))
233  {
234  HPX_ASSERT(3 == what_comment.size());
235 
236  line = detail::trim_whitespace (what_comment[1]);
237  if (line.empty())
238  continue;
239  }
240  }
241 
242  // no comments anymore: line is either section, key=val,
243  // or garbage/empty
244  boost::smatch what;
245  if (boost::regex_match(line, what, regex_qualified_entry))
246  {
247  // found a entry line
248  if (4 != what.size()) //-V112
249  {
250  line_msg("Cannot parse key/value in: ", sourcename, linenum, line);
251  }
252 
253  section* s = current; // save the section we're in
254  current = this; // start adding sections at the root
255 
256  // got the section name. It might be hierarchical, so split it up, and
257  // for each elem, check if we have it. If not, create it, and add
258  std::string sec_name (what[1]);
259  std::string::size_type pos = 0;
260  for (std::string::size_type pos1 = sec_name.find_first_of('.');
261  std::string::npos != pos1;
262  pos1 = sec_name.find_first_of ('.', pos = pos1 + 1))
263  {
264  current = current->add_section_if_new(
265  sec_name.substr(pos, pos1 - pos));
266  }
267 
268  current = current->add_section_if_new(sec_name.substr(pos));
269 
270  // add key/val to this section
271  std::unique_lock<mutex_type> l(current->mtx_);
272 
273  std::string key(what[2]);
274  if (!force_entry(key) && verify_existing && !current->has_entry(l, key))
275  {
276  line_msg ("Attempt to initialize unknown entry: ", sourcename,
277  linenum, line);
278  }
279 
280  current->add_entry (l, sec_name + "." + key, key, what[3]);
281 
282  // restore the old section
283  current = s;
284  }
285 
286  else if (boost::regex_match(line, what, regex_section))
287  {
288  // found a section line
289  if (2 != what.size())
290  {
291  line_msg("Cannot parse section in: ", sourcename, linenum, line);
292  }
293 
294  current = this; // start adding sections at the root
295 
296  // got the section name. It might be hierarchical, so split it up, and
297  // for each elem, check if we have it. If not, create it, and add
298  std::string sec_name (what[1]);
299  std::string::size_type pos = 0;
300  for (std::string::size_type pos1 = sec_name.find_first_of('.');
301  std::string::npos != pos1;
302  pos1 = sec_name.find_first_of ('.', pos = pos1 + 1))
303  {
304  current = current->add_section_if_new(
305  sec_name.substr(pos, pos1 - pos));
306  }
307 
308  current = current->add_section_if_new(sec_name.substr(pos));
309  }
310 
311  // did not match section, so might be key/val entry
312  else if ( boost::regex_match (line, what, regex_entry) )
313  {
314  // found a entry line
315  if (3 != what.size())
316  {
317  line_msg("Cannot parse key/value in: ", sourcename, linenum, line);
318  }
319 
320  // add key/val to current section
321  std::unique_lock<mutex_type> l(current->mtx_);
322 
323  std::string key(what[1]);
324  if (!force_entry(key) && verify_existing && !current->has_entry(l, key))
325  {
326  line_msg ("Attempt to initialize unknown entry: ", sourcename,
327  linenum, line);
328  }
329  current->add_entry (l, key, key, what[2]);
330  }
331  else
332  {
333  // Hmm, is not a section, is not an entry, is not empty - must be
334  // an error!
335  line_msg ("Cannot parse line at: ", sourcename, linenum, line);
336  }
337  }
338 }

450 void section::add_entry (std::unique_lock<mutex_type>& l,
451  std::string const& fullkey, std::string const& key, std::string val)
452 {
453  // first expand the full property name in the value (avoids infinite recursion)
454  expand_only(l, val, std::string::size_type(-1), get_full_name() + "." + key);
455 
456  std::string::size_type i = key.find_last_of(".");
457  if (i != std::string::npos)
458  {
459  section* current = root_;
460 
461  // make sure all sections in key exist
462  std::string sec_name = key.substr(0, i);
463 
464  std::string::size_type pos = 0;
465  for (std::string::size_type pos1 = sec_name.find_first_of('.');
466  std::string::npos != pos1;
467  pos1 = sec_name.find_first_of('.', pos = pos1 + 1))
468  {
469  current = current->add_section_if_new(l,
470  sec_name.substr(pos, pos1 - pos));
471  }
472 
473  current = current->add_section_if_new(l, sec_name.substr(pos));
474 
475  // now add this entry to the section
476  current->add_entry(l, fullkey, key.substr(i+1), val);
477  }
478  else
479  {
480  entry_map::iterator it = entries_.find(key);
481  if (it != entries_.end())
482  {
483  it->second.first = std::move(val);
484  if (!it->second.second.empty())
485  {
486  std::string value = it->second.first;
487  entry_changed_func f = it->second.second;
488 
489  hpx::util::scoped_unlock<std::unique_lock<mutex_type> > ul(l);
490  f(fullkey, value);
491  }
492  }
493  else
494  {
495  // just add this entry to the section
496  entries_[key] = entry_type(val, entry_changed_func());
497  }
498  }
499 }

588  if (!f1)

597  return result_type(std::forward<F1>(f1), std::forward<F2>(f2));

660 {

662  if (i != std::string::npos)

782 {

786  entry_map const& s_entries = second.get_entries();
787  entry_map::const_iterator end = s_entries.end();
788  for (entry_map::const_iterator i = s_entries.begin(); i != end; ++i)
789  entries_[i->first] = i->second;

792  section_map::iterator send = sections_.end();
793  for (section_map::iterator i = sections_.begin(); i != send; ++i)

795  // is there something to merge with?

808  // it is not known here, so we can't merge, but have to add it.
809  add_section (i->first, i->second, get_root());

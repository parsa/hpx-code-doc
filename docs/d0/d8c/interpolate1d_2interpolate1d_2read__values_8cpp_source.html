45  double& minval, double& maxval, double& delta,
46  std::size_t start, std::size_t end)
47  {

49  using namespace H5;

51  // Turn off the auto-printing when failure occurs
52  Exception::dontPrint();
53 
54  H5File file(datafilename, H5F_ACC_RDONLY);
55  DataSet dataset = file.openDataSet("x"); // name of data to read
56  DataSpace dataspace = dataset.getSpace();
57 
58  // number of dimensions
59  int numdims = dataspace.getSimpleExtentNdims();
60 
61  if (numdims != 1)
62  {
63  HPX_THROW_EXCEPTION(hpx::no_success, "extract_data_range",
64  "number of dimensions was not 1");
65  }
66 
67  // Get the dimension size of each dimension in the dataspace.
68  hsize_t dims[1];
69  dataspace.getSimpleExtentDims(dims, nullptr);
70  if (end == std::size_t(-1))
71  end = dims[0];
72 
73  read_values(dataset, dataspace, start, 1, &minval);
74  read_values(dataset, dataspace, end-1, 1, &maxval);
75  read_values(dataset, dataspace, start+1, 1, &delta);
76 
77  delta -= minval;
78  return dims[0]; // return size of dataset
79  }
80  catch (H5::Exception const& e) {
81  HPX_THROW_EXCEPTION(hpx::no_success, "extract_data_range",
82  e.getDetailMsg());
83  }
84  return 0; // keep compiler happy

54  void operator()()
55  {
56  double const measured =

64  private:
65  void update_lost(double lost)
66  {
67  // Based on Boost.Accumulators.
68  // mean_n = (mean_{n-1}*(n-1) + x_n) / n
69  // sigma_n ~= sqrt(((n-1)/n)*(sigma_{n-1}^2) + (1/n)*(x_n-mean_n)^2)
70 
71  ++samples_;
72 
73  mean_lost_ =
74  (mean_lost_*(samples_-1.0) + lost)/samples_;
75 
76  stdev_lost_ =
77  std::sqrt( ((samples_-1.0)/samples_)*(stdev_lost_*stdev_lost_)
78  + (1.0/samples_)*(lost-mean_lost_)*(lost-mean_lost_));
79  }

597  // create a new thread and schedule it if the initial state is equal
598  // to pending
599  void create_thread(thread_init_data& data, thread_id_type* thrd,
600  thread_state_enum initial_state, bool run_now, error_code& ec,
601  std::size_t pool_queue_num)
602  {
603  HPX_ASSERT(data.scheduler_base == this);
604 
605  if (pool_queue_num == std::size_t(-1))
606  {
607  std::size_t t = threads::detail::thread_num_tss_
608  .get_worker_thread_num();
609  pool_queue_num = this->global_to_local_thread_index(t);
610  }
611 
612  // now create the thread
613  if (data.priority == thread_priority_high ||
614  data.priority == thread_priority_high_recursive ||
615  data.priority == thread_priority_boost)
616  {
617  // boosted threads return to normal after being queued
618  if (data.priority == thread_priority_boost)
619  {
620  data.priority = thread_priority_normal;
621  }
622  // if a non worker threads started this ...
623  if (pool_queue_num >= queues_.size())
624  {
625  pool_queue_num = curr_hp_queue_++ % queues_.size();
626  }
627  //
628  std::size_t hp_queue_num = hp_queue_lookup_[pool_queue_num];
629  high_priority_queues_[hp_queue_num]->create_thread(
630  data, thrd, initial_state, run_now, ec);
631  LOG_CUSTOM_MSG("create_thread thread_priority_high "
632  << THREAD_DESC2(data, thrd) << "hp_queue_num "
633  << hexnumber(pool_queue_num) << "scheduler "
634  << hexpointer(data.scheduler_base));
635  return;
636  }
637 
638  if (data.priority == thread_priority_low)
639  {
640  low_priority_queue_.create_thread(
641  data, thrd, initial_state, run_now, ec);
642  LOG_CUSTOM_MSG("create_thread thread_priority_low "
643  << THREAD_DESC2(data, thrd) << "pool_queue_num "
644  << hexnumber(pool_queue_num) << "scheduler "
645  << hexpointer(data.scheduler_base));
646  return;
647  }
648 
649  // if a non worker thread is creating this task
650  if (pool_queue_num >= queues_.size())
651  {
652  pool_queue_num = curr_queue_++ % queues_.size();
653  }
654  //
655  queues_[pool_queue_num]->create_thread(
656  data, thrd, initial_state, run_now, ec);

658  << THREAD_DESC2(data, thrd) << "pool_queue_num "

667  {
668  // is there a high priority task to take from the queue assigned to us
669  std::size_t tq = hp_queue_lookup_[pool_queue_num];
670  auto stealing_queue = high_priority_queues_[tq];
671  bool result = stealing_queue->get_next_thread(thrd);
672 
673  // if, not how about an HP task from somewhere else?
674  if (!result) {
675  for (std::size_t i : hp_victim_queues_[pool_queue_num]) {
676  stealing_queue = high_priority_queues_[i];
677  if (stealing_queue->get_pending_queue_length() > 1) {
678  result = stealing_queue->get_next_thread(
679  thrd, false, true);
680  if (result)
681  break;
682  }
683  }
684  }
685 
686  // cross-numa stealing - steal from the queues if they are 'full'
687  if (!result) {
688  for (std::size_t i : hp_xnuma_queues_[pool_queue_num]) {
689  stealing_queue = high_priority_queues_[i];
690  if (stealing_queue->get_pending_queue_length() > 8) {
691  result = stealing_queue->get_next_thread(
692  thrd, false, true);
693  if (result) {
694  LOG_CUSTOM_MSG("xnuma steal");
695  break;
696  }
697  }
698  }
699  }
700 
701  // counter for access queries
702  stealing_queue->increment_num_pending_accesses();
703  if (result)
704  {
705  HPX_ASSERT(thrd->get_scheduler_base() == this);
706  LOG_CUSTOM_MSG("get_next_thread high priority "
707  << THREAD_DESC(thrd) << decnumber(idle_loop_count)
708  << "pool_queue_num " << decnumber(pool_queue_num));
709  return true;
710  }
711  // counter for high priority misses
712  stealing_queue->increment_num_pending_misses();
713 
714  // try to get a task from the queue associated with the requested thread
715  std::size_t queues_size = queues_.size();
716  HPX_ASSERT(pool_queue_num < queues_size);
717  thread_queue_type* this_queue = queues_[pool_queue_num];
718  {
719  bool result = this_queue->get_next_thread(thrd);
720 
721  // counter for access queries
722  this_queue->increment_num_pending_accesses();
723  if (result)
724  {
725  HPX_ASSERT(thrd->get_scheduler_base() == this);
726  LOG_CUSTOM_MSG("get_next_thread normal "
727  << THREAD_DESC(thrd) << decnumber(idle_loop_count)
728  << "pool_queue_num " << decnumber(pool_queue_num));
729  return true;
730  }
731  // counter for misses
732  this_queue->increment_num_pending_misses();
733 
734  bool have_staged = this_queue->get_staged_queue_length(
735  std::memory_order_relaxed) != 0;
736 
737  // Give up, we should have work to convert.
738  if (have_staged)
739  {
740  LOG_CUSTOM_MSG("get_next_thread have_staged"
741  << "\" (unset) "
742  << "\" " << hexpointer(thrd)
743  << decnumber(idle_loop_count) << "pool_queue_num "
744  << decnumber(pool_queue_num));
745  return false;
746  }
747  }
748 
749  // if we didn't get a task from the requested thread queue
750  // try stealing from another
751  for (std::size_t idx : victim_queues_[pool_queue_num])
752  {
753  HPX_ASSERT(idx != pool_queue_num);
754 
755  if (queues_[idx]->get_next_thread(thrd, false, true))
756  {
757  LOG_CUSTOM_MSG("get_next_thread stealing "
758  << THREAD_DESC(thrd) << "pool_queue_num "
759  << decnumber(pool_queue_num) << "victim thread "
760  << decnumber(idx));
761  queues_[idx]->increment_num_stolen_from_pending();
762  this_queue->increment_num_stolen_to_pending();
763  return true;
764  }
765  }
766 
767  // if all else failed, then try the low priority queue
768  result = low_priority_queue_.get_next_thread(thrd);
769  if (result)
770  {
771  LOG_CUSTOM_MSG("get_next_thread low priority "
772  << THREAD_DESC(thrd) << "pool_queue_num "
773  << decnumber(pool_queue_num));
774  return true;
775  }
776  // counter for low priority misses
777  low_priority_queue_.increment_num_pending_misses();
778 
779  return result;
780  }

783  void schedule_thread(threads::thread_data* thrd,
784  std::size_t pool_queue_num,
785  thread_priority priority = thread_priority_normal)
786  {
787  HPX_ASSERT(thrd->get_scheduler_base() == this);
788 
789  if (pool_queue_num == std::size_t(-1))
790  {
791  std::size_t t = threads::detail::thread_num_tss_
792  .get_worker_thread_num();

794  }
795 
796  if (priority == thread_priority_high ||
797  priority == thread_priority_high_recursive ||
798  priority == thread_priority_boost)
799  {
800  std::size_t hp_queue_num =
801  curr_hp_queue_++ % high_priority_queues_.size();

810  LOG_CUSTOM_MSG("schedule_thread low priority "
811  << THREAD_DESC(thrd) << "pool_queue_num "
812  << decnumber(pool_queue_num));
813  }
814  else
815  {
816  pool_queue_num = curr_queue_++ % queues_.size();
817  queues_[pool_queue_num]->schedule_thread(thrd);
818  }
819  }

926  // Queries the current thread count of the queues.
927  std::int64_t get_thread_count(thread_state_enum state = unknown,
928  thread_priority priority = thread_priority_default,
929  std::size_t pool_queue_num = std::size_t(-1),
930  bool reset = false) const
931  {
932  LOG_CUSTOM_MSG("get_thread_count pool_queue_num "
933  << hexnumber(pool_queue_num));
934  // Return thread count of one specific queue.
935  std::int64_t count = 0;
936  if (std::size_t(-1) != pool_queue_num)
937  {
938  HPX_ASSERT(pool_queue_num < queues_.size());
939 
940  switch (priority)
941  {
942  case thread_priority_default:
943  {
944  auto high_priority_queue = high_priority_queues_
945  [hp_queue_lookup_[pool_queue_num]];
946  count = high_priority_queue->get_thread_count(state);
947 
948  if (queues_.size() - 1 == pool_queue_num)
949  count +=
950  low_priority_queue_.get_thread_count(state);
951 
952  return count +
953  queues_[pool_queue_num]->get_thread_count(state);
954  }
955 
956  case thread_priority_low:
957  {
958  if (queues_.size() - 1 == pool_queue_num)
959  return low_priority_queue_.get_thread_count(state);
960  break;
961  }
962 
963  case thread_priority_normal:
964  return queues_[pool_queue_num]->get_thread_count(state);
965 
966  case thread_priority_boost:
967  case thread_priority_high:
968  case thread_priority_high_recursive:
969  {
970  return high_priority_queues_
971  [hp_queue_lookup_[pool_queue_num]]
972  ->get_thread_count(state);
973  break;
974  }
975 
976  default:
977  case thread_priority_unknown:
978  {
979  HPX_THROW_EXCEPTION(bad_parameter,
980  "shared_priority_scheduler::get_thread_count",
981  "unknown thread priority value "
982  "(thread_priority_unknown)");
983  return 0;
984  }
985  }
986  return 0;
987  }
988 
989  // Return the cumulative count for all queues.
990  switch (priority)
991  {
992  case thread_priority_default:
993  {
994  for (std::size_t i = 0; i != high_priority_queues_.size();
995  ++i)
996  count +=
997  high_priority_queues_[i]->get_thread_count(state);
998 
999  for (std::size_t i = 0; i != queues_.size(); ++i)
1000  count += queues_[i]->get_thread_count(state);
1001 
1002  break;
1003  }
1004 
1005  case thread_priority_low:
1006  return low_priority_queue_.get_thread_count(state);
1007 
1008  case thread_priority_normal:
1009  {
1010  for (std::size_t i = 0; i != queues_.size(); ++i)
1011  count += queues_[i]->get_thread_count(state);
1012  break;
1013  }
1014 
1015  case thread_priority_boost:
1016  case thread_priority_high:
1017  case thread_priority_high_recursive:
1018  {
1019  for (std::size_t i = 0; i != high_priority_queues_.size();
1020  ++i)
1021  count +=
1022  high_priority_queues_[i]->get_thread_count(state);
1023  break;
1024  }
1025 
1026  default:
1027  case thread_priority_unknown:
1028  {
1029  HPX_THROW_EXCEPTION(bad_parameter,
1030  "shared_priority_scheduler::get_thread_count",
1031  "unknown thread priority value "
1032  "(thread_priority_unknown)");
1033  return 0;
1034  }
1035  }
1036  return count;
1037  }

1066  virtual bool wait_or_add_new(std::size_t pool_queue_num,
1067  bool running, std::int64_t& idle_loop_count)
1068  {

1070  bool result = true;
1071 
1072  thread_queue_type* high_priority_queue =
1073  high_priority_queues_[hp_queue_lookup_[pool_queue_num]];
1074  result = high_priority_queue->wait_or_add_new(
1075  running, idle_loop_count, added) &&
1076  result;
1077  if (0 != added)
1078  {
1079  return result;
1080  }
1081 
1082  thread_queue_type* this_queue = queues_[pool_queue_num];
1083  result = this_queue->wait_or_add_new(
1084  running, idle_loop_count, added) &&
1085  result;
1086  if (0 != added)
1087  {
1088  return result;
1089  }
1090 
1091  // try all the HP queues we might steal from
1092  for (std::size_t idx : hp_victim_queues_[pool_queue_num])
1093  {
1094  result = high_priority_queue->wait_or_add_new(running,
1095  idle_loop_count, added,
1096  high_priority_queues_[idx]) &&
1097  result;
1098  if (0 != added)
1099  {
1100  high_priority_queues_[idx]
1101  ->increment_num_stolen_from_staged(added);
1102  high_priority_queue->increment_num_stolen_to_staged(
1103  added);
1104  return result;
1105  }
1106  }
1107 
1108  // try all the normal queues we might steal from
1109  for (std::size_t idx : victim_queues_[pool_queue_num])
1110  {
1111  result = this_queue->wait_or_add_new(running,
1112  idle_loop_count, added, queues_[idx]) &&
1113  result;
1114  if (0 != added)
1115  {
1116  queues_[idx]->increment_num_stolen_from_staged(added);
1117  this_queue->increment_num_stolen_to_staged(added);
1118  return result;
1119  }
1120  }
1121 
1122  result = low_priority_queue_.wait_or_add_new(
1123  running, idle_loop_count, added) &&
1124  result;
1125  if (0 != added) {}
1126 
1127  return result;
1128  }

1131  /* @TODO make sure on_start_thread is forwarded
1132  * currently it does nothing in the thread queue - but might one day
1133  * queues_[pool_queue_num]->on_start_thread(pool_queue_num);
1134  */
1135  void on_start_thread(std::size_t pool_queue_num)
1136  {
1137  LOG_CUSTOM_MSG("start thread with local thread num "
1138  << decnumber(pool_queue_num));
1139  //
1140  // @TODO. move this to a proper init function
1141  //
1142  std::lock_guard<hpx::lcos::local::spinlock> lock(init_mutex);
1143  if (!initialized_)
1144  {
1145  initialized_ = true;
1146  std::size_t num_queues = queues_.size();
1147  std::size_t hpq_size = high_priority_queues_.size();
1148  std::size_t qph = num_queues/hpq_size;
1149  std::vector<std::size_t> hp_queue_mapping_;
1150  // init hp queues with default zero queue/domain
1151  hp_queue_lookup_.resize(num_queues, 0);
1152  hp_queue_mapping_.resize(num_queues, 0);
1153  hp_victim_queues_.resize(num_queues);
1154  hp_xnuma_queues_.resize(num_queues);
1155 
1156  // assign queues for stealing according to numa domain
1157  for (std::size_t i = 0; i < num_queues; ++i)
1158  {
1159  // fill a vector with all thread ids except 'itself'
1160  // offset so that each queue steals in a staggered pattern
1161  // and don't all hit pu=0 first for example
1162  std::vector<std::size_t> thread_ids;
1163  for (std::size_t j=0; j<num_queues; ++j) {
1164  std::size_t pu = (i + j) % (int)num_queues;
1165  if (pu!=i) thread_ids.push_back(pu);
1166  }
1167  // get a list of all threads on the same numa domain
1168  std::vector<std::size_t> same = domain_threads(i, thread_ids,
1169  std::equal_to<std::size_t>());
1170  // get a list of all threads on other numa domains
1171  std::vector<std::size_t> diff = domain_threads(i, thread_ids,
1172  std::not_equal_to<std::size_t>());
1173 
1174  // always steal from queues on the same numa domain first
1175  victim_queues_[i] = same;
1176  // if we are not numa sensitive, steal from other domains
1177 // if (!numa_sensitive_) {
1178  victim_queues_[i].insert(victim_queues_[i].end(),
1179  diff.begin(), diff.end());
1180 // }
1181 
1182  // which HP queue do we use for this thread?
1183  double s = static_cast<double>(hpq_size-1);
1184  hp_victim_queues_[i].clear();
1185  hp_queue_lookup_[i] =
1186  static_cast<int>(i*(s+1)/(num_queues));
1187  hp_queue_mapping_[i] = hp_queue_lookup_[i] * qph;
1188  }
1189 
1190  // clean up mapping of HP queues to thread numbers
1191  auto last = std::unique(
1192  hp_queue_mapping_.begin(), hp_queue_mapping_.end());
1193  hp_queue_mapping_.erase(last, hp_queue_mapping_.end());
1194  //
1195  debug::output("hp_queue_lookup_ ", hp_queue_lookup_);
1196  debug::output("hp_queue_mapping_ ", hp_queue_mapping_);
1197  //
1198  for (std::size_t i = 0; i < num_queues; ++i)
1199  {
1200  auto temp_ = hp_queue_mapping_;
1201  // map i to hp queue num, and then to thread,
1202  // remove that thread from our list
1203  temp_.erase(
1204  std::remove(temp_.begin(), temp_.end(),
1205  hp_queue_mapping_[hp_queue_lookup_[i]]),
1206  temp_.end());
1207 
1208  // get a list of all HP queues on the same numa domain
1209  std::vector<std::size_t> same = domain_threads(
1210  hp_queue_mapping_[hp_queue_lookup_[i]], temp_,
1211  std::equal_to<std::size_t>());
1212 
1213  // get a list of all HP threads on other numa domains
1214  std::vector<std::size_t> diff = domain_threads(
1215  hp_queue_mapping_[hp_queue_lookup_[i]], temp_,
1216  std::not_equal_to<std::size_t>());
1217 
1218  if (same.size()>0) {
1219  //debug::output("same ", same);
1220  std::rotate(same.begin(), same.begin() + i/qph % same.size(),
1221  same.end());
1222  // convert HP queue numbers to threads
1223  for (auto q : same) {
1224  hp_victim_queues_[i].push_back(hp_queue_lookup_[q]);
1225  }
1226  }
1227 
1228  if (diff.size()>0) {
1229  //debug::output("diff ", diff);
1230  std::rotate(diff.begin(), diff.begin() + i/qph % diff.size(),
1231  diff.end());
1232  // convert HP queue numbers to threads
1233  for (auto q : diff) {
1234  hp_xnuma_queues_[i].push_back(hp_queue_lookup_[q]);
1235  }
1236  }
1237 
1238  if (!numa_sensitive_) {
1239  // no numa check - merge queues into one
1240  // hp tasks can be stolen from any hp queue
1241  hp_victim_queues_[i].insert(hp_victim_queues_[i].end(),
1242  hp_xnuma_queues_[i].begin(), hp_xnuma_queues_[i].end());
1243  hp_xnuma_queues_[i].clear();
1244  }
1245  }
1246 
1247  for (std::size_t i=0; i<num_queues; ++i) {
1248  char buff[256];
1249  sprintf(buff, "%03i", (int)i);
1250  debug::output("victim_queue " + std::string(buff) + " ",
1251  victim_queues_[i]);
1252  debug::output("hp_victim_queue " + std::string(buff) + " ",
1253  hp_victim_queues_[i]);
1254  debug::output("hp_xnuma_queues " + std::string(buff) + " ",
1255  hp_xnuma_queues_[i]);
1256  }
1257  }
1258  }

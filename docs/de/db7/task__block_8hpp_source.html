158  static void
159  on_ready(std::vector<hpx::future<void> > && results,
160  parallel::exception_list && errors)
161  {
162  for (hpx::future<void>& f: results)
163  {
164  if (f.has_exception())
165  errors.add(f.get_exception_ptr());
166  }
167  if (errors.size() != 0)
168  throw errors;
169  }

171  // return future representing the execution of all tasks
172  typename util::detail::algorithm_result<ExPolicy>::type
173  when(bool throw_on_error = false)
174  {
175  std::vector<hpx::future<void> > tasks;
176  parallel::exception_list errors;
177 
178  {
179  std::lock_guard<mutex_type> l(mtx_);
180  std::swap(tasks_, tasks);
181  std::swap(errors_, errors);
182  }
183 
184  typedef util::detail::algorithm_result<ExPolicy> result;
185 
186  if (tasks.empty() && errors.size() == 0)
187  return result::get();
188 
189  if (!throw_on_error)
190  return result::get(hpx::when_all(tasks));
191 
192  return
193  result::get(
194  hpx::dataflow(
195  hpx::util::bind(hpx::
196  util::one_shot(&task_block::on_ready),
197  hpx::util::placeholders::_1, std::move(errors)),
198  std::move(tasks)
199  ));
200  }

403  template <typename ExPolicy, typename F>
404  typename util::detail::algorithm_result<ExPolicy>::type
405  define_task_block(ExPolicy && policy, F && f)
406  {
407  static_assert(
408  parallel::execution::is_execution_policy<ExPolicy>::value,
409  "parallel::execution::is_execution_policy<ExPolicy>::value");
410 
411  typedef typename hpx::util::decay<ExPolicy>::type policy_type;
412  task_block<policy_type> trh(std::forward<ExPolicy>(policy));
413 
414  // invoke the user supplied function
415  try {
416  f(trh);
417  }
418  catch (...) {
419  detail::handle_task_block_exceptions(trh.errors_);
420  }
421 
422  // regardless of whether f(trh) has thrown an exception we need to
423  // obey the contract and wait for all tasks to join
424  return trh.when(true);
425  }

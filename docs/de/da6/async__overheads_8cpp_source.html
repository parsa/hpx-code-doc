29 hpx::future<void> spawn_level(std::size_t num_tasks)
30 {
31  std::vector<hpx::future<void> > tasks;
32  tasks.reserve(num_tasks);
33 
34  // spawn sub-levels
35  if (num_tasks > num_level_tasks && spread > 1)
36  {
37  std::size_t spawn_hierarchically = num_tasks - num_level_tasks;
38  std::size_t num_sub_tasks = 0;
39  if (spawn_hierarchically < num_level_tasks)
40  num_sub_tasks = spawn_hierarchically;
41  else
42  num_sub_tasks = spawn_hierarchically / spread;
43 
44  for (std::size_t i = 0; i != spread && spawn_hierarchically != 0; ++i)
45  {
46  std::size_t sub_spawn = (std::min)(spawn_hierarchically, num_sub_tasks);
47  spawn_hierarchically -= sub_spawn;
48  num_tasks -= sub_spawn;
49 
50  // force unwrapping
51  hpx::future<void> f = hpx::async(&spawn_level, sub_spawn);
52  tasks.push_back(std::move(f));
53  }
54  }
55 
56  // then spawn required number of tasks on this level
57  for (std::size_t i = 0; i != num_tasks; ++i)
58  tasks.push_back(hpx::async(&test_func));
59 
60  return hpx::when_all(tasks);
61 }

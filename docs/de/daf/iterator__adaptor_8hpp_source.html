74  // Iterator adaptor
75  //
76  // The idea is that when the user needs
77  // to fiddle with the reference type it is highly likely that the
78  // iterator category has to be adjusted as well. Any of the
79  // following four template arguments may be omitted or explicitly
80  // replaced by void.
81  //
82  // Value - if supplied, the value_type of the resulting iterator, unless
83  // const. If const, a conforming compiler strips const-ness for the
84  // value_type. If not supplied, iterator_traits<Base>::value_type is used
85  //
86  // Category - the traversal category of the resulting iterator. If not
87  // supplied, iterator_traversal<Base>::type is used.
88  //
89  // Reference - the reference type of the resulting iterator, and in
90  // particular, the result type of operator*(). If not supplied but
91  // Value is supplied, Value& is used. Otherwise
92  // iterator_traits<Base>::reference is used.
93  //
94  // Difference - the difference_type of the resulting iterator. If not
95  // supplied, iterator_traits<Base>::difference_type is used.
96  //
97  template <
98  typename Derived, typename Base, typename Value = void,
99  typename Category = void, typename Reference = void,
100  typename Difference = void, typename Pointer = void>
101  class iterator_adaptor

103  Derived, Base, Value, Category, Reference, Difference, Pointer
104  >::type

108  Derived, Base, Value, Category, Reference, Difference, Pointer
109  >::type base_adaptor_type;

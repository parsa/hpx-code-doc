186  std::string get_execution_environment()
187  {
188  std::vector<std::string> env;
189 
190 #if defined(HPX_WINDOWS)
191  std::size_t len = get_arraylen(_environ);
192  env.reserve(len);
193  std::copy(&_environ[0], &_environ[len], std::back_inserter(env));
194 #elif defined(linux) || defined(__linux) || defined(__linux__) || defined(__AIX__)
195  std::size_t len = get_arraylen(environ);
196  env.reserve(len);
197  std::copy(&environ[0], &environ[len], std::back_inserter(env));
198 #elif defined(__FreeBSD__)
199  std::size_t len = get_arraylen(freebsd_environ);
200  env.reserve(len);
201  std::copy(&freebsd_environ[0], &freebsd_environ[len],
202  std::back_inserter(env));
203 #elif defined(__APPLE__)
204  std::size_t len = get_arraylen(environ);
205  env.reserve(len);
206  std::copy(&environ[0], &environ[len], std::back_inserter(env));
207 #else
208 #error "Don't know, how to access the execution environment on this platform"
209 #endif
210 
211  std::sort(env.begin(), env.end());
212 
213  std::string retval = boost::str(boost::format("%d entries:\n") % env.size());
214  for (std::string const& s : env)
215  {
216  retval += " " + s + "\n";
217  }
218  return retval;
219  }

314  template <typename Exception>
315  HPX_EXPORT std::exception_ptr
316  get_exception(Exception const& e, std::string const& func,
317  std::string const& file, long line, std::string const& auxinfo)
318  {
319  if (is_of_lightweight_hpx_category(e))
320  return construct_lightweight_exception(e, func, file, line);
321 
322  std::int64_t pid = ::getpid();
323  std::string back_trace(backtrace());
324 
325  std::string state_name("not running");
326  std::string hostname;
327  hpx::runtime* rt = get_runtime_ptr();
328  if (rt)
329  {
330  state rts_state = rt->get_state();
331  state_name = get_runtime_state_name(rts_state);
332 
333  if (rts_state >= state_initialized &&
334  rts_state < state_stopped)
335  {
336  std::ostringstream strm;
337  strm << get_runtime().here();
338  hostname = strm.str();
339  }
340  }
341 
342  // if this is not a HPX thread we do not need to query neither for
343  // the shepherd thread nor for the thread id
344  error_code ec(lightweight);
345  std::uint32_t node = get_locality_id(ec);
346 
347  std::size_t shepherd = std::size_t(-1);
348  threads::thread_id_type thread_id;
349  util::thread_description thread_name;
350 
351  threads::thread_self* self = threads::get_self_ptr();
352  if (nullptr != self)
353  {
354  if (threads::threadmanager_is(state_running))
355  shepherd = hpx::get_worker_thread_num();
356 
357  thread_id = threads::get_self_id();
358  thread_name = threads::get_thread_description(thread_id);
359  }
360 
361  std::string env(get_execution_environment());
362  std::string config(configuration_string());
363 
364  return construct_exception(e, func, file, line, back_trace, node,
365  hostname, pid, shepherd,
366  reinterpret_cast<std::size_t>(thread_id.get()),
367  util::as_string(thread_name), env, config,
368  state_name, auxinfo);
369  }

446  void assertion_failed_msg(char const* msg, char const* expr,
447  char const* function, char const* file, long line)
448  {
449  if (!expect_exception_flag.load(std::memory_order_relaxed) &&
450  get_config_entry("hpx.attach_debugger", "") == "exception")
451  {
452  util::attach_debugger();
453  }
454 
455  bool threw = false;
456 
457  std::string str("assertion '" + std::string(msg) + "' failed");
458  if (expr != msg)
459  str += " (" + std::string(expr) + ")";
460 
461  try {
462  boost::filesystem::path p(hpx::util::create_path(file));
463  hpx::detail::throw_exception(
464  hpx::exception(hpx::assertion_failure, str),
465  function, p.string(), line);
466  }
467  catch (...) {
468  threw = true;
469 
470  // If the runtime pointer is available, we can safely get the prefix
471  // of this locality. If it's not available, then just terminate.
472  runtime* rt = get_runtime_ptr();
473  if (nullptr != rt) {
474  rt->report_error(std::current_exception());
475  }
476  else {
477  std::cerr << "Runtime is not available, reporting error locally. "
478  << hpx::diagnostic_information(std::current_exception())
479  << std::flush;
480  }
481  }
482 
483  // If the exception wasn't thrown, then print out the assertion message,
484  // so that the program doesn't abort without any diagnostics.
485  if (!threw) {
486  std::cerr << "Runtime is not available, reporting error locally\n"
487  "{what}: " << str << std::endl;
488  }
489  std::abort();
490  }

539  // Extract the diagnostic information embedded in the given exception and
540  // return a string holding a formatted message.
541  std::string diagnostic_information(hpx::exception_info const& xi)
542  {
543  std::ostringstream strm;
544  strm << "\n";
545 
546  std::string const* back_trace =
547  xi.get<hpx::detail::throw_stacktrace>();
548  if (back_trace && !back_trace->empty()) {
549  // FIXME: add indentation to stack frame information
550  strm << "{stack-trace}: " << *back_trace << "\n";
551  }
552 
553  std::string const* env =
554  xi.get<hpx::detail::throw_env>();
555  if (env && !env->empty())
556  strm << "{env}: " << *env;
557 
558  std::uint32_t const* locality =
559  xi.get<hpx::detail::throw_locality>();
560  if (locality)
561  strm << "{locality-id}: " << *locality << "\n";
562 
563  std::string const* hostname_ =
564  xi.get<hpx::detail::throw_hostname>();
565  if (hostname_ && !hostname_->empty())
566  strm << "{hostname}: " << *hostname_ << "\n";
567 
568  std::int64_t const* pid_ =
569  xi.get<hpx::detail::throw_pid>();
570  if (pid_ && -1 != *pid_)
571  strm << "{process-id}: " << *pid_ << "\n";
572 
573  std::string const* function =
574  xi.get<hpx::detail::throw_function>();
575  if (function)
576  strm << "{function}: " << *function << "\n";
577 
578  std::string const* file =
579  xi.get<hpx::detail::throw_file>();
580  if (file)
581  strm << "{file}: " << *file << "\n";
582 
583  long const* line =
584  xi.get<hpx::detail::throw_line>();
585  if (line)
586  strm << "{line}: " << *line << "\n";
587 
588  bool thread_info = false;
589  char const* const thread_prefix = "{os-thread}: ";
590  std::size_t const* shepherd =
591  xi.get<hpx::detail::throw_shepherd>();
592  if (shepherd && std::size_t(-1) != *shepherd) {
593  strm << thread_prefix << *shepherd;
594  thread_info = true;
595  }
596 
597  std::string thread_name = runtime::get_thread_name();
598  if (!thread_info)
599  strm << thread_prefix;
600  else
601  strm << ", ";
602  strm << thread_name << "\n";
603 
604  std::size_t const* thread_id =
605  xi.get<hpx::detail::throw_thread_id>();
606  if (thread_id && *thread_id)
607  strm << (boost::format("{thread-id}: %016x\n") % *thread_id);
608 
609  std::string const* thread_description =
610  xi.get<hpx::detail::throw_thread_name>();
611  if (thread_description && !thread_description->empty())
612  strm << "{thread-description}: " << *thread_description << "\n";
613 
614  std::string const* state =
615  xi.get<hpx::detail::throw_state>();
616  if (state)
617  strm << "{state}: " << *state << "\n";
618 
619  std::string const* auxinfo =
620  xi.get<hpx::detail::throw_auxinfo>();
621  if (auxinfo)
622  strm << "{auxinfo}: " << *auxinfo << "\n";
623 
624  // add full build information
625  strm << full_build_string();
626 
627  // Try a cast to std::exception - this should handle boost.system
628  // error codes in addition to the standard library exceptions.
629  std::exception const* se = dynamic_cast<std::exception const*>(&xi);
630  if (se)
631  strm << "{what}: " << se->what() << "\n";
632 
633  return strm.str();
634  }

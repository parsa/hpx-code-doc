388  template <typename Action>
389  inline bool
390  apply_r_sync_p(naming::address&& addr, naming::id_type const& id,
391  threads::thread_priority priority)
392  {
393  typedef typename hpx::traits::extract_action<Action>::type action_type_;
394 
395  // If remote, create a new parcel to be sent to the destination
396  // Create a new parcel with the gid, action, and arguments
397  // Send the parcel through the parcel handler
398  HPX_ASSERT(id.get_management_type() == naming::id_type::unmanaged);
399  naming::gid_type gid = id.get_gid();
400  parcelset::parcel p =
401  parcelset::detail::create_parcel::call(std::false_type(),
402  std::move(gid), complement_addr<action_type_>(addr),
403  action_type_(), priority
404  );
405 
406  parcelset::detail::parcel_await(std::move(p),
407  parcelset::write_handler_type(), 0,
408  [](parcelset::parcel&& p, parcelset::write_handler_type&&)
409  {
410  hpx::get_runtime().get_parcel_handler()
411  .sync_put_parcel(std::move(p));
412  }).apply();
413  return false; // destination is remote
414  }

34  void jacobi(
35  crs_matrix<double> const & A
36  , std::vector<double> const & b
37  , std::size_t iterations
38  , std::size_t block_size
39  )
40  {
41  typedef std::vector<double> vector_type;
42 
43  std::shared_ptr<vector_type> dst(new vector_type(b));
44  std::shared_ptr<vector_type> src(new vector_type(b));
45 
46  std::vector<range> block_ranges;
47  // pre-computing ranges for the different blocks
48  for(std::size_t i = 0; i < dst->size(); i += block_size)
49  {
50  block_ranges.push_back(
51  range(i, std::min<std::size_t>(dst->size(), i + block_size)));
52  }
53 
54  // pre-computing dependencies
55  std::vector<std::vector<std::size_t> > dependencies(block_ranges.size());
56  for(std::size_t b = 0; b < block_ranges.size(); ++b)
57  {
58  for(std::size_t i = block_ranges[b].begin(); i < block_ranges[b].end(); ++i)
59  {
60  std::size_t begin = A.row_begin(i);
61  std::size_t end = A.row_end(i);
62 
63  for(std::size_t ii = begin; ii < end; ++ii)
64  {
65  std::size_t idx = A.indices[ii];
66  for(std::size_t j = 0; j < block_ranges.size(); ++j)
67  {
68  if(block_ranges[j].begin() <= idx && idx < block_ranges[j].end())
69  {
70  if(std::find(dependencies[b].begin(),
71  dependencies[b].end(), j) == dependencies[b].end())
72  {
73  dependencies[b].push_back(j);
74  }
75  break;
76  }
77  }
78  }
79  }
80  }
81

83  std::shared_ptr<future_vector> deps_dst
84  (new future_vector(dependencies.size(), hpx::make_ready_future()));
85  std::shared_ptr<future_vector> deps_src
86  (new future_vector(dependencies.size(), hpx::make_ready_future()));
87 
88  hpx::util::high_resolution_timer t;
89  for(std::size_t iter = 0; iter < iterations; ++iter)
90  {
91  for(std::size_t block = 0; block < block_ranges.size(); ++block)
92  {
93  std::vector<std::size_t> const & deps(dependencies[block]);
94  std::vector<hpx::shared_future<void> > trigger;
95  trigger.reserve(deps.size());
96  for (std::size_t dep : deps)
97  {
98  trigger.push_back((*deps_src)[dep]);
99  }
100 
101  (*deps_dst)[block]
102  = hpx::when_all(std::move(trigger)).then(
103  hpx::launch::async,
104  hpx::util::bind(
105  jacobi_kernel_wrap
106  , block_ranges[block]
107  , std::cref(A)
108  , std::ref(*dst)
109  , std::cref(*src)
110  , std::cref(b)
111  )
112  );
113  }
114  std::swap(dst, src);
115  std::swap(deps_dst, deps_src);
116  }
117 
118  hpx::wait_all(*deps_dst);
119  hpx::wait_all(*deps_src);
120 
121  double time_elapsed = t.elapsed();
122  std::cout << dst->size() << " "
123  << ((double(dst->size() * iterations)/1e6)/time_elapsed) << " MLUPS/s\n"
124  << std::flush;
125  }

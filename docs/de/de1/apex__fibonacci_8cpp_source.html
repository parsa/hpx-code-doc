33 //[fib_func
34 std::uint64_t fibonacci(std::uint64_t n)
35 {
36  if (n < 2)
37  return n;
38 
39  // We restrict ourselves to execute the Fibonacci function locally.
40  hpx::naming::id_type const locality_id = hpx::find_here();
41 
42  // Invoking the Fibonacci algorithm twice is inefficient.
43  // However, we intentionally demonstrate it this way to create some
44  // heavy workload.
45 
46  fibonacci_action fib;
47  hpx::future<std::uint64_t> n1 =
48  hpx::async(fib, locality_id, n - 1);
49  hpx::future<std::uint64_t> n2 =
50  hpx::async(fib, locality_id, n - 2);
51 
52  return n1.get() + n2.get(); // wait for the Futures to return their values
53 }

21 struct file_wrapper
22 {
23  file_wrapper(std::string const& name, std::ios_base::openmode mode)
24  : stream_(name.c_str(), mode), mode_(mode)
25  {
26  if (!stream_.is_open())
27  throw std::runtime_error("Couldn't open file: " + name);
28 
29  std::fstream::pos_type fsize = stream_.tellg();
30  stream_.seekg(0, std::ios::end);
31  data_.resize(stream_.tellg() - fsize);
32  stream_.seekg(fsize, std::ios::beg);
33 
34  if (mode & std::ios_base::in)
35  {
36  stream_.read(data_.data(), data_.size());
37  }
38  }
39 
40  ~file_wrapper()
41  {
42  if (mode_ & std::ios_base::out)
43  {
44  stream_.write(data_.data(), data_.size());
45  }
46  }

48  void read(void* address, std::size_t count, std::size_t current) const
49  {

51  }

53  void write(void const* address, std::size_t count, std::size_t current)
54  {

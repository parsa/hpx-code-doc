52 template <typename ExPolicy, typename T>
53 void verify_values(ExPolicy && policy, hpx::partitioned_vector<T> const& v,
54  T const& val)
55 {
56  typedef typename hpx::partitioned_vector<T>::const_iterator const_iterator;
57 
58  std::size_t size = 0;
59 
60  const_iterator end = v.end();
61  for (const_iterator it = v.begin(); it != end; ++it, ++size)

184 // The returned value type has to be the same as the return type used for
185 // co_await below
186 hpx::future<sub_block> transpose_phase(
187  std::vector<block> const& A, std::vector<block>& B,
188  std::uint64_t block_order, std::uint64_t b,
189  std::uint64_t num_blocks, std::uint64_t num_local_blocks,
190  std::uint64_t block_size, std::uint64_t tile_size)
191 {
192  const std::uint64_t from_phase = b;
193  const std::uint64_t A_offset = from_phase * block_size;
194 
195  auto phase_range = boost::irange(
196  static_cast<std::uint64_t>(0), num_blocks);
197  for(std::uint64_t phase: phase_range)
198  {
199  const std::uint64_t from_block = phase;
200  const std::uint64_t B_offset = phase * block_size;
201 
202  hpx::future<sub_block> from =
203  A[from_block].get_sub_block(A_offset, block_size);
204  hpx::future<sub_block> to =
205  B[b].get_sub_block(B_offset, block_size);
206 
207  transpose(co_await from, co_await to, block_order, tile_size);
208  }
209 
210  co_return sub_block();
211 }

214 int hpx_main(boost::program_options::variables_map& vm)
215 {
216  {
217  hpx::id_type here = hpx::find_here();
218  bool root = here == hpx::find_root_locality();
219 
220  std::uint64_t num_localities = hpx::get_num_localities().get();
221 
222  std::uint64_t order = vm["matrix_size"].as<std::uint64_t>();
223  std::uint64_t iterations = vm["iterations"].as<std::uint64_t>();
224  std::uint64_t num_local_blocks = vm["num_blocks"].as<std::uint64_t>();
225  std::uint64_t tile_size = order;
226 
227  if(vm.count("tile_size"))
228  tile_size = vm["tile_size"].as<std::uint64_t>();
229 
230  verbose = vm.count("verbose") ? true : false;
231 
232  std::uint64_t bytes =
233  static_cast<std::uint64_t>(2.0 * sizeof(double) * order * order);
234 
235  std::uint64_t num_blocks = num_localities * num_local_blocks;
236 
237  std::uint64_t block_order = order / num_blocks;
238  std::uint64_t col_block_size = order * block_order;
239 
240  std::uint64_t id = hpx::get_locality_id();
241 
242  std::vector<block> A(num_blocks);
243  std::vector<block> B(num_blocks);
244 
245  std::uint64_t blocks_start = id * num_local_blocks;
246  std::uint64_t blocks_end = (id + 1) * num_local_blocks;
247 
248  // Actually allocate the block components in AGAS
249  for(std::uint64_t b = 0; b != num_blocks; ++b)
250  {
251  // Allocate block
252  if(b >= blocks_start && b != blocks_end)
253  {
254  A[b] = block(b, col_block_size, A_block_basename);
255  B[b] = block(b, col_block_size, B_block_basename);
256  }
257  // Retrieve the block by it's symbolic name
258  else
259  {
260  A[b] = block(b, A_block_basename);
261  B[b] = block(b, B_block_basename);
262  }
263  }
264 
265  if(root)
266  {
267  std::cout
268  << "Distributed HPX Matrix transpose (await): B = A^T\n"
269  << "Matrix order = " << order << "\n"
270  << "Matrix local columns = " << block_order << "\n"
271  << "Number of blocks = " << num_blocks << "\n"
272  << "Number of localities = " << num_localities << "\n";
273  if(tile_size < order)
274  std::cout << "Tile size = " << tile_size << "\n";
275  else
276  std::cout << "Untiled\n";
277  std::cout
278  << "Number of iterations = " << iterations << "\n";
279  }
280  using hpx::parallel::for_each;
281  using hpx::parallel::execution::par;
282 
283  // Fill the original matrix, set transpose to known garbage value.
284  auto range = boost::irange(blocks_start, blocks_end);
285  for_each(par, std::begin(range), std::end(range),
286  [&](std::uint64_t b)
287  {
288  std::shared_ptr<block_component> A_ptr =
289  hpx::get_ptr<block_component>(A[b].get_gid()).get();
290  std::shared_ptr<block_component> B_ptr =
291  hpx::get_ptr<block_component>(B[b].get_gid()).get();
292 
293  for(std::uint64_t i = 0; i != order; ++i)
294  {
295  for(std::uint64_t j = 0; j != block_order; ++j)
296  {
297  double col_val = COL_SHIFT * (b*block_order + j);
298  A_ptr->data_[i * block_order + j] = col_val + ROW_SHIFT * i;
299  B_ptr->data_[i * block_order + j] = -1.0;
300  }
301  }
302  }
303  );
304 
305  double errsq = 0.0;
306  double avgtime = 0.0;
307  double maxtime = 0.0;
308  double mintime = 366.0 * 24.0*3600.0; // set the minimum time to a large value;
309  // one leap year should be enough
310  for(std::uint64_t iter = 0; iter < iterations; ++iter)
311  {
312  hpx::util::high_resolution_timer t;
313 
314  auto range = boost::irange(blocks_start, blocks_end);
315 
316  const std::uint64_t block_size = block_order * block_order;
317  for_each(par, std::begin(range), std::end(range),
318  [&](std::uint64_t b)
319  {
320  transpose_phase(A, B, block_order, b,
321  num_blocks, num_local_blocks, block_size, tile_size
322  ).get();
323  });
324 
325  double elapsed = t.elapsed();
326 
327  if(iter > 0 || iterations == 1) // Skip the first iteration
328  {
329  avgtime = avgtime + elapsed;
330  maxtime = (std::max)(maxtime, elapsed);
331  mintime = (std::min)(mintime, elapsed);
332  }
333 
334  if(root)
335  errsq += test_results(order, block_order, B, blocks_start, blocks_end);
336  } // end of iter loop
337 
338  // Analyze and output results
339 
340  double epsilon = 1.e-8;
341  if(root)
342  {
343  if(errsq < epsilon)
344  {
345  std::cout << "Solution validates\n";
346  avgtime = avgtime/static_cast<double>(
347  (std::max)(iterations-1, static_cast<std::uint64_t>(1)));
348  std::cout
349  << "Rate (MB/s): " << 1.e-6 * bytes/mintime << ", "
350  << "Avg time (s): " << avgtime << ", "
351  << "Min time (s): " << mintime << ", "
352  << "Max time (s): " << maxtime << "\n";
353 
354  if(verbose)
355  std::cout << "Squared errors: " << errsq << "\n";
356  }
357  else
358  {
359  std::cout
360  << "ERROR: Aggregate squared error " << errsq
361  << " exceeds threshold " << epsilon << "\n";
362  hpx::terminate();
363  }
364  }
365  }
366 
367  return hpx::finalize();
368 }

398 void transpose(sub_block const A, sub_block B,
399  std::uint64_t block_order, std::uint64_t tile_size)
400 {
401  if(tile_size < block_order)
402  {
403  for(std::uint64_t i = 0; i != block_order; i += tile_size)
404  {
405  for(std::uint64_t j = 0; j != block_order; j += tile_size)
406  {
407  std::uint64_t max_i = (std::min)(block_order, i + tile_size);
408  std::uint64_t max_j = (std::min)(block_order, j + tile_size);
409 
410  for(std::uint64_t it = i; it != max_i; ++it)
411  {
412  for(std::uint64_t jt = j; jt != max_j; ++jt)
413  {
414  B[it + block_order * jt] = A[jt + block_order * it];
415  }
416  }
417  }
418  }
419  }
420  else
421  {
422  for(std::uint64_t i = 0; i != block_order; ++i)
423  {
424  for(std::uint64_t j = 0; j != block_order; ++j)
425  {
426  B[i + block_order * j] = A[j + block_order * i];
427  }
428  }
429  }
430 }

432 double test_results(std::uint64_t order, std::uint64_t block_order,
433  std::vector<block> & trans, std::uint64_t blocks_start,
434  std::uint64_t blocks_end)
435 {
436  using hpx::parallel::transform_reduce;
437  using hpx::parallel::execution::par;
438 
439  // Fill the original matrix, set transpose to known garbage value.
440  auto range = boost::irange(blocks_start, blocks_end);
441  double errsq =
442  transform_reduce(par, std::begin(range), std::end(range), 0.0,
443  [](double lhs, double rhs) { return lhs + rhs; },
444  [&](std::uint64_t b) -> double
445  {
446  sub_block trans_block =
447  trans[b].get_sub_block(0, order * block_order).get();
448  double errsq = 0.0;
449  for(std::uint64_t i = 0; i < order; ++i)
450  {
451  double col_val = COL_SHIFT * i;
452  for(std::uint64_t j = 0; j < block_order; ++j)
453  {
454  double diff = trans_block[i * block_order + j] -
455  (col_val + ROW_SHIFT * (b * block_order + j));
456  errsq += diff * diff;
457  }
458  }
459  return errsq;
460  }
461  );
462 
463  if(verbose)
464  std::cout << " Squared sum of differences: " << errsq << "\n";
465 
466  return errsq;

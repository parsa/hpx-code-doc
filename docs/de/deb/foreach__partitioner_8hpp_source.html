42  // The static partitioner simply spawns one chunk of iterations for
43  // each available core.
44  template <typename ExPolicy_, typename Result = void>
45  struct foreach_static_partitioner
46  {
47  template <typename ExPolicy, typename FwdIter, typename F1,
48  typename F2>
49  static FwdIter call(ExPolicy && policy, FwdIter first,
50  std::size_t count, F1 && f1, F2 && f2)
51  {
52  typedef typename
53  hpx::util::decay<ExPolicy>::type::executor_parameters_type
54  parameters_type;
55  typedef executor_parameter_traits<parameters_type>
56  parameters_traits;
57 
58  // inform parameter traits
59  scoped_executor_parameters<parameters_type> scoped_param(
60  policy.parameters());
61 
62  FwdIter last = parallel::v1::detail::next(first, count);
63 
64  std::vector<hpx::future<Result> > inititems, workitems;
65  std::list<std::exception_ptr> errors;
66 
67  try {
68  // estimates a chunk size based on number of cores used
69  typedef typename parameters_traits::has_variable_chunk_size
70  has_variable_chunk_size;
71 
72  auto shapes =
73  get_bulk_iteration_shape_idx(
74  policy, inititems, f1, first, count, 1,
75  has_variable_chunk_size());
76 
77  workitems = execution::bulk_async_execute(
78  policy.executor(),
79  partitioner_iteration<Result, F1>{std::forward<F1>(f1)},
80  std::move(shapes));
81  }
82  catch (...) {
83  handle_local_exceptions<ExPolicy>::call(
84  std::current_exception(), errors);
85  }
86 
87  // wait for all tasks to finish
88  hpx::wait_all(inititems, workitems);
89 
90  // handle exceptions
91  handle_local_exceptions<ExPolicy>::call(inititems, errors);
92  handle_local_exceptions<ExPolicy>::call(workitems, errors);
93 
94  try {
95  return f2(std::move(last));
96  }
97  catch (...) {
98  // rethrow either bad_alloc or exception_list
99  handle_local_exceptions<ExPolicy>::call(
100  std::current_exception());
101  }
102  }
103  };

111  static hpx::future<FwdIter> call(ExPolicy && policy,

123  // inform parameter traits
124  std::shared_ptr<scoped_executor_parameters>

126  scoped_executor_parameters
127  >(policy.parameters()));
128 
129  FwdIter last = parallel::v1::detail::next(first, count);
130 
131  std::vector<hpx::future<Result> > inititems, workitems;
132  std::list<std::exception_ptr> errors;
133 
134  try {
135  // estimates a chunk size based on number of cores used
136  typedef typename parameters_traits::has_variable_chunk_size
137  has_variable_chunk_size;
138 
139  auto shapes =
140  get_bulk_iteration_shape_idx(
141  policy, inititems, f1, first, count, 1,
142  has_variable_chunk_size());
143 
144  workitems = execution::bulk_async_execute(
145  policy.executor(),
146  partitioner_iteration<Result, F1>{std::forward<F1>(f1)},
147  std::move(shapes));
148  }
149  catch (std::bad_alloc const&) {
150  return hpx::make_exceptional_future<FwdIter>(
151  std::current_exception());
152  }
153  catch (...) {
154  errors.push_back(std::current_exception());
155  }
156 
157  // wait for all tasks to finish
158  return hpx::dataflow(
159  [last, errors, scoped_param, f2](
160  std::vector<hpx::future<Result> > && r1,
161  std::vector<hpx::future<Result> > && r2) mutable
162  -> FwdIter
163  {
164  HPX_UNUSED(scoped_param);
165  handle_local_exceptions<ExPolicy>::call(r1, errors);
166  handle_local_exceptions<ExPolicy>::call(r2, errors);
167 
168  return f2(std::move(last));
169  },
170  std::move(inititems), std::move(workitems));
171  }
172  };

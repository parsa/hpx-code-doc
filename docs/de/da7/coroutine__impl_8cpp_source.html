76  void coroutine_impl::operator()()
77  {
78  typedef super_type::context_exit_status context_exit_status;
79  context_exit_status status = super_type::ctx_exited_return;
80 
81  // loop as long this coroutine has been rebound
82  do
83  {
84  std::exception_ptr tinfo;
85  try
86  {
87  this->check_exit_state();
88 
89  HPX_ASSERT(this->count() > 0);
90 
91  {
92  coroutine_self* old_self = coroutine_self::get_self();
93  coroutine_self self(this, old_self);
94  reset_self_on_exit on_exit(&self, old_self);
95 
96  this->m_result_last = m_fun(*this->args());
97 
98  // if this thread returned 'terminated' we need to reset
99  // the functor and the bound arguments
100  if (this->m_result_last.first == terminated)
101  this->reset();
102  }
103 
104  // return value to other side of the fence
105  this->bind_result(&this->m_result_last);
106  }
107  catch (exit_exception const&) {
108  status = super_type::ctx_exited_exit;
109  tinfo = std::current_exception();
110  this->reset(); // reset functor
111  }
112  catch (boost::exception const&) {
113  status = super_type::ctx_exited_abnormally;
114  tinfo = std::current_exception();
115  this->reset();
116  }
117  catch (std::exception const&) {
118  status = super_type::ctx_exited_abnormally;
119  tinfo = std::current_exception();
120  this->reset();
121  }
122  catch (...) {
123  status = super_type::ctx_exited_abnormally;
124  tinfo = std::current_exception();
125  this->reset();
126  }
127 
128  this->do_return(status, std::move(tinfo));
129 
130  } while (this->m_state == super_type::ctx_running);
131 
132  // should not get here, never
133  HPX_ASSERT(this->m_state == super_type::ctx_running);
134  }

219  coroutine_impl* coroutine_impl::allocate(
220  thread_id_repr_type id, std::ptrdiff_t stacksize)
221  {
222  // start looking at the matching heap
223  std::size_t const heap_num = std::size_t(id) / 32; //-V112
224  std::size_t const heap_count = get_heap_count(stacksize);
225 
226  // look through all heaps to find an available coroutine object
227  coroutine_impl* p = get_heap(heap_num, stacksize).allocate();
228  if (!p)
229  {
230  for (std::size_t i = 1; i != heap_count && !p; ++i)
231  {
232  p = get_heap(heap_num + i, stacksize).allocate();
233  }
234  }
235  return p;
236  }

31  std::size_t prefetch_distance_factor = 2;
32  std::vector<double> c(10007, 1.0);
33 
34  std::vector<std::size_t> range(10007);
35  std::iota(range.begin(), range.end(), 0);
36 
37  auto ctx = hpx::parallel::util::make_prefetcher_context(
38  range.begin(), range.end(), prefetch_distance_factor, c);
39 
40  hpx::parallel::for_each(std::forward<ExPolicy>(policy),
41  ctx.begin(), ctx.end(),
42  [&](std::size_t i)

44  c[i] = 42.1;
45  });
46 
47  // verify values

49  std::for_each(std::begin(c), std::end(c),

22 {

36 std::ptrdiff_t structure_t::find(std::ptrdiff_t i) const
37 {
38  assert(i>=0 && i<N);
39  if (B == 0) return -1;
40  std::ptrdiff_t b0 = 0, b1 = B-1;
41  auto loopinv = [&]() { return b0>=0 && b1<B && b0<=b1; };
42  auto loopvar = [&]() { return b1 - b0; };
43  assert(loopinv());
44  (void)loopinv;
45  std::ptrdiff_t old_loopvar = loopvar();
46  while (b0 < b1 && i>=begin[b0] && i<end[b1]) {
47  std::ptrdiff_t b = (b0 + b1)/2;
48  if (i < end[b]) {
49  b1 = b;
50  } else {
51  b0 = b + 1;
52  }
53  assert(loopinv());
54  auto next_loopvar = loopvar();
55  assert(next_loopvar >= 0 && next_loopvar < old_loopvar);
56  old_loopvar = next_loopvar;
57  }
58  if (b0 == b1 && i>=begin[b0] && i<end[b1]) {
59  // found
60  return b0;
61  }
62  // not found
63  return -1;
64 }

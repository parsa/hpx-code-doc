42  bool apply_single(parcel &p)
43  {
44  archive_.reset();
45  archive_ << p;
46 
47  // We are doing a fixed point iteration until we are sure that the
48  // serialization process requires nothing more to wait on ...
49  // Things where we need waiting:
50  // - (shared_)future<id_type>: when the future wasn't ready yet, we
51  // need to do another await round for the id splitting
52  // - id_type: we need to await, if and only if, the credit of the
53  // needs to split.
54  if(preprocess_.has_futures())
55  {
56  auto this_ = this->shared_from_this();
57  preprocess_([this_](){ this_->apply(); });
58  return false;
59  }
60  archive_.flush();
61  p.size() = preprocess_.size() + overhead_;
62  p.num_chunks() = archive_.get_num_chunks();
63  hpx::serialization::detail::preprocess::split_gids_map split_gids;
64  std::swap(split_gids, preprocess_.split_gids_);
65  p.set_split_gids(std::move(split_gids));
66 
67  return true;
68  }

152 double perform_2n_iterations()
153 {
154  std::vector<coroutine_type*> coroutines;
155  std::vector<std::uint64_t> indices;
156 
157  coroutines.reserve(contexts);
158  indices.reserve(iterations);
159 
160  std::mt19937_64 prng(seed);
161  std::uniform_int_distribution<std::uint64_t>
162  dist(0, contexts - 1);
163 
164  kernel k;
165 
166  for (std::uint64_t i = 0; i < contexts; ++i)
167  {
168  coroutine_type* c = new coroutine_type(k);
169  coroutines.push_back(c);
170  }
171 
172  for (std::uint64_t i = 0; i < iterations; ++i)
173  indices.push_back(dist(prng));
174 
176  // Warmup
177  for (std::uint64_t i = 0; i < iterations; ++i)
178  {
179  (*coroutines[indices[i]])(wait_signaled);
180  }
181 
182  hpx::util::high_resolution_timer t;
183 
184  for (std::uint64_t i = 0; i < iterations; ++i)
185  {
186  (*coroutines[indices[i]])(wait_signaled);
187  }
188 
189  double elapsed = t.elapsed();
190 
191  for (std::uint64_t i = 0; i < contexts; ++i)
192  {
193  delete coroutines[i];
194  }
195 
196  coroutines.clear();
197 
198  return elapsed;
199 }

80  private:
81  template<typename Func>
82  int poll_verbs_event_channel(int fd, Func &&f)
83  {
84  const int eventChannel = 0;
85  const int numFds = 1;
86  //
87  scoped_lock lock(polling_mutex);
88  //
89  pollfd pollInfo[numFds];
90  int polltimeout = 0; // seconds*1000; // 10000 == 10 sec
91 
92  pollInfo[eventChannel].fd = fd;
93  pollInfo[eventChannel].events = POLLIN;
94  pollInfo[eventChannel].revents = 0;
95 
96  // Wait for an event on one of the descriptors.
97  int rc = poll(pollInfo, numFds, polltimeout);
98 
99  // If there were no events/messages
100  if (rc == 0) {
101  return 0;
102  }
103 
104  // There was an error so log the failure and try again.
105  if (rc == -1) {
106  int err = errno;
107  // an interrupt is ok, not an epic fail
108  if (err == EINTR) {
109  LOG_TRACE_MSG("poll returned EINTR, continuing ...");
110  return 0;
111  }
112  rdma_error e(err, "poll_verbs_event_channel failed");
113  throw e;
114  }
115 
116  if (pollInfo[eventChannel].revents & POLLIN) {
117  LOG_TRACE_MSG("input event available on event channel");
118  f();
119  }
120  return 1;
121  }

124  // ----------------------------------------------------------------------------
125  // If the expected event is received, it is acked, otherwise the

128  // Communication event details are returned in the rdma_cm_event structure.

136  static int get_event(
137  struct rdma_event_channel *channel,
138  event_ack_type ack,
139  rdma_cm_event_type *event,
140  struct rdma_cm_event *&cm_event)
141  {
142  cm_event = nullptr;
143  // This operation can block if there are no pending events available.
144  // (So only call it after the event poll says there is an event waiting)
145  if (event)
146  {
147  LOG_DEVEL_MSG("waiting for " << rdma_event_str(*event)
148  << " on event channel " << hexnumber(channel->fd));
149  }
150  else
151  {
152  LOG_DEVEL_MSG("waiting for on event channel " << hexnumber(channel->fd));
153  }
154 
155  int rc = rdma_get_cm_event(channel, &cm_event);
156  if (rc != 0) {
157  int err = errno;
158  LOG_ERROR_MSG("error getting rdma cm event: "
159  << rdma_error::error_string(err));
160  return -1;
161  }
162  uint32_t qpnum = (cm_event->id->qp) ? cm_event->id->qp->qp_num : 0;
163  LOG_DEVEL_MSG("got " << rdma_event_str(cm_event->event)
164  << " on event channel " << hexnumber(channel->fd)
165  << ": on qp " << decnumber(qpnum));
166 
167  // we have to ack events, even when they are not the ones we wanted
168  if (event && cm_event->event != *event) {
169  LOG_ERROR_MSG("mismatch " << rdma_event_str(cm_event->event)
170  << " not " << rdma_event_str(*event));
171  if (ack==do_ack_event) ack_event(cm_event);
172  return -1;
173  }
174  // Acknowledge the event.
175  if (ack==do_ack_event) return ack_event(cm_event);
176  else return 0;
177  }
178 
179  // ----------------------------------------------------------------------------
180  static int ack_event(struct rdma_cm_event *cm_event)
181  {
182  if (cm_event == nullptr) {
183  LOG_ERROR_MSG("nullptr rdma cm event : cannot ack");
184  return ENOENT;
185  }
186 
187  LOG_DEVEL_MSG("ack rdma cm event " << rdma_event_str(cm_event->event)
188  << " for rdma cm id " << cm_event->id);
189 
190  int err = rdma_ack_cm_event(cm_event);
191  if (err != 0) {
192  rdma_error e(errno, std::string("Failed acking event ") +
193  rdma_event_str(cm_event->event));
194  throw e;
195  }
196  return 0;
197  }

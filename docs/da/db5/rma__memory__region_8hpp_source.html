44  // --------------------------------------------------------------------
45  // construct a memory region object by registering an existing address buffer
46  rma_memory_region(provider_domain *pd, const void *buffer, const uint64_t length)
47  {
48  address_ = static_cast<char *>(const_cast<void *>(buffer));
49  base_addr_ = address_;
50  size_ = length;
51  used_space_ = length;
52  flags_ = BLOCK_USER;
53 
54  int ret = traits::rma_memory_region_traits<RegionProvider>::register_memory(
55  pd, const_cast<void*>(buffer), length,
56  traits::rma_memory_region_traits<RegionProvider>::flags(),
57  0, (uint64_t)address_, 0, &(region_), nullptr);
58 
59  if (ret) {
60  LOG_ERROR_MSG(
61  "error registering region "
62  << hexpointer(buffer) << hexlength(length));
63  throw std::runtime_error("error in memory registration");
64  }
65  else {
66  LOG_DEBUG_MSG(
67  "OK registering region "
68  << hexpointer(buffer) << hexpointer(address_)
69  << "desc " << hexpointer(fi_mr_desc(region_))
70  << "rkey " << hexpointer(fi_mr_key(region_))
71  << "length " << hexlength(size_));
72  }
73  }
74 
75  // --------------------------------------------------------------------
76  // allocate a block of size length and register it
77  int allocate(provider_domain *pd, uint64_t length)
78  {
79  // Allocate storage for the memory region.
80  void *buffer = new char[length];
81  if (buffer != nullptr) {
82  LOG_DEBUG_MSG("allocated storage for memory region with malloc OK "
83  << hexnumber(length));
84  }
85  address_ = static_cast<char*>(buffer);
86  base_addr_ = static_cast<char*>(buffer);
87  size_ = length;
88  used_space_ = 0;
89 
90  int ret = traits::rma_memory_region_traits<RegionProvider>::register_memory(
91  pd, const_cast<void*>(buffer), length,
92  traits::rma_memory_region_traits<RegionProvider>::flags(),
93  0, (uint64_t)address_, 0, &(region_), nullptr);
94 
95  if (ret) {
96  LOG_ERROR_MSG(
97  "error registering region "
98  << hexpointer(buffer) << hexlength(length));
99  throw std::runtime_error("error in memory registration");
100  }
101  else {
102  LOG_DEBUG_MSG(
103  "OK registering region "
104  << hexpointer(buffer) << hexpointer(address_)
105  << "desc " << hexpointer(fi_mr_desc(region_))
106  << "rkey " << hexpointer(fi_mr_key(region_))
107  << "length " << hexlength(size_));
108  }
109 
110  LOG_DEBUG_MSG("allocated/registered memory region " << hexpointer(this)
111  << "with desc " << hexnumber(get_desc())
112  << "at address " << hexpointer(get_address())
113  << "with length " << hexlength(get_size()));
114  return 0;
115  }
116 
117  // --------------------------------------------------------------------
118  // destroy the region and memory according to flag settings
119  ~rma_memory_region()
120  {
121  if (get_partial_region()) return;
122  release();
123  }
124 
125  // --------------------------------------------------------------------
126  // Deregister and free the memory region.
127  // returns 0 when successful, -1 otherwise
128  int release(void)
129  {
130  if (region_ != nullptr) {
131  LOG_TRACE_MSG("About to release memory region with desc "
132  << hexpointer(get_desc()));
133  // get these before deleting/unregistering (for logging)
134  const void *buffer = get_base_address();
135  LOG_EXCLUSIVE(
136  uint32_t length = get_size();
137  );
138  //
139  if (traits::rma_memory_region_traits<RegionProvider>::
140  unregister_memory(region_))
141  {
142  LOG_ERROR_MSG("Error, fi_close mr failed\n");
143  return -1;
144  }
145  else {
146  LOG_DEBUG_MSG("deregistered memory region with desc "
147  << hexpointer(get_desc())
148  << "at address " << hexpointer(buffer)
149  << "with length " << hexlength(length));
150  }
151  if (!get_user_region()) {
152  delete [](static_cast<const char*>(buffer));
153  }
154  region_ = nullptr;
155  }
156  return 0;
157  }

210  // --------------------------------------------------------------------
211  // flags used for management of lifetime
212  enum {
213  BLOCK_USER = 1,
214  BLOCK_TEMP = 2,
215  BLOCK_PARTIAL = 4,
216  };
217 
218  // --------------------------------------------------------------------
219  // A user allocated region use memory allocted by the user.
220  // on destruction, the memory is unregisterd, but not deleted
221  inline void set_user_region() {
222  flags_ |= BLOCK_USER;
223  }
224 
225  inline bool get_user_region() const {
226  return (flags_ & BLOCK_USER) == BLOCK_USER;
227  }
228 
229  // --------------------------------------------------------------------
230  // A temp region is one that the memory pool is not managing
231  // so it is unregistered and deleted when returned to the pool and not reused
232  inline void set_temp_region() {
233  flags_ |= BLOCK_TEMP;
234  }
235 
236  inline bool get_temp_region() const {
237  return (flags_ & BLOCK_TEMP) == BLOCK_TEMP;
238  }
239 
240  // --------------------------------------------------------------------
241  // a partial region is a subregion of a larger memory region
242  // on destruction, it is not unregistered or deleted as the 'parent' region
243  // will delete many partial regions on destruction
244  inline void set_partial_region() {
245  flags_ |= BLOCK_PARTIAL;
246  }
247 
248  inline bool get_partial_region() const {
249  return (flags_ & BLOCK_PARTIAL) == BLOCK_PARTIAL;
250  }

207  void io_service_work()
208  {
209  std::size_t k = 0;
210  // We only execute work on the IO service while HPX is starting
211  while(hpx::is_starting())
212  {
213  bool has_work = sender_.background_work();
214  has_work = receiver_.background_work() || has_work;
215  if(has_work)

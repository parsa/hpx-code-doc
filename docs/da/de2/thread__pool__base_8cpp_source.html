66  void thread_pool_base::init_pool_time_scale()
67  {
68  // scale timestamps to nanoseconds
69  std::uint64_t base_timestamp = util::hardware::timestamp();
70  std::uint64_t base_time = util::high_resolution_clock::now();
71  std::uint64_t curr_timestamp = util::hardware::timestamp();
72  std::uint64_t curr_time = util::high_resolution_clock::now();
73 
74  while ((curr_time - base_time) <= 100000)
75  {
76  curr_timestamp = util::hardware::timestamp();
77  curr_time = util::high_resolution_clock::now();
78  }
79 
80  if (curr_timestamp - base_timestamp != 0)
81  {
82  timestamp_scale_ = double(curr_time - base_time) /
83  double(curr_timestamp - base_timestamp);
84  }
85  }

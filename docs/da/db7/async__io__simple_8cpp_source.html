6 // The purpose of this example is to demonstrate how it is possible to
7 // schedule an IO task onto one of the IO-threads in HPX (which are OS-threads)

18 // this function will be executed by a dedicated OS thread
19 void do_async_io(char const* string_to_write, int& result)
20 {
21  // This IO operation will possibly block the IO thread in the
22  // kernel.
23  std::cout << "OS-thread: " << string_to_write << std::endl;
24 
25  result = 0;
26 }

28 // This function will be executed by an HPX thread
29 int async_io(char const* string_to_write)
30 {
31  int result = 0;
32 
33  {
34  // Get a reference to one of the IO specific HPX io_service objects ...
35  hpx::threads::executors::io_pool_executor scheduler;
36 
37  // ... and schedule the handler to run on one of its OS-threads.
38  scheduler.add(hpx::util::bind(&do_async_io,
39  string_to_write, std::ref(result)));
40 
41  // Note that the destructor of the scheduler object will wait for
42  // the scheduled task to finish executing. This might be
43  // undesireable, in which case the technique demonstrated in
44  // the example async_io_low_level.cpp is preferrable.
45  }
46 
47  return result; // this will be executed only after result has been set
48 }

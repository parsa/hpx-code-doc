38  template <typename InIter>
39  void sequential_rotate_helper(InIter first, InIter new_first,
40  InIter last)
41  {
42  InIter next = new_first;
43  while (first != next)
44  {
45  std::iter_swap(first++, next++);
46  if (next == last)
47  {
48  next = new_first;
49  }
50  else if (first == new_first)
51  {
52  new_first = next;
53  }
54  }
55  }

57  template <typename InIter>
58  inline std::pair<InIter, InIter>
59  sequential_rotate(InIter first, InIter new_first, InIter last)
60  {
61  if (first != new_first && new_first != last)
62  sequential_rotate_helper(first, new_first, last);
63 
64  std::advance(first, std::distance(new_first, last));
65  return std::make_pair(first, last);
66  }

68  template <typename ExPolicy, typename FwdIter>
69  hpx::future<std::pair<FwdIter, FwdIter> >
70  rotate_helper(ExPolicy policy, FwdIter first, FwdIter new_first,
71  FwdIter last)
72  {
73  typedef std::false_type non_seq;
74 
75  auto p =
76  execution::parallel_task_policy()
77  .on(policy.executor())
78  .with(policy.parameters());
79 
80  detail::reverse<FwdIter> r;
81  return dataflow(
82  [=](hpx::future<FwdIter>&& f1, hpx::future<FwdIter>&& f2) mutable
83  -> hpx::future<std::pair<FwdIter, FwdIter> >
84  {
85  // propagate exceptions
86  f1.get(); f2.get();
87 
88  hpx::future<FwdIter> f = r.call(p, non_seq(), first, last);
89  return f.then(
90  [=] (hpx::future<FwdIter> && f) mutable
91  -> std::pair<FwdIter, FwdIter>
92  {
93  f.get(); // propagate exceptions
94  std::advance(first, std::distance(new_first, last));
95  return std::make_pair(first, last);
96  });
97  },
98  r.call(p, non_seq(), first, new_first),
99  r.call(p, non_seq(), new_first, last));
100  }

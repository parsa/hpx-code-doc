79 
80  // Physical sizes is equal to logical sizes if physical sizes are
81  // not defined
82  list_type & hw_sizes_ = hw_sizes.size() ? hw_sizes : sw_sizes;
83 
84  // Check that sizes of the view are valid regarding its dimension
85  HPX_ASSERT_MSG(sw_sizes.size() == N, \
86  "Defined co-sizes must match the partitioned_vector_view " \
87  "dimension");
88 
89  // Generate two mixed radix basis
90  fill_basis(hw_sizes_, hw_basis_, indices() );
91  fill_basis(sw_sizes, sw_basis_, indices() );
92 
93  // Compute the needed size for the described view
94  std::ptrdiff_t limit = 0;
95  std::size_t idx = 0;
96  for(std::size_t const & i : sw_sizes)
97  {
98  limit += (i-1) * hw_basis_[idx];
99  idx++;
100  }
101 
102  // Check that combined sizes doesn't overflow the used space
103  HPX_ASSERT_MSG(limit <= std::distance(begin,last), \
104  "Space dedicated to the described partitioned_vector_view " \
105  "is too small");
106 
107  // Update end_
108  end_ += hw_basis_[N-1] * sw_sizes.begin()[N-1];
109  cend_ += hw_basis_[N-1] * sw_sizes.begin()[N-1];
110  }

112  private:
113  // Update view basis from the partitioned_vector_view sizes
114  template<std::size_t... I>
115  void fill_basis(
116  list_type const & sizes,
117  std::array<std::size_t,N+1> & basis,
118  hpx::util::detail::pack_c<std::size_t, I...>) const
119  {
120  basis[0] = 1;
121 
122  std::size_t tmp = 1;
123  auto in = sizes.begin();
124 
125  (void)std::initializer_list<int>{
126  (static_cast<void>( basis[I+1] = tmp *= (*in), in++)
127  , 0)... };

130  template<typename... I>
131  std::size_t offset_solver(I ... index) const
132  {
133  // Check that the subscript is valid regarding the view dimension
134  static_assert( sizeof...(I) == N, \
135  "Subscript must match the partitioned_vector_view " \
136  "dimension");
137 
138  // Check that all the elements are of integral type
139  static_assert(
140  util::detail::all_of<
141  typename std::is_integral<I>::type ... >::value,
142  "One or more elements in subscript is not integral");
143 
144  std::size_t offset = 0;
145  std::size_t i = 0;
146 
147  (void)std::initializer_list<int>
148  { ( static_cast<void>(
149  offset += ((std::size_t)index) * hw_basis_[i++])
150  , 0 )...
151  };
152 
153  // Check that the solved index doesn't overflow the used space
154  HPX_ASSERT_MSG( offset< hw_basis_.back(), \
155  "*Invalid partitioned_vector_view subscript");

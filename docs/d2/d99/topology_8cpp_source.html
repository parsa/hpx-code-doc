56  mask_type topology::get_service_affinity_mask(
57  mask_cref_type used_processing_units, error_code& ec) const
58  {
59  // We bind the service threads to the first NUMA domain. This is useful
60  // as the first NUMA domain is likely to have the PCI controllers etc.
61  mask_cref_type machine_mask = this->get_numa_node_affinity_mask(0, ec);
62  if (ec || !any(machine_mask))
63  return mask_type();
64 
65  if (&ec != &throws)
66  ec = make_success_code();
67 
68  mask_type res = ~used_processing_units & machine_mask;
69 
70  return (!any(res)) ? machine_mask : res;
71  }

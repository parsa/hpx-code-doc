144  // Create a new GID (if called for the first time), assign this
145  // GID to this instance of a component and register this gid
146  // with the AGAS service
147  //
148  // Returns he global id (GID) assigned to this instance of a component
149  naming::gid_type get_base_gid(
150  naming::gid_type const& assign_gid = naming::invalid_gid) const
151  {
152  if (!gid_)
153  {
154  naming::address addr(get_current_address());
155  if (!assign_gid)
156  {
157  gid_ = hpx::detail::get_next_id();
158  if (!applier::bind_gid_local(gid_, addr))
159  {
160  std::ostringstream strm;
161  strm << "failed to bind id " << gid_
162  << "to locality: " << hpx::get_locality();
163 
164  gid_ = naming::invalid_gid; // invalidate GID
165 
166  HPX_THROW_EXCEPTION(duplicate_component_address,
167  "component_base<Component>::get_base_gid",
168  strm.str());
169  }
170  }
171  else
172  {
173  applier::applier& appl = hpx::applier::get_applier();
174  gid_ = assign_gid;
175  naming::detail::strip_credits_from_gid(gid_);
176 
177  if (!agas::bind(
178  launch::sync, gid_, addr, appl.get_locality_id()))
179  {
180  std::ostringstream strm;
181  strm << "failed to rebind id " << gid_
182  << "to locality: " << hpx::get_locality();
183 
184  gid_ = naming::invalid_gid; // invalidate GID
185 
186  HPX_THROW_EXCEPTION(duplicate_component_address,
187  "component_base<Component>::get_base_gid",
188  strm.str());
189  }
190  }
191  }
192 
193  std::unique_lock<naming::gid_type::mutex_type> l(gid_.get_mutex());
194 
195  if (!naming::detail::has_credits(gid_))
196  {
197  naming::gid_type gid = gid_;
198  return gid;
199  }
200 
201  // on first invocation take all credits to avoid a self reference
202  naming::gid_type gid = gid_;
203 
204  naming::detail::strip_credits_from_gid(
205  const_cast<naming::gid_type&>(gid_));
206 
207  HPX_ASSERT(naming::detail::has_credits(gid));
208 
209  // We have to assume this credit was split as otherwise the gid
210  // returned at this point will control the lifetime of the
211  // component.
212  naming::detail::set_credit_split_mask_for_gid(gid);
213  return gid;
214  }

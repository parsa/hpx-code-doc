236  threads::thread_result_type
237  runtime_impl::run_helper(
238  util::function_nonser<runtime::hpx_main_function_type> func, int& result)
239  {
240  lbt_ << "(2nd stage) runtime_impl::run_helper: launching pre_main";
241 
242  // Change our thread description, as we're about to call pre_main
243  threads::set_thread_description(threads::get_self_id(), "pre_main");
244 
245  // Finish the bootstrap
246  result = hpx::pre_main(mode_);
247  if (result) {
248  lbt_ << "runtime_impl::run_helper: bootstrap "
249  "aborted, bailing out";
250  return threads::thread_result_type(threads::terminated, nullptr);
251  }
252 
253  lbt_ << "(4th stage) runtime_impl::run_helper: bootstrap complete";
254  set_state(state_running);
255 
256  parcel_handler_.enable_alternative_parcelports();
257 
258  // reset all counters right before running main, if requested
259  if (get_config_entry("hpx.print_counter.startup", "0") == "1")
260  {
261  bool reset = false;
262  if (get_config_entry("hpx.print_counter.reset", "0") == "1")
263  reset = true;
264 
265  error_code ec(lightweight); // ignore errors
266  evaluate_active_counters(reset, "startup", ec);
267  }
268 
269  // Connect back to given latch if specified
270  std::string connect_back_to(
271  get_config_entry("hpx.on_startup.wait_on_latch", ""));
272  if (!connect_back_to.empty())
273  {
274  // inform launching process that this locality is up and running
275  hpx::lcos::latch l;
276  l.connect_to(connect_back_to);
277  l.count_down_and_wait();
278  }
279 
280  // Now, execute the user supplied thread function (hpx_main)
281  if (!!func) {
282  // Change our thread description, as we're about to call hpx_main
283  threads::set_thread_description(threads::get_self_id(), "hpx_main");
284 
285  // Call hpx_main
286  result = func();
287  }
288  return threads::thread_result_type(threads::terminated, nullptr);
289  }

291  int runtime_impl::start(
292  util::function_nonser<hpx_main_function_type> const& func, bool blocking)
293  {
294 #if defined(_WIN64) && defined(_DEBUG) && !defined(HPX_HAVE_FIBER_BASED_COROUTINES)
295  // needs to be called to avoid problems at system startup
296  // see: http://connect.microsoft.com/VisualStudio/feedback/ViewFeedback.aspx?FeedbackID=100319
297  _isatty(0);
298 #endif
299  // {{{ early startup code - local
300 
301  // initialize instrumentation system
302  util::apex_init();
303 
304  LRT_(info) << "cmd_line: " << get_config().get_cmd_line();
305 
306  lbt_ << "(1st stage) runtime_impl::start: booting locality " << here();
307 
308  // start runtime_support services
309  runtime_support_->run();
310  lbt_ << "(1st stage) runtime_impl::start: started "
311  "runtime_support component";
312 
313 #ifdef HPX_HAVE_IO_POOL
314  // start the io pool
315  io_pool_.run(false);
316  lbt_ << "(1st stage) runtime_impl::start: started the application "
317  "I/O service pool";
318 #endif
319  // start the thread manager
320  thread_manager_->run();
321  lbt_ << "(1st stage) runtime_impl::start: started threadmanager";
322  // }}}
323 
324  // invoke the AGAS v2 notifications
325  agas::get_big_boot_barrier().trigger();
326 
327  // {{{ launch main
328  // register the given main function with the thread manager
329  lbt_ << "(1st stage) runtime_impl::start: launching run_helper "
330  "HPX thread";
331 
332  threads::thread_init_data data(
333  util::bind(&runtime_impl::run_helper, this, func,
334  std::ref(result_)),
335  "run_helper", 0, threads::thread_priority_normal, std::size_t(-1),
336  threads::get_stack_size(threads::thread_stacksize_large));
337 
338  threads::thread_id_type id = threads:: invalid_thread_id;
339  thread_manager_->register_thread(data, id);
340  this->runtime::starting();
341  // }}}
342 
343  // block if required
344  if (blocking)
345  return wait(); // wait for the shutdown_action to be executed
346 
347  // Register this thread with the runtime system to allow calling certain
348  // HPX functionality from the main thread.
349  init_tss("main-thread", 0, "", false);
350 
351  return 0; // return zero as we don't know the outcome of hpx_main yet
352  }

411  int runtime_impl::wait()
412  {
413  LRT_(info) << "runtime_impl: about to enter wait state";
414 
415  // start the wait_helper in a separate thread
416  compat::mutex mtx;
417  compat::condition_variable cond;
418  bool running = false;
419 
420  compat::thread t (util::bind(
421  &runtime_impl::wait_helper,
422  this, std::ref(mtx), std::ref(cond), std::ref(running)
423  ));
424 
425  // wait for the thread to run
426  {
427  std::unique_lock<compat::mutex> lk(mtx);
428  while (!running)
429  cond.wait(lk);
430  }
431 
432  // use main thread to drive main thread pool
433  main_pool_.thread_run(0);
434 
435  // block main thread
436  t.join();
437 
438  LRT_(info) << "runtime_impl: exiting wait state";
439  return result_;
440  }

443  // First half of termination process: stop thread manager,
444  // schedule a task managed by timer_pool to initiate second part
445  void runtime_impl::stop(bool blocking)
446  {
447  LRT_(warning) << "runtime_impl: about to stop services";
448 
449  // flush all parcel buffers, stop buffering parcels at this point
450  //parcel_handler_.do_background_work(true);
451 
452  // execute all on_exit functions whenever the first thread calls this
453  this->runtime::stopping();
454 
455  // stop runtime_impl services (threads)
456  thread_manager_->stop(false); // just initiate shutdown
457 
458  if (threads::get_self_ptr())
459  {
460  // schedule task on separate thread to execute stopped() below
461  // this is necessary as this function (stop()) might have been called
462  // from a HPX thread, so it would deadlock by waiting for the thread
463  // manager
464  compat::mutex mtx;
465  compat::condition_variable cond;
466  std::unique_lock<compat::mutex> l(mtx);
467 
468  compat::thread t(util::bind(&runtime_impl::stopped, this, blocking,
469  std::ref(cond), std::ref(mtx)));
470  cond.wait(l);
471 
472  t.join();
473  }
474  else
475  {
476  runtime_support_->stopped(); // re-activate shutdown HPX-thread
477  thread_manager_->stop(blocking); // wait for thread manager
478 
479  // this disables all logging from the main thread
480  deinit_tss();
481 
482  LRT_(info) << "runtime_impl: stopped all services";
483  }
484 
485  // stop the rest of the system
486  parcel_handler_.stop(blocking); // stops parcel pools as well
487 #ifdef HPX_HAVE_IO_POOL
488  io_pool_.stop(); // stops io_pool_ as well
489 #endif
490  deinit_tss();
491  }

513  void runtime_impl::report_error(
514  std::size_t num_thread, std::exception_ptr const& e)
515  {
516  // Early and late exceptions, errors outside of HPX-threads
517  if (!threads::get_self_ptr() || !threads::threadmanager_is(state_running))
518  {
519  // report the error to the local console
520  detail::report_exception_and_continue(e);
521 
522  // store the exception to be able to rethrow it later
523  {
524  std::lock_guard<compat::mutex> l(mtx_);
525  exception_ = e;
526  }
527  lcos::barrier::get_global_barrier().detach();
528 
529  // initiate stopping the runtime system
530  runtime_support_->notify_waiting_main();
531  stop(false);
532 
533  return;
534  }

646  void runtime_impl::init_tss_ex(
647  std::string const& locality, char const* context, std::size_t num,
648  char const* postfix, bool service_thread, error_code& ec)
649  {
650  // initialize our TSS
651  this->runtime::init_tss();
652 
653  // initialize applier TSS
654  applier_.init_tss();
655 
656  // set the thread's name, if it's not already set
657  if (nullptr == runtime::thread_name_.get())
658  {
659  std::string* fullname = new std::string(locality);
660  if (!locality.empty())
661  *fullname += "/";
662  *fullname += context;
663  if (postfix && *postfix)
664  *fullname += postfix;
665  *fullname += "#" + std::to_string(num);
666  runtime::thread_name_.reset(fullname);
667 
668  char const* name = runtime::thread_name_.get()->c_str();
669 
670  // initialize thread mapping for external libraries (i.e. PAPI)
671  thread_support_->register_thread(name, ec);
672 
673  // initialize coroutines context switcher
674  hpx::threads::coroutines::thread_startup(name);
675 
676  // register this thread with any possibly active Intel tool
677  HPX_ITT_THREAD_SET_NAME(name);
678 
679  // set thread name as shown in Visual Studio
680  util::set_thread_name(name);
681 
682 #if defined(HPX_HAVE_APEX)
683  if (std::strstr(name, "worker") != nullptr)
684  apex::register_thread(name);
685 #endif
686  }
687 
688  // if this is a service thread, set its service affinity
689  if (service_thread)
690  {
691  // FIXME: We don't set the affinity of the service threads on BG/Q,

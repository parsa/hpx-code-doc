36 struct gva_cache_key
37 {

49  explicit gva_cache_key(
50  hpx::naming::gid_type const& id, std::uint64_t count = 1)
51  : key_(hpx::naming::detail::get_stripped_gid(id),
52  hpx::naming::detail::get_stripped_gid(id) + (count - 1))
53  {
54  HPX_ASSERT(count);
55  }
56 
57  hpx::naming::gid_type get_gid() const
58  {
59  return key_.first;
60  }
61 
62  std::uint64_t get_count() const
63  {
64  hpx::naming::gid_type const size = key_.second - key_.first;
65  HPX_ASSERT(size.get_msb() == 0);
66  return size.get_lsb();
67  }
68 
69  friend bool operator<(gva_cache_key const& lhs, gva_cache_key const& rhs)
70  {
71  return lhs.key_.second < rhs.key_.first;
72  }
73 
74  friend bool operator==(gva_cache_key const& lhs, gva_cache_key const& rhs)
75  {
76  // Direct hit
77  if (lhs.key_ == rhs.key_)
78  {
79  return true;
80  }
81 
82  // Is lhs in rhs?
83  if (1 == lhs.get_count() && 1 != rhs.get_count())
84  {
85  return rhs.key_.first <= lhs.key_.first &&
86  lhs.key_.second <= rhs.key_.second;
87  }
88 
89  // Is rhs in lhs?
90  else if (1 != lhs.get_count() && 1 == rhs.get_count())
91  {
92  return lhs.key_.first <= rhs.key_.first &&
93  rhs.key_.second <= lhs.key_.second;
94  }
95 
96  return false;
97  }
98 };

162 void test_insert(gva_cache_type& cache, std::size_t num_entries)
163 {
164  hpx::naming::gid_type locality = hpx::get_locality();
165  std::uint32_t ct = hpx::components::component_invalid;
166 
167  std::vector<std::uint64_t> timings;
168  timings.reserve(num_entries);
169 
170  for (std::size_t i = 0; i != num_entries; ++i)
171  {
172  gva_cache_key key(hpx::detail::get_next_id(), 1);
173  hpx::agas::gva value(locality, ct, 1, std::uint64_t(0), 0);
174 
175  std::uint64_t t = hpx::util::high_resolution_clock::now();
176 
177  cache.insert(key, value);
178 
179  timings.push_back(hpx::util::high_resolution_clock::now() - t);
180  }
181 
182  calculate_histogram("insert", timings);
183 }

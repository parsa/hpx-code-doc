68  bool performance_counter_set::find_counter(counter_info const& info,
69  bool reset, error_code& ec)
70  {
71  // keep only local counters if requested
72  if (print_counters_locally_)
73  {
74  counter_path_elements p;
75  get_counter_path_elements(info.fullname_, p, ec);
76  if (ec) return false;
77 
78  if (p.parentinstanceindex_ != hpx::get_locality_id())
79  {
80  if (&ec != &throws)
81  ec = make_success_code();
82  return true;
83  }
84  }
85 
86  naming::id_type id = get_counter(info.fullname_, ec);
87  if (HPX_UNLIKELY(!id))
88  {
89  HPX_THROWS_IF(ec, bad_parameter,
90  "performance_counter_set::find_counter",
91  boost::str(boost::format(
92  "unknown performance counter: '%1%' (%2%)") %
93  info.fullname_ % ec.get_message()));
94  return false;
95  }
96 
97  {
98  std::unique_lock<mutex_type> l(mtx_);
99  infos_.push_back(info);
100  ids_.push_back(id);
101  reset_.push_back(reset ? 1 : 0);
102  }
103 
104  if (&ec != &throws)
105  ec = make_success_code();
106 
107  return true;
108  }

110  void performance_counter_set::add_counters(std::string const& name,
111  bool reset, error_code& ec)
112  {
113  using util::placeholders::_1;
114  using util::placeholders::_2;
115 
116  discover_counter_func func = util::bind(
117  &performance_counter_set::find_counter, this, _1, reset, _2);
118 
119  // do INI expansion on counter name
120  std::string n(name);
121  util::expand(n);
122 
123  // find matching counter types
124  discover_counter_type(n, std::move(func), discover_counters_full, ec);
125  if (ec) return;
126 
127  HPX_ASSERT(ids_.size() == infos_.size());
128  }

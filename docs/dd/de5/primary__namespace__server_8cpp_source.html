318 bool primary_namespace::bind_gid(
319  gva g
320  , naming::gid_type id
321  , naming::gid_type locality
322  )
323 { // {{{ bind_gid implementation
324  util::scoped_timer<std::atomic<std::int64_t> > update(
325  counter_data_.bind_gid_.time_
326  );
327  counter_data_.increment_bind_gid_count();
328  using hpx::util::get;
329 
330  naming::detail::strip_internal_bits_from_gid(id);
331 
332  std::unique_lock<mutex_type> l(mutex_);
333 
334  gva_table_type::iterator it = gvas_.lower_bound(id)
335  , begin = gvas_.begin()
336  , end = gvas_.end();
337 
338  if (it != end)
339  {
340  // If we got an exact match, this is a request to update an existing
341  // binding (e.g. move semantics).
342  if (it->first == id)
343  {
344  gva& gaddr = it->second.first;
345  naming::gid_type& loc = it->second.second;
346 
347  // Check for count mismatch (we can't change block sizes of
348  // existing bindings).
349  if (HPX_UNLIKELY(gaddr.count != g.count))
350  {
351  // REVIEW: Is this the right error code to use?
352  l.unlock();
353 
354  HPX_THROW_EXCEPTION(bad_parameter
355  , "primary_namespace::bind_gid"
356  , "cannot change block size of existing binding");
357  }
358 
359  if (HPX_UNLIKELY(components::component_invalid == g.type))
360  {
361  l.unlock();
362 
363  HPX_THROW_EXCEPTION(bad_parameter
364  , "primary_namespace::bind_gid"
365  , boost::str(boost::format(
366  "attempt to update a GVA with an invalid type, "
367  "gid(%1%), gva(%2%), locality(%3%)")
368  % id % g % locality));
369  }
370 
371  if (HPX_UNLIKELY(!locality))
372  {
373  l.unlock();
374 
375  HPX_THROW_EXCEPTION(bad_parameter
376  , "primary_namespace::bind_gid"
377  , boost::str(boost::format(
378  "attempt to update a GVA with an invalid locality id, "
379  "gid(%1%), gva(%2%), locality(%3%)")
380  % id % g % locality));
381  }
382 
383  // Store the new endpoint and offset
384  gaddr.prefix = g.prefix;
385  gaddr.type = g.type;
386  gaddr.lva(g.lva());
387  gaddr.offset = g.offset;
388  loc = locality;
389 
390  l.unlock();
391 
392  LAGAS_(info) << (boost::format(
393  "primary_namespace::bind_gid, gid(%1%), gva(%2%), "
394  "locality(%3%), response(repeated_request)")
395  % id % g % locality);
396 
397  return false;
398  }
399 
400  // We're about to decrement the iterator it - first, we
401  // check that it's safe to do this.
402  else if (it != begin)
403  {
404  --it;
405 
406  // Check that a previous range doesn't cover the new id.
407  if (HPX_UNLIKELY((it->first + it->second.first.count) > id))
408  {
409  // REVIEW: Is this the right error code to use?
410  l.unlock();
411 
412  HPX_THROW_EXCEPTION(bad_parameter
413  , "primary_namespace::bind_gid"
414  , "the new GID is contained in an existing range");
415  }
416  }
417  }
418 
419  else if (HPX_LIKELY(!gvas_.empty()))
420  {
421  --it;
422 
423  // Check that a previous range doesn't cover the new id.
424  if ((it->first + it->second.first.count) > id)
425  {
426  // REVIEW: Is this the right error code to use?
427  l.unlock();
428 
429  HPX_THROW_EXCEPTION(bad_parameter
430  , "primary_namespace::bind_gid"
431  , "the new GID is contained in an existing range");
432  }
433  }
434 
435  naming::gid_type upper_bound(id + (g.count - 1));
436 
437  if (HPX_UNLIKELY(id.get_msb() != upper_bound.get_msb()))
438  {
439  l.unlock();
440 
441  HPX_THROW_EXCEPTION(internal_server_error
442  , "primary_namespace::bind_gid"
443  , "MSBs of lower and upper range bound do not match");
444  }
445 
446  if (HPX_UNLIKELY(components::component_invalid == g.type))
447  {
448  l.unlock();
449 
450  HPX_THROW_EXCEPTION(bad_parameter
451  , "primary_namespace::bind_gid"
452  , boost::str(boost::format(
453  "attempt to insert a GVA with an invalid type, "
454  "gid(%1%), gva(%2%), locality(%3%)")
455  % id % g % locality));
456  }
457 
458  // Insert a GID -> GVA entry into the GVA table.
459  if (HPX_UNLIKELY(!util::insert_checked(gvas_.insert(
460  std::make_pair(id, std::make_pair(g, locality))))))
461  {
462  l.unlock();
463 
464  HPX_THROW_EXCEPTION(lock_error
465  , "primary_namespace::bind_gid"
466  , boost::str(boost::format(
467  "GVA table insertion failed due to a locking error or "
468  "memory corruption, gid(%1%), gva(%2%)")
469  % id % g % locality));
470  }
471 
472  l.unlock();
473 
474  LAGAS_(info) << (boost::format(
475  "primary_namespace::bind_gid, gid(%1%), gva(%2%), locality(%3%)")
476  % id % g % locality);
477 
478  return true;
479 } // }}}

656 std::pair<naming::gid_type, naming::gid_type> primary_namespace::allocate(
657  std::uint64_t count
658  )
659 { // {{{ allocate implementation
660  util::scoped_timer<std::atomic<std::int64_t> > update(
661  counter_data_.allocate_.time_
662  );
663  counter_data_.increment_allocate_count();
664 
665  std::uint64_t const real_count = (count) ? (count - 1) : (0);
666 
667  // Just return the prefix
668  // REVIEW: Should this be an error?
669  if (0 == count)
670  {
671  LAGAS_(info) << (boost::format(
672  "primary_namespace::allocate, count(%1%), "
673  "lower(%1%), upper(%3%), prefix(%4%), response(repeated_request)")
674  % count % next_id_ % next_id_
675  % naming::get_locality_id_from_gid(next_id_));
676 
677  return std::make_pair(next_id_, next_id_);
678  }
679 
680  // Compute the new allocation.
681  naming::gid_type lower(next_id_ + 1);
682  naming::gid_type upper(lower + real_count);
683 
684  // Check for overflow.
685  if (upper.get_msb() != lower.get_msb())
686  {
687  // Check for address space exhaustion (we currently use 86 bits of

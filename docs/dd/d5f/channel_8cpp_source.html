109 dispatched_work(hpx::lcos::channel<int> jobs, hpx::lcos::channel<> done)
110 {
111  int received_jobs = 0;
112  bool was_closed = false;
113 
114  while(true)
115  {
116  hpx::error_code ec(hpx::lightweight);
117  int job = jobs.get(hpx::launch::sync, ec);
118  (void)job;
119 
120  if (!ec)
121  {
122  ++received_jobs;
123  }
124  else
125  {
126  was_closed = true;
127  done.set();
128  break;
129  }
130  }
131 
132  return std::make_pair(received_jobs, was_closed);
133 }

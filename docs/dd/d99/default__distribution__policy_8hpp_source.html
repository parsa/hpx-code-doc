141  template <typename Component, typename ...Ts>
142  hpx::future<std::vector<bulk_locality_result> >
143  bulk_create(std::size_t count, Ts&&... vs) const
144  {
145  using components::stub_base;
146 
147  if (localities_.size() > 1)
148  {
149  // schedule creation of all objects across given localities
150  std::vector<hpx::future<std::vector<hpx::id_type> > > objs;
151  objs.reserve(localities_.size());
152  for (hpx::id_type const& loc: localities_)
153  {
154  objs.push_back(stub_base<Component>::bulk_create_async(
155  loc, get_num_items(count, loc), vs...));
156  }
157 
158  // consolidate all results
159  return hpx::dataflow(hpx::launch::sync,
160  [=](std::vector<hpx::future<std::vector<hpx::id_type> > > && v)
161  mutable -> std::vector<bulk_locality_result>
162  {
163  HPX_ASSERT(localities_.size() == v.size());
164 
165  std::vector<bulk_locality_result> result;
166  result.reserve(v.size());
167 
168  for (std::size_t i = 0; i != v.size(); ++i)
169  {
170  result.emplace_back(
171  std::move(localities_[i]), v[i].get()
172  );
173  }
174  return result;
175  },
176  std::move(objs));
177  }
178 
179  // handle special cases
180  hpx::id_type id = get_next_target();
181 
182  hpx::future<std::vector<hpx::id_type> > f =
183  stub_base<Component>::bulk_create_async(
184  id, count, std::forward<Ts>(vs)...);
185 
186  return f.then(hpx::launch::sync,
187  [id](hpx::future<std::vector<hpx::id_type> > && f)
188  -> std::vector<bulk_locality_result>
189  {
190  std::vector<bulk_locality_result> result;
191  result.emplace_back(id, f.get());
192  return result;
193  });
194  }

284  protected:
286  std::size_t
287  get_num_items(std::size_t items, hpx::id_type const& loc) const
288  {
289  // make sure the given id is known to this distribution policy
290  HPX_ASSERT(
291  std::find(localities_.begin(), localities_.end(), loc) !=
292  localities_.end() ||
293  (localities_.empty() && loc == hpx::find_here())
294  );
295 
296  // this distribution policy places an equal number of items onto
297  // each locality
298  std::size_t locs = (std::max)(std::size_t(1), localities_.size());
299 
300  // the overall number of items to create is smaller than the number
301  // of localities
302  if (items < locs)
303  {
304  auto it = std::find(localities_.begin(), localities_.end(), loc);
305  std::size_t num_loc = std::distance(localities_.begin(), it);
306  return (items < num_loc) ? 1 : 0;
307  }
308 
309  // the last locality might get less items
310  if (localities_.size() > 1 && loc == localities_.back())
311  return items - detail::round_to_multiple(items, locs, locs-1);
312 
313  // otherwise just distribute evenly
314  return (items + locs - 1) / locs;
315  }

56 // -------------------------------------------------------------------------
57 // a simple cublas wrapper helper object that can be used to synchronize
58 // cublas calls with an hpx future.
59 // -------------------------------------------------------------------------
60 template<typename T>
61 struct cublas_helper
62 {
63 public:
64  using future_type = hpx::future<void>;
65 
66 #ifdef HPX_CUBLAS_DEMO_WITH_ALLOCATOR
67  using allocator_type = typename hpx::compute::cuda::allocator<T>;
68  using vector_type = typename hpx::compute::vector<T, allocator_type>;
69 #endif
70 
71  // construct a cublas stream
72  cublas_helper(std::size_t device=0) : target_(device) {
73  handle_ = 0;
74  stream_ = target_.native_handle().get_stream();
75  cublas_error(cublasCreate(&handle_));
76  }
77 
78  ~cublas_helper() {
79  cublas_error(cublasDestroy(handle_));
80  }
81 
82  // This is a simple wrapper for any cublas call, pass in the same arguments
83  // that you would use for a cublas call except the cublas handle which is omitted
84  // as the wrapper will supply that for you
85  template <typename Func, typename ...Args>
86  void operator()(Func && cublas_function, Args&&... args)
87  {
88  // make sue we run on the correct device
89  cuda_error(cudaSetDevice(target_.native_handle().get_device()));
90 
91  // make sure this operation takes place on our stream
92  cublas_error(cublasSetStream(handle_, stream_));
93 
94  // insert the cublas handle in the arg list and call the cublas function
95  cublas_error(cublas_function(handle_, std::forward<Args>(args)...));
96  }
97 
98  template <typename ...Args>
99  void copy_async(Args&&... args)
100  {
101  // make sue we run on the correct device
102  cuda_error(cudaSetDevice(target_.native_handle().get_device()));
103 
104  // insert the uda stream in the arg list and call the cuda memcpy
105  cuda_error(cudaMemcpyAsync (std::forward<Args>(args)..., stream_));
106  }
107 
108  // get the future to synchronize this cublas stream with
109  future_type get_future() { return target_.get_future(); }
110 
111  // return a copy of the cublas handle
112  cublasHandle_t handle() { return handle_; }
113 
114  // return a reference to the compute::cuda object owned by this class
115  hpx::compute::cuda::target & target() { return target_; }
116 
117  static void cublas_error(cublasStatus_t err) {
118  if (err != CUBLAS_STATUS_SUCCESS) {
119  std::stringstream temp;
120  temp << "cublas function returned error code " << _cudaGetErrorEnum(err);
121  throw std::runtime_error(temp.str());
122  }
123  }
124 
125  static void cuda_error(cudaError_t err) {
126  if (err != cudaSuccess) {
127  std::stringstream temp;
128  temp << "cuda function returned error code " << cudaGetErrorString(err);
129  throw std::runtime_error(temp.str());
130  }
131  }
132 
133 private:
134  cublasHandle_t handle_;
135  cudaStream_t stream_;
136  hpx::compute::cuda::target target_;
137 };

145 // -------------------------------------------------------------------------
146 // Compute reference data set matrix multiply on CPU
147 // C = A * B
148 // @param C reference data, computed but preallocated
149 // @param A matrix A as provided to device
150 // @param B matrix B as provided to device
151 // @param hA height of matrix A
152 // @param wB width of matrix B
153 // -------------------------------------------------------------------------
154 template <typename T>
155 void matrixMulCPU(T *C, const T *A, const T *B,
156  unsigned int hA, unsigned int wA, unsigned int wB)
157 {
158  hpx::parallel::for_loop(
159  hpx::parallel::execution::par, 0, hA,
160  [&](int i) {
161  for (unsigned int j = 0; j < wB; ++j) {
162  T sum = 0;
163  for (unsigned int k = 0; k < wA; ++k) {
164  T a = A[i * wA + k];
165  T b = B[k * wB + j];
166  sum += a * b;
167  }
168  C[i * wB + j] = (T)sum;
169  }
170  });
171 }

173 // -------------------------------------------------------------------------
174 // Compute the L2 norm difference between two arrays
175 inline bool
176 compare_L2_err(const float *reference, const float *data,
177  const unsigned int len, const float epsilon)
178 {
179  HPX_ASSERT(epsilon >= 0);
180 
181  float error = 0;
182  float ref = 0;
183 
184  hpx::parallel::for_loop(
185  hpx::parallel::execution::par, 0, len,
186  [&](int i) {
187  float diff = reference[i] - data[i];
188  error += diff * diff;
189  ref += reference[i] * reference[i];
190  }
191  );
192 
193  float normRef = sqrtf(ref);
194  if (std::fabs(ref) < 1e-7f) {
195  return false;
196  }
197 
198  float normError = sqrtf(error);
199  error = normError / normRef;
200  bool result = error < epsilon;
201  return result;
202 }

235 // Run a simple test matrix multiply using CUBLAS
236 // -------------------------------------------------------------------------
237 template <typename T>
238 void matrixMultiply(sMatrixSize &matrix_size, std::size_t device, std::size_t iterations)
239 {
240  using hpx::parallel::execution::par;
241 
242  // Allocate host memory for matrices A and B
243  unsigned int size_A = matrix_size.uiWA * matrix_size.uiHA;
244  unsigned int size_B = matrix_size.uiWB * matrix_size.uiHB;
245  unsigned int size_C = matrix_size.uiWC * matrix_size.uiHC;
246 
247  std::vector<T> h_A(size_A);
248  std::vector<T> h_B(size_B);
249  std::vector<T> h_C(size_C);
250  std::vector<T> h_CUBLAS(size_C);
251 
252  // Fill A and B with random numbers
253  auto randfunc = [](T &x) { x = rand() / (T)RAND_MAX; };
254  hpx::parallel::for_each(par, h_A.begin(), h_A.end(), randfunc);
255  hpx::parallel::for_each(par, h_B.begin(), h_B.end(), randfunc);
256 
257  // create a cublas helper object we'll use to futurize the cuda events
258  cublas_helper<T> cublas(device);
259 
260  using cublas_future = typename cublas_helper<T>::future_type;
261 
262 #ifdef HPX_CUBLAS_DEMO_WITH_ALLOCATOR
263  // for convenience
264  using device_allocator = typename cublas_helper<T>::allocator_type;
265  using device_vector = typename cublas_helper<T>::vector_type;
266  // The policy used in the parallel algorithms
267  auto policy = hpx::parallel::execution::par;
268 
269  // Create a cuda allocator
270  device_allocator alloc(cublas.target());
271 
272  // Allocate device memory
273  device_vector d_vA(size_A, alloc);
274  device_vector d_vB(size_B, alloc);
275  device_vector d_vC(size_C, alloc);
276 
277  // copy host memory to device
278  hpx::parallel::copy(policy, h_A.begin(), h_A.end(), d_vA.begin());
279  hpx::parallel::copy(policy, h_B.begin(), h_B.end(), d_vB.begin());
280 
281  // just to make the rest of code the same for both cases
282  T *d_A=d_vA.device_data();
283  T *d_B=d_vB.device_data();
284  T *d_C=d_vC.device_data();
285 
286 #else
287  T *d_A, *d_B, *d_C;
288  cublas_helper<T>::cuda_error(
289  cudaMalloc((void **) &d_A, size_A*sizeof(T)));
290 
291  cublas_helper<T>::cuda_error(
292  cudaMalloc((void **) &d_B, size_B*sizeof(T)));
293 
294  cublas_helper<T>::cuda_error(
295  cudaMalloc((void **) &d_C, size_C*sizeof(T)));
296 
297  // adding async copy operations into the stream before cublas calls puts
298  // the copies in the queue before the matrix operations.
299  cublas.copy_async(
300  d_A, h_A.data(), size_A*sizeof(T), cudaMemcpyHostToDevice);
301 
302  cublas.copy_async(
303  d_B, h_B.data(), size_B*sizeof(T), cudaMemcpyHostToDevice);
304 
305  // we can call get_future multiple times on the cublas helper.
306  // Each one returns a new future that will be set ready when the stream event
307  // for this point is triggered
308  auto copy_future = cublas.get_future().then([](cublas_future &&f){
309  std::cout << "The async host->device copy operation completed" << std::endl;
310  });
311 
312 #endif
313 
314  std::cout << "Computing result using CUBLAS...\n";
315  const T alpha = 1.0f;
316  const T beta = 0.0f;
317 
318  // Perform warmup operation with cublas
319  // note cublas is column major ordering : transpose the order
320  hpx::util::high_resolution_timer t1;
321  //
322  std::cout << "calling CUBLAS...\n";
323  cublas(
324  &cublasSgemm,
325  CUBLAS_OP_N, CUBLAS_OP_N,
326  matrix_size.uiWB, matrix_size.uiHA, matrix_size.uiWA,
327  &alpha,
328  d_B, matrix_size.uiWB,
329  d_A, matrix_size.uiWA,
330  &beta,
331  d_C, matrix_size.uiWA);
332 
333  // wait until the operation completes
334  cublas.get_future().get();
335 
336  double us1 = t1.elapsed_microseconds();
337  std::cout << "warmup: elapsed_microseconds " << us1 << std::endl;
338 
339  // once the future has been retrieved, the next call to
340  // get_future will create a new event and a new future so
341  // we can reuse the same cublas wrapper object and stream if we want
342 
343  hpx::util::high_resolution_timer t2;
344  for (std::size_t j=0; j<iterations; j++) {
345  cublas(
346  &cublasSgemm,
347  CUBLAS_OP_N, CUBLAS_OP_N,
348  matrix_size.uiWB, matrix_size.uiHA, matrix_size.uiWA,
349  &alpha,
350  d_B, matrix_size.uiWB,
351  d_A, matrix_size.uiWA,
352  &beta,
353  d_C, matrix_size.uiWA);
354  }
355  // get a future for when the stream reaches this point (matrix operations complete)
356  auto matrix_finished = cublas.get_future();
357 
358 #ifndef HPX_CUBLAS_DEMO_WITH_ALLOCATOR
359  // when the matrix operations complete, copy the result to the host
360  cublas.copy_async(
361  h_CUBLAS.data(), d_C, size_C*sizeof(T), cudaMemcpyDeviceToHost);
362 
363  // and get another future when the copy back is done
364  auto copy_finished = cublas.get_future();
365 
366 #endif
367 
368  // attach a continuation to the cublas future
369  auto new_future = matrix_finished.then([&](cublas_future &&f)
370  {
371  double us2 = t2.elapsed_microseconds();
372  std::cout << "actual: elapsed_microseconds " << us2
373  << " iterations " << iterations << std::endl;
374 
375  // Compute and print the performance
376  double usecPerMatrixMul = us2 / iterations;
377  double flopsPerMatrixMul = 2.0 * (double)matrix_size.uiWA *
378  (double)matrix_size.uiHA * (double)matrix_size.uiWB;
379  double gigaFlops = (flopsPerMatrixMul * 1.0e-9) / (usecPerMatrixMul / 1e6);
380  printf(
381  "Performance = %.2f GFlop/s, Time = %.3f msec/iter, Size = %.0f Ops\n",
382  gigaFlops,
383  1e-3*usecPerMatrixMul,
384  flopsPerMatrixMul);
385  }
386  );
387 
388  // wait for the timing to complete, and then do a CPU comparison
389  auto finished = new_future.then([&](cublas_future &&f) {
390  // compute reference solution on the CPU
391  std::cout << "\nComputing result using host CPU...\n";
392 #ifdef HPX_CUBLAS_DEMO_WITH_ALLOCATOR
393  // copy result from device to host
394  hpx::parallel::copy(policy, d_C.begin(), d_C.end(), h_CUBLAS.begin());
395 #else
396  // just wait for the device->host copy to complete if it hasn't already
397  copy_finished.get();
398 #endif
399 
400  // compute reference solution on the CPU
401  // allocate storage for the CPU result
402  std::vector<T> reference(size_C);
403 
404  hpx::util::high_resolution_timer t3;
405  matrixMulCPU<T>(reference.data(), h_A.data(), h_B.data(),
406  matrix_size.uiHA, matrix_size.uiWA, matrix_size.uiWB);
407  double us3 = t3.elapsed_microseconds();
408  std::cout << "CPU elapsed_microseconds (1 iteration) " << us3 << std::endl;
409 
410  // check result (CUBLAS)
411  bool resCUBLAS = compare_L2_err(reference.data(), h_CUBLAS.data(), size_C, 1e-6);
412  if (resCUBLAS != true) {
413  throw std::runtime_error("matrix CPU/GPU comparison error");
414  }
415  // if the result was incorrect, we throw an exception, so here it's ok
416  std::cout << "\nComparing CUBLAS Matrix Multiply with CPU results: OK \n";
417  });
418 
419  finished.get();
420 #ifndef HPX_CUBLAS_DEMO_WITH_ALLOCATOR
421  cudaFree(d_A);
422  cudaFree(d_B);
423  cudaFree(d_C);
424 #endif
425 }

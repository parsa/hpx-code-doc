105  virtual bool get_next_thread(std::size_t num_thread, bool running,
106  std::int64_t& idle_loop_count, threads::thread_data*& thrd)
107  {
108  // Drain the queue of the disabled OS thread
109  while (disabled(num_thread)) {
110  thread_queue_type* q = queues_[num_thread];
111  while (q->get_next_thread(thrd)) {
112  this->wait_or_add_new(num_thread, running, idle_loop_count) ;
113  this->schedule_thread(thrd, num_thread, thread_priority_normal);
114  }
115 
116  std::unique_lock<compat::mutex> l(mtx_);
117  cond_.wait(l);
118  }
119 
120  // grab work if available
121  return this->base_type::get_next_thread(num_thread,
122  running, idle_loop_count, thrd);
123  }

127  void schedule_thread(threads::thread_data* thrd, std::size_t num_thread,
128  thread_priority priority = thread_priority_normal)
129  {
130  // Loop and find a thread that is not disabled
131  if (std::size_t(-1) == num_thread)
132  num_thread = curr_queue_++ % queues_.size();
133  else
134  {
135  for(std::size_t tid=0; tid<queues_.size(); tid++) {
136  // Loop and find a thread that is not disabled
137  if (!disabled(tid)) {
138  num_thread = tid;
139  break;
140  }
141  }
142  }
143 
144  HPX_ASSERT(num_thread < queues_.size());
145 
146  queues_[num_thread]->schedule_thread(thrd);
147  }

168  void disable(std::size_t shepherd)

218  std::size_t wtid = hpx::get_worker_thread_num();
219 
220  std::size_t cnt = 0;
221  std::size_t tid_start = 0;
222  while ( cnt < num_threads ) {
223  for (std::size_t i=tid_start; i<disabled_os_threads_.size();
224  i += num_hw_threads)
225  {
226  if (!disabled_os_threads_[i] && i != wtid) {
227  disabled_os_threads_[i] = true;
228  cnt++;
229  if (cnt == num_threads) return;

232  tid_start++;
233  }

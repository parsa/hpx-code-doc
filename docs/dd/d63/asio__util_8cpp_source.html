47  bool get_endpoint(std::string const& addr, std::uint16_t port,
48  boost::asio::ip::tcp::endpoint& ep)
49  {
50  using namespace boost::asio::ip;
51  boost::system::error_code ec;
52  address_v4 addr4 = address_v4::from_string(addr.c_str(), ec);
53  if (!ec) { // it's an IPV4 address
54  ep = tcp::endpoint(address(addr4), port);
55  return true;
56  }
57 
58  address_v6 addr6 = address_v6::from_string(addr.c_str(), ec);
59  if (!ec) { // it's an IPV6 address
60  ep = tcp::endpoint(address(addr6), port);
61  return true;
62  }
63  return false;
64  }

107  }

150  // Addresses are supposed to have the format <hostname>[:port]
151  bool split_ip_address(std::string const& v, std::string& host,
152  std::uint16_t& port)
153  {
154  std::string::size_type p = v.find_first_of(":");
155 
156  std::string tmp_host;
157  std::uint16_t tmp_port = 0;
158 
159  try {
160  if (p != std::string::npos) {
161  tmp_host = v.substr(0, p);
162  tmp_port = boost::lexical_cast<std::uint16_t>(v.substr(p+1));
163  }
164  else {
165  tmp_host = v;
166  }
167 
168  if (!tmp_host.empty()) {
169  host = tmp_host;
170  if (tmp_port)
171  port = tmp_port;
172  }
173  }
174  catch (boost::bad_lexical_cast const& /*e*/) {
175  // port number is invalid
176  return false;
177  }
178  return true;
179  }

183  // Take an ip v4 or v6 address and "standardize" it for comparison checks
184  // note that this code doesn't work as expected if we use the boost
185  // inet_pton functions on linux. see issue #2177 for further info
186  std::string cleanup_ip_address(const std::string &addr)
187  {
188  char buf[sizeof(struct in6_addr)];
189  int i=0, domain[2] = {AF_INET, AF_INET6};
190  char str[INET6_ADDRSTRLEN];
191 
192 #if defined(HPX_WINDOWS)
193  unsigned long scope_id;
194  boost::system::error_code ec;
195 #endif
196 
197  for (i=0; i<2; ++i) {
198 #if defined(HPX_WINDOWS)
199  int s = boost::asio::detail::socket_ops::inet_pton(
200  domain[i], &addr[0], buf, &scope_id, ec);
201  if (s>0 && !ec) break;
202 #else
203  int s = inet_pton(domain[i], &addr[0], buf);
204  if (s>0) break;
205 #endif
206  }
207  if (i==2) {
208  HPX_THROW_EXCEPTION(bad_parameter, "cleanup_ip_address",
209  "Invalid IP address string");
210  }
211 
212 #if defined(HPX_WINDOWS)
213  if (boost::asio::detail::socket_ops::inet_ntop(
214  domain[i], buf, str, INET6_ADDRSTRLEN, scope_id, ec) == 0) {
215 #else
216  if (inet_ntop(domain[i], buf, str, INET6_ADDRSTRLEN) == nullptr) {
217 #endif
218  HPX_THROW_EXCEPTION(bad_parameter, "cleanup_ip_address",
219  "inet_ntop failure");
220  }
221  return std::string(str);
222  }

44  // The static partitioner simply spawns one chunk of iterations for
45  // each available core.
46  template <typename ExPolicy_, typename R, typename Result1,
47  typename Result2>
48  struct static_scan_partitioner
49  {
50  template <typename ExPolicy, typename FwdIter, typename T,
51  typename F1, typename F2, typename F3, typename F4>
52  static R call(ExPolicy && policy, FwdIter first,
53  std::size_t count, T && init, F1 && f1, F2 && f2, F3 && f3,
54  F4 && f4)
55  {
56  typedef typename
57  hpx::util::decay<ExPolicy>::type::executor_parameters_type
58  parameters_type;
59  typedef executor_parameter_traits<parameters_type>
60  parameters_traits;
61 
62  // inform parameter traits
63  scoped_executor_parameters<parameters_type> scoped_param(
64  policy.parameters());
65 
66  std::vector<hpx::shared_future<Result1> > workitems;
67  std::vector<hpx::future<Result2> > finalitems;
68  std::list<std::exception_ptr> errors;
69 
70  try {
71  // pre-initialize first intermediate result
72  workitems.push_back(make_ready_future(std::forward<T>(init)));
73 
74  HPX_ASSERT(count > 0);
75  FwdIter first_ = first;
76  std::size_t count_ = count;
77 
78  // estimate a chunk size based on number of cores used
79  typedef typename parameters_traits::has_variable_chunk_size
80  has_variable_chunk_size;
81 
82  auto shape = get_bulk_iteration_shape(policy, workitems,
83  f1, first, count, 1, has_variable_chunk_size());
84 
85  // schedule every chunk on a separate thread
86  std::size_t size = hpx::util::size(shape);
87  workitems.reserve(size + 1);
88  finalitems.reserve(size);
89 
90  // If the size of count was enough to warrant testing for a
91  // chunk, pre-initialize second intermediate result and
92  // start f3.
93  if (workitems.size() == 2)
94  {
95  HPX_ASSERT(count_ > count);
96 
97  hpx::shared_future<Result1> curr = workitems[1];
98  finalitems.push_back(dataflow(hpx::launch::sync,
99  f3, first_, count_ - count, workitems[0], curr));
100 
101  workitems[1] = dataflow(hpx::launch::sync,
102  f2, workitems[0], curr);
103  }
104 
105  // Schedule first step of scan algorithm, step 2 is
106  // performed as soon as the current partition and the
107  // partition to the left is ready.
108  for(auto const& elem: shape)
109  {
110  FwdIter it = hpx::util::get<0>(elem);
111  std::size_t size = hpx::util::get<1>(elem);
112 
113  hpx::shared_future<Result1> prev = workitems.back();
114  auto curr = execution::async_execute(
115  policy.executor(), f1, it, size).share();
116 
117  finalitems.push_back(dataflow(hpx::launch::sync,
118  f3, it, size, prev, curr));
119 
120  workitems.push_back(dataflow(hpx::launch::sync,
121  f2, prev, curr));
122  }
123  }
124  catch (...) {
125  handle_local_exceptions<ExPolicy>::call(
126  std::current_exception(), errors);
127  }
128 
129  // wait for all tasks to finish
130  hpx::wait_all(finalitems, workitems);
131 
132  // always rethrow if 'errors' is not empty or 'workitems' or
133  // 'finalitems' have an exceptional future
134  handle_local_exceptions<ExPolicy>::call(workitems, errors);
135  handle_local_exceptions<ExPolicy>::call(finalitems, errors);
136 
137  try {
138  return f4(std::move(workitems), std::move(finalitems));
139  }
140  catch (...) {
141  // rethrow either bad_alloc or exception_list
142  handle_local_exceptions<ExPolicy>::call(
143  std::current_exception());
144  }
145  }
146  };

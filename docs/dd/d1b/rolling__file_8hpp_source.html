95  void restart() {
96  namespace fs = boost::filesystem;
97 
98  if ( m_flags.initial_erase()) {
99  for ( unsigned idx = 0; idx < m_flags.file_count(); ++idx) {
100  boost::system::error_code ec;
101  if ( fs::exists( file_name(idx), ec) && !ec)
102  fs::remove( file_name(idx) );
103  }
104  }
105 
106  // see what file to start from
107  if ( m_flags.start_where_size_not_exceeded() ) {
108  for ( m_cur_idx = 0; m_cur_idx < m_flags.file_count(); ++m_cur_idx ) {
109  boost::system::error_code ec;
110  if ( fs::exists( file_name(m_cur_idx), ec) && !ec) {
111  if ( fs::file_size( file_name(m_cur_idx))
112  < m_flags.max_size_bytes() )
113  // file hasn't reached max size
114  break;
115  }
116  else
117  // file not found, we'll create it now
118  break;
119  }
120 
121  if ( m_cur_idx >= m_flags.file_count())
122  // all files are too full (we'll overwrite the first one)
123  m_cur_idx = 0;
124  }
125 
126  recreate_file();
127  }
128 
129  std::string file_name(unsigned idx) {
130  std::ostringstream out;
131  if ( idx > 0)
132  out << m_name_prefix << "." << (idx+1);
133  else
134  out << m_name_prefix;
135  return out.str();
136  }

138  void recreate_file() {
139  // many thanks to Benjamin de Dardel!
140  namespace fs = boost::filesystem;
141  m_out = std::shared_ptr< std::basic_ofstream<char_type> >
142  (new std::basic_ofstream<char_type>( file_name(m_cur_idx).c_str(),
143  m_flags.extra_flags() | std::ios_base::out | std::ios_base::app));
144  if ( fs::file_size( file_name(m_cur_idx)) > m_flags.max_size_bytes()) {
145  // this file is already full - clear it first
146  m_out = std::shared_ptr< std::basic_ofstream<char_type>
147  >(new std::basic_ofstream<char_type>( file_name(m_cur_idx).c_str(),
148  m_flags.extra_flags() | std::ios_base::out | std::ios_base::trunc));
149  }
150  }
151 
152  template<class msg_type> void write( const msg_type& msg) {
153  convert_dest::write(msg, (*m_out) );
154  if ( m_flags.flush_each_time())
155  m_out->flush();
156 
157  if ( m_out->tellp() > m_flags.max_size_bytes()) {
158  m_cur_idx = (m_cur_idx + 1) % m_flags.file_count();
159  recreate_file();
160  }
161  }
162 
163  void flush() {
164  m_out->flush();
165  }

88  template <typename T>
89  class unlimited_channel : public channel_impl_base<T>
90  {
91  typedef hpx::lcos::local::spinlock mutex_type;
92 
93  public:
94  HPX_NON_COPYABLE(unlimited_channel);
95 
96  public:
97  unlimited_channel()
98  : get_generation_(0), set_generation_(0), closed_(false)
99  {}
100 
101  protected:
102  hpx::future<T> get(std::size_t generation, bool blocking)
103  {
104  std::unique_lock<mutex_type> l(mtx_);
105 
106  if (buffer_.empty())
107  {
108  if (closed_)
109  {
110  l.unlock();
111  return hpx::make_exceptional_future<T>(
112  HPX_GET_EXCEPTION(hpx::invalid_status,
113  "hpx::lcos::local::channel::get",
114  "this channel is empty and was closed"));
115  }
116 
117  if (blocking && this->use_count() == 1)
118  {
119  l.unlock();
120  return hpx::make_exceptional_future<T>(
121  HPX_GET_EXCEPTION(hpx::invalid_status,
122  "hpx::lcos::local::channel::get",
123  "this channel is empty and is not accessible "
124  "by any other thread causing a deadlock"));
125  }
126  }
127 
128  ++get_generation_;
129  if (generation == std::size_t(-1))
130  generation = get_generation_;
131 
132  if (closed_)
133  {
134  // the requested item must be available, otherwise this
135  // would create a deadlock
136  hpx::future<T> f;
137  if (!buffer_.try_receive(generation, &f))
138  {
139  l.unlock();
140  return hpx::make_exceptional_future<T>(
141  HPX_GET_EXCEPTION(hpx::invalid_status,
142  "hpx::lcos::local::channel::get",
143  "this channel is closed and the requested value"
144  "has not been received yet"));
145  }
146  return f;
147  }
148 
149  return buffer_.receive(generation);
150  }
151 
152  bool try_get(std::size_t generation, hpx::future<T>* f = nullptr)
153  {
154  std::lock_guard<mutex_type> l(mtx_);
155 
156  if (buffer_.empty() && closed_)
157  return false;
158 
159  ++get_generation_;
160  if (generation == std::size_t(-1))
161  generation = get_generation_;
162 
163  if (f != nullptr)
164  *f = buffer_.receive(generation);
165 
166  return true;
167  }
168 
169  hpx::future<void> set(std::size_t generation, T && t)
170  {
171  std::unique_lock<mutex_type> l(mtx_);
172  if(closed_)
173  {
174  l.unlock();
175  return hpx::make_exceptional_future<void>(
176  HPX_GET_EXCEPTION(hpx::invalid_status,
177  "hpx::lcos::local::channel::set",
178  "attempting to write to a closed channel"));
179  }
180 
181  ++set_generation_;
182  if (generation == std::size_t(-1))
183  generation = set_generation_;
184 
185  buffer_.store_received(generation, std::move(t), &l);
186  return hpx::make_ready_future();
187  }
188 
189  void close()
190  {
191  std::unique_lock<mutex_type> l(mtx_);
192  if(closed_)
193  {
194  l.unlock();
195  HPX_THROW_EXCEPTION(hpx::invalid_status,
196  "hpx::lcos::local::channel::close",
197  "attempting to close an already closed channel");
198  return;
199  }
200 
201  closed_ = true;
202 
203  if (buffer_.empty())
204  return;
205 
206  std::exception_ptr e;
207 
208  {
209  util::scoped_unlock<std::unique_lock<mutex_type> > ul(l);
210  e = HPX_GET_EXCEPTION(hpx::future_cancelled,
211  "hpx::lcos::local::close",
212  "canceled waiting on this entry");
213  }
214 
215  // all pending requests which can't be satisfied have to be
216  // canceled at this point
217  buffer_.cancel_waiting(e);
218  }
219 
220  private:
221  mutable mutex_type mtx_;
222  receive_buffer<T, no_mutex> buffer_;
223  std::size_t get_generation_;
224  std::size_t set_generation_;
225  bool closed_;
226  };

235  private:
236  template <typename T1>
237  void set(T1 && val)
238  {
239  val_ = std::forward<T1>(val);
240  empty_ = false;
241  push_active_ = false;
242  }
243  void set_deferred(T && val)
244  {
245  val_ = std::move(val);
246  empty_ = false;
247  push_active_ = false;
248  }

250  T get()
251  {
252  empty_ = true;
253  pop_active_ = false;
254  return std::move(val_);
255  }

257  template <typename T1>
258  local::packaged_task<void()> push_pt(T1 && val)
259  {
260  return local::packaged_task<void()>(
261  util::deferred_call(
262  &one_element_queue_async::set_deferred, this,
263  std::forward<T1>(val)));
264  }
265  local::packaged_task<T()> pop_pt()
266  {
267  return local::packaged_task<T()>(
268  util::deferred_call(&one_element_queue_async::get, this));
269  }
270 
271  public:
272  one_element_queue_async()
273  : empty_(true), push_active_(false), pop_active_(false)
274  {}
275 
276  template <typename T1, typename Lock>
277  hpx::future<void> push(T1 && val, Lock& l)
278  {
279  HPX_ASSERT_OWNS_LOCK(l);
280  if (!empty_)
281  {
282  if (push_active_)
283  {
284  l.unlock();
285  return hpx::make_exceptional_future<void>(
286  HPX_GET_EXCEPTION(hpx::invalid_status,
287  "hpx::lcos::local::detail::"
288  "one_element_queue_async::push",
289  "attempting to write to a busy queue"));
290  }
291 
292  push_ = push_pt(std::forward<T1>(val));
293  push_active_ = true;
294  return push_.get_future();
295  }
296 
297  set(std::forward<T1>(val));
298  if (pop_active_)
299  {
300  pop_(); // trigger waiting pop
301  }
302  return hpx::make_ready_future();
303  }
304 
305  template <typename Lock>
306  void cancel(std::exception_ptr const& e, Lock& l)
307  {
308  HPX_ASSERT_OWNS_LOCK(l);
309  if (pop_active_)
310  {
311  pop_.set_exception(e);
312  pop_active_ = false;
313  }
314  }
315 
316  template <typename Lock>
317  hpx::future<T> pop(Lock& l)
318  {
319  HPX_ASSERT_OWNS_LOCK(l);
320  if (empty_)
321  {
322  if (pop_active_)
323  {
324  l.unlock();
325  return hpx::make_exceptional_future<T>(
326  HPX_GET_EXCEPTION(hpx::invalid_status,
327  "hpx::lcos::local::detail::"
328  "one_element_queue_async::pop",
329  "attempting to read from an empty queue"));
330  }
331 
332  pop_ = pop_pt();
333  pop_active_ = true;
334  return pop_.get_future();
335  }
336 
337  T val = get();
338  if (push_active_)
339  {
340  push_(); // trigger waiting push
341  }
342  return hpx::make_ready_future(val);
343  }
344 
345  template <typename Lock>
346  bool is_empty(Lock& l) const
347  {
348  HPX_ASSERT_OWNS_LOCK(l);
349  return empty_;
350  }
351 
352  template <typename Lock>
353  bool has_pending_request(Lock& l) const
354  {
355  HPX_ASSERT_OWNS_LOCK(l);
356  return push_active_;
357  }
358 
359  private:
360  T val_;
361  local::packaged_task<void()> push_;
362  local::packaged_task<T()> pop_;
363  bool empty_;
364  bool push_active_;
365  bool pop_active_;
366  };

369  template <typename T>
370  class one_element_channel : public channel_impl_base<T>
371  {
372  typedef hpx::lcos::local::spinlock mutex_type;
373 
374  public:
375  HPX_NON_COPYABLE(one_element_channel);
376 
377  public:
378  one_element_channel()
379  : closed_(false)
380  {}
381 
382  protected:
383  hpx::future<T> get(std::size_t, bool blocking)
384  {
385  std::unique_lock<mutex_type> l(mtx_);
386 
387  if (buffer_.is_empty(l) && !buffer_.has_pending_request(l))
388  {
389  if (closed_)
390  {
391  l.unlock();
392  return hpx::make_exceptional_future<T>(
393  HPX_GET_EXCEPTION(hpx::invalid_status,
394  "hpx::lcos::local::channel::get",
395  "this channel is empty and was closed"));
396  }
397 
398  if (blocking && this->use_count() == 1)
399  {
400  l.unlock();
401  return hpx::make_exceptional_future<T>(
402  HPX_GET_EXCEPTION(hpx::invalid_status,
403  "hpx::lcos::local::channel::get",
404  "this channel is empty and is not accessible "
405  "by any other thread causing a deadlock"));
406  }
407  }
408 
409  hpx::future<T> f = buffer_.pop(l);
410  if (closed_ && !f.is_ready())
411  {
412  // the requested item must be available, otherwise this
413  // would create a deadlock
414  l.unlock();
415  return hpx::make_exceptional_future<T>(
416  HPX_GET_EXCEPTION(hpx::invalid_status,
417  "hpx::lcos::local::channel::get",
418  "this channel is closed and the requested value"
419  "has not been received yet"));
420  }
421 
422  return f;
423  }
424 
425  bool try_get(std::size_t, hpx::future<T>* f = nullptr)
426  {
427  std::unique_lock<mutex_type> l(mtx_);
428 
429  if (buffer_.is_empty(l) && !buffer_.has_pending_request(l) && closed_)
430  return false;
431 
432  if (f != nullptr)
433  {
434  *f = buffer_.pop(l);
435  }
436  return true;
437  }
438 
439  hpx::future<void> set(std::size_t, T && t)
440  {
441  std::unique_lock<mutex_type> l(mtx_);
442 
443  if (closed_)
444  {
445  l.unlock();
446  return hpx::make_exceptional_future<void>(
447  HPX_GET_EXCEPTION(hpx::invalid_status,
448  "hpx::lcos::local::channel::set",
449  "attempting to write to a closed channel"));
450  }
451 
452  return buffer_.push(std::move(t), l);
453  }
454 
455  void close()
456  {
457  std::unique_lock<mutex_type> l(mtx_);
458 
459  if (closed_)
460  {
461  l.unlock();
462  HPX_THROW_EXCEPTION(hpx::invalid_status,
463  "hpx::lcos::local::channel::close",
464  "attempting to close an already closed channel");
465  return;
466  }
467 
468  closed_ = true;
469 
470  if (buffer_.is_empty(l) || !buffer_.has_pending_request(l))
471  return;
472 
473  // all pending requests which can't be satisfied have to be
474  // canceled at this point
475  std::exception_ptr e;
476  {
477  util::scoped_unlock<std::unique_lock<mutex_type> > ul(l);
478  e = std::exception_ptr(
479  HPX_GET_EXCEPTION(hpx::future_cancelled,
480  "hpx::lcos::local::close",
481  "canceled waiting on this entry"));
482  }
483  buffer_.cancel(std::move(e), l);
484  }
485 
486  void set_exception(std::exception_ptr e)
487  {
488  std::unique_lock<mutex_type> l(mtx_);
489  closed_ = true;
490 
491  if (!buffer_.is_empty(l))
492  buffer_.cancel(e, l);
493  }
494 
495  private:
496  mutable mutex_type mtx_;
497  one_element_queue_async<T> buffer_;
498  bool closed_;
499  };

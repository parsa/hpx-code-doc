179  template <typename Executor_, typename Result, typename F,
180  typename Iter, typename ... Ts>
181  static hpx::future<void>
182  spawn(Executor_& sched, std::vector<hpx::future<Result> >& results,
183  std::size_t base, std::size_t size, std::size_t num_tasks,
184  F const& func, Iter it, Ts const&... ts)
185  {
186  const std::size_t num_spread = 4;
187 
188  if (size > num_tasks)
189  {
190  // spawn hierarchical tasks
191  std::size_t chunk_size = (size + num_spread) / num_spread - 1;
192  chunk_size = (std::max)(chunk_size, num_tasks);
193 
194  std::vector<hpx::future<void> > tasks;
195  tasks.reserve(num_spread);
196 
197  hpx::future<void> (*spawn_func)(
198  Executor_&, std::vector<hpx::future<Result> >&,
199  std::size_t, std::size_t, std::size_t, F const&, Iter,
200  Ts const&...
201  ) = &executor_traits::spawn;
202 
203  while (size != 0)
204  {
205  std::size_t curr_chunk_size = (std::min)(chunk_size, size);
206 
207  hpx::future<void> f = hpx::async(
208  spawn_func, std::ref(sched), std::ref(results), base,
209  curr_chunk_size, num_tasks, std::ref(func), it,
210  std::ref(ts)...);
211  tasks.push_back(std::move(f));
212 
213  base += curr_chunk_size;
214  it = hpx::parallel::v1::detail::next(it, curr_chunk_size);
215  size -= curr_chunk_size;
216  }
217 
218  HPX_ASSERT(size == 0);
219 
220  return hpx::when_all(tasks);
221  }
222 
223  // spawn all tasks sequentially
224  HPX_ASSERT(base + size <= results.size());
225 
226  for (std::size_t i = 0; i != size; ++i, ++it)
227  {
228  results[base + i] = hpx::async(sched, func, *it, ts...);
229  }
230 
231  return hpx::make_ready_future();
232  }

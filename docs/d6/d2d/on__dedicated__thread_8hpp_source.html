235  int sleep_ms = 0;
236  while ( true) {
237  { scoped_lock lk( non_const_context_base::context().cs);
238  // refresh it - just in case it got changed...
239  sleep_ms = non_const_context_base::context().write_period_ms;

246  if ( non_const_context_base::context().is_paused) {

254  if ( non_const_context_base::context().is_paused)

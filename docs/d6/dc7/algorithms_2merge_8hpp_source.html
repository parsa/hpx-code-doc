47  // sequential merge with projection function.
48  template <typename InIter1, typename InIter2, typename OutIter,
49  typename Comp, typename Proj1, typename Proj2>
50  hpx::util::tuple<InIter1, InIter2, OutIter>
51  sequential_merge(InIter1 first1, InIter1 last1,
52  InIter2 first2, InIter2 last2,
53  OutIter dest, Comp && comp, Proj1 && proj1, Proj2 && proj2)
54  {
55  using hpx::util::invoke;
56 
57  if (first1 != last1 && first2 != last2)
58  {
59  while (true)
60  {
61  if (invoke(comp,
62  invoke(proj2, *first2),
63  invoke(proj1, *first1)))
64  {
65  *dest++ = *first2++;
66  if (first2 == last2)
67  break;
68  }
69  else
70  {
71  *dest++ = *first1++;
72  if (first1 == last1)
73  break;
74  }
75  }
76  }
77  dest = std::copy(first1, last1, dest);
78  dest = std::copy(first2, last2, dest);
79 
80  return hpx::util::make_tuple(last1, last2, dest);
81  }

84  {
85  // upper_bound with projection function.
86  template<typename RandIter, typename Type, typename Comp, typename Proj>
87  static RandIter
88  call(RandIter first, RandIter last, const Type& value,
89  Comp comp, Proj proj)
90  {
91  typedef typename std::iterator_traits<RandIter>::difference_type
92  difference_type;
93 
94  using hpx::util::invoke;
95 
96  difference_type count = std::distance(first, last);
97 
98  while (count > 0)
99  {
100  difference_type step = count / 2;
101  RandIter mid = std::next(first, step);
102 
103  if (!invoke(comp, value, invoke(proj, *mid)))
104  {
105  first = ++mid;
106  count -= step + 1;
107  }
108  else
109  {
110  count = step;
111  }
112  }
113 
114  return first;
115  }

157  template <typename ExPolicy,
158  typename RandIter1, typename RandIter2, typename RandIter3,
159  typename Comp, typename Proj1, typename Proj2, typename BinarySearchHelper,
160  HPX_CONCEPT_REQUIRES_(
161  hpx::traits::is_random_access_iterator<RandIter1>::value &&
162  hpx::traits::is_random_access_iterator<RandIter2>::value &&
163  hpx::traits::is_random_access_iterator<RandIter3>::value)>
164  void
165  parallel_merge_helper(ExPolicy policy,
166  RandIter1 first1, RandIter1 last1,
167  RandIter2 first2, RandIter2 last2,
168  RandIter3 dest, Comp comp,
169  Proj1 proj1, Proj2 proj2, BinarySearchHelper)
170  {
171  using hpx::util::invoke;
172 
173  const std::size_t threshold = 65536ul;
174  HPX_ASSERT(threshold >= 1ul);
175 
176  std::size_t size1 = last1 - first1;
177  std::size_t size2 = last2 - first2;
178 
179  // Perform sequential merge if data size is smaller than threshold.
180  if (size1 + size2 <= threshold)
181  {
182  sequential_merge(first1, first1 + size1,
183  first2, first2 + size2, dest, comp, proj1, proj2);
184  return;
185  }
186 
187  // Let size1 is bigger than size2 always.
188  if (size1 < size2)
189  {
190  // For stability of algorithm, must switch binary search methods
191  // when swapping size1 and size2.
192  parallel_merge_helper(policy,
193  first2, last2, first1, last1, dest, comp, proj2, proj1,
194  typename BinarySearchHelper::another_type());
195  return;
196  }
197 
198  HPX_ASSERT(size1 >= size2);
199  HPX_ASSERT(size1 >= 1ul);
200 
201  RandIter1 mid1 = first1 + size1 / 2;
202  RandIter2 boundary2 = BinarySearchHelper::call(
203  first2, last2, invoke(proj1, *mid1), comp, proj2);
204  RandIter3 target = dest + (mid1 - first1) + (boundary2 - first2);
205 
206  *target = *mid1;
207 
208  hpx::future<void> fut = execution::async_execute(policy.executor(),
209  [&]() -> void
210  {
211  // Process leftside ranges.
212  parallel_merge_helper(policy,
213  first1, mid1, first2, boundary2,
214  dest, comp, proj1, proj2, BinarySearchHelper());
215  });
216 
217  try {
218  // Process rightside ranges.
219  parallel_merge_helper(policy,
220  mid1 + 1, last1, boundary2, last2,
221  target + 1, comp, proj1, proj2, BinarySearchHelper());
222  }
223  catch (...) {
224  fut.wait();
225 
226  std::vector<hpx::future<void>> futures(2);
227  futures[0] = std::move(fut);
228  futures[1] = hpx::make_exceptional_future<void>(
229  std::current_exception());
230 
231  std::list<std::exception_ptr> errors;
232  util::detail::handle_local_exceptions<ExPolicy>::call(
233  futures, errors);
234 
235  // Not reachable.
236  HPX_ASSERT(false);
237  }
238 
239  fut.get();
240  }

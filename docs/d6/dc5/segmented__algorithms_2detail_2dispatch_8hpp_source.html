362  dispatch(id_type const& id, Algo && algo, ExPolicy const& policy,
363  IsSeq is_seq, Args&&... args)
364  {
365  // synchronously invoke remote operation
366  future<typename hpx::util::decay<Algo>::type::result_type> f =
367  dispatch_async(id, std::forward<Algo>(algo), policy, is_seq,
368  std::forward<Args>(args)...);
369  f.wait();
370 
371  // handle any remote exceptions
372  if (f.has_exception())

375  parallel::util::detail::handle_remote_exceptions<
376  ExPolicy
377  >::call(f.get_exception_ptr(), errors);
378 
379  HPX_ASSERT(errors.empty());
380  throw exception_list(std::move(errors));
381  }
382  return f.get();

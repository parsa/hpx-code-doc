208  get(std::string const& symbol_name, error_code& ec = throws) const
209  {
210  const_cast<dll&>(*this).LoadLibrary(ec);
211  // make sure everything is initialized
212  if (ec) return std::pair<SymbolType, Deleter>();

239  // Open the library. Yes, we do it on every access to
240  // a symbol, the LoadLibrary function increases the refcnt of the dll
241  // so in the end the dll class holds one refcnt and so does every
242  // symbol.
243 
244  dlerror(); // Clear the error state.
245  HMODULE handle = MyLoadLibrary(
246  (dll_name.empty() ? nullptr : dll_name.c_str()));
247  if (!handle) {
248  std::ostringstream str;
249  str << "Hpx.Plugin: Could not open shared library '"
250  << dll_name << "' (dlerror: " << dlerror() << ")";
251 
252  lock.unlock();
253 
254  // report error

256  "plugin::get", str.str());

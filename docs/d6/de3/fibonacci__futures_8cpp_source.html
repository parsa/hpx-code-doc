79 hpx::future<std::uint64_t> fibonacci_future_one(std::uint64_t n)
80 {
81  // if we know the answer, we return a future encapsulating the final value
82  if (n < 2)
83  return hpx::make_ready_future(n);
84  if (n < threshold)
85  return hpx::make_ready_future(fibonacci_serial(n));
86 
87  // asynchronously launch the calculation of one of the sub-terms
88  // attach a continuation to this future which is called asynchronously on
89  // its completion and which calculates the other sub-term
90  return hpx::async(&fib, n-1).then(fibonacci_future_one_continuation(n));
91 }

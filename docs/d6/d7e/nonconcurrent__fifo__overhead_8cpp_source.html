145 template <typename Fifo>
146 std::pair<double, double>
147 bench_fifo(Fifo& fifo, std::uint64_t local_iterations)
148 {
150  // Push.
151  typename Fifo::value_type seed;
152 
153  std::pair<double, double> elapsed(0.0, 0.0);
154 
155  // Start the clock.
156  high_resolution_timer t;
157 
158  for ( std::uint64_t block = 0
159  ; block < (local_iterations / blocksize)
160  ; ++block)
161  {
162  // Restart the clock.
163  t.restart();
164 
165  for (std::uint64_t i = 0; i < blocksize; ++i)
166  {
167  push(fifo, seed);
168  }
169 
170  elapsed.first += t.elapsed();
171 
173  // Pop.
174 
175  // Restart the clock.
176  t.restart();
177 
178  for (std::uint64_t i = 0; i < blocksize; ++i)
179  {
180  pop(fifo);
181  }
182 
183  elapsed.second += t.elapsed();
184  }
185 
186  return elapsed;
187 }

220 {

223  std::vector<std::pair<double, double> >
224  elapsed_lockfree(threads, std::pair<double, double>(0.0, 0.0));

238  if (thread.joinable())
239  thread.join();
240  }
241

243  std::pair<double, double> total_elapsed_lockfree(0.0, 0.0);
244 
245  for (std::uint64_t i = 0; i < elapsed_control.size(); ++i)
246  {

248  total_elapsed_control.second += elapsed_control[i].second;
249 
250  total_elapsed_lockfree.first += elapsed_lockfree[i].first;
251  total_elapsed_lockfree.second += elapsed_lockfree[i].second;
252  }
253 
254  // Print out the results.
255  print_results(vm, total_elapsed_control, total_elapsed_lockfree);
256 
257  return 0;

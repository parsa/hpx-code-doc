36  template<typename BasePlugin, typename DeleterType>
37  std::pair<abstract_factory<BasePlugin> *, dll_handle>
38  get_abstract_factory_static(get_plugins_list_type f, DeleterType d,
39  std::string const &class_name, std::string const& libname = "",
40  error_code& ec = throws)
41  {
42  typedef typename std::remove_pointer<get_plugins_list_type>
43  ::type PointedType;
44 
45  exported_plugins_type& e = *f();
46  std::string clsname(class_name);
47  boost::algorithm::to_lower(clsname);
48 
49  typename exported_plugins_type::iterator it = e.find(clsname);
50  if (it != e.end()) {
51  abstract_factory<BasePlugin>** xw =
52  boost::unsafe_any_cast<abstract_factory<BasePlugin> *>
53  (&(*it).second);
54 
55  if (!xw) {
56  HPX_THROWS_IF(ec, filesystem_error,
57  "get_abstract_factory_static",
58  "Hpx.Plugin: Can't cast to the right factory type\n");
59  return std::pair<abstract_factory<BasePlugin> *, dll_handle>();
60  }
61 
62  abstract_factory<BasePlugin> *w = *xw;
63  return make_pair(w, boost::shared_ptr<PointedType>(f, d));
64  }
65  else {
66  std::ostringstream str;
67  str << "Hpx.Plugin: Class '" << class_name
68  << "' was not found";
69 
70  if (!libname.empty())
71  str << " in the shared library '" << libname << "'.";
72 
73  if (!e.empty()) {
74  str << " Existing classes: ";
75 
76  bool first = true;
77  typename exported_plugins_type::iterator end = e.end();
78  for (typename exported_plugins_type::iterator jt = e.begin();
79  jt != end; ++jt)
80  {
81  if (first) {
82  str << "'" << (*jt).first << "'";
83  first = false;
84  }
85  else {
86  str << ", '" << (*jt).first << "'";
87  }
88  }
89  str << ".";
90  }
91  else {
92  str << " No classes exist.";
93  }
94 
95  HPX_THROWS_IF(ec, filesystem_error,
96  "get_abstract_factory_static",
97  str.str());
98  return std::pair<abstract_factory<BasePlugin> *, dll_handle>();
99  }
100  }

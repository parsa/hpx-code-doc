152  discover_counters(std::string const& name, error_code& ec)
153  {
154  std::vector<performance_counter> counters;
155 
156  std::vector<counter_info> infos;
157  counter_status status = discover_counter_type(name, infos,
158  discover_counters_full, ec);
159  if (!status_is_valid(status) || ec)
160  return counters;
161 
162  try {
163  counters.reserve(infos.size());
164  for (counter_info const& info : infos)
165  {
166  performance_counter counter(info.fullname_);
167  counters.push_back(counter);
168  }
169  }
170  catch (hpx::exception const& e) {
171  HPX_RETHROWS_IF(ec, e, "discover_counters");
172  }
173 
174  return counters;
175  }

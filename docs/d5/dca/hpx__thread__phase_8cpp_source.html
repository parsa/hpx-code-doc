51 void lock_and_wait(
52  mutex& m
53  , barrier& b0
54  , barrier& b1
55  , value_type& entry
56  , std::size_t wait
57 ) {
58  // Wait for all hpxthreads in this iteration to be created.
59  b0.wait();
60 
61  const thread_id_type this_ = get_self_id();
62 
63  while (true)
64  {
65  // Try to acquire the mutex.
66  std::unique_lock<mutex> l(m, std::try_to_lock);
67 
68  if (l.owns_lock())
69  {
70  entry = value_type(this_, get_thread_phase(this_));
71  break;
72  }
73 
74  // Schedule a wakeup.
75  set_thread_state(this_, milliseconds(30), pending);
76 
77  // Suspend this HPX thread.
78  hpx::this_thread::suspend(suspended);
79  }
80 
81  // Make hpx_main wait for us to finish.
82  b1.wait();
83 }

107 
108  for (std::size_t i = 0; i < iterations; ++i)
109  {
110  std::cout << "iteration: " << i << "\n";
111 
112  // Have the fifo preallocate storage.
113  fifo_type hpxthreads(hpxthread_count);
114 
115  // Allocate the mutexes.
116  std::vector<mutex> m(mutex_count);
117  barrier b0(hpxthread_count + 1), b1(hpxthread_count + 1);
118 
119  for (std::size_t j = 0; j < hpxthread_count; ++j)
120  {
121  // Compute the mutex to be used for this thread.
122  const std::size_t index = j % mutex_count;
123 
124  register_thread(hpx::util::bind
125  (&lock_and_wait, std::ref(m[index])
126  , std::ref(b0)
127  , std::ref(b1)
128  , std::ref(hpxthreads[j])
129  , wait)
130  , "lock_and_wait");
131  }
132 
133  // Tell all hpxthreads that they can start running.
134  b0.wait();
135 
136  // Wait for all hpxthreads to finish.
137  b1.wait();
138 
139  // {{{ Print results for this iteration.
140  for(value_type &entry: hpxthreads)

143  }

145  }

181  {
182  std::lock_guard<mutex_type> lock(mtx_);
183 
184  // Check if this key already exists in the cache.
185  typename cache_type::iterator const it = cache_.find(l);
186 
187  // Check if this key already exists in the cache.
188  if (it != cache_.end())
189  {
190  // Key exists in cache.
191 
192  // Update LRU meta data.
193  key_tracker_.splice(
194  key_tracker_.end()
195  , key_tracker_
196  , lru_reference(it->second)
197  );
198

200  // remove the oldest one and return it.
201  if (!cached_connections(it->second).empty())
202  {

204  connection_type result = connections.front();

236  bool get_or_reserve(key_type const& l, connection_type& conn,
237  bool force_insert = false)
238  {
239  std::lock_guard<mutex_type> lock(mtx_);
240 
241  typename cache_type::iterator const it = cache_.find(l);
242 
243  // Check if this key already exists in the cache.
244  if (it != cache_.end())
245  {
246  // Key exists in cache.
247 
248  // Update LRU meta data.
249  key_tracker_.splice(
250  key_tracker_.end()
251  , key_tracker_
252  , lru_reference(it->second)
253  );
254 
255  // If connections to the locality are available in the cache,
256  // remove the oldest one and return it.
257  if (!cached_connections(it->second).empty())
258  {
259  value_type& connections = cached_connections(it->second);
260  conn = connections.front();
261  connections.pop_front();
262 
263 #if defined(HPX_TRACK_STATE_OF_OUTGOING_TCP_CONNECTION)
264  conn->set_state(Connection::state_reinitialized);
265 #endif
266  ++hits_;
267  check_invariants();
268  return true;
269  }
270 
271  // Otherwise, if we have less connections for this locality
272  // than the maximum, try to reserve space in the cache for a new
273  // connection.
274  if (num_existing_connections(it->second) <
275  max_num_connections(it->second) ||
276  force_insert)
277  {
278  // See if we have enough space or can make space available.
279 
280  // Note that if we don't have any space and there are no
281  // outstanding connections for this locality, we grow the
282  // cache size beyond its limit (hoping that it will be
283  // reduced in size next time some connection is handed back
284  // to the cache).
285 
286  if (!free_space() &&
287  num_existing_connections(it->second) != 0 &&
288  !force_insert)
289  {
290  // If we can't find or make space, give up.
291  ++misses_;
292  check_invariants();
293  return false;
294  }
295 
296  // Make sure the input connection shared_ptr doesn't hold
297  // anything.
298  conn.reset();
299 
300  // Increase the per-locality and overall connection counts.
301  increment_connection_count(it->second);
302 
303  // Statistics
304  ++insertions_;
305  check_invariants();
306  return true;
307  }
308 
309  // We've reached the maximum number of connections for this
310  // locality, and none of them are checked into the cache, so
311  // we have to give up.
312  ++misses_;
313  check_invariants();
314  return false;
315  }
316 
317  // Key (locality) isn't in cache.
318 
319  // See if we have enough space or can make space available.
320 
321  // Note that we ignore the outcome of free_space() here as we have
322  // to guarantee to have space for the new connection as there are
323  // no connections outstanding for this locality. If free_space
324  // fails we grow the cache size beyond its limit (hoping that it
325  // will be reduced in size next time some connection is handed back
326  // to the cache).
327  free_space();
328 
329  // Update LRU meta data.
330  typename key_tracker_type::iterator kt =
331  key_tracker_.insert(key_tracker_.end(), l);
332 
333  cache_.insert(std::make_pair(
334  l, util::make_tuple(
335  value_type(), 1, max_connections_per_locality_, kt
336  ))
337  );
338 
339  // Make sure the input connection shared_ptr doesn't hold anything.
340  conn.reset();
341 
342  // Increase the overall connection counts.
343  ++connections_;
344 
345  ++insertions_;
346  check_invariants();
347  return true;
348  }

354  void reclaim(key_type const& l, connection_type const& conn)
355  {
356  std::lock_guard<mutex_type> lock(mtx_);
357 
358  // Search for an entry for this key.
359  typename cache_type::iterator const ct = cache_.find(l);
360 
361  if (ct != cache_.end()) {
362  // Update LRU meta data.
363  key_tracker_.splice(
364  key_tracker_.end()
365  , key_tracker_
366  , lru_reference(ct->second)
367  );
368 
369  // Return the connection back to the cache only if the number
370  // of connections does not need to be shrunk.
371  if (num_existing_connections(ct->second) <=
372  max_num_connections(ct->second))
373  {
374  // Add the connection to the entry.
375  cached_connections(ct->second).push_back(conn);
376 
377  ++reclaims_;
378 
379 #if defined(HPX_TRACK_STATE_OF_OUTGOING_TCP_CONNECTION)
380  conn->set_state(Connection::state_reclaimed);
381 #endif
382  }
383  else
384  {
385  // Adjust the number of existing connections for this key.
386  decrement_connection_count(ct->second);
387 
388  // do the accounting
389  ++evictions_;
390 
391  // the connection itself will go out of scope on return
392 #if defined(HPX_TRACK_STATE_OF_OUTGOING_TCP_CONNECTION)
393  conn->set_state(Connection::state_deleting);
394 #endif
395  }
396 
397  // FIXME: Again, this should probably throw instead of asserting,
398  // as invariants could be invalidated here due to caller error.
399  check_invariants();
400  }
401 // else {
402 // // Key should already exist in the cache. FIXME: This should
403 // // probably throw as could easily be triggered by caller error.
404 // HPX_ASSERT(shutting_down_);
405 // }
406  }

544  void check_invariants() const
545  {
546 #if defined(HPX_DEBUG)
547  typedef typename cache_type::const_iterator const_iterator;
548 
549  size_type in_cache_count = 0, total_count = 0;
550  const_iterator end = cache_.end();
551  for (const_iterator ct = cache_.begin(); ct != end; ++ct)
552  {
553  cache_value_type const& val = ct->second;
554 
555  std::size_t num_connections = cached_connections(val).size();
556  std::size_t num_existing = num_existing_connections(val);
557 
558  // The separate item counter has to properly count all the
559  // existing elements, not only those in the cache entry.
560  HPX_ASSERT(num_connections <= num_existing);
561 
562  // Count all connections (both those in the cache and those
563  // checked out of the cache).
564  in_cache_count += num_connections;
565  total_count += num_existing;
566  }
567 
568  // Overall connection count should be larger than or equal to the
569  // number of entries in the cache.
570  HPX_ASSERT(in_cache_count <= connections_);
571 
572  // Overall connection count should be equal to the sum of connection
573  // counts for all localities.
574  HPX_ASSERT(total_count == connections_);
575 
576  // The list of key trackers should have the same size as the cache.
577  HPX_ASSERT(key_tracker_.size() == cache_.size());
578 #endif
579  }

586  bool free_space()
587  {
588  // If the cache isn't full, just return true.
589  if (connections_ < max_connections_)
590  return true;
591 
592  // Find the least recently used key.
593  typename key_tracker_type::iterator kt = key_tracker_.begin();
594 
595  while (connections_ >= max_connections_)
596  {
597  // Find the least recently used keys data.
598  typename cache_type::iterator ct = cache_.find(*kt);
599  HPX_ASSERT(ct != cache_.end());
600 
601  // If the entry is empty, ignore it and try the next least
602  // recently used entry.
603  if (cached_connections(ct->second).empty())
604  {
605  // Remove the key if its connection count is zero.
606  if (0 == num_existing_connections(ct->second)) {
607  cache_.erase(ct);
608  key_tracker_.erase(kt);
609  kt = key_tracker_.begin();
610  }
611  else {
612  // REVIEW: Should we reorder key_tracker_ to speed up
613  // the eviction?
614  ++kt;
615  }
616 
617  // If we've gone through key_tracker_ and haven't found
618  // anything evict-able, then all the entries must be
619  // currently checked out.
620  if (key_tracker_.end() == kt)
621  return false;
622 
623  continue;
624  }
625 
626  // Remove the oldest connection.
627  cached_connections(ct->second).pop_front();
628 
629  // Adjust the overall and per-locality connection count.
630  decrement_connection_count(ct->second);
631 
632  // Statistics
633  ++evictions_;
634  }
635 
636  return true;
637  }

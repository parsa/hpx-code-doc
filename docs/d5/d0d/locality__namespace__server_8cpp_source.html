43 // register all performance counter types exposed by this component
44 void locality_namespace::register_counter_types(
45  error_code& ec
46  )
47 {
48  using util::placeholders::_1;
49  using util::placeholders::_2;
50  boost::format help_count(
51  "returns the number of invocations of the AGAS service '%s'");
52  boost::format help_time(
53  "returns the overall execution time of the AGAS service '%s'");
54  performance_counters::create_counter_func creator(
55  util::bind(&performance_counters::agas_raw_counter_creator, _1, _2
56  , agas::server::locality_namespace_service_name));
57 
58  for (std::size_t i = 0;
59  i != detail::num_locality_namespace_services;
60  ++i)
61  {
62  // global counters are handled elsewhere
63  if (detail::locality_namespace_services[i].code_ ==
64  locality_ns_statistics_counter)
65  continue;
66 
67  std::string name(detail::locality_namespace_services[i].name_);
68  std::string help;
69  std::string::size_type p = name.find_last_of('/');
70  HPX_ASSERT(p != std::string::npos);
71 
72  if (detail::locality_namespace_services[i].target_ ==
73  detail::counter_target_count)
74  help = boost::str(help_count % name.substr(p+1));
75  else
76  help = boost::str(help_time % name.substr(p+1));
77 
78  performance_counters::install_counter_type(
79  agas::performance_counter_basename + name
80  , performance_counters::counter_raw
81  , help
82  , creator
83  , &performance_counters::locality0_counter_discoverer
84  , HPX_PERFORMANCE_COUNTER_V1
85  , detail::locality_namespace_services[i].uom_
86  , ec
87  );
88  if (ec) return;
89  }
90 }

167 std::uint32_t locality_namespace::allocate(
168  parcelset::endpoints_type const& endpoints
169  , std::uint64_t count
170  , std::uint32_t num_threads
171  , naming::gid_type suggested_prefix
172  )
173 { // {{{ allocate implementation
174  util::scoped_timer<std::atomic<std::int64_t> > update(
175  counter_data_.allocate_.time_
176  );
177  counter_data_.increment_allocate_count();
178 
179  using hpx::util::get;
180 
181  std::unique_lock<mutex_type> l(mutex_);
182 
183 #if defined(HPX_DEBUG)
184  for (partition_table_type::value_type const& partition : partitions_)
185  {
186  HPX_ASSERT(get<0>(partition.second) != endpoints);
187  }
188 #endif
189  // Check for address space exhaustion.
190  if (HPX_UNLIKELY(0xFFFFFFFE < partitions_.size())) //-V104
191  {
192  l.unlock();
193 
194  HPX_THROW_EXCEPTION(internal_server_error
195  , "locality_namespace::allocate"
196  , "primary namespace has been exhausted");
197  }
198 
199  // Compute the locality's prefix.
200  std::uint32_t prefix = naming::invalid_locality_id;
201 
202  // check if the suggested prefix can be used instead of the next
203  // free one
204  std::uint32_t suggested_locality_id =
205  naming::get_locality_id_from_gid(suggested_prefix);
206 
207  partition_table_type::iterator it = partitions_.end();
208  if (suggested_locality_id != naming::invalid_locality_id)
209  {
210  it = partitions_.find(suggested_locality_id);
211 
212  if(it == partitions_.end())
213  {
214  prefix = suggested_locality_id;
215  }
216  else
217  {
218  do {
219  prefix = prefix_counter_++;
220  it = partitions_.find(prefix);
221  } while (it != partitions_.end());
222  }
223  }
224  else
225  {
226  do {
227  prefix = prefix_counter_++;
228  it = partitions_.find(prefix);
229  } while (it != partitions_.end());
230  }
231 
232  // We need to create an entry in the partition table for this
233  // locality.
234  if(HPX_UNLIKELY(!util::insert_checked(partitions_.insert(
235  std::make_pair(prefix, partition_type(endpoints, num_threads))), it)))
236  {
237  l.unlock();
238 
239  HPX_THROW_EXCEPTION(lock_error
240  , "locality_namespace::allocate"
241  , boost::str(boost::format(
242  "partition table insertion failed due to a locking "
243  "error or memory corruption, endpoint(%1%), "
244  "prefix(%2%)") % endpoints % prefix));
245  }
246 
247 
248  // Now that we've inserted the locality into the partition table
249  // successfully, we need to put the locality's GID into the GVA
250  // table so that parcels can be sent to the memory of a locality.
251  if (primary_)
252  {
253  naming::gid_type id(naming::get_gid_from_locality_id(prefix));
254  gva const g(id, components::component_runtime_support, count);
255 
256  if(!primary_->bind_gid(g, id, id))
257  {
258  HPX_THROW_EXCEPTION(bad_request
259  , "locality_namespace::allocate"
260  , boost::str(boost::format(
261  "unable to bind prefix(%1%) to a gid") % prefix));
262  }
263  return prefix;
264  }
265 
266  LAGAS_(info) << (boost::format(
267  "locality_namespace::allocate, ep(%1%), count(%2%), "
268  "prefix(%3%)")
269  % endpoints % count % prefix);
270 
271  return prefix;
272 } // }}}

296 void locality_namespace::free(naming::gid_type locality)
297 { // {{{ free implementation
298  util::scoped_timer<std::atomic<std::int64_t> > update(
299  counter_data_.free_.time_
300  );
301  counter_data_.increment_free_count();
302 
303  using hpx::util::get;
304 
305  // parameters
306  std::uint32_t prefix = naming::get_locality_id_from_gid(locality);
307 
308  std::unique_lock<mutex_type> l(mutex_);
309 
310  partition_table_type::iterator pit = partitions_.find(prefix)
311  , pend = partitions_.end();
312 
313  if (pit != pend)
314  {
315  /*
316  // Wipe the locality from the tables.
317  naming::gid_type locality =
318  naming::get_gid_from_locality_id(get<0>(pit->second));
319 
320  // first remove entry from reverse partition table
321  prefixes_.erase(get<0>(pit->second));
322  */
323 
324  // now remove it from the main partition table
325  partitions_.erase(pit);
326 
327  if (primary_)
328  {
329  l.unlock();
330 
331  // remove primary namespace
332  {
333  naming::gid_type service(HPX_AGAS_PRIMARY_NS_MSB,
334  HPX_AGAS_PRIMARY_NS_LSB);
335  primary_->unbind_gid(
336  1, naming::replace_locality_id(service, prefix));
337  }
338 
339  // remove symbol namespace
340  {
341  naming::gid_type service(HPX_AGAS_SYMBOL_NS_MSB,
342  HPX_AGAS_SYMBOL_NS_LSB);
343  primary_->unbind_gid(
344  1, naming::replace_locality_id(service, prefix));
345  }
346 
347  // remove locality itself
348  {
349  primary_->unbind_gid(0, locality);
350  }
351  }
352 
353  /*
354  LAGAS_(info) << (boost::format(
355  "locality_namespace::free, ep(%1%)")
356  % ep);
357  */
358  }
359 
360  /*
361  LAGAS_(info) << (boost::format(
362  "locality_namespace::free, ep(%1%), "
363  "response(no_success)")
364  % ep);
365  */
366 } // }}}

452 naming::gid_type locality_namespace::statistics_counter(std::string name)
453 { // {{{ statistics_counter implementation
454  LAGAS_(info) << "locality_namespace::statistics_counter";
455 
456  performance_counters::counter_path_elements p;
457  performance_counters::get_counter_path_elements(name, p);
458 
459  if (p.objectname_ != "agas")
460  {
461  HPX_THROW_EXCEPTION(bad_parameter,
462  "locality_namespace::statistics_counter",
463  "unknown performance counter (unrelated to AGAS)");
464  }
465 
466  namespace_action_code code = invalid_request;
467  detail::counter_target target = detail::counter_target_invalid;
468  for (std::size_t i = 0;
469  i != detail::num_locality_namespace_services;
470  ++i)
471  {
472  if (p.countername_ == detail::locality_namespace_services[i].name_)
473  {
474  code = detail::locality_namespace_services[i].code_;
475  target = detail::locality_namespace_services[i].target_;
476  break;
477  }
478  }
479 
480  if (code == invalid_request || target == detail::counter_target_invalid)
481  {
482  HPX_THROW_EXCEPTION(bad_parameter,
483  "locality_namespace::statistics_counter",
484  "unknown performance counter (unrelated to AGAS)");
485  }
486 
487  typedef locality_namespace::counter_data cd;
488 
489  using util::placeholders::_1;
490  util::function_nonser<std::int64_t(bool)> get_data_func;
491  if (target == detail::counter_target_count)
492  {
493  switch (code) {
494  case locality_ns_allocate:
495  get_data_func = util::bind(&cd::get_allocate_count,
496  &counter_data_, _1);
497  break;
498  case locality_ns_resolve_locality:
499  get_data_func = util::bind(&cd::get_resolve_locality_count,
500  &counter_data_, _1);
501  break;
502  case locality_ns_free:
503  get_data_func = util::bind(&cd::get_free_count,
504  &counter_data_, _1);
505  break;
506  case locality_ns_localities:
507  get_data_func = util::bind(&cd::get_localities_count,
508  &counter_data_, _1);
509  break;
510  case locality_ns_num_localities:
511  get_data_func = util::bind(&cd::get_num_localities_count,
512  &counter_data_, _1);
513  break;
514  case locality_ns_num_threads:
515  get_data_func = util::bind(&cd::get_num_threads_count,
516  &counter_data_, _1);
517  break;
518  case locality_ns_statistics_counter:
519  get_data_func = util::bind(&cd::get_overall_count, &counter_data_, _1);
520  break;
521  default:
522  HPX_THROW_EXCEPTION(bad_parameter
523  , "locality_namespace::statistics"
524  , "bad action code while querying statistics");
525  }
526  }
527  else {
528  HPX_ASSERT(detail::counter_target_time == target);
529  switch (code) {
530  case locality_ns_allocate:
531  get_data_func = util::bind(&cd::get_allocate_time,
532  &counter_data_, _1);
533  break;
534  case locality_ns_resolve_locality:
535  get_data_func = util::bind(&cd::get_resolve_locality_time,
536  &counter_data_, _1);
537  break;
538  case locality_ns_free:
539  get_data_func = util::bind(&cd::get_free_time, &counter_data_, _1);
540  break;
541  case locality_ns_localities:
542  get_data_func = util::bind(&cd::get_localities_time,
543  &counter_data_, _1);
544  break;
545  case locality_ns_num_localities:
546  get_data_func = util::bind(&cd::get_num_localities_time,
547  &counter_data_, _1);
548  break;
549  case locality_ns_num_threads:
550  get_data_func = util::bind(&cd::get_num_threads_time, &counter_data_, _1);
551  break;
552  case locality_ns_statistics_counter:
553  get_data_func = util::bind(&cd::get_overall_time, &counter_data_, _1);
554  break;
555  default:
556  HPX_THROW_EXCEPTION(bad_parameter
557  , "locality_namespace::statistics"
558  , "bad action code while querying statistics");
559  }
560  }
561 
562  performance_counters::counter_info info;
563  performance_counters::get_counter_type(name, info);
564 
565  performance_counters::complement_counter_info(info);
566 
567  using performance_counters::detail::create_raw_counter;
568  naming::gid_type gid = create_raw_counter(info, get_data_func, hpx::throws);
569 
570  return naming::detail::strip_credits_from_gid(gid);
571 }

29  inline std::string
30  trim_whitespace (std::string const &s)
31  {
32  typedef std::string::size_type size_type;
33 
34  size_type first = s.find_first_not_of(" \t");
35  if (std::string::npos == first)
36  return std::string();
37 
38  size_type last = s.find_last_not_of(" \t");
39  return s.substr(first, last - first + 1);
40  }

43  // All command line options which are normally formatted as --hpx:foo
44  // should be usable as --hpx:N:foo, where N is the node number this
45  // option should be exclusively used for.
46  bool handle_node_specific_option(std::string const& s, std::size_t node,
47  std::pair<std::string, std::string>& opt)
48  {
49  // any option not starting with --hpx: will be handled elsewhere
50  char hpx_prefix[] = "--hpx:";
51  std::string::size_type const hpx_prefix_len = sizeof(hpx_prefix)-1;
52  if (s.size() < hpx_prefix_len || s.find(hpx_prefix) != 0 ||
53  !std::isdigit(s[hpx_prefix_len]))
54  {
55  return false;
56  }
57 
58  // any --hpx: option without a second ':' is handled elsewhere as well
59  std::string::size_type p = s.find_first_of(":", hpx_prefix_len);
60  if (p == std::string::npos)
61  return false;
62 
63  if (hpx::util::safe_lexical_cast(
64  s.substr(hpx_prefix_len, p-hpx_prefix_len),
65  std::size_t(-1)) == node)
66  {
67  // this option is for the current locality only
68  std::string::size_type p1 = s.find_first_of('=', p);
69  if (p1 != std::string::npos) {
70  // the option has a value
71  std::string o("hpx:" + trim_whitespace(s.substr(p+1, p1-p-1)));
72  std::string v(trim_whitespace(s.substr(p1+1)));
73  opt = std::make_pair(o, v);
74  }
75  else {
76  // no value
77  std::string o("hpx:" + trim_whitespace(s.substr(p+1)));
78  opt = std::make_pair(o, std::string());
79  }
80  return true;
81  }
82 
83  // This option is specifically not for us, so we return an option
84  // which will be silently ignored.
85  opt = std::make_pair(std::string("hpx:ignore"), std::string());
86  return true;
87  }

90  // Handle aliasing of command line options based on information stored
91  // in the ini-configuration
92  std::pair<std::string, std::string> handle_aliasing(
93  util::section const& ini, std::string const& option)
94  {
95  std::pair<std::string, std::string> result;
96 
97  std::string opt(trim_whitespace(option));
98  if (opt.size() < 2 || opt[0] != '-')
99  return result;
100 
101  util::section const* sec = ini.get_section("hpx.commandline.aliases");
102  if (nullptr == sec)
103  return result; // no alias mappings are defined
104 
105  // we found shortcut option definitions, try to find mapping
106  std::string expand_to;
107  std::string::size_type start_at = 2;
108  bool long_option = false;
109  if (opt.size() > 2 && opt[1] != '-') {
110  // short option with value: first two letters have to match
111  expand_to = trim_whitespace(sec->get_entry(opt.substr(0, start_at), ""));
112  }
113  else {
114  // short option (no value) or long option
115  if (opt[1] == '-') {
116  start_at = opt.find_last_of("=");
117  long_option = true;
118  }
119 
120  if (start_at != std::string::npos) {
121  expand_to = trim_whitespace(
122  sec->get_entry(opt.substr(0, start_at), ""));
123  }
124  else {
125  expand_to = trim_whitespace(sec->get_entry(opt, ""));
126  }
127  }
128 
129  if (expand_to.size() < 2 || expand_to.substr(0, 2) != "--")
130  return result; // no sensible alias is defined for this option
131  expand_to.erase(0, 2);
132 
133  std::string::size_type p = expand_to.find_first_of('=');
134  if (p != std::string::npos) {
135  // the option alias defines its own value
136  std::string o(trim_whitespace(expand_to.substr(0, p)));
137  std::string v(trim_whitespace(expand_to.substr(p+1)));
138  result = std::make_pair(o, v);
139  }
140  else if (start_at != std::string::npos && start_at < opt.size()) {
141  // extract value from original option
142  result = std::make_pair(expand_to,
143  opt.substr(start_at + (long_option ? 1 : 0)));
144  }
145  else {
146  // no value
147  result = std::make_pair(expand_to, std::string());
148  }
149 
150  return result;
151  }

154  // Additional command line parser which interprets '@something' as an
155  // option "options-file" with the value "something". Additionally we
156  // resolve defined command line option aliases.
157  struct option_parser
158  {
159  option_parser(util::section const& ini, std::size_t node)
160  : ini_(ini), node_(node)
161  {}
162 
163  std::pair<std::string, std::string> operator()(std::string const& s) const
164  {
165  // handle special syntax for configuration files @filename
166  if ('@' == s[0])
167  return std::make_pair(std::string("hpx:options-file"), s.substr(1));
168 
169  // handle node specific options
170  std::pair<std::string, std::string> opt;
171  if (handle_node_specific_option(s, node_, opt))
172  return opt;
173 
174  // handle aliasing, if enabled
175  if (ini_.get_entry("hpx.commandline.aliasing", "1") == "1")
176  return handle_aliasing(ini_, s);
177 
178  return opt;
179  }
180 
181  util::section const& ini_;
182  std::size_t node_;
183  };

197  // Read all options from a given config file, parse and add them to the
198  // given variables_map
199  bool read_config_file_options(std::string const &filename,
200  boost::program_options::options_description const &desc,
201  boost::program_options::variables_map &vm,
202  util::section const& rtcfg, std::size_t node, int error_mode)
203  {
204  std::ifstream ifs(filename.c_str());
205  if (!ifs.is_open()) {
206  if (error_mode & util::report_missing_config_file) {
207  std::cerr << "hpx::init: command line warning: command line "
208  "options file not found (" << filename << ")"
209  << std::endl;
210  }
211  return false;
212  }
213 
214  std::vector<std::string> options;
215  std::string line;
216  while (std::getline(ifs, line)) {
217  // skip empty lines
218  std::string::size_type pos = line.find_first_not_of(" \t");
219  if (pos == std::string::npos)
220  continue;
221 
222  // strip leading and trailing whitespace
223  line = trim_whitespace(line);
224 
225  // skip comment lines
226  if ('#' != line[0]) {
227  std::string::size_type p1 = line.find_first_of(" \t");
228  if (p1 != std::string::npos) {
229  // rebuild the line connecting the parts with a '='
230  line = trim_whitespace(line.substr(0, p1)) + '=' +
231  trim_whitespace(line.substr(p1));
232  }
233  options.push_back(line);
234  }
235  }
236 
237  // add options to parsed settings
238  if (options.size() > 0) {
239  using boost::program_options::value;
240  using boost::program_options::store;
241  using boost::program_options::basic_command_line_parser;
242  using boost::program_options::command_line_parser;
243  using namespace boost::program_options::command_line_style;
244 
245  store(detail::get_commandline_parser(
246  command_line_parser(options)
247  .options(desc)
248  .style(unix_style)
249  .extra_parser(detail::option_parser(rtcfg, node)),
250  error_mode).run(),
251  vm);
252  notify(vm);
253  }
254  return true;
255  }

257  // try to find a config file somewhere up the filesystem hierarchy
258  // starting with the input file path. This allows to use a general
259  // <app_name>.cfg file for all executables in a certain project.
260  void handle_generic_config_options(std::string appname,
261  boost::program_options::variables_map& vm,
262  boost::program_options::options_description const& desc_cfgfile,
263  util::section const& ini, std::size_t node, int error_mode)
264  {
265  if (appname.empty())
266  return;
267 
268  boost::filesystem::path dir (boost::filesystem::initial_path());
269  boost::filesystem::path app (appname);
270  appname = boost::filesystem::basename(app.filename());
271 
272  // walk up the hierarchy, trying to find a file <appname>.cfg
273  while (!dir.empty()) {
274  boost::filesystem::path filename = dir / (appname + ".cfg");
275  bool result = read_config_file_options(filename.string(),
276  desc_cfgfile, vm, ini, node,
277  error_mode & ~util::report_missing_config_file);
278  if (result)
279  break; // break on the first options file found
280 
281  dir = dir.parent_path(); // chop off last directory part
282  }
283  }

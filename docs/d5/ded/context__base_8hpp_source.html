314  // Resume coroutine.
315  // Pre: The coroutine must be ready.
316  // Post: The coroutine relinquished control. It might be ready, waiting
317  // or exited.
318  // Throws:- 'waiting' if the coroutine entered the wait state,
319  // - 'coroutine_exited' if the coroutine exited by an uncaught
320  // 'exit_exception'.
321  // - 'abnormal_exit' if the coroutine was exited by another
322  // uncaught exception.
323  // Note, it guarantees that the coroutine is resumed. Can throw only
324  // on return.
325  void invoke()
326  {
327  HPX_ASSERT(is_ready());
328  do_invoke();
329  // TODO: could use a binary or here to eliminate
330  // shortcut evaluation (and a branch), but maybe the compiler is
331  // smart enough to do it anyway as there are no side effects.
332  if (m_exit_status || m_state == ctx_waiting)
333  {
334  if (m_state == ctx_waiting)
335  throw coroutines::waiting();
336  if (m_exit_status == ctx_exited_return)
337  return;
338  if (m_exit_status == ctx_exited_abnormally)
339  {
340  std::rethrow_exception(m_type_info);
341  //std::type_info const* tinfo = nullptr;
342  //std::swap(m_type_info, tinfo);
343  //throw abnormal_exit(tinfo ? *tinfo :
344  // typeid(unknown_exception_tag));
345  }
346  else if (m_exit_status == ctx_exited_exit)
347  throw coroutine_exited();
348  else {
349  HPX_ASSERT(0 && "unknown exit status");
350  }
351  }
352  }

354  // Put coroutine in ready state and relinquish control
355  // to caller until resumed again.
356  // Pre: Coroutine is running.
357  // Exit not pending.
358  // Operations not pending.
359  // Post: Coroutine is running.
360  // Throws: exit_exception, if exit is pending *after* it has been
361  // resumed.
362  void yield()
363  {
364  HPX_ASSERT(m_exit_state < ctx_exit_signaled); //prevent infinite loops
365  HPX_ASSERT(running());
366  HPX_ASSERT(!pending());
367 
368  m_state = ctx_ready;
369  do_yield();
370 
371  HPX_ASSERT(running());
372  check_exit_state();
373  }

375 #if defined(HPX_HAVE_THREAD_OPERATIONS_COUNT)
376  //
377  // If n > 0, put the coroutine in the wait state
378  // then relinquish control to caller.
379  // If n = 0 do nothing.
380  // The coroutine will remain in the wait state until
381  // is signaled 'n' times.
382  // Pre: 0 <= n < pending()
383  // Coroutine is running.
384  // Exit not pending.
385  // Post: Coroutine is running.
386  // The coroutine has been signaled 'n' times unless an exit
387  // has been signaled.
388  // Throws: exit_exception.
389  // FIXME: currently there is a BIG problem. A coroutine cannot
390  // be exited as long as there are futures pending.
391  // The exit_exception would cause the future to be destroyed and
392  // an assertion to be generated. Removing an assertion is not a
393  // solution because we would leak the coroutine impl. The callback
394  // bound to the future in fact hold a reference to it. If the coroutine
395  // is exited the callback cannot be called.
396  void wait(int n)
397  {
398  HPX_ASSERT(!(n < 0));
399  HPX_ASSERT(m_exit_state < ctx_exit_signaled); //prevent infinite loop
400  HPX_ASSERT(running());
401  HPX_ASSERT(!(pending() < n));
402 
403  if (n == 0) return;
404  m_wait_counter = n;
405 
406  m_state = ctx_waiting;
407  do_yield();
408 
409  HPX_ASSERT(m_state == ctx_running);
410  check_exit_state();
411  HPX_ASSERT(m_wait_counter == 0);
412  }
413 #endif

417  // Cannot be called if there are pending operations.

389  // load information about dynamically discovered plugins
390  std::vector<std::shared_ptr<plugins::plugin_registry_base> >
391  runtime_configuration::load_modules()
392  {
393  typedef std::vector<std::shared_ptr<plugins::plugin_registry_base> >
394  plugin_list_type;
395 
396  namespace fs = boost::filesystem;
397 
398  // try to build default ini structure from shared libraries in default
399  // installation location, this allows to install simple components
400  // without the need to install an ini file
401  // split of the separate paths from the given path list
402  typedef boost::tokenizer<boost::char_separator<char> > tokenizer_type;
403 
404  std::string component_path(
405  get_entry("hpx.component_path", HPX_DEFAULT_COMPONENT_PATH));
406 
407  std::string component_path_suffixes(
408  get_entry("hpx.component_path_suffixes", "/lib/hpx"));
409 
410  // protect against duplicate paths
411  std::set<std::string> component_paths;
412 
413  // list of base names avoiding to load a module more than once
414  std::map<std::string, fs::path> basenames;
415 
416  boost::char_separator<char> sep (HPX_INI_PATH_DELIMITER);
417  tokenizer_type tok_path(component_path, sep);
418  tokenizer_type tok_suffixes(component_path_suffixes, sep);
419  tokenizer_type::iterator end_path = tok_path.end();
420  tokenizer_type::iterator end_suffixes = tok_suffixes.end();
421  plugin_list_type plugin_registries;
422 
423  for (tokenizer_type::iterator it = tok_path.begin(); it != end_path; ++it)
424  {
425  std::string p = *it;
426  for(tokenizer_type::iterator jt = tok_suffixes.begin();
427  jt != end_suffixes; ++jt)
428  {
429  std::string path(p);
430  path += *jt;
431 
432  if (!path.empty()) {
433  fs::path this_p(path);
434  boost::system::error_code fsec;
435  fs::path canonical_p = util::canonical_path(this_p, fsec);
436  if (fsec)
437  canonical_p = this_p;
438 
439  std::pair<std::set<std::string>::iterator, bool> p =
440  component_paths.insert(
441  util::native_file_string(canonical_p));
442 
443  if (p.second) {
444  // have all path elements, now find ini files in there...
445  fs::path this_path (hpx::util::create_path(*p.first));
446  if (fs::exists(this_path, fsec) && !fsec) {
447  plugin_list_type tmp_regs =
448  util::init_ini_data_default(
449  this_path.string(), *this, basenames, modules_);
450 
451  std::copy(tmp_regs.begin(), tmp_regs.end(),
452  std::back_inserter(plugin_registries));
453  }
454  }
455  }
456  }
457  }
458 
459  // read system and user ini files _again_, to allow the user to
460  // overwrite the settings from the default component ini's.
461  util::init_ini_data_base(*this, hpx_ini_file);
462 
463  // let the command line override the config file.
464  if (!cmdline_ini_defs.empty())
465  parse("<command line definitions>", cmdline_ini_defs, true, false);
466 
467  // merge all found ini files of all components
468  util::merge_component_inis(*this);
469 
470  need_to_call_pre_initialize = true;
471 
472  // invoke reconfigure
473  reconfigure();
474 
475  return plugin_registries;
476  }

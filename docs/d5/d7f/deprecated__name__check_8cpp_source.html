138  void deprecated_name_check::inspect(
139  const string & library_name,
140  const path & full_path, // example: c:/foo/boost/filesystem/path.hpp
141  const string & contents) // contents of file to be inspected
142  {
143  std::string::size_type p = contents.find( "hpxinspect:" "nodeprecatedname" );
144  if (p != string::npos)
145  {
146  // ignore this directive here (it is handled below) if it is followed
147  // by a ':'
148  if (p == contents.size() - 27 ||
149  (contents.size() > p + 27 && contents[p + 27] != ':'))
150  {
151  return;
152  }
153  }
154 
155  std::set<std::string> found_names;
156 
157  // for all given names, check whether any is used
158  for (deprecated_names_regex_data const& d : regex_data)
159  {
160  boost::sregex_iterator cur(contents.begin(), contents.end(), d.pattern), end;
161  for(; cur != end; ++cur)
162  {
163  auto m = *cur;
164  if (m[1].matched)
165  {
166  // avoid errors to be reported twice
167  std::string found_name(m[1].first, m[1].second);
168 
169  if (found_names.find(found_name) == found_names.end())
170  {
171  std::string tag("hpxinspect:" "nodeprecatedname:" + found_name);
172  if (contents.find(tag) != string::npos)
173  continue;
174 
175  // name was found
176  found_names.insert(found_name);
177 
178  auto it = contents.begin();
179  auto match_it = m[1].first;
180  auto line_start = it;
181 
182  string::size_type line_number = 1;
183  for (; it != match_it; ++it)
184  {
185  if (string::traits_type::eq(*it, '\n'))
186  {
187  ++line_number;
188  line_start = it + 1; // could be end()
189  }
190  }
191 
192  ++m_errors;
193  error(library_name, full_path, string(name())
194  + " deprecated name ("
195  + found_name
196  + ") on line "
197  + linelink(full_path, boost::lexical_cast<string>(line_number))
198  + ", use " + m.format(d.data->use_instead)
199  + " instead");
200  }
201  }
202  }
203  }
204  }

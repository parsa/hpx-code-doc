113

123  p.instanceindex_ = 0;
124  status = get_counter_name(p, i.fullname_, ec);
125  if (!status_is_valid(status) || !f(i, ec) || ec)
126  return false;

129  return false;
130  }
131 
132  if (&ec != &hpx::throws)
133  ec = hpx::make_success_code();
134

139  // Creation function for explicit sine performance counter. It's purpose is
140  // to create and register a new instance of the given name (or reuse an
141  // existing instance).
142  hpx::naming::gid_type explicit_sine_counter_creator(
143  hpx::performance_counters::counter_info const& info, hpx::error_code& ec)
144  {
145  // verify the validity of the counter instance name
146  hpx::performance_counters::counter_path_elements paths;
147  get_counter_path_elements(info.fullname_, paths, ec);
148  if (ec) return hpx::naming::invalid_gid;
149 
150  if (paths.parentinstance_is_basename_) {
151  HPX_THROWS_IF(ec, hpx::bad_parameter,
152  "sine::explicit_sine_counter_creator",
153  "invalid counter instance parent name: " +
154  paths.parentinstancename_);
155  return hpx::naming::invalid_gid;
156  }
157 
158  // create individual counter
159  if (paths.instancename_ == "instance" && paths.instanceindex_ != -1) {
160  // make sure parent instance name is set properly
161  hpx::performance_counters::counter_info complemented_info = info;
162  complement_counter_info(complemented_info, info, ec);
163  if (ec) return hpx::naming::invalid_gid;
164 
165  // create the counter as requested
166  hpx::naming::gid_type id;
167  try {
168  // create the 'sine' performance counter component locally, we
169  // only get here if this instance does not exist yet
170  id = hpx::components::server::construct<sine_counter_type>(
171  complemented_info);
172  }
173  catch (hpx::exception const& e) {
174  if (&ec == &hpx::throws)
175  throw;
176  ec = make_error_code(e.get_error(), e.what());
177  return hpx::naming::invalid_gid;
178  }
179 
180  if (&ec != &hpx::throws)
181  ec = hpx::make_success_code();
182  return id;
183  }
184 
185  HPX_THROWS_IF(ec, hpx::bad_parameter,
186  "sine::explicit_sine_counter_creator",
187  "invalid counter instance name: " + paths.instancename_);
188  return hpx::naming::invalid_gid;
189  }

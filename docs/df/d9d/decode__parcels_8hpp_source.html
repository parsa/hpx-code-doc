36  template <typename Buffer>
37  std::vector<serialization::serialization_chunk> decode_chunks(Buffer & buffer)
38  {
39  typedef typename Buffer::transmission_chunk_type transmission_chunk_type;
40 
41  std::vector<serialization::serialization_chunk> chunks;
42 
43  std::size_t num_zero_copy_chunks =
44  static_cast<std::size_t>(
45  static_cast<std::uint32_t>(buffer.num_chunks_.first));
46 
47  if (num_zero_copy_chunks != 0)
48  {
49  // decode chunk information
50  std::size_t num_non_zero_copy_chunks =
51  static_cast<std::size_t>(
52  static_cast<std::uint32_t>(buffer.num_chunks_.second));
53 
54  chunks.resize(num_zero_copy_chunks + num_non_zero_copy_chunks);
55 
56  // place the zero-copy chunks at their spots first
57  for (std::size_t i = 0; i != num_zero_copy_chunks; ++i)
58  {
59  transmission_chunk_type& c = buffer.transmission_chunks_[i];
60  std::size_t first = static_cast<std::size_t>(
61  static_cast<std::uint64_t>(c.first));
62  std::size_t second = static_cast<std::size_t>(
63  static_cast<std::uint64_t>(c.second));
64 
65  HPX_ASSERT(buffer.chunks_[i].size() == second);
66 
67  chunks[first] = serialization::create_pointer_chunk(
68  buffer.chunks_[i].data(), second);
69  }
70 
71  std::size_t index = 0;
72  for (std::size_t i = num_zero_copy_chunks;
73  i != num_zero_copy_chunks + num_non_zero_copy_chunks;
74  ++i)
75  {
76  transmission_chunk_type& c = buffer.transmission_chunks_[i];
77  std::size_t first = static_cast<std::size_t>(
78  static_cast<std::uint64_t>(c.first));
79  std::size_t second = static_cast<std::size_t>(
80  static_cast<std::uint64_t>(c.second));
81 
82  // find next free entry
83  while (chunks[index].size_ != 0)
84  ++index;
85 
86  // place the index based chunk at the right spot
87  chunks[index] = serialization::create_index_chunk(first, second);
88  ++index;
89  }
90 #if defined(HPX_DEBUG)
91  // make sure that all spots have been populated
92  for (std::size_t i = 0;
93  i != num_zero_copy_chunks + num_non_zero_copy_chunks;
94  ++i)
95  {
96  HPX_ASSERT(chunks[i].size_ != 0);
97  }
98 #endif
99  }
100 
101  return chunks;
102  }

105  template <typename Parcelport, typename Buffer>
106  void decode_message_with_chunks(
107  Parcelport & pp
108  , Buffer buffer
109  , std::size_t parcel_count
110  , std::vector<serialization::serialization_chunk> &chunks
111  , std::size_t num_thread = -1
112  )
113  {
114  std::size_t inbound_data_size = static_cast<std::size_t>(
115  static_cast<std::uint64_t>(buffer.data_size_));
116 
117  // protect from un-handled exceptions bubbling up
118  try {
119  try {
120  // mark start of serialization
121  util::high_resolution_timer timer;
122  std::int64_t overall_add_parcel_time = 0;
123  performance_counters::parcels::data_point& data =
124  buffer.data_point_;
125 
126  {
127  std::vector<parcel> deferred_parcels;
128  // De-serialize the parcel data
129  serialization::input_archive archive(buffer.data_,
130  inbound_data_size, &chunks);
131 
132  if(parcel_count == 0)
133  {
134  archive >> parcel_count; //-V128
135  }
136  if (parcel_count > 1)
137  deferred_parcels.reserve(parcel_count);
138 
139  for(std::size_t i = 0; i != parcel_count; ++i)
140  {
141  bool deferred_schedule = parcel_count > 1;
142 
143 #if defined(HPX_HAVE_PARCELPORT_ACTION_COUNTERS)
144  std::size_t archive_pos = archive.current_pos();
145  std::int64_t serialize_time = timer.elapsed_nanoseconds();
146 #endif
147  // de-serialize parcel and add it to incoming parcel queue
148  parcel p;
149  // deferred_schedule will be set to false if the action
150  // to be loaded is a non direct action. If we only got
151  // one parcel to decode, deferred_schedule will be
152  // preset to false and the direct action will be called
153  // directly
154  bool migrated = p.load_schedule(archive, num_thread,
155  deferred_schedule);
156 
157  std::int64_t add_parcel_time = timer.elapsed_nanoseconds();
158 
159 #if defined(HPX_HAVE_PARCELPORT_ACTION_COUNTERS)
160  performance_counters::parcels::data_point action_data;
161  action_data.bytes_ = archive.current_pos() - archive_pos;
162  action_data.serialization_time_ =
163  add_parcel_time - serialize_time;
164  action_data.num_parcels_ = 1;
165  pp.add_received_data(p.get_action()->get_action_name(),
166  action_data);
167 #endif
168 
169  // make sure this parcel ended up on the right locality
170  naming::gid_type const& here = hpx::get_locality();
171  if (hpx::get_runtime_ptr() && here &&
172  (naming::get_locality_id_from_gid(
173  p.destination_locality()) !=
174  naming::get_locality_id_from_gid(here)))
175  {
176  std::ostringstream os;
177  os << "parcel destination does not match "
178  "locality which received the parcel ("
179  << here << "), " << p;
180  HPX_THROW_EXCEPTION(invalid_status,
181  "hpx::parcelset::decode_message",
182  os.str());
183  return;
184  }
185 
186  if (migrated)
187  {
188  naming::resolver_client& client =
189  hpx::naming::get_agas_client();
190  client.route(
191  std::move(p),
192  &parcelset::detail::parcel_route_handler,
193  threads::thread_priority_normal);
194  }
195  // If we got a direct action,
196  else if (deferred_schedule)
197  deferred_parcels.push_back(std::move(p));
198 
199  // be sure not to measure add_parcel as serialization time
200  overall_add_parcel_time += timer.elapsed_nanoseconds() -
201  add_parcel_time;
202  }
203 
204  // complete received data with parcel count
205  data.num_parcels_ = parcel_count;
206  data.raw_bytes_ = archive.bytes_read();
207 
208  if (!deferred_parcels.empty())
209  {
210  for (std::size_t i = 1; i != deferred_parcels.size(); ++i)
211  {
212  // schedule all but the first parcel on a new thread.
213  hpx::applier::register_thread_nullary(
214  util::bind(
215  util::one_shot(
216  [num_thread](parcel&& p)
217  {
218  p.schedule_action(num_thread);
219  }
220  ), std::move(deferred_parcels[i])),
221  "schedule_parcel",
222  threads::pending, true, threads::thread_priority_boost,
223  num_thread, threads::thread_stacksize_default);
224  }
225  // If we are the first deferred parcel, we don't need to spin
226  // a new thread...
227  deferred_parcels[0].schedule_action(num_thread);
228  }
229  }
230 
231  // store the time required for serialization
232  data.serialization_time_ = timer.elapsed_nanoseconds() -
233  overall_add_parcel_time;
234 
235  pp.add_received_data(data);
236  }
237  catch (hpx::exception const& e) {
238  LPT_(error)
239  << "decode_message: caught hpx::exception: "
240  << e.what();
241  hpx::report_error(std::current_exception());
242  }
243  catch (boost::system::system_error const& e) {
244  LPT_(error)
245  << "decode_message: caught boost::system::error: "
246  << e.what();
247  hpx::report_error(std::current_exception());
248  }
249  catch (boost::exception const&) {
250  LPT_(error)
251  << "decode_message: caught boost::exception.";
252  hpx::report_error(std::current_exception());
253  }
254  catch (std::exception const& e) {
255  // We have to repackage all exceptions thrown by the
256  // serialization library as otherwise we will loose the
257  // e.what() description of the problem, due to slicing.
258  hpx::throw_with_info(
259  hpx::exception(serialization_error, e.what()));
260  }
261  }
262  catch (...) {
263  LPT_(error)
264  << "decode_message: caught unknown exception.";
265  hpx::report_error(std::current_exception());
266  }
267  }

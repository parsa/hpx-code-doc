79  template <typename Buffer>
80  void encode_finalize(Buffer & buffer, std::size_t arg_size)
81  {
82  buffer.size_ = buffer.data_.size();
83  buffer.data_size_ = arg_size;
84 
85  LPT_(debug) << binary_archive_content(buffer);
86 
87  performance_counters::parcels::data_point& data = buffer.data_point_;
88  data.bytes_ = buffer.data_.size();
89  data.raw_bytes_ = arg_size;
90 
91  // prepare chunk data for transmission, the transmission_chunks data
92  // first holds all zero-copy, then all non-zero-copy chunk infos
93  typedef typename Buffer::transmission_chunk_type
94  transmission_chunk_type;
95  typedef typename Buffer::count_chunks_type
96  count_chunks_type;
97 
98  std::vector<transmission_chunk_type>& chunks =
99  buffer.transmission_chunks_;
100 
101  chunks.clear();
102  chunks.reserve(buffer.chunks_.size());
103 
104  std::size_t index = 0;
105  for (serialization::serialization_chunk& c : buffer.chunks_)
106  {
107  if (c.type_ == serialization::chunk_type_pointer)
108  chunks.push_back(transmission_chunk_type(index, c.size_));
109  ++index;
110  }
111 
112  buffer.num_chunks_ = count_chunks_type(
113  static_cast<std::uint32_t>(chunks.size()),
114  static_cast<std::uint32_t>(buffer.chunks_.size() - chunks.size())
115  );
116 
117  if (!chunks.empty()) {
118  // the remaining number of chunks are non-zero-copy
119  for (serialization::serialization_chunk& c : buffer.chunks_)
120  {
121  if (c.type_ == serialization::chunk_type_index) {
122  chunks.push_back(
123  transmission_chunk_type(c.data_.index_, c.size_));
124  }
125  }
126  }
127  }
128  }

130  template <typename Buffer>
131  std::size_t
132  encode_parcels(parcelport& pp,
133  parcel const * ps, std::size_t num_parcels, Buffer & buffer,
134  int archive_flags_, std::uint64_t max_outbound_size)
135  {
136  HPX_ASSERT(buffer.data_.empty());
137  // collect argument sizes from parcels
138  std::size_t num_chunks = 0;
139  std::size_t arg_size = 0;
140  std::size_t parcels_sent = 0;
141  std::size_t parcels_size = 1;
142 
143  if(num_parcels != std::size_t(-1))
144  {
145  arg_size = sizeof(std::int64_t);
146  parcels_size = num_parcels;
147  }
148 
149  // guard against serialization errors
150  try {
151  try {
152  std::unique_ptr<serialization::binary_filter> filter(
153  ps[0].get_serialization_filter());
154 
155  int archive_flags = archive_flags_;
156  if (filter.get() != nullptr)
157  archive_flags |= serialization::enable_compression;
158 
159 
160  // preallocate data
161  for (; parcels_sent != parcels_size; ++parcels_sent)
162  {
163  if (arg_size >= max_outbound_size)
164  break;
165  arg_size += ps[parcels_sent].size();
166  num_chunks += ps[parcels_sent].num_chunks();
167  }
168 
169  buffer.data_.reserve(arg_size);
170 
171  buffer.chunks_.reserve(num_chunks);
172 
173  // mark start of serialization
174  util::high_resolution_timer timer;
175 
176  {
177  // Serialize the data
178  if (filter.get() != nullptr)
179  filter->set_max_length(buffer.data_.capacity());
180 
181  serialization::output_archive archive(
182  buffer.data_
183  , archive_flags
184  , &buffer.chunks_
185  , filter.get());
186 
187  if(num_parcels != std::size_t(-1))
188  archive << parcels_sent; //-V128
189 
190  for(std::size_t i = 0; i != parcels_sent; ++i)
191  {
192 #if defined(HPX_HAVE_PARCELPORT_ACTION_COUNTERS)
193  std::size_t archive_pos = archive.current_pos();
194  std::int64_t serialize_time =
195  timer.elapsed_nanoseconds();
196 #endif
197 
198  LPT_(debug) << ps[i];
199  archive.set_split_gids(ps[i].split_gids());
200  archive << ps[i];
201 
202 #if defined(HPX_HAVE_PARCELPORT_ACTION_COUNTERS)
203  performance_counters::parcels::data_point action_data;
204  action_data.bytes_ = archive.current_pos() - archive_pos;
205  action_data.serialization_time_ =
206  timer.elapsed_nanoseconds() - serialize_time;
207  action_data.num_parcels_ = 1;
208  pp.add_sent_data(
209  ps[i].get_action()->get_action_name(),
210  action_data);
211 #endif
212  }
213  archive.flush();
214  arg_size = archive.bytes_written();
215  }
216 
217  // store the time required for serialization
218  buffer.data_point_.serialization_time_ =
219  timer.elapsed_nanoseconds();
220  }
221  catch (hpx::exception const& e) {
222  LPT_(fatal)
223  << "encode_parcels: "
224  "caught hpx::exception: "
225  << e.what();
226  hpx::report_error(std::current_exception());
227  return 0;
228  }

230  LPT_(fatal)
231  << "encode_parcels: "
232  "caught boost::system::error: "
233  << e.what();
234  hpx::report_error(std::current_exception());
235  return 0;
236  }
237  catch (boost::exception const&) {
238  LPT_(fatal)
239  << "encode_parcels: "
240  "caught boost::exception";
241  hpx::report_error(std::current_exception());
242  return 0;
243  }
244  catch (std::exception const& e) {
245  // We have to repackage all exceptions thrown by the
246  // serialization library as otherwise we will loose the
247  // e.what() description of the problem, due to slicing.
248  hpx::throw_with_info(
249  hpx::exception(serialization_error, e.what()));
250  return 0;
251  }
252  }
253  catch (...) {
254  LPT_(fatal)
255  << "encode_parcels: "
256  "caught unknown exception";
257  hpx::report_error(std::current_exception());
258  return 0;
259  }
260 
261  buffer.data_point_.num_parcels_ = parcels_sent;
262  detail::encode_finalize(buffer, arg_size);
263 
264  return parcels_sent;
265  }
266  }

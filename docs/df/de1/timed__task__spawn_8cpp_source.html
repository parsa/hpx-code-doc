293 void stage_workers(
294  std::uint64_t target_thread
295  , std::uint64_t local_tasks
296  , stage_worker_function stage_worker
297  )
298 {
299  std::uint64_t num_thread = hpx::get_worker_thread_num();
300 
301  if (num_thread != target_thread)
302  {
303  register_work(hpx::util::bind(&stage_workers
304  , target_thread
305  , local_tasks
306  , stage_worker)
307  , "stage_workers"
308  , hpx::threads::pending
309  , hpx::threads::thread_priority_normal
310  , target_thread
311  );
312  return;
313  }
314 
315  for (std::uint64_t i = 0; i < local_tasks;)
316  {
317  for (std::uint64_t j = 0; j < suspend_step; ++j)
318  {
319  stage_worker(target_thread, true);
320  ++i;
321  }
322  for (std::uint64_t j = 0; j < no_suspend_step; ++j)
323  {
324  stage_worker(target_thread, false);
325  ++i;
326  }
327  }

331 int hpx_main(
332  variables_map& vm
333  )
334 {
335  {
336  if (vm.count("no-header"))
337  header = false;
338 
339  if (vm.count("csv-header"))
340  csv_header = true;
341 
342  if (0 == tasks)
343  throw std::invalid_argument("count of 0 tasks specified\n");
344 
345  if (suspended_tasks > tasks)
346  throw std::invalid_argument(
347  "suspended tasks must be smaller than tasks\n");
348 
349  std::uint64_t const os_thread_count = get_os_thread_count();
350 
352  stage_worker_function stage_worker;
353 
354  if ("static-balanced-stackbased" == distribution)
355  stage_worker = &stage_worker_static_balanced_stackbased;
356  else if ("static-imbalanced" == distribution)
357  stage_worker = &stage_worker_static_imbalanced;
358  else if ("round-robin" == distribution)
359  stage_worker = &stage_worker_round_robin;
360  else
361  throw std::invalid_argument(
362  "invalid distribution type specified (valid options are "
363  "\"static-balanced\", \"static-imbalanced\" or \"round-robin\")"
364  );
365 
367  std::uint64_t tasks_per_feeder = 0;
368  //std::uint64_t total_tasks = 0;
369  std::uint64_t suspended_tasks_per_feeder = 0;
370  std::uint64_t total_suspended_tasks = 0;
371 
372  if ("strong" == scaling)
373  {
374  if (tasks % os_thread_count)
375  throw std::invalid_argument(
376  "tasks must be cleanly divisable by OS-thread count\n");
377 
378  if (suspended_tasks % os_thread_count)
379  throw std::invalid_argument(
380  "suspended tasks must be cleanly divisable by OS-thread "
381  "count\n");
382 
383  tasks_per_feeder = tasks / os_thread_count;
384  //total_tasks = tasks;
385  suspended_tasks_per_feeder = suspended_tasks / os_thread_count;
386  total_suspended_tasks = suspended_tasks;
387  }
388  else if ("weak" == scaling)
389  {
390  tasks_per_feeder = tasks;
391  //total_tasks = tasks * os_thread_count;
392  suspended_tasks_per_feeder = suspended_tasks;
393  total_suspended_tasks = suspended_tasks * os_thread_count;
394  }
395  else
396  throw std::invalid_argument(
397  "invalid scaling type specified (valid options are \"strong\" "
398  "or \"weak\")");
399 
401  if (suspended_tasks != 0)
402  {
403  std::uint64_t gcd = boost::math::gcd(tasks_per_feeder
404  , suspended_tasks_per_feeder);
405 
406  suspend_step = suspended_tasks_per_feeder / gcd;
407  // We check earlier to make sure that there are never more
408  // suspended tasks than tasks requested.
409  no_suspend_step = (tasks_per_feeder / gcd) - suspend_step;
410  }
411 
413  std::vector<std::string> counter_shortnames;
414  std::vector<std::string> counters;
415  if (vm.count("counter"))
416  {
417  std::vector<std::string> raw_counters =
418  vm["counter"].as<std::vector<std::string> >();
419 
420  for (std::uint64_t i = 0; i < raw_counters.size(); ++i)
421  {
422  std::vector<std::string> entry;
423  boost::algorithm::split(entry, raw_counters[i],
424  boost::algorithm::is_any_of(","),
425  boost::algorithm::token_compress_on);
426 
427  HPX_ASSERT(entry.size() == 2);
428 
429  counter_shortnames.push_back(entry[0]);
430  counters.push_back(entry[1]);
431  }
432  }
433 
434  std::shared_ptr<hpx::util::activate_counters> ac;
435  if (!counters.empty())
436  ac.reset(new hpx::util::activate_counters(counters));
437 
439  // Start the clock.
440  high_resolution_timer t;
441 
442  if (ac)
443  ac->reset_counters();
444 
445  // This needs to stay here; we may have suspended as recently as the
446  // performance counter reset (which is called just before the staging
447  // function).
448  std::uint64_t const num_thread = hpx::get_worker_thread_num();
449 
450  for (std::uint64_t i = 0; i < os_thread_count; ++i)
451  {
452  if (num_thread == i) continue;
453 
454  register_work(hpx::util::bind(&stage_workers
455  , i
456  , tasks_per_feeder
457  , stage_worker
458  )
459  , "stage_workers"
460  , hpx::threads::pending
461  , hpx::threads::thread_priority_normal
462  , i
463  );
464  }
465 
466  stage_workers(num_thread, tasks_per_feeder, stage_worker);
467 
468  double warmup_estimate = t.elapsed();
469 
470  // Schedule a low-priority thread; when it is executed, it checks to
471  // make sure all the tasks (which are normal priority) have been
472  // executed, and then it
473  hpx::lcos::local::barrier finished(2);
474 
475  register_work(hpx::util::bind(&wait_for_tasks
476  , std::ref(finished)
477  , total_suspended_tasks
478  )
479  , "wait_for_tasks", hpx::threads::pending
480  , hpx::threads::thread_priority_low);
481 
482  finished.wait();
483 
484  // Stop the clock
485  double time_elapsed = t.elapsed();
486 
487  print_results(os_thread_count, time_elapsed, warmup_estimate
488  , counter_shortnames, ac);
489  }
490 
491  if (suspended_tasks != 0)
492  // Force termination of all suspended tasks.
493  hpx::get_runtime().get_thread_manager().abort_all_suspended_threads();
494 
495  return finalize();
496 }

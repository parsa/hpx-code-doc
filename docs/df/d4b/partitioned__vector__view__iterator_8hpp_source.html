37  template<std::size_t... I>
38  std::size_t increment_solver( std::size_t dist,
39  hpx::util::detail::pack_c<std::size_t, I...> ) const
40  {
41  std::size_t max = N-1;
42  std::size_t offset = 0;
43  std::size_t carry = dist;
44  std::size_t tmp;
45 
46  // More expensive than a usual incrementation but did not find another
47  // solution
48  (void)std::initializer_list<int>
49  { ( static_cast<void>(
50  carry -= tmp = (carry/sw_basis_[max-I]) * sw_basis_[max-I],
51  offset += (tmp/sw_basis_[max-I]) * hw_basis_[max-I]),
52  0 )...
53  };
54 
55  return offset;
56  }

256  // Number masks:
257  // Vectors of non-negative integers
258  // Indicating which architecture object each PU belongs to.
259  // For example, numa_node_numbers[0] indicates which numa node
260  // number PU #0 (zero-based index) belongs to
261  std::vector<std::size_t> socket_numbers_;
262  std::vector<std::size_t> numa_node_numbers_;
263  std::vector<std::size_t> core_numbers_;
264 // std::vector<std::size_t> pu_numbers_; // (empty vector)
265 
266  // Affinity masks: vectors of bitmasks
267  // - Length of the vector: number of PUs of the machine
268  // - Elements of the vector:
269  // Bitmasks of length equal to the number of PUs of the machine.
270  // The bitmasks indicate which PUs belong to which resource.
271  // For example, core_affinity_masks[0] is a bitmask, where the
272  // elements = 1 indicate the PUs that belong to the core on which
273  // PU #0 (zero-based index) lies.
274  mask_type machine_affinity_mask_;
275  std::vector<mask_type> socket_affinity_masks_;
276  std::vector<mask_type> numa_node_affinity_masks_;

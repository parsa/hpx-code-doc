30 HPX_NOINLINE std::uint64_t fibonacci_serial(std::uint64_t n)
31 {
32  if (n < 2)
33  return n;
34  return fibonacci_serial(n-1) + fibonacci_serial(n-2);
35 }

38 hpx::future<std::uint64_t> fibonacci(std::uint64_t n)
39 {
40  if (n < 2) return hpx::make_ready_future(n);
41  if (n < threshold) return hpx::make_ready_future(fibonacci_serial(n));
42 
43  hpx::future<std::uint64_t> lhs_future = hpx::async(&fibonacci, n-1);
44  hpx::future<std::uint64_t> rhs_future = fibonacci(n-2);
45 
46  return
47  hpx::dataflow(
48  hpx::util::unwrapping(
49  [](std::uint64_t lhs, std::uint64_t rhs)
50  {
51  return lhs + rhs;
52  })
53  , std::move(lhs_future), std::move(rhs_future)
54  );
55 }

58 int hpx_main(boost::program_options::variables_map& vm)
59 {
60  // extract command line argument, i.e. fib(N)
61  std::uint64_t n = vm["n-value"].as<std::uint64_t>();
62  std::string test = vm["test"].as<std::string>();
63  std::uint64_t max_runs = vm["n-runs"].as<std::uint64_t>();
64 
65  if (max_runs == 0) {
66  std::cerr << "fibonacci_dataflow: wrong command line argument value for "
67  "option 'n-runs', should not be zero" << std::endl;
68  return hpx::finalize(); // Handles HPX shutdown
69  }
70 
71  threshold = vm["threshold"].as<unsigned int>();
72  if (threshold < 2 || threshold > n) {
73  std::cerr << "fibonacci_dataflow: wrong command line argument value for "
74  "option 'threshold', should be in between 2 and n-value"
75  ", value specified: " << threshold << std::endl;
76  return hpx::finalize(); // Handles HPX shutdown
77  }
78 
79  bool executed_one = false;
80  std::uint64_t r = 0;
81 
82  if (test == "all" || test == "0")
83  {
84  // Keep track of the time required to execute.
85  std::uint64_t start = hpx::util::high_resolution_clock::now();
86 
87  for (std::uint64_t i = 0; i != max_runs; ++i)
88  {
89  // serial execution
90  r = fibonacci_serial(n);
91  }
92 
93 // double d = double(hpx::util::high_resolution_clock::now() - start) / 1.e9;
94  std::uint64_t d = hpx::util::high_resolution_clock::now() - start;
95  char const* fmt = "fibonacci_serial(%1%) == %2%,"
96  "elapsed time:,%3%,[s]\n";
97  std::cout << (boost::format(fmt) % n % r % (d / max_runs));
98 
99  executed_one = true;
100  }
101 
102  if (test == "all" || test == "1")
103  {
104  // Keep track of the time required to execute.
105  std::uint64_t start = hpx::util::high_resolution_clock::now();
106 
107  for (std::uint64_t i = 0; i != max_runs; ++i)
108  {
109  // Create a future for the whole calculation, execute it locally,
110  // and wait for it.
111  r = fibonacci(n).get();
112  }
113 
114 // double d = double(hpx::util::high_resolution_clock::now() - start) / 1.e9;
115  std::uint64_t d = hpx::util::high_resolution_clock::now() - start;
116  char const* fmt = "fibonacci_await(%1%) == %2%,"
117  "elapsed time:,%3%,[s]\n";
118  std::cout << (boost::format(fmt) % n % r % (d / max_runs));
119 
120  executed_one = true;
121  }
122 
123  if (!executed_one)
124  {
125  std::cerr << "fibonacci_dataflow: wrong command line argument value for "
126  "option 'tests', should be either 'all' or a number between zero "
127  "and 1, value specified: " << test << std::endl;
128  }
129 
130  return hpx::finalize(); // Handles HPX shutdown
131 }

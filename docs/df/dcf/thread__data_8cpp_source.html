97  bool thread_data::interruption_point(bool throw_on_interrupt)
98  {
99  // We do not protect enabled_interrupt_ and requested_interrupt_
100  // from concurrent access here (which creates a benign data race) in
101  // order to avoid infinite recursion. This function is called by
102  // this_thread::suspend which causes problems if the lock would call
103  // suspend itself.
104  if (enabled_interrupt_ && requested_interrupt_)
105  {
106  // Verify that there are no more registered locks for this
107  // OS-thread. This will throw if there are still any locks
108  // held.
109  util::force_error_on_lock();
110 
111  // now interrupt this thread
112  if (throw_on_interrupt)
113  throw hpx::thread_interrupted();
114 
115  return true;
116  }
117  return false;
118  }

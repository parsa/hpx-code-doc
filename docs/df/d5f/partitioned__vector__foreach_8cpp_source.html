49 template <typename Policy, typename Vector>
50 std::uint64_t foreach_vector(Policy && policy, Vector const& v)
51 {
52  typedef typename Vector::value_type value_type;
53 
54  std::uint64_t start = hpx::util::high_resolution_clock::now();
55 
56  for (int i = 0; i != test_count; ++i)
57  {
58  hpx::parallel::for_each(
59  std::forward<Policy>(policy),
60  std::begin(v), std::end(v), wait_op<Vector>()
61  );
62  }
63 
64  return (hpx::util::high_resolution_clock::now() - start) / test_count;
65 }

40  // Unrolled loops for loading and storing streams of bytes.
41 
42  template <typename T, std::size_t n_bytes,
43  bool sign = std::is_signed<T>::value>
44  struct unrolled_byte_loops
45  {
46  typedef unrolled_byte_loops<T, n_bytes - 1, sign> next;
47 
48  static T load_big(const unsigned char* bytes)
49  { return *(bytes - 1) | (next::load_big(bytes - 1) << 8); }
50  static T load_little(const unsigned char* bytes)
51  { return *bytes | (next::load_little(bytes + 1) << 8); }
52 
53  static void store_big(char* bytes, T value)
54  {
55  *(bytes - 1) = static_cast<char>(value);
56  next::store_big(bytes - 1, value >> 8);
57  }
58  static void store_little(char* bytes, T value)
59  {
60  *bytes = static_cast<char>(value);
61  next::store_little(bytes + 1, value >> 8);
62  }
63  };
64 
65  template <typename T>
66  struct unrolled_byte_loops<T, 1, false>
67  {
68  static T load_big(const unsigned char* bytes)
69  { return *(bytes - 1); }
70  static T load_little(const unsigned char* bytes)
71  { return *bytes; }
72  static void store_big(char* bytes, T value)
73  { *(bytes - 1) = static_cast<char>(value); }
74  static void store_little(char* bytes, T value)
75  { *bytes = static_cast<char>(value); }
76 
77  };
78 
79  template <typename T>
80  struct unrolled_byte_loops<T, 1, true>
81  {
82  static T load_big(const unsigned char* bytes)
83  { return *reinterpret_cast<const signed char*>(bytes - 1); }
84  static T load_little(const unsigned char* bytes)
85  { return *reinterpret_cast<const signed char*>(bytes); }
86  static void store_big(char* bytes, T value)
87  { *(bytes - 1) = static_cast<char>(value); }
88  static void store_little(char* bytes, T value)
89  { *bytes = static_cast<char>(value); }
90  };
91 
92  template <typename T, std::size_t n_bytes>
93  inline
94  T load_big_endian(const void* bytes)
95  {
96  return unrolled_byte_loops<T, n_bytes>::load_big
97  (static_cast<const unsigned char*>(bytes) + n_bytes);
98  }
99 
100  template <typename T, std::size_t n_bytes>
101  inline
102  T load_little_endian(const void* bytes)
103  {
104  return unrolled_byte_loops<T, n_bytes>::load_little
105  (static_cast<const unsigned char*>(bytes));
106  }
107 
108  template <typename T, std::size_t n_bytes>
109  inline
110  void store_big_endian(void* bytes, T value)
111  {
112  unrolled_byte_loops<T, n_bytes>::store_big
113  (static_cast<char*>(bytes) + n_bytes, value);
114  }

117  inline
118  void store_little_endian(void* bytes, T value)
119  {

140  // Taking an integer type as a parameter provides a nice way to pass both

144  template <typename T, std::size_t n_bits>
145  class endian< big, T, n_bits, unaligned >
146  : cover_operators< endian< big, T, n_bits >, T >
147  {
148  static_assert((n_bits/8)*8 == n_bits, "(n_bits/8)*8 == n_bits");
149  public:
150  typedef T value_type;
151  endian() {}
152  endian(T i) { detail::store_big_endian<T, n_bits/8>(bytes, i); }
153  operator T() const
154  { return detail::load_big_endian<T, n_bits/8>(bytes); }
155  private:

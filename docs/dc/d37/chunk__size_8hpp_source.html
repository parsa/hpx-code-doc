72  HPX_ASSERT(0 != max_chunks);

74  FwdIter last = begin;
75  std::advance(last, count);
76 
77  Stride stride = parallel::v1::detail::abs(s);
78  auto test_function =
79  [&]() -> std::size_t
80  {
81  std::size_t test_chunk_size = count / 100;
82  if (test_chunk_size == 0)
83  return 0;
84 
85  if (stride != 1)
86  {
87  // rounding up
88  test_chunk_size = (std::max)(std::size_t(stride),
89  ((test_chunk_size + stride - 1) / stride) * stride);
90  }
91 
92  add_ready_future(workitems, f1, begin, test_chunk_size);
93 
94  // modifies 'test_chunk_size'
95  begin = parallel::v1::detail::next(

97 
98  count -= test_chunk_size;
99  return test_chunk_size;

103  traits::get_chunk_size(policy.parameters(),
104  policy.executor(), test_function, cores, count);
105 
106  // we should not consider more chunks than we have elements
107  max_chunks = (std::min)(max_chunks, count);
108 
109  // we should not make chunks smaller than what's determined by the
110  // max chunk size
111  chunk_size = (std::max)(chunk_size,
112  (count + max_chunks - 1) / max_chunks);

70  template <typename ExPolicy, typename RanIter1, typename RanIter2,
71  typename FwdIter, typename F, typename Combiner, typename SetOp>
72  typename util::detail::algorithm_result<ExPolicy, FwdIter>::type
73  set_operation(ExPolicy policy,
74  RanIter1 first1, RanIter1 last1, RanIter2 first2, RanIter2 last2,
75  FwdIter dest, F && f, Combiner && combiner, SetOp && setop)
76  {
77  typedef typename std::iterator_traits<RanIter1>::difference_type
78  difference_type1;
79  typedef typename std::iterator_traits<RanIter2>::difference_type
80  difference_type2;
81 
82  // allocate intermediate buffers
83  difference_type1 len1 = std::distance(first1, last1);
84  difference_type2 len2 = std::distance(first2, last2);
85 
86  typedef typename set_operations_buffer<FwdIter>::type buffer_type;
87  boost::shared_array<buffer_type> buffer(
88  new buffer_type[combiner(len1, len2)]);
89 
90  std::size_t cores = execution::processing_units_count(
91  policy.executor(), policy.parameters());
92 
93  std::size_t step = (len1 + cores - 1) / cores;
94  boost::shared_array<set_chunk_data> chunks(new set_chunk_data[cores]);
95 
96  // fill the buffer piecewise
97  return parallel::util::partitioner<ExPolicy, FwdIter, void>::call(
98  policy, chunks.get(), cores,
99  // first step, is applied to all partitions
100  [=](set_chunk_data* curr_chunk, std::size_t part_size) -> void
101  {
102  HPX_ASSERT(part_size == 1);
103 
104  // find start in sequence 1
105  std::size_t start1 = (curr_chunk - chunks.get()) * step;
106  std::size_t end1 = (std::min)(start1 + step, std::size_t(len1));
107 
108  bool first_partition = (start1 == 0);
109  bool last_partition = (end1 == std::size_t(len1));
110 
111  // all but the last chunk require special handling
112  if (!last_partition)
113  {
114  // this chunk will be handled by the next one if all
115  // elements of this partition are equal
116  if (!f(first1[start1], first1[end1 + 1]))
117  return;
118 
119  // move backwards to find earliest element which is equal to
120  // the last element of the current chunk
121  while (end1 != 0 && !f(first1[end1 - 1], first1[end1]))
122  --end1;
123  }
124 
125  // move backwards to find earliest element which is equal to
126  // the first element of the current chunk
127  while (start1 != 0 && !f(first1[start1 - 1], first1[start1]))
128  --start1;
129 
130  // find start and end in sequence 2
131  std::size_t start2 = 0;
132  if (!first_partition)
133  {
134  start2 =
135  std::lower_bound(
136  first2, first2 + len2, first1[start1], f
137  ) - first2;
138  }
139 
140  std::size_t end2 = len2;
141  if (!last_partition)
142  {
143  end2 =
144  std::lower_bound(
145  first2 + start2, first2 + len2, first1[end1], f
146  ) - first2;
147  }
148 
149  // perform requested set-operation into the proper place of the
150  // intermediate buffer
151  curr_chunk->start = combiner(start1, start2);
152  auto buffer_dest = buffer.get() + curr_chunk->start;
153  curr_chunk->len =
154  setop(first1 + start1, first1 + end1,
155  first2 + start2, first2 + end2, buffer_dest, f
156  ) - buffer_dest;
157  },
158  // second step, is executed after all partitions are done running
159  [buffer, chunks, cores, dest](std::vector<future<void> >&&) -> FwdIter
160  {
161  // accumulate real length
162  set_chunk_data* chunk = chunks.get();
163  chunk->start_index = 0;
164  for (size_t i = 1; i != cores; ++i)
165  {
166  set_chunk_data* curr_chunk = chunk++;
167  chunk->start_index =
168  curr_chunk->start_index + curr_chunk->len;
169  }
170 
171  // finally, copy data to destination
172  parallel::util::foreach_partitioner<
173  hpx::parallel::execution::parallel_policy
174  >::call(execution::par, chunks.get(), cores,
175  [buffer, dest](
176  set_chunk_data* chunk, std::size_t, std::size_t)
177  {
178  std::copy(buffer.get() + chunk->start,
179  buffer.get() + chunk->start + chunk->len,
180  dest + chunk->start_index);
181  },
182  [](set_chunk_data* last) -> set_chunk_data*
183  {
184  return last;
185  });
186 
187  return dest;
188  });
189  }

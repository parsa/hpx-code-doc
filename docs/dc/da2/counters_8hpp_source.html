448  template <typename T>
449  T get_value(error_code& ec = throws) const
450  {
451  if (!status_is_valid(status_)) {
452  HPX_THROWS_IF(ec, invalid_status,
453  "counter_value::get_value<T>",
454  "counter value is in invalid status");
455  return T();
456  }
457 
458  T val = static_cast<T>(value_);
459 
460  if (scaling_ != 1) {
461  if (scaling_ == 0) {
462  HPX_THROWS_IF(ec, uninitialized_value,
463  "counter_value::get_value<T>",
464  "scaling should not be zero");
465  return T();
466  }
467 
468  // calculate and return the real counter value
469  if (scale_inverse_)
470  return val / static_cast<T>(scaling_);
471 
472  return val * static_cast<T>(scaling_);
473  }
474  return val;
475  }

95  naming::id_type symbol_namespace::symbol_namespace_locality(std::string const& key)
96  {
97  std::uint32_t hash_value = 0;
98  if (key.size() < 2 || key[1] != '0' || key[0] != '/')
99  {
100  // keys starting with '/0' have to go to node 0
101  util::jenkins_hash hash;
102  hash_value = hash(key) % get_initial_num_localities();
103  }
104  return naming::id_type(get_service_instance(hash_value),
105  naming::id_type::unmanaged);
106  }

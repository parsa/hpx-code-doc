109  // ---------------------------------------------------------------------------
110  int poll_completion(struct ibv_wc *completion)
111  {
112  int nc = ibv_poll_cq(completionQ_, 1, completion);
113  if (nc < 0) {
114  rdma_error e(EINVAL, "ibv_poll_cq() failed");
115  LOG_ERROR_MSG(decnumber(completionQ_->handle)
116  << "error polling completion queue: "
117  << rdma_error::error_string(e.error_code()));
118  throw e;
119  }
120  if (nc > 0) {
121  if (completion->status != IBV_WC_SUCCESS) {
122  // if the completion was from a flushed queue pair
123  // then it was not a fatal error
124  if (completion->status == IBV_WC_WR_FLUSH_ERR) {
125  LOG_DEVEL_MSG("Received a flushed work completion on qp "
126  << decnumber(completion->qp_num));
127  return -1;
128  }
129  LOG_ERROR_MSG(
130  "CQ " << decnumber(completionQ_->handle)
131  << "work completion status '"
132  << ibv_wc_status_str(completion->status)
133  << "' for operation " << wc_opcode_str(completion->opcode)
134  << " (" << completion->opcode << ") "
135  << hexpointer(completion->wr_id)
136  << "qpnum " << decnumber(completion->qp_num)
137  );
138  std::terminate();
139  }
140  else {
141  LOG_TRACE_MSG(
142  "CQ " << decnumber(completionQ_->handle)
143  << "work completion status '"
144  << ibv_wc_status_str(completion->status)
145  << "' for operation " << wc_opcode_str(completion->opcode)
146  << " (" << completion->opcode << ")");
147  }
148 
149  LOG_TRACE_MSG(
150  "CQ " << decnumber(completionQ_->handle)
151  << "removing " << hexpointer(completion->wr_id)
152  << verbs_completion_queue::wc_opcode_str(completion->opcode));
153  }
154  return nc;
155  }

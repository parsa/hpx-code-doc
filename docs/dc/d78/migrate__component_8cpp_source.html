126 bool test_migrate_component(hpx::id_type source, hpx::id_type target)
127 {
128  // create component on given locality
129  test_client t1 = hpx::new_<test_client>(source, 42);
130  HPX_TEST_NEQ(hpx::naming::invalid_id, t1.get_id());
131 
132  // the new object should live on the source locality
133  HPX_TEST_EQ(t1.call(), source);
134  HPX_TEST_EQ(t1.get_data(), 42);
135 
136  try {
137  // migrate t1 to the target
138  test_client t2(hpx::components::migrate(t1, target));
139 
140  // wait for migration to be done
141  HPX_TEST_NEQ(hpx::naming::invalid_id, t2.get_id());
142 
143  // the migrated object should have the same id as before
144  HPX_TEST_EQ(t1.get_id(), t2.get_id());
145 
146  // the migrated object should live on the target now
147  HPX_TEST_EQ(t2.call(), target);
148  HPX_TEST_EQ(t2.get_data(), 42);
149  }
150  catch (hpx::exception const& e) {
151  hpx::cout << hpx::get_error_what(e) << std::endl;
152  return false;
153  }
154 
155  return true;
156 }

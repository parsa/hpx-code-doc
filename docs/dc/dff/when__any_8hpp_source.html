345  {
346  // set callback functions to executed when future is ready
347  set_on_completed_callback(*this);
348 
349  // if one of the requested futures is already set, our
350  // callback above has already been called often enough, otherwise
351  // we suspend ourselves
352  if (!goal_reached_on_calling_thread_)

357  }

360  HPX_ASSERT(index_.load() != when_any_result<Sequence>::index_error());
361

364  }

603  template <typename Func>
604  size_type erase(Func const& ep = policies::always<storage_value_type>())
605  {
606  update_on_exit update(statistics_, statistics::method_erase_entry);
607 
608  size_type erased = 0;
609  for (heap_iterator it = entry_heap_.begin();
610  it != entry_heap_.end(); )
611  {
612  iterator sit = *it;
613 
614  // check if this item needs to be erased
615  // do not remove this entry from the cache if either the
616  // function object or the entries' remove function return false
617  typename storage_type::value_type& val = *sit;
618  if (ep(val) && val.second.remove()) {
619  // update the current size and the overall size of the
620  // removed items
621  size_type entry_size = (*(*it)).second.get_size();
622  current_size_ -= entry_size;
623  erased += entry_size;
624 
625  // we remove the element manually, forcing the heap to be
626  // rebuilt at the end
627  it = entry_heap_.erase(it);
628 
629  // remove the cache entry
630  store_.erase(sit);
631 
632  // update statistics
633  statistics_.got_eviction();
634  }
635  else {
636  // do not remove this item from cache
637  ++it;
638  }
639  }
640 
641  // reorder heap based on the changed entry list
642  std::make_heap(entry_heap_.begin(), entry_heap_.end(),
643  update_policy_);

693  protected:
695  // Free some space in the cache
696  bool free_space (long num_free)
697  {
698  if (entry_heap_.empty())
699  return false;
700 
701  bool is_heap = true;
702  for (heap_iterator it = entry_heap_.begin();
703  num_free > 0 && it != entry_heap_.end(); )
704  {
705  iterator sit = *it;
706  if (!(*sit).second.remove()) {
707  ++it; // do not remove this entry from the cache
708  }
709  else {
710  size_type entry_size = (*(*it)).second.get_size();
711 
712  if (it == entry_heap_.begin()) {
713  // if we're at the top of the list, just pop the item
714  // this doesn't disturb the heap property of the heap
715  ++it;
716  std::pop_heap(entry_heap_.begin(), entry_heap_.end(),
717  update_policy_);
718  entry_heap_.pop_back();
719  }
720  else {
721  // otherwise we remove the element manually, forcing
722  // the heap to be rebuilt at the end
723  it = entry_heap_.erase(it);
724  is_heap = false;
725  }
726 
727  // remove the cache entry
728  store_.erase(sit);
729  num_free -= static_cast<long>(entry_size);
730  current_size_ -= entry_size;
731 
732  // update statistics
733  statistics_.got_eviction();
734  }
735  }
736 
737  // reorder heap based on the changed entry list
738  if (!is_heap) {
739  std::make_heap(entry_heap_.begin(), entry_heap_.end(),
740  update_policy_);
741  }
742

207  {

212  void on_ready(
213  std::size_t i,
214  typename traits::detail::shared_state_ptr_for<T>::type const&
215  state)
216  {
217  try {
218  typedef typename traits::future_traits<T>::type result_type;
219  result_type* result = state->get_result();
220  this->base_type::set_value(std::move((*result)[i]));

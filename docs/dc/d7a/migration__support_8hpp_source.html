66  // Pinning functionality
67  void pin()
68  {
69  std::lock_guard<mutex_type> l(mtx_);
70  HPX_ASSERT(pin_count_ != ~0x0u);
71  if (pin_count_ != ~0x0u)
72  ++pin_count_;
73  }
74  void unpin()
75  {
76  // make sure to always grab the AGAS lock first
77  agas::mark_as_migrated(this->gid_,
78  [this]() mutable -> std::pair<bool, hpx::future<void> >
79  {
80  std::unique_lock<mutex_type> l(mtx_);
81  HPX_ASSERT(this->pin_count_ != 0);
82  if (this->pin_count_ != ~0x0u)
83  {
84  if (--this->pin_count_ == 0)
85  {
86  // trigger pending migration if this was the last
87  // unpin and a migration operation is pending
88  if (trigger_migration_.valid() && was_marked_for_migration_)
89  {
90  was_marked_for_migration_ = false;
91 
92  hpx::lcos::local::promise<void> p;
93  std::swap(p, trigger_migration_);
94 
95  l.unlock();
96 
97  p.set_value();
98  return std::make_pair(true, make_ready_future());
99  }
100  }
101  }
102  return std::make_pair(false, make_ready_future());
103  });

118  hpx::future<void> mark_as_migrated(hpx::id_type const& to_migrate)
119  {
120  // we need to first lock the AGAS migrated objects table, only then
121  // access (lock) the object
122  return agas::mark_as_migrated(to_migrate.get_gid(),
123  [this]() mutable -> std::pair<bool, hpx::future<void> >
124  {
125  std::unique_lock<mutex_type> l(mtx_);
126 
127  // make sure that no migration is currently in flight
128  if (was_marked_for_migration_)
129  {
130  l.unlock();
131  return std::make_pair(false,
132  hpx::make_exceptional_future<void>(
133  HPX_GET_EXCEPTION(invalid_status,
134  "migration_support::mark_as_migrated",
135  "migration operation is already in flight")
136  ));
137  }
138 
139  if (1 == pin_count_)
140  {
141  // all is well, migration can be triggered now
142  return std::make_pair(true, make_ready_future());
143  }
144 
145  // delay migrate operation until pin count goes to zero
146  was_marked_for_migration_ = true;
147  hpx::future<void> f = trigger_migration_.get_future();
148 
149  l.unlock();
150  return std::make_pair(true, std::move(f));
151  });

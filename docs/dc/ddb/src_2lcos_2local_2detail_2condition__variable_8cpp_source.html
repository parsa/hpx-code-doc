105  void condition_variable::notify_all(
106  std::unique_lock<mutex_type> lock, threads::thread_priority priority,
107  error_code& ec)
108  {
109  HPX_ASSERT(lock.owns_lock());
110 
111  // swap the list
112  queue_type queue;
113  queue.swap(queue_);
114 
115  if (!queue.empty())
116  {
117  // update reference to queue for all queue entries
118  for (queue_entry& qe : queue)
119  qe.q_ = &queue;
120 
121  do {
122  threads::thread_id_repr_type id = queue.front().id_;
123 
124  // remove item from queue before error handling
125  queue.front().id_ = threads::invalid_thread_id_repr;
126  queue.pop_front();
127 
128  if (HPX_UNLIKELY(id == threads::invalid_thread_id_repr))
129  {
130  prepend_entries(lock, queue);
131  lock.unlock();
132 
133  HPX_THROWS_IF(ec, null_thread_id,
134  "condition_variable::notify_all",
135  "null thread id encountered");
136  return;
137  }
138 
139  error_code local_ec;
140  {
141  util::ignore_while_checking<std::unique_lock<mutex_type> > il(&lock);
142  threads::set_thread_state(threads::thread_id_type(
143  reinterpret_cast<threads::thread_data*>(id)),
144  threads::pending, threads::wait_signaled, priority, local_ec);
145  }
146 
147  if (local_ec)
148  {
149  prepend_entries(lock, queue);
150  lock.unlock();
151 
152  if (&ec != &throws)
153  {
154  ec = std::move(local_ec);
155  }
156  else
157  {
158  std::rethrow_exception(
159  hpx::detail::access_exception(local_ec));
160  }
161  return;
162  }
163 
164  } while (!queue.empty());
165  }
166

168  ec = make_success_code();

179  std::unique_lock<mutex_type>& lock,
180  char const* description, error_code& ec)
181  {
182  HPX_ASSERT(threads::get_self_ptr() != nullptr);
183  HPX_ASSERT(lock.owns_lock());
184 
185  // enqueue the request and block this thread
186  queue_entry f(threads::get_self_id().get(), &queue_);
187  queue_.push_back(f);
188 
189  reset_queue_entry r(f, queue_);
190  threads::thread_state_ex_enum reason = threads::wait_unknown;
191  {
192  // yield this thread
193  util::unlock_guard<std::unique_lock<mutex_type> > ul(lock);
194  reason = this_thread::suspend(threads::suspended, description, ec);
195  if (ec) return threads::wait_unknown;
196  }
197 
198  return (f.id_ != threads::invalid_thread_id_repr) ?
199  threads::wait_timeout : reason;

229  {

236

240

125  inline int make_context(::ucontext_t* ctx, void* stack, std::ptrdiff_t size,
126  void (*startfunc)(void*), void* startarg,
127  ::ucontext_t* exitto = nullptr)
128  {
129  int error = ::getcontext(ctx);
130  if (error)
131  return error;
132 
133  ctx->uc_stack.ss_sp = (char*)stack;
134  ctx->uc_stack.ss_size = size;
135  ctx->uc_link = exitto;
136 
137  typedef void (*ctx_main)();
138  //makecontext can't fail.
139  ::makecontext(ctx,
140  (ctx_main)(startfunc),
141  1,
142  startarg);
143  return 0;
144  }

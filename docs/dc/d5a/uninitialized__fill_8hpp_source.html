84  typename util::detail::algorithm_result<ExPolicy>::type
85  parallel_sequential_uninitialized_fill_n(ExPolicy && policy,
86  Iter first, std::size_t count, T const& value)
87  {

89  return util::detail::algorithm_result<ExPolicy>::get();

101  mutable -> partition_result_type
102  {
103  return std::make_pair(it,

113  // didn't fail, but only if at least one failed
114  [](partition_result_type && r) -> void
115  {
116  while (r.first != r.second)
117  {
118  (*r.first).~value_type();

120  }

28  // mix -- mix 3 32-bit values reversibly.
29  //
30  // This is reversible, so any information in (a,b,c) before mix() is
31  // still in (a,b,c) after mix().
32  //
33  // If four pairs of (a,b,c) inputs are run through mix(), or through
34  // mix() in reverse, there are at least 32 bits of the output that
35  // are sometimes the same for one pair and different for another pair.
36  // This was tested for:
37  // * pairs that differed by one bit, by two bits, in any combination
38  // of top bits of (a,b,c), or in any combination of bottom bits of
39  // (a,b,c).
40  // * "differ" is defined as +, -, ^, or ~^. For + and -, I transformed
41  // the output delta to a Gray code (a^(a>>1)) so a string of 1's (as
42  // is commonly produced by subtraction) look like a single 1-bit
43  // difference.
44  // * the base values were pseudorandom, all zero but one bit set, or
45  // all zero plus a counter that starts at zero.
46  //
47  // Some k values for my "a-=c; a^=rot(c,k); c+=b;" arrangement that
48  // satisfy this are
49  // 4 6 8 16 19 4
50  // 9 15 3 18 27 15
51  // 14 9 3 7 17 3
52  // Well, "9 15 3 18 27 15" didn't quite get 32 bits diffing
53  // for "differ" defined as + with a one-bit base and a two-bit delta. I
54  // used http://burtleburtle.net/bob/hash/avalanche.html to choose
55  // the operations, constants, and arrangements of the variables.
56  //
57  // This does not achieve avalanche. There are input bits of (a,b,c)
58  // that fail to affect some output bits of (a,b,c), especially of a. The
59  // most thoroughly mixed value is c, but it doesn't really even achieve
60  // avalanche in c.
61  //
62  // This allows some parallelism. Read-after-writes are good at doubling
63  // the number of bits affected, so the goal of mixing pulls in the opposite
64  // direction as the goal of parallelism. I did what I could. Rotates
65  // seem to cost as much as shifts on every machine I could lay my hands
66  // on, and rotates are much kinder to the top and bottom bits, so I used
67  // rotates.
68  // ------------------------------------------------------------------------
69  template <typename T>
70  inline
71  void mix(T& a, T& b, T& c)
72  {
73  a -= b; a -= c; a ^= (c >> 13);
74  b -= c; b -= a; b ^= (a << 8);
75  c -= a; c -= b; c ^= (b >> 13);
76  a -= b; a -= c; a ^= (c >> 12);
77  b -= c; b -= a; b ^= (a << 16);
78  c -= a; c -= b; c ^= (b >> 5);
79  a -= b; a -= c; a ^= (c >> 3);
80  b -= c; b -= a; b ^= (a << 10);
81  c -= a; c -= b; c ^= (b >> 15);
82  }
83  }

142  // hash() -- hash a variable-length key into a 32-bit value
143  // k : the key (the unaligned variable-length array of bytes)
144  // len : the length of the key, counting by bytes
145  // Returns a 32-bit value. Every bit of the key affects every bit of
146  // the return value. Every 1-bit and 2-bit delta achieves avalanche.
147  // About 6*len+35 instructions.
148  //
149  // The best hash table sizes are powers of 2. There is no need to do
150  // mod a prime (mod is sooo slow!). If you need less than 32 bits,
151  // use a bitmask. For example, if you need only 10 bits, do
152  // h = (h & hashmask(10));
153  // In which case, the hash table should have hashsize(10) elements.
154  //
155  // If you are hashing n strings (cmph_uint8 **)k, do it like this:
156  // for (i=0, h=0; i<n; ++i) h = hash( k[i], len[i], h);
157  //
158  // By Bob Jenkins, 1996. bob_jenkins@burtleburtle.net. You may use this
159  // code any way you wish, private, educational, or commercial. It's free.
160  //
161  // See http://burtleburtle.net/bob/hash/evahash.html
162  // Use for hash table lookup, or anything where one collision in 2^^32 is
163  // acceptable. Do NOT use for cryptographic purposes.
164  size_type hash (const char *k, std::size_t length) const
165  {
166  size_type a, b, c;
167  std::size_t len = length;
168 
169  /* Set up the internal state */
170  a = b = 0x9e3779b9; /* the golden ratio; an arbitrary value */
171  c = seed_; /* the previous hash value - seed in our case */
172 
173  /*---------------------------------------- handle most of the key */
174  while (len >= 12)
175  {
176  a += (k[0] + ((size_type)k[1] << 8)
177  + ((size_type)k[2] << 16)
178  + ((size_type)k[3] << 24));
179  b += (k[4] + ((size_type)k[5] << 8)
180  + ((size_type)k[6] << 16)
181  + ((size_type)k[7] << 24));
182  c += (k[8] + ((size_type)k[9] << 8)
183  + ((size_type)k[10] << 16)
184  + ((size_type)k[11] << 24));
185  detail::mix(a, b, c);
186  k += 12;
187  len -= 12;
188  }
189 
190  /*------------------------------------- handle the last 11 bytes */
191  c += (size_type)length;
192  switch(len) /* all the case statements fall through */
193  {
194  case 11:
195  c += ((size_type)k[10] << 24);
196  HPX_FALLTHROUGH;
197  case 10:
198  c += ((size_type)k[9] << 16);
199  HPX_FALLTHROUGH;
200  case 9:
201  c += ((size_type)k[8] << 8);
202  HPX_FALLTHROUGH;
203  /* the first byte of c is reserved for the length */
204  case 8:
205  b += ((size_type)k[7] << 24);
206  HPX_FALLTHROUGH;
207  case 7:
208  b += ((size_type)k[6] << 16);
209  HPX_FALLTHROUGH;
210  case 6:
211  b += ((size_type)k[5] << 8);
212  HPX_FALLTHROUGH;
213  case 5:
214  b += k[4];
215  HPX_FALLTHROUGH;
216  case 4:
217  a += ((size_type)k[3] << 24);
218  HPX_FALLTHROUGH;
219  case 3:
220  a += ((size_type)k[2] << 16);
221  HPX_FALLTHROUGH;
222  case 2:
223  a += ((size_type)k[1] << 8);
224  HPX_FALLTHROUGH;
225  case 1:
226  a += k[0];
227  /* case 0: nothing left to add */
228  }
229 
230  detail::mix(a, b, c);
231  return c; /* report the result */
232  }

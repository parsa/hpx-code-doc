28  T const& val)

32  HPX_TEST(traits::is_segmented_iterator::value);
33 
34  typedef typename hpx::partitioned_vector<T>::const_iterator const_iterator;
35  typedef hpx::traits::segmented_iterator_traits<const_iterator> const_traits;
36  HPX_TEST(const_traits::is_segmented_iterator::value);
37 
38  HPX_TEST_EQ(v.size(), size);
39  for(std::size_t i = 0; i != size; ++i)
40  {
41  HPX_TEST_EQ(v[i], val);
42  v[i] = T(i+1);
43  HPX_TEST_EQ(v[i], T(i+1));
44  }
45 
46  // test normal iteration
47  std::size_t count = 0;
48  std::size_t i = 42;
49  for (iterator it = v.begin(); it != v.end(); ++it, ++i, ++count)
50  {
51  HPX_TEST_NEQ(*it, val);
52  *it = T(i);
53  HPX_TEST_EQ(*it, T(i));
54  }
55  HPX_TEST_EQ(count, size);
56 
57  count = 0;
58  i = 42;
59  for (const_iterator cit = v.cbegin(); cit != v.cend(); ++cit, ++i, ++count)
60  {
61  HPX_TEST_EQ(*cit, T(i));
62  }
63  HPX_TEST_EQ(count, size);
64 }

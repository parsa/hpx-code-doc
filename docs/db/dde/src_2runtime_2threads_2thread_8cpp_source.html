111  threads::thread_result_type thread::thread_function_nullary(
112  util::unique_function_nonser<void()> const& func)
113  {
114  try {
115  // Now notify our calling thread that we started execution.
116  func();
117  }
118  catch (hpx::thread_interrupted const&) { //-V565
119  /* swallow this exception */
120  }
121  catch (hpx::exception const&) {
122  // Verify that there are no more registered locks for this
123  // OS-thread. This will throw if there are still any locks
124  // held.
125  util::force_error_on_lock();
126 
127  // run all callbacks attached to the exit event for this thread
128  run_thread_exit_callbacks();
129 
130  throw; // rethrow any exception except 'thread_interrupted'
131  }
132 
133  // Verify that there are no more registered locks for this
134  // OS-thread. This will throw if there are still any locks
135  // held.
136  util::force_error_on_lock();
137 
138  // run all callbacks attached to the exit event for this thread
139  run_thread_exit_callbacks();
140 
141  return threads::thread_result_type(threads::terminated, nullptr);
142  }

178  void thread::join()
179  {
180  std::unique_lock<mutex_type> l(mtx_);
181 
182  if(!joinable_locked())
183  {
184  terminate("thread::join", "trying to join a non joinable thread");
185  }
186 
187  native_handle_type this_id = threads::get_self_id();
188  if (this_id == id_)
189  {
190  HPX_THROW_EXCEPTION(thread_resource_error, "thread::join",
191  "hpx::thread: trying joining itself");
192  return;
193  }
194  this_thread::interruption_point();
195 
196 
197  // register callback function to be called when thread exits
198  if (threads::add_thread_exit_callback(id_,
199  util::bind(&resume_thread, this_id)))
200  {
201  // wait for thread to be terminated
202  util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
203  this_thread::suspend(threads::suspended, "thread::join");
204  }
205 
206  detach_locked(); // invalidate this object
207  }

61 int check(int fd)
62 {
63  std::string out;
64  std::vector<double> cnt;
65  while (true)
66  {
67  char buf[1024];
68  ssize_t n = read(fd, buf, 1024);
69  if (n > 0)
70  {
71  out += std::string(buf, buf + n);
72  size_t pos = out.find("invalid");
73  if (pos != out.npos)
74  return 0;
75 
76  pos = out.find(counter_name);
77  if (pos != out.npos)
78  { // counter name found
79  out.erase(0, pos);
80  pos = out.find('\n');
81  if (pos != out.npos)
82  { // this is complete line; extract counter value
83  size_t cpos = out.rfind(',', pos);
84  std::cerr << out.substr(0, pos+1);
85  std::string value = out.substr(cpos+1, pos-cpos-1);
86  if (value == "invalid")
87  {
88  cnt.push_back(-1.0);
89  }
90  else
91  {
92  cnt.push_back(boost::lexical_cast<double>(value));
93  }
94  if (cnt.size() == 5) break;
95  out.erase(0, pos+1);
96  }
97  }
98  }
99  else
100  throw std::runtime_error("truncated input; didn't get all counter values");
101  }
102  // bail out if perf counter isn't available ...
103  if (close_enough(cnt[0], -1.0, 5.0))
104  return 0;
105 
106  // since printing affects the counts, the relative error bounds need to be
107  // increased compared to the "basic_functions" test
108  bool pass = close_enough(cnt[0], nstores, 5.0) &&
109  (cnt[1] >= cnt[0]) && close_enough(cnt[0], cnt[1], 5.0) &&
110  close_enough(cnt[2], 2.0*cnt[0], 5.0) &&
111  close_enough(cnt[3], cnt[0], 5.0);
112 
113 
114  std::cerr << (pass? "PASSED": "FAILED") << ".\n";
115 
116  return pass? 0: 1;
117 }

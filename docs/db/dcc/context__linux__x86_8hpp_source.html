188  template<typename Functor>
189  x86_linux_context_impl(Functor& cb, std::ptrdiff_t stack_size = -1)
190  : m_stack_size(stack_size == -1
191  ? static_cast<std::ptrdiff_t>(default_stack_size)
192  : stack_size),
193  m_stack(nullptr)
194  {
195  if (0 != (m_stack_size % EXEC_PAGESIZE))
196  {
197  throw std::runtime_error(
198  boost::str(boost::format(
199  "stack size of %1% is not page aligned, page size is %2%")
200  % m_stack_size % EXEC_PAGESIZE));
201  }
202 
203  if (0 >= m_stack_size)
204  {
205  throw std::runtime_error(
206  boost::str(boost::format("stack size of %1% is invalid") %
207  m_stack_size));
208  }
209 
210  m_stack = posix::alloc_stack(static_cast<std::size_t>(m_stack_size));
211  HPX_ASSERT(m_stack);
212  posix::watermark_stack(m_stack, static_cast<std::size_t>(m_stack_size));
213 
214  typedef void fun(Functor*);
215  fun * funp = trampoline;
216 
217  m_sp = (static_cast<void**>(m_stack)
218  + static_cast<std::size_t>(m_stack_size) / sizeof(void*))
219  - context_size;
220 
221  m_sp[backup_cb_idx] = m_sp[cb_idx] = &cb;
222  m_sp[backup_funp_idx] = m_sp[funp_idx] = nasty_cast<void*>(funp);
223 
224 #if defined(HPX_HAVE_VALGRIND) && !defined(NVALGRIND)
225  {
226  void * eos = static_cast<char*>(m_stack) + m_stack_size;
227  m_sp[valgrind_id_idx] = reinterpret_cast<void*>(
228  VALGRIND_STACK_REGISTER(m_stack, eos));
229  }
230 #endif
231 
232 #if defined(HPX_HAVE_THREAD_STACKOVERFLOW_DETECTION)
233  segv_stack.ss_sp = valloc(SEGV_STACK_SIZE);
234  segv_stack.ss_flags = 0;
235  segv_stack.ss_size = SEGV_STACK_SIZE;
236 
237  std::memset(&action, '\0', sizeof(action));
238  action.sa_flags = SA_SIGINFO|SA_ONSTACK; //SA_STACK
239  action.sa_sigaction = &x86_linux_context_impl::sigsegv_handler;
240 
241  sigaltstack(&segv_stack, nullptr);
242  sigfillset(&action.sa_mask);
243  sigaction(SIGSEGV, &action, nullptr);
244 #endif
245  }

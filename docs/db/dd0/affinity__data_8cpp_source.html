108  std::size_t affinity_data::init(util::command_line_handling const& cfg_)
109  {
110  num_threads_ = cfg_.num_threads_;
111  std::size_t num_system_pus = hardware_concurrency();
112  auto& topo = resource::get_partitioner().get_topology();
113 
114  // initialize from command line
115  std::size_t pu_offset = get_pu_offset(cfg_);
116  if (pu_offset == std::size_t(-1))
117  {
118  pu_offset_ = 0;
119  }
120  else
121  {
122  pu_offset_ = pu_offset;
123  }
124 
125  if(num_system_pus > 1)
126  pu_step_ = get_pu_step(cfg_) % num_system_pus;
127 
128  affinity_domain_ = hpx::detail::get_affinity_domain(cfg_);
129  pu_nums_.clear();
130 
131  std::size_t const used_cores = cfg_.rtcfg_.get_first_used_core();
132  std::size_t max_cores =
133  hpx::util::safe_lexical_cast<std::size_t>(
134  get_config_entry("hpx.cores", 0), 0);
135 
136 #if defined(HPX_HAVE_HWLOC)
137  std::string affinity_desc;
138  hpx::detail::get_affinity_description(cfg_, affinity_desc);
139  if (affinity_desc == "none")
140  {
141  // don't use any affinity for any of the os-threads
142  threads::resize(no_affinity_, num_system_pus);
143  for (std::size_t i = 0; i != num_threads_; ++i)
144  threads::set(no_affinity_, i);
145  }
146  else if (!affinity_desc.empty())
147  {
148  affinity_masks_.clear();
149  affinity_masks_.resize(num_threads_, 0);
150 
151  for (std::size_t i = 0; i != num_threads_; ++i)
152  threads::resize(affinity_masks_[i], num_system_pus);
153 
154  parse_affinity_options(affinity_desc, affinity_masks_,
155  used_cores, max_cores, num_threads_, pu_nums_);
156 
157  std::size_t num_initialized = count_initialized(affinity_masks_);
158  if (num_initialized != num_threads_) {
159  HPX_THROW_EXCEPTION(bad_parameter,
160  "affinity_data::affinity_data",
161  boost::str(
162  boost::format("The number of OS threads requested "
163  "(%1%) does not match the number of threads to "
164  "bind (%2%)") % num_threads_ % num_initialized));
165  }
166  }
167  else if (pu_offset == std::size_t(-1))
168  {
169  // calculate the pu offset based on the used cores, but only if its
170  // not explicitly specified
171  for(std::size_t num_core = 0; num_core != used_cores; ++num_core)
172  {
173  pu_offset_ += topo.get_number_of_core_pus(num_core);
174  }
175  }
176 #endif
177 
178  // correct used_cores from config data if appropriate
179  if (used_cores_ == 0)
180  {
181  used_cores_ = used_cores;
182  }
183 
184  pu_offset_ %= num_system_pus;
185  init_cached_pu_nums(num_system_pus);
186 
187  std::vector<std::size_t> cores;
188  cores.reserve(num_threads_);
189  for (std::size_t i = 0; i != num_threads_; ++i)
190  {
191  std::size_t add_me = topo.get_core_number(get_pu_num(i));
192  cores.push_back(add_me);
193  }
194 
195  std::sort(cores.begin(), cores.end());
196  std::vector<std::size_t>::iterator it =
197  std::unique(cores.begin(), cores.end());
198  cores.erase(it, cores.end());
199 
200  std::size_t num_unique_cores = cores.size();
201  return (std::max)(num_unique_cores, max_cores);
202  }

204  mask_cref_type affinity_data::get_pu_mask(std::size_t global_thread_num) const
205  {
206  // get a topology instance
207  topology const& topology = resource::get_partitioner().get_topology();
208 
209  // --hpx:bind=none disables all affinity
210  if (threads::test(no_affinity_, global_thread_num))
211  {
212  static mask_type m = get_full_machine_mask(num_threads_);
213  return m;
214  }
215 
216  // if we have individual, predefined affinity masks, return those
217  if (!affinity_masks_.empty())
218  return affinity_masks_[global_thread_num];
219 
220  // otherwise return mask based on affinity domain
221  std::size_t pu_num = get_pu_num(global_thread_num);
222  if (0 == std::string("pu").find(affinity_domain_))
223  {
224  // The affinity domain is 'processing unit', just convert the
225  // pu-number into a bit-mask.
226  return topology.get_thread_affinity_mask(pu_num);
227  }
228  if (0 == std::string("core").find(affinity_domain_))
229  {
230  // The affinity domain is 'core', return a bit mask corresponding
231  // to all processing units of the core containing the given
232  // pu_num.
233  return topology.get_core_affinity_mask(pu_num);
234  }
235  if (0 == std::string("numa").find(affinity_domain_))
236  {
237  // The affinity domain is 'numa', return a bit mask corresponding
238  // to all processing units of the NUMA domain containing the
239  // given pu_num.
240  return topology.get_numa_node_affinity_mask(pu_num);
241  }
242 
243  // The affinity domain is 'machine', return a bit mask corresponding
244  // to all processing units of the machine.
245  HPX_ASSERT(0 == std::string("machine").find(affinity_domain_));
246  return topology.get_machine_affinity_mask();
247  }

290  // means of adding a processing unit after initialization
291  void affinity_data::add_punit(std::size_t virt_core, std::size_t thread_num)
292  {
293  std::size_t num_system_pus = hardware_concurrency();
294 
295  // initialize affinity_masks and set the mask for the given virt_core
296  if (affinity_masks_.empty())
297  {
298  affinity_masks_.resize(num_threads_);
299  for (std::size_t i = 0; i != num_threads_; ++i)
300  threads::resize(affinity_masks_[i], num_system_pus);
301  }
302  threads::set(affinity_masks_[virt_core], thread_num);
303 
304  // find first used pu, which is then stored as the pu_offset
305  std::size_t first_pu = std::size_t(-1);
306  for (std::size_t i = 0; i != num_threads_; ++i)
307  {
308  std::size_t first = threads::find_first(affinity_masks_[i]);
309  first_pu = (std::min)(first_pu, first);
310  }
311  if (first_pu != std::size_t(-1))
312  pu_offset_ = first_pu;
313 
314  init_cached_pu_nums(num_system_pus);
315  }

329  std::size_t affinity_data::get_pu_num(std::size_t num_thread,
330  std::size_t hardware_concurrency) const
331  {
332  // The offset shouldn't be larger than the number of available
333  // processing units.
334  HPX_ASSERT(pu_offset_ < hardware_concurrency);
335 
336  // The distance between assigned processing units shouldn't be zero
337  HPX_ASSERT(pu_step_ > 0 && pu_step_ <= hardware_concurrency);
338 
339  // We 'scale' the thread number to compute the corresponding
340  // processing unit number.
341  //
342  // The base line processing unit number is computed from the given
343  // pu-offset and pu-step.
344  std::size_t num_pu = pu_offset_ + pu_step_ * num_thread;
345 
346  // We add an additional offset, which allows to 'roll over' if the
347  // pu number would get larger than the number of available
348  // processing units. Note that it does not make sense to 'roll over'
349  // farther than the given pu-step.
350  std::size_t offset = (num_pu / hardware_concurrency) % pu_step_;
351 
352  // The resulting pu number has to be smaller than the available
353  // number of processing units.
354  return (num_pu + offset) % hardware_concurrency;
355  }

59  template <typename Scheduler>
60  thread_pool_os_executor<Scheduler>::thread_pool_os_executor(
61  std::size_t num_punits, std::string const& affinity_desc)
62  : scheduler_(nullptr),
63  executor_name_(get_unique_name()),
64  notifier_(get_notification_policy(executor_name_.c_str())),
65  pool_(nullptr),
66  num_threads_(num_punits)
67  {
68  if (num_punits > hpx::threads::hardware_concurrency())
69  {
70  HPX_THROW_EXCEPTION(bad_parameter,
71  "thread_pool_os_executor<Scheduler>::thread_pool_os_executor",
72  "max_punit shouldn't be larger than number of available "
73  "OS-threads");
74  return;
75  }
76 
77  std::unique_ptr<Scheduler> scheduler(new Scheduler(num_punits));
78  scheduler_ = scheduler.get();
79 
80  pool_.reset(new threads::detail::scheduled_thread_pool<Scheduler>(
81  std::move(scheduler), notifier_, 0, executor_name_.c_str()));
82 
83  std::unique_lock<mutex_type> lk(mtx_);
84  pool_->init(num_threads_, 0);
85 
86  if (!pool_->run(lk, num_threads_))
87  {
88  HPX_THROW_EXCEPTION(invalid_status,
89  "thread_pool_os_executor<Scheduler>::thread_pool_os_executor",
90  "couldn't start thread_pool");
91  }
92  }

160  template <typename Scheduler>
161  void thread_pool_os_executor<Scheduler>::add(closure_type && f,
162  util::thread_description const& desc,
163  threads::thread_state_enum initial_state,
164  bool run_now, threads::thread_stacksize stacksize, error_code& ec)
165  {
166  // create a new thread
167  thread_init_data data(util::bind(
168  util::one_shot(&thread_pool_os_executor::thread_function_nullary),
169  std::move(f)), desc);
170  data.stacksize = threads::get_stack_size(stacksize);
171 
172  threads::thread_id_type id = threads::invalid_thread_id;
173  pool_->create_thread(data, id, initial_state, run_now, ec);
174  if (ec) return;

176  HPX_ASSERT(invalid_thread_id != id || !run_now);
177 
178  if (&ec != &throws)
179  ec = make_success_code();
180  }

182  // Schedule given function for execution in this executor no sooner
183  // than time abs_time. This call never blocks, and may violate
184  // bounds on the executor's queue size.
185  template <typename Scheduler>
186  void thread_pool_os_executor<Scheduler>::add_at(
187  util::steady_clock::time_point const& abs_time,
188  closure_type && f, util::thread_description const& desc,
189  threads::thread_stacksize stacksize, error_code& ec)
190  {
191  // create a new suspended thread
192  thread_init_data data(util::bind(
193  util::one_shot(&thread_pool_os_executor::thread_function_nullary),
194  std::move(f)), desc);
195  data.stacksize = threads::get_stack_size(stacksize);
196 
197  threads::thread_id_type id = threads::invalid_thread_id;
198  pool_->create_thread(data, id, suspended, true, ec);
199  if (ec) return;
200 
201  HPX_ASSERT(invalid_thread_id != id); // would throw otherwise
202 
203  // now schedule new thread for execution
204  pool_->set_state(abs_time, id, pending, wait_timeout,
205  thread_priority_normal, ec);
206  if (ec) return;
207 
208  if (&ec != &throws)
209  ec = make_success_code();
210  }

212  // Schedule given function for execution in this executor no sooner

214  // violate bounds on the executor's queue size.
215  template <typename Scheduler>

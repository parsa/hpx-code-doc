283  std::size_t handle_num_threads(util::manage_config& cfgmap,
284  boost::program_options::variables_map& vm,
285  util::batch_environment& env, bool using_nodelist, bool initial)
286  {
287  std::size_t batch_threads = env.retrieve_number_of_threads();
288  std::size_t default_threads = thread::hardware_concurrency();
289  std::string threads_str = cfgmap.get_value<std::string>(
290  "hpx.os_threads", "");
291 
292  if ("all" == threads_str)
293  {
294  if (batch_threads == std::size_t(-1))
295  batch_threads = thread::hardware_concurrency();
296  else
297  default_threads = batch_threads;
298 
299  cfgmap.config_["hpx.os_threads"] =
300  std::to_string(batch_threads);
301  }
302  else if (batch_threads != std::size_t(-1))
303  {
304  default_threads = batch_threads;
305  }
306 
307  std::size_t threads = cfgmap.get_value<std::size_t>(
308  "hpx.os_threads", default_threads);
309 
310  if (vm.count("hpx:threads"))
311  {
312  threads_str = vm["hpx:threads"].as<std::string>();
313  if ("all" == threads_str)
314  {
315  if (batch_threads == std::size_t(-1))
316  {
317  batch_threads = thread::hardware_concurrency();
318  }
319  threads = batch_threads; //-V101
320  }
321  else
322  {
323  threads = hpx::util::safe_lexical_cast<std::size_t>(threads_str);
324  }
325 
326  if (threads == 0)
327  {
328  throw hpx::detail::command_line_error("Number of --hpx:threads "
329  "must be greater than 0");
330  }
331 
332 #if defined(HPX_HAVE_MAX_CPU_COUNT)
333  if (threads > HPX_HAVE_MAX_CPU_COUNT)
334  {
335  throw hpx::detail::command_line_error("Requested more than "
336  HPX_PP_STRINGIZE(HPX_HAVE_MAX_CPU_COUNT)" --hpx:threads "
337  "to use for this application, use the option "
338  "-DHPX_WITH_MAX_CPU_COUNT=<N> when configuring HPX.");
339  }
340 #endif
341  }
342 
343  if (!initial && env.found_batch_environment() &&
344  using_nodelist && (threads > batch_threads))
345  {
346  detail::report_thread_warning(env.get_batch_name(),
347  threads, batch_threads);
348  }
349  return threads;
350  }

353  std::size_t get_number_of_default_cores(util::batch_environment& env)
354  {
355  threads::topology& top = threads::create_topology();
356 
357  std::size_t batch_threads = env.retrieve_number_of_threads();
358  std::size_t num_cores = top.get_number_of_cores();
359  if(batch_threads == std::size_t(-1))
360  return num_cores;
361 
362  // assuming we assign the first N cores ...
363  std::size_t core = 0;
364  for(; core < num_cores; ++core)
365  {
366  batch_threads -= top.get_number_of_core_pus(core);
367  if(batch_threads == 0) break;
368  }
369 
370  return core + 1;
371  }

432  bool command_line_handling::handle_arguments(util::manage_config& cfgmap,
433  boost::program_options::variables_map& vm,
434  std::vector<std::string>& ini_config, std::size_t& node, bool initial)
435  {
436  // verify that no networking options were used if networking was
437  // disabled
438  detail::check_networking_options(vm);
439 
440  using namespace boost::assign;
441 
442  bool debug_clp = node != std::size_t(-1) && vm.count("hpx:debug-clp");
443 
444  if (vm.count("hpx:ini")) {
445  std::vector<std::string> cfg =
446  vm["hpx:ini"].as<std::vector<std::string> >();
447  std::copy(cfg.begin(), cfg.end(), std::back_inserter(ini_config));
448  cfgmap.add(cfg);
449  }
450 
451  // create host name mapping
452  util::map_hostnames mapnames(debug_clp);
453 
454  if (vm.count("hpx:ifsuffix"))
455  mapnames.use_suffix(vm["hpx:ifsuffix"].as<std::string>());
456  if (vm.count("hpx:ifprefix"))
457  mapnames.use_prefix(vm["hpx:ifprefix"].as<std::string>());
458 
459  // The AGAS host name and port number are pre-initialized from
460  //the command line
461  std::string agas_host =
462  cfgmap.get_value<std::string>("hpx.agas.address",
463  rtcfg_.get_entry("hpx.agas.address", ""));
464  std::uint16_t agas_port =
465  cfgmap.get_value<std::uint16_t>("hpx.agas.port",
466  boost::lexical_cast<std::uint16_t>(
467  rtcfg_.get_entry("hpx.agas.port", HPX_INITIAL_IP_PORT)
468  ));
469 
470  if (vm.count("hpx:agas")) {
471  if (!util::split_ip_address(
472  vm["hpx:agas"].as<std::string>(), agas_host, agas_port))
473  {
474  std::cerr
475  << "hpx::init: command line warning: illegal port "
476  "number given, using default value instead."
477  << std::endl;
478  }
479  }
480 
481  // Check command line arguments.
482 
483  if (vm.count("hpx:iftransform")) {
484  util::sed_transform iftransform(vm["hpx:iftransform"].as<std::string>());
485 
486  // Check for parsing failures
487  if (!iftransform) {
488  throw hpx::detail::command_line_error(boost::str(boost::format(
489  "Could not parse --hpx:iftransform argument '%1%'") %
490  vm["hpx:iftransform"].as<std::string>()));
491  }
492 
493  typedef util::map_hostnames::transform_function_type
494  transform_function_type;
495  mapnames.use_transform(transform_function_type(iftransform));
496  }
497 
498  bool using_nodelist = false;
499  bool enable_batch_env = false;
500 
501  std::vector<std::string> nodelist;
502 
503 #if defined(HPX_HAVE_NETWORKING)
504  if(vm.count("hpx:nodefile"))
505  {
506  if (vm.count("hpx:nodes")) {
507  throw hpx::detail::command_line_error(
508  "Ambiguous command line options. "
509  "Do not specify more than one of the --hpx:nodefile and "
510  "--hpx:nodes options at the same time.");
511  }
512  std::string node_file = vm["hpx:nodefile"].as<std::string>();
513  ini_config += "hpx.nodefile!=" + node_file;
514  std::ifstream ifs(node_file.c_str());
515  if (ifs.is_open())
516  {
517  if (debug_clp)
518  std::cerr << "opened: " << node_file << std::endl;
519  std::string line;
520  while (std::getline(ifs, line)) {
521  if (!line.empty()) {
522  nodelist.push_back(line);
523  }
524  }
525  }
526  else {
527  if (debug_clp)
528  std::cerr << "failed opening: " << node_file << std::endl;
529 
530  // raise hard error if node file could not be opened
531  throw hpx::detail::command_line_error(boost::str(boost::format(
532  "Could not open nodefile: '%s'") % node_file));
533  }
534  }
535  else if (vm.count("hpx:nodes")) {
536  nodelist = vm["hpx:nodes"].as<std::vector<std::string> >();
537  }
538 
539  enable_batch_env = (cfgmap.get_value<int>("hpx.ignore_batch_env", 0)
540  + vm.count("hpx:ignore-batch-env")) == 0;
541 #endif
542 
543  util::batch_environment env(nodelist, rtcfg_, debug_clp, enable_batch_env);
544 
545 #if defined(HPX_HAVE_NETWORKING)
546  if(!nodelist.empty())
547  {
548  using_nodelist = true;
549  ini_config += "hpx.nodes!=" + env.init_from_nodelist(
550  nodelist, agas_host);
551  }
552 
553  // let the batch environment decide about the AGAS host
554  agas_host = env.agas_host_name(
555  agas_host.empty() ? HPX_INITIAL_IP_ADDRESS : agas_host);
556 #endif
557 
558  // handle number of cores and threads
559  num_threads_ = detail::handle_num_threads(
560  cfgmap, vm, env, using_nodelist, initial);
561  num_cores_ = detail::handle_num_cores(cfgmap, vm, num_threads_, env);
562 
563  bool expect_connections = false;
564  bool run_agas_server = false;
565  std::string hpx_host;
566  std::uint16_t initial_hpx_port = 0;
567  std::uint16_t hpx_port = 0;
568 
569 #if defined(HPX_HAVE_NETWORKING)
570  // handling number of localities, those might have already been initialized
571  // from MPI environment
572  num_localities_ = detail::handle_num_localities(cfgmap, vm, env,
573  using_nodelist, num_localities_, initial);
574 
575  // Determine our network port, use arbitrary port if running on one
576  // locality.
577  hpx_host = cfgmap.get_value<std::string>("hpx.parcel.address",
578  env.host_name(
579  rtcfg_.get_entry("hpx.parcel.address", HPX_INITIAL_IP_ADDRESS)
580  ));
581 
582  // we expect dynamic connections if:
583  // - --hpx:expect-connecting-localities or
584  // - hpx.expect_connecting_localities=1 is given, or
585  // - num_localities > 1
586  expect_connections =
587  cfgmap.get_value<int>("hpx.expect_connecting_localities",
588  num_localities_ > 1 ? 0 : 1) ? true : false;
589 
590  if (vm.count("hpx:expect-connecting-localities"))
591  expect_connections = true;
592 
593  ini_config += std::string("hpx.expect_connecting_localities=") +
594  (expect_connections ? "1" : "0");
595 
596  if (num_localities_ != 1 || expect_connections)
597  {
598  initial_hpx_port =
599  boost::lexical_cast<std::uint16_t>(
600  rtcfg_.get_entry("hpx.parcel.port", HPX_INITIAL_IP_PORT));
601  }
602 
603  hpx_port = cfgmap.get_value<std::uint16_t>("hpx.parcel.port",
604  initial_hpx_port);
605 
606  run_agas_server = vm.count("hpx:run-agas-server") != 0;
607  if (node == std::size_t(-1))
608  node = env.retrieve_node_number();
609 #else
610  num_localities_ = 1;
611  node = 0;
612 #endif
613 
614  // If the user has not specified an explicit runtime mode we
615  // retrieve it from the command line.
616  if (hpx::runtime_mode_default == rtcfg_.mode_) {
617 #if defined(HPX_HAVE_NETWORKING)
618  // The default mode is console, i.e. all workers need to be
619  // started with --worker/-w.
620  rtcfg_.mode_ = hpx::runtime_mode_console;
621  if (vm.count("hpx:console") + vm.count("hpx:worker") +
622  vm.count("hpx:connect") > 1)
623  {
624  throw hpx::detail::command_line_error(
625  "Ambiguous command line options. "
626  "Do not specify more than one of --hpx:console, "
627  "--hpx:worker, or --hpx:connect");
628  }
629 
630  // In these cases we default to executing with an empty
631  // hpx_main, except if specified otherwise.
632  if (vm.count("hpx:worker")) {
633  rtcfg_.mode_ = hpx::runtime_mode_worker;
634 
635 #if !defined(HPX_HAVE_RUN_MAIN_EVERYWHERE)
636  // do not execute any explicit hpx_main except if asked
637  // otherwise
638  if (!vm.count("hpx:run-hpx-main") &&
639  !cfgmap.get_value<int>("hpx.run_hpx_main", 0))
640  {

1146  int command_line_handling::call(
1147  boost::program_options::options_description const& desc_cmdline,
1148  int argc, char** argv)
1149  {
1150  // set the flag signaling that command line parsing has been done
1151  cmd_line_parsed_ = true;
1152 
1153  util::manage_config cfgmap(ini_config_);
1154 
1155  std::vector<std::shared_ptr<plugins::plugin_registry_base> >
1156  plugin_registries = rtcfg_.load_modules();
1157 
1158  // insert the pre-configured ini settings after loading modules
1159  for (std::string const& e : ini_config_)
1160  rtcfg_.parse("<user supplied config>", e, true, false);
1161 
1162  // Initial analysis of the command line options. This is
1163  // preliminary as it will not take into account any aliases as
1164  // defined in any of the runtime configuration files.
1165  {
1166  // Boost V1.47 and before do not properly reset a variables_map
1167  // when calling vm.clear(). We work around that problems by
1168  // creating a separate instance just for the preliminary
1169  // command line handling.
1170  boost::program_options::variables_map prevm;
1171  if (!util::parse_commandline(rtcfg_, desc_cmdline, argc,
1172  argv, prevm, std::size_t(-1), util::allow_unregistered,
1173  rtcfg_.mode_))
1174  {
1175  return -1;
1176  }
1177 
1178  // handle all --hpx:foo options, determine node
1179  std::vector<std::string> ini_config; // will be discarded
1180  if (!handle_arguments(cfgmap, prevm, ini_config, node_, true))
1181  return -2;
1182 
1183  // re-initialize runtime configuration object
1184  if (prevm.count("hpx:config"))
1185  rtcfg_.reconfigure(prevm["hpx:config"].as<std::string>());
1186  else
1187  rtcfg_.reconfigure("");
1188 
1189  // Make sure any aliases defined on the command line get used
1190  // for the option analysis below.
1191  std::vector<std::string> cfg;
1192  if (prevm.count("hpx:ini")) {
1193  cfg = prevm["hpx:ini"].as<std::vector<std::string> >();
1194  cfgmap.add(cfg);
1195  }
1196 
1197  // append ini options from command line
1198  std::copy(ini_config_.begin(), ini_config_.end(),
1199  std::back_inserter(cfg));
1200 
1201  rtcfg_.reconfigure(cfg);
1202  }
1203 
1204  // Re-run program option analysis, ini settings (such as aliases)
1205  // will be considered now.
1206 
1207  parcelset::parcelhandler::init(&argc, &argv, *this);
1208  for (std::shared_ptr<plugins::plugin_registry_base>& reg : plugin_registries)
1209  {
1210  reg->init(&argc, &argv, *this);
1211  }
1212 
1213  // minimally assume one locality and this is the console
1214  if (node_ == std::size_t(-1))
1215  node_ = 0;
1216 
1217  // Now re-parse the command line using the node number (if given).
1218  // This will additionally detect any --hpx:N:foo options.
1219  boost::program_options::options_description help;
1220  std::vector<std::string> unregistered_options;
1221 
1222  if (!util::parse_commandline(rtcfg_, desc_cmdline,
1223  argc, argv, vm_, node_, util::allow_unregistered, rtcfg_.mode_,
1224  &help, &unregistered_options))
1225  {
1226  return -1;
1227  }
1228 
1229  // break into debugger, if requested
1230  handle_attach_debugger();

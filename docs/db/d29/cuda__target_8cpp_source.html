113  void future_data::init(cudaStream_t stream)
114  {
115  // Hold on to the shared state on behalf of the cuda runtime
116  // right away as the callback could be called immediately.
117  lcos::detail::intrusive_ptr_add_ref(this);
118 
119  cudaError_t error = cudaStreamAddCallback(
120  stream, stream_callback, this, 0);
121  if (error != cudaSuccess)
122  {
123  // callback was not called, release object
124  lcos::detail::intrusive_ptr_release(this);
125 
126  // report error
127  HPX_THROW_EXCEPTION(kernel_error,
128  "cuda::detail::future_data::future_data()",
129  std::string("cudaStreamAddCallback failed: ") +
130  cudaGetErrorString(error));
131  }

135  void target::native_handle_type::init_processing_units()
136  {
137  cudaDeviceProp props;
138  cudaError_t error = cudaGetDeviceProperties(&props, device_);
139  if (error != cudaSuccess)
140  {
141  // report error
142  HPX_THROW_EXCEPTION(kernel_error,
143  "cuda::default_executor::processing_units_count()",
144  std::string("cudaGetDeviceProperties failed: ") +
145  cudaGetErrorString(error));
146  }
147 
148  std::size_t mp = props.multiProcessorCount;
149  switch(props.major)
150  {
151  case 2: // Fermi
152  if (props.minor == 1) {
153  mp = mp * 48;
154  }
155  else {
156  mp = mp * 32;
157  }
158  break;
159  case 3: // Kepler
160  mp = mp * 192;
161  break;
162  case 5: // Maxwell
163  mp = mp * 128;
164  break;
165  case 6: // Pascal
166  mp = mp * 64;
167  break;
168  default:
169  break;
170  }
171  processing_units_ = mp;
172  processor_family_ = props.major;
173  processor_name_ = props.name;
174  }

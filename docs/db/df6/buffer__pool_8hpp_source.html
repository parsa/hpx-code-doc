17  // This class holds shared_ptr of vector<T, Allocator> with a power of two
18  template <typename T, typename Allocator = std::allocator<T> >

26  shared_buffer_type get_buffer(size_type size)
27  {
28  size_type capacity = next_power_of_two(size);
29  typename buffer_map_type::iterator it = buffers_.find(capacity);
30  shared_buffer_type res;
31  if(it == buffers_.end() || it->second.empty())
32  {
33  res.reset(new buffer_type());
34  res->reserve(capacity);
35  }

42  }

70  static size_type next_power_of_two(size_type size)
71  {
72  // Check if we already have a power of two
73  // http://graphics.stanford.edu/~seander/bithacks.html#DetermineIfPowerOf2
74  if(size && !(size & (size - 1)))
75  return size;
76  size--;

29  // Provides thread-safe initialization of a single static instance of T.
30  //
31  // This instance is guaranteed to be constructed on static storage in a
32  // thread-safe manner, on the first call to the constructor of static_.
33  //
34  // Requirements:
35  // T is default constructible or has one argument
36  // T::T() MUST not throw!
37  // this is a requirement of boost::call_once.
38  //
39  // In addition this type registers global construction and destruction
40  // functions used by the HPX runtime system to reinitialize the held data
41  // structures.
42  template <typename T, typename Tag = T, std::size_t N = 1>
43  struct HPX_EXPORT_REINITIALIZABLE_STATIC reinitializable_static;

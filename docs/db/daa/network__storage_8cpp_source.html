33 //
34 // This is a test program which reads and writes chunks of memory to storage
35 // distributed across localities.
36 //
37 // The principal problem can be summarized as follows:
38 //
39 // Put data into remote memory
40 // 1 Pass user data pointer into serialize_buffer (0 copy)
41 // 2 Copy data into parcelport for transmission. (1 copy)
42 // 3 Transmit data into remote parcelport buffer
43 // 4 Copy data from parcelport into serialize_buffer (2 copy)
44 // 5 Copy data from serialize buffer into remote host storage (3 copy ?)
45 //
46 // Get data from remote memory
47 // 1 Allocate temporary buffer and copy from host storage into it (1 copy)
48 // 2 Wrap storage in serialize_buffer and give to parcelport for return
49 // 3 Copy data from serialize_buffer into parcelport for transmission (2 copy)
50 // 4 Receive data into local parcelport buffer
51 // 5 Copy parcelport buffer to serialize_buffer(user data pointer) (3 copy)
52 //
53 // The ideal situation would be as follows
54 //
55 // Put into remote memory
56 // 1 Pass user data pointer into serialize_buffer (0 copy)
57 // 2 Copy data into parcelport for transmission. (1 copy)
58 // 3 Transmit data into remote parcelport buffer
59 // 4 Copy data from parcelport into serialize_buffer(user data pointer) (2 copy)
60 //
61 // Get data from remote memory
62 // 1 Request serialize buffer from parcelport (0 copy)
63 // 2 Copy from storage into serialize_buffer (1 copy)
64 // 4 Receive data into local parcelport buffer
65 // 5 Copy parcelport buffer to serialize_buffer(user data pointer) (2 copy)
66 //
67 // To make each process run a main function and participate in the test,
68 // use a command line of the kind (no mpiexec assumed)
69 //
70 // network_storage.exe -l%1 -%%x --hpx:run-hpx-main --hpx:threads=4
71 //
72 // (+or more) where %l is num_ranks, and %%x is rank of current process

216 // A custom allocator which takes a pointer in its constructor and then returns
217 // this pointer in response to any allocate request. It is here to try to fool
218 // the hpx serialization into copying directly into a user provided buffer
219 // without copying from a result into another buffer.
220 //
221 template <typename T>
222 class pointer_allocator
223 {
224 public:
225  typedef T value_type;
226  typedef T* pointer;
227  typedef const T* const_pointer;
228  typedef T& reference;
229  typedef const T& const_reference;
230  typedef std::size_t size_type;
231  typedef std::ptrdiff_t difference_type;
232 
233  pointer_allocator() noexcept
234  : pointer_(nullptr), size_(0)
235  {
236  }
237 
238  pointer_allocator(pointer p, size_type size) noexcept
239  : pointer_(p), size_(size)
240  {
241  }
242 
243  pointer address(reference value) const { return &value; }
244  const_pointer address(const_reference value) const { return &value; }
245 
246  pointer allocate(size_type n, void const* hint = nullptr)
247  {
248  HPX_ASSERT(n == size_);
249  return static_cast<T*>(pointer_);
250  }
251 
252  void deallocate(pointer p, size_type n)
253  {
254  HPX_ASSERT(p == pointer_ && n == size_);
255  }
256 
257 private:
258  // serialization support
259  friend class hpx::serialization::access;
260 
261  template <typename Archive>
262  void load(Archive& ar, unsigned int const version)
263  {
264  std::size_t t = 0;
265  ar >> size_ >> t;
266  pointer_ = reinterpret_cast<pointer>(t);
267  }
268 
269  template <typename Archive>
270  void save(Archive& ar, unsigned int const version) const
271  {
272  std::size_t t = reinterpret_cast<std::size_t>(pointer_);
273  ar << size_ << t;
274  }
275 
276  HPX_SERIALIZATION_SPLIT_MEMBER()
277 
278 private:
279  pointer pointer_;
280  size_type size_;
281 };

318 //----------------------------------------------------------------------------
319 //
320 // Two actions which are called from remote or local localities
321 //
322 // Copy to storage, just invokes the local copy function and returns the future
323 // from it
324 //
325 // Copy from storage allocates a buffer for the return memory and then wraps
326 // it into a serialize buffer which is returned and passed into the local
327 // process.
328 //
329 // This unfortunately means memory is copied from the storage, into a buffer
330 // and not zero copied as we would like.
331 // I have not been successful in removing this copy.
332 namespace Storage {
333  //------------------------------------------------------------------------
334  // A PUT into memory on this locality from a requester sending a
335  // general_buffer_type
336  async_mem_result_type CopyToStorage(general_buffer_type const& srcbuffer,
337  uint32_t address, uint64_t length)
338  {
339  boost::shared_array<char> src = srcbuffer.data_array();
340  return copy_to_local_storage(src.get(), address, length);
341  }
342 
343  //------------------------------------------------------------------------
344  // A GET from memory on this locality is returned to the requester in the
345  // transfer_buffer_type
346  hpx::future<transfer_buffer_type> CopyFromStorage(
347  uint32_t address, uint64_t length, std::size_t remote_buffer)
348  {
349  // we must allocate a temporary buffer to copy from storage into
350  // we can't use the remote buffer supplied because it is a handle to memory on
351  // the (possibly) remote node. We allocate here using
352  // a nullptr deleter so the array will
353  // not be released by the shared_pointer.
354  //
355  // The memory must be freed after final use.
356  std::allocator<char> local_allocator;
357  boost::shared_array<char> local_buffer(local_allocator.allocate(length),
358  [](char*){
359  DEBUG_OUTPUT(6, "Not deleting memory");
360  }
361  );
362 
363  // allow the storage class to asynchronously copy the data into buffer
364 #ifdef ASYNC_MEMORY
365  async_mem_result_type fut =
366 #endif
367  copy_from_local_storage(local_buffer.get(), address, length);
368  DEBUG_OUTPUT(6, "create local buffer count " << local_buffer.use_count());
369 
370  // wrap the remote buffer pointer in an allocator for return
371  pointer_allocator<char> return_allocator(
372  reinterpret_cast<char*>(remote_buffer), length);
373 
374  // lock the mutex, will be unlocked by the transfer buffer's deleter
375 // storage_mutex.lock();
376 #ifdef ASYNC_MEMORY
377  return fut.then(
378  hpx::launch::sync,
379  // return the data in a transfer buffer
380  [=](hpx::future<int> fut) -> transfer_buffer_type {
381  DEBUG_OUTPUT(6, ".then local buffer count " << local_buffer.use_count());
382  return transfer_buffer_type(
383  local_buffer.get(), length,
384  transfer_buffer_type::take,
385  [](char *p) {
386  release_storage_lock(p);
387  },
388  return_allocator);
389  }
390  );
391 #else
392  return hpx::make_ready_future<transfer_buffer_type>(
393  transfer_buffer_type(
394  local_buffer.get(), length,
395  transfer_buffer_type::take,
396  [](char *p) {
397  release_storage_lock(p);
398  },
399  return_allocator
400  ));
401 #endif
402  }
403 } // namespace storage

455  throw std::runtime_error("Remote put/get failed");

460  ++fut;
461  }
462  }
463  }
464  }

466  }
467  return num_removed;

484 // Test speed of write/put
485 void test_write(
486  uint64_t rank, uint64_t nranks, uint64_t num_transfer_slots,
487  std::mt19937& gen, std::uniform_int_distribution<>& random_rank,
488  std::uniform_int_distribution<>& random_slot,
489  test_options &options
490  )

495  //

497  //
498  DEBUG_OUTPUT(1, "Passed Barrier at start of write on rank " << rank);
499  //

501  hpx::util::simple_profiler level1("Write function", rank==0 && !options.warmup);
502  //
503  bool active = (rank==0) || (rank>0 && options.all2all);
504  if (rank==0) std::cout << "Iteration ";
505  for (std::uint64_t i = 0; active && i < options.iterations; i++) {
506  hpx::util::simple_profiler iteration(level1, "Iteration");
507  if (rank==0) {
508  if (i%10==0) {
509  std::cout << "x" << std::flush;

511  else {
512  std::cout << "." << std::flush;
513  }
514  }

516  DEBUG_OUTPUT(1, "Starting iteration " << i << " on rank " << rank);
517 
518 #ifdef USE_CLEANING_THREAD

520  // start a thread which will clear any completed futures from our list.
521  //
522  FuturesActive = true;
523  hpx::future<int> cleaner = hpx::async(RemoveCompletions);
524 #endif

526 #ifdef USE_PARCELPORT_THREAD
527  //
528  // start a thread which will keep the parcelport active.
529  //
530  FuturesActive = true;
531  hpx::future<int> background = hpx::async(background_work);
532 #endif
533 
534  // used to track callbacks to free buffers for async_cb
535  uint64_t buffer_index = 0;
536 
537  //
538  // Start main message sending loop
539  //
540  for (uint64_t i = 0; i < num_transfer_slots; i++) {
541  hpx::util::simple_profiler prof_setup(iteration, "Setup slots");
542  uint64_t send_rank;
543  if (options.distribution==0) {
544  // pick a random locality to send to
545  send_rank = random_rank(gen);
546  while (options.nolocal && send_rank==rank) {
547  send_rank = random_rank(gen);
548  }
549  }
550  else {
551  send_rank = static_cast<uint64_t>((rank + i) % nranks);
552  while (options.nolocal && send_rank==rank) {
553  send_rank = random_rank(gen);
554  }
555  }
556 
557  // get the pointer to the current packet send buffer
558  char *buffer = &local_storage[i*options.transfer_size_B];
559  // Get the HPX locality from the dest rank
560  hpx::id_type locality = hpx::naming::get_id_from_locality_id(send_rank);
561  // pick a random slot to write our data into
562  int memory_slot = random_slot(gen);
563  uint32_t memory_offset = static_cast<uint32_t>
564  (memory_slot*options.transfer_size_B);
565  DEBUG_OUTPUT(5,
566  "Rank " << rank << " sending block " << i << " to rank " << send_rank
567  );
568  prof_setup.done();
569 
570  // Execute a PUT on whatever locality we chose
571  // Create a serializable memory buffer ready for sending.
572  // Do not copy any data. Protect this with a mutex to ensure the
573  // background thread removing completed futures doesn't collide
574  {
575  hpx::util::simple_profiler prof_put(iteration, "Put");
576  DEBUG_OUTPUT(5,
577  "Put from rank " << rank << " on rank " << send_rank
578  );
579 #ifdef USE_CLEANING_THREAD
580  ++FuturesWaiting[send_rank];
581  std::lock_guard<hpx::lcos::local::spinlock> lk(FuturesMutex);
582 #endif
583 
584  std::shared_ptr<general_buffer_type> temp_buffer =
585  std::make_shared<general_buffer_type>(
586  static_cast<char*>(buffer), options.transfer_size_B,
587  general_buffer_type::reference);
588  using hpx::util::placeholders::_1;
589  using hpx::util::placeholders::_2;
590  {
591  scoped_lock lock(keep_alive_mutex);
592  keep_alive_buffers[buffer_index] = temp_buffer;
593  }
594  auto temp_future =
595  hpx::async_cb(hpx::launch::fork, actWrite, locality,
596  hpx::util::bind(&async_callback, buffer_index, _1, _2),
597  *temp_buffer,
598  memory_offset, options.transfer_size_B
599  ).then(
600  hpx::launch::sync,
601  [send_rank](hpx::future<int> &&fut) -> int {
602  int result = fut.get();
603  --FuturesWaiting[send_rank];
604  return result;
605  });
606  buffer_index++;
607 
608  ActiveFutures[send_rank].push_back(std::move(temp_future));
609  }
610  }
611  DEBUG_OUTPUT(2, "Exited transfer loop on rank " << rank);
612 
613 #ifdef USE_CLEANING_THREAD
614  int removed = 0;
615  {
616  hpx::util::simple_profiler prof_clean(iteration, "Cleaning Wait");
617  // tell the cleaning thread it's time to stop
618  FuturesActive = false;
619  // wait for cleanup thread to terminate before we reduce any remaining futures
620  removed = cleaner.get();

622  }
623 #endif
624 
625 #ifdef USE_PARCELPORT_THREAD
626  // tell the parcelport thread it's time to stop
627  FuturesActive = false;
628  background.get();
629 #endif
630  //
631  hpx::util::simple_profiler prof_move(iteration, "Moving futures");
632  std::vector<hpx::future<int> > final_list;
633  for (uint64_t i = 0; i < nranks; i++) {
634  // move the contents of intermediate vector into final list
635  final_list.reserve(final_list.size() + ActiveFutures[i].size());
636  std::move(ActiveFutures[i].begin(), ActiveFutures[i].end(),
637  std::back_inserter(final_list));
638  ActiveFutures[i].clear();
639  }
640  prof_move.done();
641  //
642  hpx::util::simple_profiler fwait(iteration, "Future wait");
643 
644  DEBUG_ONLY(int numwait = static_cast<int>(final_list.size());)
645 
646  DEBUG_OUTPUT(1, "Waiting for when_all future on rank " << rank);
647  hpx::future<int> result = when_all(final_list).then(hpx::launch::sync, reduce);
648  result.get();
649 #ifdef USE_CLEANING_THREAD
650  int total = numwait+removed;
651 #endif
652  DEBUG_OUTPUT(3, "Future wait, rank " << rank << " waiting on " << numwait);
653  fwait.done();
654  }
655  if (rank==0) std::cout << std::endl;
656  DEBUG_OUTPUT(2, "Exited iterations loop on rank " << rank);
657 
658  hpx::util::simple_profiler prof_barrier(level1, "Final Barrier");
659  hpx::lcos::barrier::synchronize();
660  //

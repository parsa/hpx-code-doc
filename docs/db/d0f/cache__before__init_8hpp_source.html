107  bool is_cache_turned_off() const {
108  if ( m_is_caching_off)
109  return true; // cache has been turned off
110 
111  // now we go the slow way - use mutex to see if cache is turned off
112  mutex::scoped_lock lk(m_cs);
113  m_is_caching_off = !(m_cache.is_using_cache);
114  return m_is_caching_off;

117  template<class writer_type>
118  void turn_cache_off(const writer_type & writer_) {
119  if ( is_cache_turned_off() )
120  return; // already turned off
121 
122  {
123  mutex::scoped_lock lk(m_cs);
124  m_cache.is_using_cache = false;
125  }
126 
127  // dump messages
128  typename cache::message_array msgs;
129  {
130  mutex::scoped_lock lk(m_cs);
131  std::swap( m_cache.msgs, msgs);
132  }
133  for ( typename cache::message_array::iterator b = msgs.begin(),
134  e = msgs.end(); b != e; ++b) {
135  if ( !(b->is_enabled) )
136  // no filter
137  writer_( b->string );
138  else if ( b->is_enabled() )
139  // filter enabled
140  writer_( b->string );
141  }
142  }

144  void add_msg(const msg_type & msg) const {
145  mutex::scoped_lock lk(m_cs);
146  // note : last_enabled can be null, if we don't want to use filters
147  // (HPX_LOG_BEFORE_INIT_USE_LOG_ALL)
148  is_enabled_func func = m_cache.threads[ get_thread_id() ].last_enabled ;
149  m_cache.msgs.push_back( message(func, msg) );
150  }

194  std::int64_t parcelport::get_pending_parcels_count(bool /*reset*/)
195  {
196  std::lock_guard<lcos::local::spinlock> l(mtx_);
197  std::int64_t count = 0;
198  for (auto && p : pending_parcels_)
199  {
200  count += hpx::util::get<0>(p.second).size();
201  HPX_ASSERT(
202  hpx::util::get<0>(p.second).size() ==
203  hpx::util::get<1>(p.second).size());
204  }
205  return count;
206  }

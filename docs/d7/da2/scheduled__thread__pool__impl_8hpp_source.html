210  template <typename Scheduler>
211  bool hpx::threads::detail::scheduled_thread_pool<Scheduler>::run(
212  std::unique_lock<compat::mutex>& l, std::size_t pool_threads)
213  {
214  HPX_ASSERT(l.owns_lock());
215 
216  LTM_(info) //-V128
217  << "run: " << id_.name()
218  << " number of processing units available: " //-V128
219  << threads::hardware_concurrency();
220  LTM_(info) //-V128
221  << "run: " << id_.name() << " creating " << pool_threads
222  << " OS thread(s)"; //-V128
223 
224  if (0 == pool_threads)
225  {
226  HPX_THROW_EXCEPTION(
227  bad_parameter, "run", "number of threads is zero");
228  }
229 
230  if (!threads_.empty() ||
231  sched_->Scheduler::has_reached_state(state_running))
232  {
233  return true; // do nothing if already running
234  }
235 
236  init_perf_counter_data(pool_threads);
237  this->init_pool_time_scale();
238 
239  LTM_(info) << "run: " << id_.name()
240  << " timestamp_scale: " << timestamp_scale_; //-V128
241 
242  // run threads and wait for initialization to complete
243  std::size_t thread_num = 0;
244  std::shared_ptr<compat::barrier> startup =
245  std::make_shared<compat::barrier>(pool_threads + 1);
246  try
247  {
248  auto const& rp = resource::get_partitioner();
249 
250  for (; thread_num != pool_threads; ++thread_num)
251  {
252  std::size_t global_thread_num = this->thread_offset_ + thread_num;
253  threads::mask_cref_type mask = rp.get_pu_mask(global_thread_num);
254 
255  // thread_num ordering: 1. threads of default pool
256  // 2. threads of first special pool
257  // 3. etc.
258  // get_pu_mask expects index according to ordering of masks
259  // in affinity_data::affinity_masks_
260  // which is in order of occupied PU
261  LTM_(info) //-V128
262  << "run: " << id_.name() << " create OS thread "
263  << global_thread_num //-V128
264  << ": will run on processing units within this mask: "
265  << std::hex << HPX_CPU_MASK_PREFIX << mask;
266 
267  // create a new thread
268  add_processing_unit(thread_num, global_thread_num, startup);
269 
270  // set the new threads affinity (on Windows systems)
271  if (!any(mask))
272  {
273  LTM_(debug) //-V128
274  << "run: " << id_.name()
275  << " setting thread affinity on OS thread " //-V128
276  << global_thread_num << " was explicitly disabled.";
277  }
278  }
279 
280  // wait for all threads to have started up
281  startup->wait();
282 
283  HPX_ASSERT(pool_threads == std::size_t(thread_count_.load()));
284  }
285  catch (std::exception const& e)
286  {
287  LTM_(always) << "run: " << id_.name()
288  << " failed with: " << e.what();
289 
290  // trigger the barrier
291  pool_threads -= thread_num;
292  while (pool_threads-- != 0)
293  startup->wait();
294 
295  stop_locked(l);
296  threads_.clear();
297 
298  return false;
299  }
300 
301  LTM_(info) << "run: " << id_.name() << " running";
302  return true;
303  }

305  template <typename Scheduler>
306  void hpx::threads::detail::scheduled_thread_pool<Scheduler>::thread_func(
307  std::size_t thread_num, std::size_t global_thread_num,
308  std::shared_ptr<compat::barrier> startup)
309  {
310  auto const& rp = resource::get_partitioner();
311  topology const& topo = rp.get_topology();
312 
313  // Set the affinity for the current thread.
314  threads::mask_cref_type mask = rp.get_pu_mask(global_thread_num);
315 
316  if (LHPX_ENABLED(debug))
317  topo.write_to_log();
318 
319  error_code ec(lightweight);
320  if (any(mask))
321  {
322  topo.set_thread_affinity_mask(mask, ec);
323  if (ec)
324  {
325  LTM_(warning) //-V128
326  << "thread_func: " << id_.name()
327  << " setting thread affinity on OS thread " //-V128
328  << global_thread_num
329  << " failed with: " << ec.get_message();
330  }
331  }
332  else
333  {
334  LTM_(debug) //-V128
335  << "thread_func: " << id_.name()
336  << " setting thread affinity on OS thread " //-V128
337  << global_thread_num << " was explicitly disabled.";
338  }
339 
340  // Setting priority of worker threads to a lower priority, this
341  // needs to
342  // be done in order to give the parcel pool threads higher
343  // priority
344  if ((mode_ & policies::reduce_thread_priority) &&
345  any(mask & used_processing_units_))
346  {
347  topo.reduce_thread_priority(ec);
348  if (ec)
349  {
350  LTM_(warning) //-V128
351  << "thread_func: " << id_.name()
352  << " reducing thread priority on OS thread " //-V128
353  << global_thread_num
354  << " failed with: " << ec.get_message();
355  }
356  }
357 
358  // manage the number of this thread in its TSS
359  init_tss_helper<Scheduler> tss_helper(
360  *this, thread_num, this->thread_offset_);
361 
362  // wait for all threads to start up before before starting HPX work
363  ++thread_count_;
364  startup->wait();
365 
366  LTM_(info) //-V128
367  << "thread_func: " << id_.name()
368  << " starting OS thread: " << thread_num; //-V128
369 
370  // set state to running
371  std::atomic<hpx::state>& state =
372  sched_->Scheduler::get_state(thread_num);
373  hpx::state oldstate = state.exchange(state_running);
374 
375  HPX_ASSERT(oldstate <= state_running);
376 
377  try
378  {
379  try
380  {
381  manage_active_thread_count count(thread_count_);
382 
383  // run the work queue
384  hpx::threads::coroutines::prepare_main_thread main_thread;
385 
386  // run main Scheduler loop until terminated
387  detail::scheduling_counters counters(
388  executed_threads_[thread_num],
389  executed_thread_phases_[thread_num],
390  tfunc_times_[thread_num],
391  exec_times_[thread_num],
392  idle_loop_counts_[thread_num],
393  busy_loop_counts_[thread_num],
394  tasks_active_[thread_num]);
395 
396  detail::scheduling_callbacks callbacks(
397  util::bind( //-V107
398  &policies::scheduler_base::idle_callback,
399  std::ref(sched_), global_thread_num),
400  detail::scheduling_callbacks::callback_type());
401 
402  if (mode_ & policies::do_background_work)
403  {
404  callbacks.background_ = util::bind( //-V107
405  &policies::scheduler_base::background_callback,
406  std::ref(sched_), global_thread_num);
407  }
408 
409  sched_->Scheduler::set_scheduler_mode(mode_);
410  detail::scheduling_loop(
411  thread_num, *sched_, counters, callbacks);
412 
413  // the OS thread is allowed to exit only if no more HPX
414  // threads exist or if some other thread has terminated
415  HPX_ASSERT(
416  !sched_->Scheduler::get_thread_count(
417  unknown, thread_priority_default, thread_num) ||
418  sched_->Scheduler::get_state(thread_num) == state_terminating);
419  }
420  catch (hpx::exception const& e)
421  {
422  LFATAL_ //-V128
423  << "thread_func: " << id_.name()
424  << " thread_num:" << global_thread_num //-V128
425  << " : caught hpx::exception: " << e.what()
426  << ", aborted thread execution";
427 
428  report_error(global_thread_num, std::current_exception());
429  return;
430  }
431  catch (boost::system::system_error const& e)
432  {
433  LFATAL_ //-V128
434  << "thread_func: " << id_.name()
435  << " thread_num:" << global_thread_num //-V128
436  << " : caught boost::system::system_error: " << e.what()
437  << ", aborted thread execution";
438 
439  report_error(global_thread_num, std::current_exception());
440  return;
441  }
442  catch (std::exception const& e)
443  {
444  // Repackage exceptions to avoid slicing.
445  hpx::throw_with_info(
446  hpx::exception(unhandled_exception, e.what()));
447  }
448  }
449  catch (...)
450  {
451  LFATAL_ //-V128
452  << "thread_func: " << id_.name()
453  << " thread_num:" << global_thread_num //-V128
454  << " : caught unexpected " //-V128
455  "exception, aborted thread execution";
456 
457  report_error(global_thread_num, std::current_exception());
458  return;
459  }
460 
461  LTM_(info) //-V128
462  << "thread_func: " << id_.name()
463  << " thread_num: " << global_thread_num
464  << " : ending OS thread, " //-V128
465  "executed "
466  << executed_threads_[global_thread_num] << " HPX threads";
467  }

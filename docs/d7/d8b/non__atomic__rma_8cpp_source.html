7 // This is the non-atomic version of the random memory access. The array length
8 // is given by the variable array_length and N is the number of random accesses
9 // to this array.
10 //
11 // At present, the entire array is placed in the root locality while all the random
12 // accesses and writes occur on the remote locality.

14 // Note that this is a *non-atomic* example.

72 int hpx_main(po::variables_map &vm)
73 {

81 
82

84  naming::id_type that_prefix;
85 
86  if (!localities.empty()) {
87  // If there is at least one such remote locality, store the first gid in the list
88  that_prefix = localities[0];
89  } else {
90  that_prefix = this_prefix;
91  }
92 
93 
94  {
95  // Create a timer so see how its done
96  util::high_resolution_timer t;
97 
98  std::vector<lcos::future<naming::id_type> > n;
99 
100  int array_length = 6;
101  for (int i=0;i<array_length;i++) {
102  n.push_back(hpx::async<set_initialdata_action>(this_prefix,i));
103  }
104 
105  srand( time(nullptr) );
106 
107  std::vector<lcos::future< void > > future_update;
108  int N = 10; // number of random accesses to the array
109  for (int i=0;i<N;i++) {
110  int rn = rand() % array_length;
111  std::cout << " Random number element accessed: " << rn << std::endl;
112  naming::id_type tmp = n[rn].get();
113  future_update.push_back(hpx::async<update_action>(that_prefix,tmp));
114  }
115 
116  hpx::wait_all(future_update);
117 
118  for (int i=0;i<array_length;i++) {
119  components::access_memory_block<data>
120  result( components::stubs::memory_block::get_data(n[i].get()) );
121  std::cout << " Result index: " << i << " value : "
122  << result->val_ << std::endl;
123  }
124 
125  // What is the elapsed time?
126  elapsed = t.elapsed();
127 
128  // Print out a completion message. The correct answer is 12 for this example.
129  std::cout << "Achieved result of " << result << " in "
130  << elapsed << " seconds."<< std::endl;
131  }
132 
133  // Initiate shutdown of the runtime systems on all localities
134  hpx::finalize(5.0);
135  return 0;

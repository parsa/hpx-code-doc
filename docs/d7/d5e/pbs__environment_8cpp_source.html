64  void pbs_environment::read_nodefile(std::vector<std::string> & nodelist,
65  bool debug, util::runtime_configuration const& cfg)
66  {
67  char *node_file = std::getenv("PBS_NODEFILE");
68  if(!node_file)
69  {
70  valid_ = false;
71  return;

73 
74  std::ifstream ifs(node_file);
75  if (ifs.is_open())
76  {
77  std::set<std::string> nodes;
78  typedef std::set<std::string>::iterator nodes_iterator;
79 
80  bool fill_nodelist = nodelist.empty();
81 
82  if (debug)

84

90  nodes_iterator it = nodes.find(line);
91  if (it == nodes.end())
92  {

94  if (fill_nodelist)
95  {
96  nodelist.push_back(line);
97  }
98  }
99  }
100  }

102  }
103  else

107 
108 #if defined(HPX_HAVE_PARCELPORT_MPI)

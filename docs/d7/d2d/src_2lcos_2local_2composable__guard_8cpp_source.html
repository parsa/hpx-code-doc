88  static void run_guarded(guard& g, detail::guard_task* task)
89  {
90  HPX_ASSERT(task != nullptr);
91  task->check();
92  detail::guard_task* prev = g.task.exchange(task);
93  if (prev != nullptr) {
94  prev->check();
95  detail::guard_task* zero = nullptr;
96  if (!prev->next.compare_exchange_strong(zero, task)) {
97  run_composable(task);
98  free(prev);
99  }
100  } else {
101  run_composable(task);
102  }
103  }

109  stage_task_cleanup(stage_data* sd_, std::size_t n_) : sd(sd_), n(n_) {}
110  ~stage_task_cleanup() {
111  detail::guard_task* zero = nullptr;
112  // The tasks on the other guards had single_task marked,
113  // so they haven't had their next field set yet. Setting
114  // the next field is necessary if they are going to
115  // continue processing.
116  for (std::size_t k=0; k<n; k++) {
117  detail::guard_task* lt = sd->stages[k];
118  lt->check();
119  HPX_ASSERT(!lt->single_guard);
120  zero = nullptr;
121  if (!lt->next.compare_exchange_strong(zero, lt)) {
122  HPX_ASSERT(zero != lt);
123  run_composable(zero);
124  free(lt);
125  }
126  }
127  delete sd;
128  }

173  // This class exists so that a destructor is
174  // used to perform cleanup. By using a destructor
175  // we ensure the code works even if exceptions are
176  // thrown.
177  struct run_composable_cleanup
178  {
179  detail::guard_task* task;
180  run_composable_cleanup(detail::guard_task* task_) : task(task_) {}
181  ~run_composable_cleanup() {
182  detail::guard_task* zero = nullptr;
183  // If single_guard is false, then this is one of the
184  // setup tasks for a multi-guarded task. By not setting
185  // the next field, we halt processing on items queued
186  // to this guard.
187  HPX_ASSERT(task != nullptr);
188  task->check();
189  if (!task->next.compare_exchange_strong(zero, task)) {
190  HPX_ASSERT(zero != nullptr);
191  run_composable(zero);
192  free(task);
193  }
194  }
195  };

125  template<class T>
126  bool discover_events(counter_path_elements& cpe,
127  std::set<std::string> const& tdesc,
128  hpx::performance_counters::counter_info& info,
129  T& gen,
130  hpx::performance_counters::discover_counter_func const& f,
131  hpx::error_code& ec)
132  {
133  typename boost::generator_iterator_generator<T>::type gi =
134  boost::make_generator_iterator(gen);
135  for ( ; *gi != 0; ++gi)
136  {
137  std::set<std::string>::const_iterator it;
138  // iterate over known thread names
139  for (it = tdesc.begin(); it != tdesc.end(); ++it)
140  {
141  cpe.instancename_ = *it;
142  cpe.countername_ = (*gi)->symbol;
143 
144  hpx::performance_counters::counter_status status =
145  get_counter_name(cpe, info.fullname_, ec);
146  if (!status_is_valid(status)) return false;
147  std::string evstr((*gi)->long_descr);
148  info.helptext_ = "returns the count of occurrences of \""+
149  evstr+"\" in "+(*it)+" instance";
150  if (!f(info, std::ref(ec)) || ec) return false;
151  }
152  }
153  return true;
154  }

156  // discover available PAPI counters
157  bool discover_papi_counters_helper(
158  hpx::performance_counters::counter_info const& info,
159  hpx::performance_counters::discover_counter_func const& f,
160  hpx::performance_counters::discover_counters_mode mode, hpx::error_code& ec)
161  {
162  hpx::performance_counters::counter_info cnt_info = info;
163 
164  // decompose the counter name
165  counter_path_elements p;
166  hpx::performance_counters::counter_status status =
167  get_counter_path_elements(info.fullname_, p, ec);
168  if (!status_is_valid(status)) return false;
169 
170  if (p.objectname_ == "papi" && p.parentinstancename_.empty() &&
171  p.instancename_.empty() && p.countername_.empty())
172  { // discover all available PAPI counters
173  // obtain known OS thread labels
174  std::set<std::string> tdesc;
175  find_threads(tdesc, mode);
176 
177  // fill in common path segments for all counters
178  counter_path_elements cpe;
179  cpe.objectname_ = p.objectname_;
180  cpe.parentinstancename_ = "locality#*";
181  cpe.parentinstanceindex_ = -1;
182  cpe.instanceindex_ = -1;
183 
184  // enumerate PAPI presets
185  {
186  util::all_preset_info_gen gen;
187  if (!discover_events(cpe, tdesc, cnt_info, gen, f, ec))
188  return false;
189  }
190  // enumerate PAPI native events
191  for (int ci = 0; ci < PAPI_num_components(); ++ci)
192  {
193  util::native_info_gen gen(ci);
194  if (!discover_events(cpe, tdesc, cnt_info, gen, f, ec))
195  return false;
196  }
197  }
198  else
199  { // no wildcards expected here
200  if (p.instanceindex_ < 0 || p.parentinstanceindex_ < 0) {
201  if (!f(cnt_info, ec) || ec)
202  return false;
203  }
204  else {
205  hpx::util::thread_mapper& tm = get_runtime().get_thread_mapper();
206  std::string lab = p.instancename_+"#"+
207  std::to_string(p.instanceindex_);
208  if (p.objectname_ == "papi" &&
209  p.parentinstancename_ == "locality" &&
210  tm.get_thread_index(lab) != tm.invalid_index &&
211  !p.countername_.empty())
212  { // validate specific PAPI event
213  int code;
214  if (PAPI_event_name_to_code(const_cast<char *>
215  (p.countername_.c_str()), &code) != PAPI_OK)
216  return false;
217  hpx::performance_counters::counter_status status =
218  get_counter_name(p, cnt_info.fullname_, ec);
219  if (!status_is_valid(status)) return false;
220  // this is just validation, no helptext required
221  if (!f(cnt_info, ec) || ec) return false;
222  }
223  // unsupported path components, let runtime handle this
224  else if (!f(cnt_info, ec) || ec) {
225  return false;
226  }
227  }
228  }
229 
230  if (&ec != &hpx::throws)
231  ec = hpx::make_success_code();
232  return true;
233  }

248  // enable multiplexing with specified period for thread tix
249  bool enable_multiplexing(int evset, int interval)
250  {
251  switch (PAPI_get_multiplex(evset))
252  {
253  case PAPI_OK: // not yet multiplexed
254  if (PAPI_assign_eventset_component(evset, 0) != PAPI_OK)
255  return false;
256  if (!interval)
257  { // enable multiplexing with default interval
258  return PAPI_set_multiplex(evset) != PAPI_OK;
259  }
260  break;
261  case PAPI_EINVAL: // already multiplexed
262  if (!interval) return true;
263  break; // still need to change interval
264  default: // error
265  return false;
266  }
267  // force the requested interval
268  PAPI_option_t popt;
269  popt.multiplex.eventset = evset;
270  popt.multiplex.ns = interval;
271  popt.multiplex.flags = PAPI_MULTIPLEX_DEFAULT;
272  return PAPI_set_opt(PAPI_MULTIPLEX, &popt) == PAPI_OK;
273  }

61  "component_instance_counter_creator",

64  return naming::invalid_gid;
65  }
66 
67  hpx::util::function_nonser<std::int64_t()> f =
68  util::bind(&get_instance_count, type);
69  return create_raw_counter(info, std::move(f), ec);
70  }
71  break;
72

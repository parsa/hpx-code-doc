63  { // {{{ gva_cache_key implementation
64  private:
65  typedef std::pair<naming::gid_type, naming::gid_type> key_type;
66 
67  key_type key_;
68 
69  public:
70  gva_cache_key()
71  : key_()
72  {
73  }
74 
75  explicit gva_cache_key(
76  naming::gid_type const& id, std::uint64_t count = 1)
77  : key_(naming::detail::get_stripped_gid(id),
78  naming::detail::get_stripped_gid(id) + (count - 1))
79  {
80  HPX_ASSERT(count);
81  }
82 
83  naming::gid_type get_gid() const
84  {
85  return key_.first;
86  }
87 
88  std::uint64_t get_count() const
89  {
90  naming::gid_type const size = key_.second - key_.first;
91  HPX_ASSERT(size.get_msb() == 0);
92  return size.get_lsb();
93  }
94 
95  friend bool operator<(
96  gva_cache_key const& lhs, gva_cache_key const& rhs)
97  {
98  return lhs.key_.second < rhs.key_.first;
99  }
100 
101  friend bool operator==(
102  gva_cache_key const& lhs, gva_cache_key const& rhs)
103  {
104  // Direct hit
105  if (lhs.key_ == rhs.key_)
106  {
107  return true;
108  }
109 
110  // Is lhs in rhs?
111  if (1 == lhs.get_count() && 1 != rhs.get_count())
112  {
113  return rhs.key_.first <= lhs.key_.first &&
114  lhs.key_.second <= rhs.key_.second;
115  }
116 
117  // Is rhs in lhs?
118  else if (1 != lhs.get_count() && 1 == rhs.get_count())
119  {
120  return lhs.key_.first <= rhs.key_.first &&
121  rhs.key_.second <= lhs.key_.second;
122  }
123 
124  return false;
125  }

372 parcelset::endpoints_type const & addressing_service::resolve_locality(
373  naming::gid_type const & gid
374  , error_code& ec
375  )
376 { // {{{
377  std::unique_lock<mutex_type> l(resolved_localities_mtx_);
378  resolved_localities_type::iterator it = resolved_localities_.find(gid);
379  if (it == resolved_localities_.end() || it->second.empty())
380  {
381  // The locality hasn't been requested to be resolved yet. Do it now.
382  parcelset::endpoints_type endpoints;
383  {
384  hpx::util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
385  endpoints = locality_ns_->resolve_locality(gid);
386  if (endpoints.empty())
387  {
388  std::stringstream strm;
389  strm << "couldn't resolve the given target locality ("
390  << gid << ")";
391  l.unlock();
392 
393  HPX_THROWS_IF(ec, bad_parameter,
394  "addressing_service::resolve_locality",
395  strm.str());
396  return resolved_localities_[naming::invalid_gid];
397  }
398  }
399 
400  // Search again ... might have been added by a different thread already
401  it = resolved_localities_.find(gid);
402  if (it == resolved_localities_.end())
403  {
404  if(HPX_UNLIKELY(!util::insert_checked(resolved_localities_.insert(
405  std::make_pair(
406  gid
407  , endpoints
408  )
409  ), it)))
410  {
411  l.unlock();
412 
413  HPX_THROWS_IF(ec, internal_server_error
414  , "addressing_service::resolve_locality"
415  , "resolved locality insertion failed "
416  "due to a locking error or memory corruption");
417  return resolved_localities_[naming::invalid_gid];
418  }
419  }
420  else if (it->second.empty() && !endpoints.empty())
421  {
422  resolved_localities_[gid] = endpoints;
423  }
424  }
425  return it->second;
426 } // }}}

1460 
1461  // Some examples of calculating the compensated credits below
1462  //
1463  // case pending credits remaining sent to compensated
1464  // no decref decrefs AGAS credits
1465  // ------+---------+---------+------------+--------+-------------
1466  // 1 0 10 0 0 10
1467  // 2 10 9 1 0 10
1468  // 3 10 10 0 0 10
1469  // 4 10 11 0 1 10

1472  bool has_pending_incref = false;
1473  std::int64_t pending_decrefs = 0;

1477

1479

1481  if (matches != refcnt_requests_->end())
1482  {
1483  pending_decrefs = matches->second;
1484  matches->second += credit;
1485

1487 
1488  // If the given incref was fully compensated by a pending decref
1489  // (i.e. match_data is less than 0) then there is no need
1490  // to do anything more.
1491  if (matches->second > 0)

1493  // credit > decrefs (case no 4): store the remaining incref to
1494  // be handled below.
1495  pending_incref = mapping(matches->first, matches->second);
1496  has_pending_incref = true;

1499  }
1500  else if (matches->second == 0)

1503  // pending decref, just remove the pending decref request.
1504  refcnt_requests_->erase(matches);
1505  }
1506  else
1507  {
1508  // credit < decref (case no. 2): do nothing
1509  }
1510  }
1511  else
1512  {
1513  // case no. 1
1514  pending_incref = mapping(raw, credit);
1515  has_pending_incref = true;
1516  }
1517  }
1518

1520  {
1521  // no need to talk to AGAS, acknowledge the incref immediately
1522  return hpx::make_ready_future(pending_decrefs);
1523  }
1524 
1525  naming::gid_type const e_lower = pending_incref.first;
1526 
1527  lcos::future<std::int64_t> f =
1528  primary_ns_.increment_credit(pending_incref.second, e_lower, e_lower);
1529

1531  using util::placeholders::_1;

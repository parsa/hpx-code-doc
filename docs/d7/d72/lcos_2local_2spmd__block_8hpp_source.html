80  void sync_images(std::set<std::size_t> const & images) const
81  {
82  using lock_type = std::lock_guard<mutex_type>;
83 
84  table_type & brs = barriers_.get();
85  typename table_type::iterator it;
86 
87  // Critical section
88  {
89  lock_type lk(mtx_);
90  it = brs.find(images);
91 
92  if(it == brs.end())
93  {
94  it = brs.insert({images,
95  std::make_shared<barrier_type>(images.size())}).first;
96  }
97  }
98 
99  if( images.find(image_id_) != images.end() )
100  {
101  it->second->wait();
102  }
103  }

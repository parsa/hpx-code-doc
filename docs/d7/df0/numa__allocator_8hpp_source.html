74  // memory allocation
75  pointer allocate(size_type cnt,
76  typename std::allocator<void>::const_pointer = nullptr)
77  {
78  // allocate memory
79  pointer p = reinterpret_cast<pointer>(topo_.allocate(cnt * sizeof(T)));
80 
81  // first touch policy, distribute evenly onto executors
82  std::size_t part_size = cnt / executors_.size();
83  std::vector<hpx::future<void> > first_touch;
84  first_touch.reserve(executors_.size());
85 
86  for (std::size_t i = 0; i != executors_.size(); ++i)
87  {
88  pointer begin = p + i * part_size;
89  pointer end = begin + part_size;
90  first_touch.push_back(
91  hpx::parallel::for_each(
92  hpx::parallel::execution::par(hpx::parallel::execution::task)
93  .on(executors_[i])
94  .with(hpx::parallel::static_chunk_size()),
95  begin, end,
96 #if defined(HPX_DEBUG)
97  [this, i]
98 #else
99  []
100 #endif
101  (T& val)
102  {
103  // touch first byte of every object
104  *reinterpret_cast<char*>(&val) = 0;
105 
106 #if defined(HPX_DEBUG)
107  // make sure memory was placed appropriately
108  hpx::threads::mask_type mem_mask =
109  topo_.get_thread_affinity_mask_from_lva(
110  reinterpret_cast<hpx::naming::address_type>(&val));
111 
112  std::size_t thread_num = hpx::get_worker_thread_num();
113  hpx::threads::mask_cref_type thread_mask =
114  hpx::parallel::execution::get_pu_mask(
115  executors_[i], topo_, thread_num);
116 
117  HPX_ASSERT(threads::mask_size(mem_mask) ==
118  threads::mask_size(thread_mask));
119  HPX_ASSERT(threads::bit_and(mem_mask, thread_mask,
120  threads::mask_size(mem_mask)));
121 #endif
122  })
123  );

63 double ireceive(hpx::naming::id_type dest, std::size_t size, std::size_t window_size)
64 {
65  int loop = LOOP_SMALL;
66  int skip = SKIP_SMALL;
67 
68  if (size > LARGE_MESSAGE_SIZE) {
69  loop = LOOP_LARGE;
70  skip = SKIP_LARGE;
71  }
72 
73  // align used buffers on page boundaries
74  unsigned long align_size = getpagesize();
75  HPX_ASSERT(align_size <= MAX_ALIGNMENT);
76 
77  char* aligned_send_buffer = align_buffer(send_buffer, align_size);
78  std::memset(aligned_send_buffer, 'a', size);
79 
80  hpx::util::high_resolution_timer t;
81 
82  isend_action send;
83  for (int i = 0; i != loop + skip; ++i) {
84  // do not measure warm up phase
85  if (i == skip)
86  t.restart();
87 
88  typedef hpx::serialization::serialize_buffer<char> buffer_type;
89 
90  using hpx::parallel::for_each;
91  using hpx::parallel::execution::par;
92 
93  std::size_t const start = 0;
94 
95  auto range = boost::irange(start, window_size);
96  for_each(par, std::begin(range), std::end(range),
97  [&](std::uint64_t j)
98  {
99  send(dest, buffer_type(aligned_send_buffer, size,
100  buffer_type::reference));
101  });
102  }
103 
104  double elapsed = t.elapsed();
105  return (elapsed * 1e6) / (2 * loop * window_size);
106 }

118 void run_benchmark(boost::program_options::variables_map & vm)
119 {
120  std::vector<hpx::id_type> localities = hpx::find_all_localities();
121  std::size_t window_size = vm["window-size"].as<std::size_t>();
122  std::size_t pairs = localities.size() / 2;
123 
124  for (std::size_t size = 1; size <= MAX_MSG_SIZE; size *= 2)
125  {
126  std::vector<hpx::future<double> > benchmarks;
127 
128  for (std::uint32_t locality_id = 0; locality_id !=
129  localities.size(); ++locality_id)
130  {
131  ireceive_action receive;
132 
133  std::size_t partner = 0;
134  if(locality_id < pairs)
135  partner = locality_id + pairs;
136  else
137  partner = locality_id - pairs;
138 
139  benchmarks.push_back(hpx::async(receive,
140  localities[locality_id], localities[partner], size,
141  window_size));
142  }
143 
144  double total_latency = std::accumulate(
145  benchmarks.begin(), benchmarks.end(), 0.0,
146  [](double sum, hpx::future<double>& f)
147  {
148  return sum + f.get();
149  });
150 
151  hpx::cout << std::left << std::setw(10) << size
152  << total_latency / (2. * pairs)
153  << hpx::endl << hpx::flush;
154  }

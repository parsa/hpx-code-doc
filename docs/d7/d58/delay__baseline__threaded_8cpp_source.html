131 int app_main(
132  variables_map& vm
133  )
134 {
135  if (0 == tasks)
136  throw std::invalid_argument("error: count of 0 tasks specified\n");
137 
138  std::vector<double> elapsed(threads - 1);
139  std::vector<compat::thread> workers;
140  hpx::compat::barrier b(threads - 1);
141 
142  for (std::uint32_t i = 0; i != threads - 1; ++i)
143  {
144  workers.push_back(compat::thread(invoke_n_workers,
145  std::ref(b), std::ref(elapsed[i]), tasks));
146  }
147 
148  double total_elapsed = 0;
149 
150  invoke_n_workers_nowait(total_elapsed, tasks);
151 
152  for (compat::thread& thread : workers)
153  {
154  if (thread.joinable())
155  thread.join();
156  }
157 
158  for (std::uint64_t i = 0; i < elapsed.size(); ++i)
159  {
160  //cout << i << " " << elapsed[i] << "\n";
161  total_elapsed += elapsed[i];
162  }
163 
164  // Print out the results.
165  print_results(vm, total_elapsed / double(threads)
166  , (total_elapsed * 1e6) / double(tasks * threads));
167 
168  return 0;
169 }

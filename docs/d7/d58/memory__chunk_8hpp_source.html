159  char *allocate(size_type size)
160  {
161  std::lock_guard<mutex_type> l(mtx_);
162  check_invariants_locked();
163 
164  if(!data_) charge();
165 
166  iterator result = 0;
167 
168  HPX_ASSERT(size <= chunk_size_);
169 
170  typedef free_list_type::iterator free_iterator;
171  free_iterator it = free_list_.lower_bound(size);
172  if(it != free_list_.end())
173  {
174  result = it->second;
175  HPX_ASSERT(it->first >= size);
176  if(it->first == size)
177  {
178  free_list_.erase(it);
179  check_invariants_locked(result, size);
180  return result;
181  }
182  size_type new_size = it->first - size;
183  iterator new_free_block = it->second + size;
184  free_list_.erase(it);
185  if(new_free_block + new_size == current_)
186  {
187  current_ = new_free_block;
188  allocated_ -= new_size;
189  }
190  else
191  {
192  free_list_.insert(std::make_pair(new_size, new_free_block));
193  }
194  check_invariants_locked(result, size);
195  return result;
196  }
197 
198  if(size + allocated_ <= chunk_size_)
199  {
200  result = current_;
201  current_ += size;
202  allocated_ += size;
203  check_invariants_locked(result, size);
204  return result;
205  }
206  check_invariants_locked(0, size);
207  return 0;
208  }

210  bool deallocate(char * p, size_type size)
211  {
212  std::lock_guard<mutex_type> l(mtx_);
213  HPX_ASSERT(data_);
214  if(!contains_locked(p))
215  return false;
216 
217  HPX_ASSERT(p != current_);
218 
219  check_invariants_locked(p, size);
220 
221  if(current_ - size == p)
222  {
223  allocated_ -= size;
224  current_ -= size;
225  check_invariants_locked(0, size);
226  return true;
227  }
228 
229  iterator p_end = p + size;
230  for(free_list_type::iterator jt = free_list_.begin();
231  jt != free_list_.end(); ++jt)
232  {
233  // check if the chunk to be deleted is left to the chunk in
234  // the free list
235  iterator chunk_begin = jt->second;
236  if(p_end == chunk_begin)
237  {
238  size_type new_size = size + jt->first;
239  free_list_.erase(jt);
240  if(p + new_size == current_)
241  {
242  current_ -= new_size;
243  allocated_ -= new_size;
244  check_invariants_locked(0, size);
245  return true;
246  }
247  free_list_.insert(std::make_pair(new_size, p));
248  check_invariants_locked(0, size);
249  return true;
250  }
251 
252  // check if the chunk to be deleted is right to the chunk in
253  // the free list
254  iterator chunk_end = chunk_begin + jt->first;
255  if(p == chunk_end)
256  {
257  size_type new_size = size + jt->first;
258  iterator new_free = jt->second;
259  free_list_.erase(jt);
260  if(new_free + new_size == current_)
261  {
262  current_ -= new_size;
263  allocated_ -= new_size;
264  check_invariants_locked(0, size);
265  return true;
266  }
267  free_list_.insert(std::make_pair(new_size, new_free));
268  check_invariants_locked(0, size);
269  return true;
270  }
271  // Since we can't have overlapping regions, we don't need more
272  // checks ...
273  }
274  free_list_.insert(std::make_pair(size, p));
275  check_invariants_locked(0, size);
276  return true;
277  }

65  // helper class for switching thread state in and out during execution
66  class switch_status
67  {
68  public:
69  switch_status (thread_data* t, thread_state prev_state)
70  : thread_(t), prev_state_(prev_state),
71  next_thread_id_(nullptr),
72  need_restore_state_(t->set_state_tagged(active, prev_state_, orig_state_))
73  {}
74 
75  ~switch_status ()
76  {
77  if (need_restore_state_)
78  store_state(prev_state_);
79  }
80 
81  bool is_valid() const { return need_restore_state_; }
82 
83  // allow to change the state the thread will be switched to after
84  // execution
85  thread_state operator=(thread_result_type && new_state)
86  {
87  prev_state_ = thread_state(new_state.first,
88  prev_state_.state_ex(), prev_state_.tag() + 1);
89  next_thread_id_ = std::move(new_state.second);
90  return prev_state_;
91  }
92 
93  // Get the state this thread was in before execution (usually pending),
94  // this helps making sure no other worker-thread is started to execute this
95  // HPX-thread in the meantime.
96  thread_state_enum get_previous() const
97  {
98  return prev_state_.state();
99  }
100 
101  // This restores the previous state, while making sure that the
102  // original state has not been changed since we started executing this
103  // thread. The function returns true if the state has been set, false
104  // otherwise.
105  bool store_state(thread_state& newstate)
106  {
107  disable_restore();
108  if (thread_->restore_state(prev_state_, orig_state_)) {
109  newstate = prev_state_;
110  return true;
111  }
112  return false;
113  }
114 
115  // disable default handling in destructor
116  void disable_restore() { need_restore_state_ = false; }
117 
118  thread_data* get_next_thread() const
119  {
120  // we know that the thread-id is just the pointer to the thread_data
121  return reinterpret_cast<thread_data*>(next_thread_id_.get());
122  }
123 
124  private:
125  thread_data* thread_;
126  thread_state prev_state_;
127  thread_state orig_state_;
128  thread_id_type next_thread_id_;
129  bool need_restore_state_;
130  };

290  template <typename SchedulingPolicy>
291  thread_id_type create_background_thread(SchedulingPolicy& scheduler,
292  scheduling_callbacks& callbacks, std::shared_ptr<bool>& background_running,
293  std::size_t num_thread, std::int64_t& idle_loop_count)
294  {
295  thread_id_type background_thread;
296  background_running.reset(new bool(true));
297  thread_init_data background_init(
298  [&, background_running](thread_state_ex_enum) -> thread_result_type
299  {
300  while(*background_running)
301  {
302  if (callbacks.background_())
303  {
304  // we only update the idle_loop_count if
305  // background_running is true. If it was false, this task
306  // was given back to the scheduler.
307  if (*background_running)
308  idle_loop_count = 0;
309  }
310  hpx::this_thread::suspend(hpx::threads::pending,
311  "background_work");
312  }
313  return thread_result_type(terminated, nullptr);
314  },
315  hpx::util::thread_description("background_work"),
316  0,
317  thread_priority_high_recursive,
318  num_thread,
319  get_stack_size(thread_stacksize_large),
320  &scheduler);
321 
322  // Create in suspended to prevent the thread from being scheduled
323  // directly...
324  scheduler.SchedulingPolicy::create_thread(background_init,
325  &background_thread, suspended, true, hpx::throws, num_thread);
326  HPX_ASSERT(background_thread);
327  // We can now set the state to pending
328  background_thread->set_state(pending);
329  return background_thread;
330  }

332  // This function tries to invoke the background work thread. It returns
333  // false when we need to give the background thread back to scheduler
334  // and create a new one that is supposed to be executed inside the
335  // scheduling_loop, true otherwise
336  template <typename SchedulingPolicy>
337  bool call_background_thread(thread_id_type& background_thread,
338  thread_data*& next_thrd, SchedulingPolicy& scheduler, std::size_t num_thread,
339  bool running)
340  {
341  if (HPX_UNLIKELY(background_thread))
342  {
343  thread_state state = background_thread->get_state();
344  thread_state_enum state_val = state.state();
345 
346  if (HPX_LIKELY(pending == state_val))
347  {
348  {
349  // tries to set state to active (only if state is still
350  // the same as 'state')
351  detail::switch_status thrd_stat (background_thread.get(), state);
352  if (HPX_LIKELY(thrd_stat.is_valid() &&
353  thrd_stat.get_previous() == pending))
354  {
355 #if defined(HPX_HAVE_APEX)
356  // get the APEX data pointer, in case we are resuming the
357  // thread and have to restore any leaf timers from
358  // direct actions, etc.
359 
360  // the address of tmp_data is getting stored
361  // by APEX during this call
362  util::apex_wrapper apex_profiler(
363  background_thread->get_description(),
364  background_thread->get_apex_data());
365 
366  thrd_stat = (*background_thread)();
367 
368  if (thrd_stat.get_previous() == terminated)
369  {
370  apex_profiler.stop();
371  }
372  else
373  {
374  apex_profiler.yield();
375  }
376 #else
377  thrd_stat = (*background_thread)();
378 #endif
379  thread_data *next = thrd_stat.get_next_thread();
380  if (next != nullptr && next != background_thread.get())
381  {
382  if (next_thrd == nullptr)
383  {
384  next_thrd = next;
385  }
386  else
387  {
388  scheduler.SchedulingPolicy::schedule_thread(
389  next, num_thread);
390  }
391  }
392  }
393  thrd_stat.store_state(state);
394  state_val = state.state();
395 
396  if (HPX_LIKELY(state_val == pending_boost))
397  {
398  background_thread->set_state(pending);
399  }
400  else if(terminated == state_val)
401  {
402  std::int64_t busy_count = 0;
403  scheduler.SchedulingPolicy::destroy_thread(
404  background_thread.get(), busy_count);
405  background_thread.reset();
406  }
407  else if(suspended == state_val)
408  {
409  return false;
410  }
411  }
412  return true;
413  }
414  // This should never be reached ... we should only deal with pending
415  // here.
416  HPX_ASSERT(false);
417  }
418  return true;
419  }

421  template <typename SchedulingPolicy>
422  void scheduling_loop(std::size_t num_thread, SchedulingPolicy& scheduler,
423  scheduling_counters& counters, scheduling_callbacks& params)
424  {
425  std::atomic<hpx::state>& this_state = scheduler.get_state(num_thread);
426 
427  util::itt::stack_context ctx; // helper for itt support
428  util::itt::domain domain = hpx::get_thread_itt_domain();
429  util::itt::id threadid(domain, &scheduler);
430  util::itt::string_handle task_id("task_id");
431  util::itt::string_handle task_phase("task_phase");
432 
433 // util::itt::frame_context fctx(domain);
434 
435  std::int64_t& idle_loop_count = counters.idle_loop_count_;
436  std::int64_t& busy_loop_count = counters.busy_loop_count_;
437 
438  idle_collect_rate idle_rate(counters.tfunc_time_, counters.exec_time_);
439  tfunc_time_wrapper tfunc_time_collector(idle_rate);
440 
441  scheduler.SchedulingPolicy::start_periodic_maintenance(this_state);
442 
443  // spin for some time after queues have become empty
444  bool may_exit = false;
445  thread_data* thrd = nullptr;
446  thread_data* next_thrd = nullptr;
447 
448  std::shared_ptr<bool> background_running = nullptr;
449  thread_id_type background_thread = nullptr;
450 
451  if ((scheduler.get_scheduler_mode() & policies::do_background_work) &&
452  num_thread < params.max_background_threads_ &&
453  !params.background_.empty())
454  {
455  background_thread = create_background_thread(scheduler, params,
456  background_running, num_thread, idle_loop_count);
457  }
458 
459  while (true) {
460  // Get the next HPX thread from the queue
461  thrd = next_thrd;
462  bool running = this_state.load(
463  std::memory_order_relaxed) < state_stopping;
464 
465  if (HPX_LIKELY(thrd ||
466  scheduler.SchedulingPolicy::get_next_thread(
467  num_thread, running, idle_loop_count, thrd)))
468  {
469  tfunc_time_wrapper tfunc_time_collector(idle_rate);
470 
471  idle_loop_count = 0;
472  ++busy_loop_count;
473 
474  may_exit = false;
475 
476  // Only pending HPX threads will be executed.
477  // Any non-pending HPX threads are leftovers from a set_state()
478  // call for a previously pending HPX thread (see comments above).
479  thread_state state = thrd->get_state();
480  thread_state_enum state_val = state.state();
481 
482  detail::write_old_state_log(num_thread, thrd, state_val);
483 
484  if (HPX_LIKELY(pending == state_val))
485  {
486  // switch the state of the thread to active and back to
487  // what the thread reports as its return value
488 
489  {
490  // tries to set state to active (only if state is still
491  // the same as 'state')
492  detail::switch_status thrd_stat (thrd, state);
493  if (HPX_LIKELY(thrd_stat.is_valid() &&
494  thrd_stat.get_previous() == pending))
495  {
496  tfunc_time_wrapper tfunc_time_collector(idle_rate);
497 
498  // thread returns new required state
499  // store the returned state in the thread
500  {
501  is_active_wrapper utilization(counters.is_active_);
502 #if HPX_HAVE_ITTNOTIFY != 0 && !defined(HPX_HAVE_APEX)
503  util::itt::caller_context cctx(ctx);
504 // util::itt::undo_frame_context undoframe(fctx);
505  util::itt::task task(domain, thrd->get_description());
506  task.add_metadata(task_id, thrd);
507  task.add_metadata(task_phase, thrd->get_thread_phase());
508 #endif
509  // Record time elapsed in thread changing state
510  // and add to aggregate execution time.
511  exec_time_wrapper exec_time_collector(idle_rate);
512 
513 #if defined(HPX_HAVE_APEX)
514  // get the APEX data pointer, in case we are resuming the
515  // thread and have to restore any leaf timers from
516  // direct actions, etc.
517 
518  // the address of tmp_data is getting stored
519  // by APEX during this call
520  util::apex_wrapper apex_profiler(
521  thrd->get_description(),
522  thrd->get_apex_data());
523 
524  thrd_stat = (*thrd)();
525 
526  if (thrd_stat.get_previous() == terminated)
527  {
528  apex_profiler.stop();
529  }
530  else
531  {
532  apex_profiler.yield();
533  }
534 #else
535  thrd_stat = (*thrd)();
536 #endif
537  }
538 
539 #ifdef HPX_HAVE_THREAD_CUMULATIVE_COUNTS
540  ++counters.executed_thread_phases_;
541 #endif
542  }
543  else {
544  // some other worker-thread got in between and started
545  // executing this HPX-thread, we just continue with
546  // the next one
547  thrd_stat.disable_restore();
548  detail::write_new_state_log_warning(
549  num_thread, thrd, state_val, "no execution");
550  continue;
551  }
552 
553  // store and retrieve the new state in the thread
554  if (HPX_UNLIKELY(!thrd_stat.store_state(state))) {
555  // some other worker-thread got in between and changed
556  // the state of this thread, we just continue with
557  // the next one
558  detail::write_new_state_log_warning(
559  num_thread, thrd, state_val, "no state change");
560  continue;
561  }
562 
563  state_val = state.state();
564 
565  // any exception thrown from the thread will reset its
566  // state at this point
567 
568  // handle next thread id if given (switch directly to
569  // this thread)
570  next_thrd = thrd_stat.get_next_thread();
571  }
572 
573  //detail::write_new_state_log_debug(num_thread, thrd,
574  // state_val, "normal");
575 
576  // Re-add this work item to our list of work items if the HPX
577  // thread should be re-scheduled. If the HPX thread is suspended
578  // now we just keep it in the map of threads.
579  if (HPX_UNLIKELY(state_val == pending))
580  {
581  if (HPX_LIKELY(next_thrd == nullptr)) {
582  // schedule other work
583  scheduler.SchedulingPolicy::wait_or_add_new(
584  num_thread, running, idle_loop_count);
585  }
586 
587  // schedule this thread again, make sure it ends up at
588  // the end of the queue
589  scheduler.SchedulingPolicy::schedule_thread_last(thrd,
590  num_thread);
591  scheduler.SchedulingPolicy::do_some_work(num_thread);
592  }
593  else if (HPX_UNLIKELY(state_val == pending_boost))
594  {
595  thrd->set_state(pending);
596 
597  if (HPX_LIKELY(next_thrd == nullptr))
598  {
599  // reschedule this thread right away if the
600  // background work will be triggered
601  if (HPX_UNLIKELY(
602  busy_loop_count > params.max_busy_loop_count_))
603  {
604  next_thrd = thrd;
605  }
606  else
607  {
608  // schedule other work
609  scheduler.SchedulingPolicy::wait_or_add_new(
610  num_thread, running, idle_loop_count);
611 
612  // schedule this thread again immediately with
613  // boosted priority
614  scheduler.SchedulingPolicy::schedule_thread(
615  thrd, num_thread, thread_priority_boost);
616  scheduler.SchedulingPolicy::do_some_work(
617  num_thread);
618  }
619  }
620  else if (HPX_LIKELY(next_thrd != thrd))
621  {
622  // schedule this thread again immediately with
623  // boosted priority
624  scheduler.SchedulingPolicy::schedule_thread(
625  thrd, num_thread, thread_priority_boost);
626  scheduler.SchedulingPolicy::do_some_work(
627  num_thread);
628  }
629  }
630  }
631  else if (HPX_UNLIKELY(active == state_val)) {
632  LTM_(warning) << "tfunc(" << num_thread << "): " //-V128
633  "thread(" << thrd->get_thread_id().get() << "), "
634  "description(" << thrd->get_description() << "), "
635  "rescheduling";
636 
637  // re-schedule thread, if it is still marked as active
638  // this might happen, if some thread has been added to the
639  // scheduler queue already but the state has not been reset
640  // yet
641  //
642  // REVIEW: Passing a specific target thread may set off
643  // the round robin queuing.
644  scheduler.SchedulingPolicy::schedule_thread(thrd, num_thread);
645  }
646 
647  // Remove the mapping from thread_map_ if HPX thread is depleted
648  // or terminated, this will delete the HPX thread as all
649  // references go out of scope.
650  // REVIEW: what has to be done with depleted HPX threads?
651  if (HPX_LIKELY(state_val == depleted || state_val == terminated))
652  {
653 #ifdef HPX_HAVE_THREAD_CUMULATIVE_COUNTS
654  ++counters.executed_threads_;
655 #endif
656  scheduler.SchedulingPolicy::destroy_thread(thrd, busy_loop_count);
657  }
658  }
659 
660  // if nothing else has to be done either wait or terminate
661  else {
662  ++idle_loop_count;
663 
664  if (scheduler.SchedulingPolicy::wait_or_add_new(
665  num_thread, running, idle_loop_count))
666  {
667  // clean up terminated threads one more time before existing
668  if (scheduler.SchedulingPolicy::cleanup_terminated(true))
669  {
670  // if this is an inner scheduler, exit immediately
671  if (!(scheduler.get_scheduler_mode() & policies::delay_exit))
672  {
673  this_state.store(state_stopped);

732  }
733  }

752  }

57  template <int SIZE>
58  struct header
59  {
60  static constexpr unsigned int header_block_size = sizeof(detail::header_block);
61  static constexpr unsigned int data_size_ = SIZE - header_block_size;
62  static const unsigned int chunk_flag = 0x01;
63  static const unsigned int message_flag = 0x02;
64  //
65  private:
66  detail::header_block message_header;
67  std::array<char, data_size_> data_;
68 
69  public:
70  //
71  template <typename Buffer>
72  header(Buffer const & buffer, uint32_t tag)
73  {
74  message_header.flags = 0;
75  message_header.tag = tag;
76  message_header.size = static_cast<uint32_t>(buffer.size_);
77  message_header.num_chunks =
78  std::make_pair(buffer.num_chunks_.first, buffer.num_chunks_.second);
79 
80  // find out how much space is needed for chunk information
81  const std::vector<serialization::serialization_chunk>&
82  chunks = buffer.chunks_;
83  size_t chunkbytes = chunks.size() *
84  sizeof(serialization::serialization_chunk);
85  // can we send the chunk info inside the header
86  if (chunkbytes <= data_size_) {
87  message_header.flags |= chunk_flag;
88  message_header.chunk_info.piggyback.offset = 0;
89  message_header.chunk_info.piggyback.bytes = chunkbytes;
90  std::memcpy(&data_[0], chunks.data(), chunkbytes);
91  LOG_DEBUG_MSG("Chunkbytes is " << decnumber(chunkbytes) <<
92  "header_block_size "
93  << decnumber(sizeof(detail::header_block)));
94  }
95  else {
96  message_header.flags &= ~chunk_flag;
97  chunkbytes = 0;
98  }
99 
100  // the end of header position will be start of piggyback data
101  message_header.message_info.piggyback.offset = chunkbytes;
102 
103  // can we send main message chunk as well as other information
104  if (buffer.data_.size() <= (data_size_ - chunkbytes)) {
105  message_header.flags |= message_flag;
106  }
107  else {
108  message_header.flags &= ~message_flag;
109  }
110  }

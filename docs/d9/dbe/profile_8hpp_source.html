107  enum type {
108  gather = 0,
109  writer,
110  filter,
111  on_other_thread,

113  last_ = on_other_thread
114  };

116  typedef ::hpx::util::function_nonser<void(const std::string&)> log_function;
117  static void nothing(const std::string&) {}
118 
119  static compute & inst() {
120  static compute c;

122  }

124  void add_period(const ::std::int64_t & mcs, type t) {
125  // note : this will work even after this object has been destroyed
126  // (since it's a global)
127  m_cpu_times[t] += mcs;
128  }
129 
130  void log_results(log_function l) {
131  m_log = l;
132  }
133 protected:
134  compute() {
135  m_log = &nothing;
136  }
137  ~compute() {
138  dump_results();
139  }
140 private:
141  void dump_results() {
142  std::ostringstream out;
143  out << "gather time: "
144  << (m_cpu_times[gather] / 1000000) << "."
145  << (m_cpu_times[gather] % 1000000) << " seconds " << std::endl;
146  out << "write time: "
147  << (m_cpu_times[writer] / 1000000) << "."
148  << (m_cpu_times[writer] % 1000000) << " seconds " << std::endl;
149  out << "filter time: "
150  << (m_cpu_times[filter] / 1000000) << "."
151  << (m_cpu_times[filter] % 1000000) << " seconds " << std::endl;
152  out << "otherthread time: "
153  << (m_cpu_times[on_other_thread] / 1000000) << "."
154  << (m_cpu_times[on_other_thread] % 1000000) << " seconds " << std::endl;
155  m_log( out.str() );
156  }
157 
158 private:
159  compute(const compute&);
160  void operator=(const compute&);
161 
162 private:
163  // the CPU time taken for each activity
164  //
165  // note: we don't use std::vector;
166  // we want this to be available even after destruction
167  // (if logs are used after they've been destroyed
168  std::int64_t m_cpu_times[ last_ + 1];
169 
170  // where should we dump the results?
171  log_function m_log;
172 };

519  // initialization of arrays of wrappers.
520  template <typename T = Component>
521  static typename std::enable_if<
522  std::is_default_constructible<T>::value, value_type*
523  >::type
524  create(std::size_t count = 1)
525  {
526  // allocate the memory
527  void* p = heap_type::alloc(count);
528  if (nullptr == p) {
529  HPX_THROW_STD_EXCEPTION(std::bad_alloc(),
530  "managed_component::create");
531  }
532 
533  if (1 == count)
534  return new (p) value_type();
535 
536  // call constructors
537  std::size_t succeeded = 0;
538  try {
539  value_type* curr = reinterpret_cast<value_type*>(p);
540  for (std::size_t i = 0; i != count; ++i, ++curr) {
541  // call placement new, might throw
542  new (curr) value_type();
543  ++succeeded;
544  }
545  }
546  catch (...) {
547  // call destructors for successfully constructed objects
548  value_type* curr = reinterpret_cast<value_type*>(p);
549  for (std::size_t i = 0; i != succeeded; ++i)
550  {
551  curr->finalize();
552  curr->~derived_type();
553  ++curr;
554  }
555  heap_type::free(p, count); // free memory
556  throw; // rethrow
557  }
558  return reinterpret_cast<value_type*>(p);
559  }
560 
562  // de-allocation of arrays of wrappers
563  static void destroy(value_type* p, std::size_t count = 1)
564  {
565  if (nullptr == p || 0 == count)
566  return; // do nothing if given a nullptr pointer
567 
568  // call destructors for all managed_component instances
569  value_type* curr = p;
570  for (std::size_t i = 0; i != count; ++i)
571  {
572  curr->finalize();
573  curr->~derived_type();
574  ++curr;
575  }
576 
577  // free memory itself
578  heap_type::free(p, count);
579  }

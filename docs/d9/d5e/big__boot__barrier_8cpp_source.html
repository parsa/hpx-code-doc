390 // remote call to AGAS
391 void register_worker(registration_header const& header)
392 {
393  // This lock acquires the bbb mutex on creation. When it goes out of scope,
394  // its dtor calls big_boot_barrier::notify().
395  big_boot_barrier::scoped_lock lock(get_big_boot_barrier());
396 
397  runtime& rt = get_runtime();
398  naming::resolver_client& agas_client = rt.get_agas_client();
399 
400  if (HPX_UNLIKELY(agas_client.is_connecting()))
401  {
402  HPX_THROW_EXCEPTION(
403  internal_server_error
404  , "agas::register_worker"
405  , "a locality in connect mode cannot be an AGAS server.");
406  }
407 
408  if (HPX_UNLIKELY(!agas_client.is_bootstrap()))
409  {
410  HPX_THROW_EXCEPTION(
411  internal_server_error
412  , "agas::register_worker"
413  , "registration parcel received by non-bootstrap locality.");
414  }
415 
416  naming::gid_type prefix = header.prefix;
417  if (prefix != naming::invalid_gid && naming::get_locality_id_from_gid(prefix) == 0)
418  {
419  HPX_THROW_EXCEPTION(internal_server_error
420  , "agas::register_worker"
421  , boost::str(
422  boost::format("worker node (%s) can't suggest locality_id zero, "
423  "this is reserved for the console") %
424  header.endpoints));
425  return;
426  }
427 
428  if (!agas_client.register_locality(header.endpoints, prefix, header.num_threads))
429  {
430  HPX_THROW_EXCEPTION(internal_server_error
431  , "agas::register_worker"
432  , boost::str(
433  boost::format("attempt to register locality %s more than once") %
434  header.endpoints));
435  return;
436  }
437 
438 // naming::gid_type primary_ns_gid(
439 // primary_namespace::get_service_instance(prefix));
440 // naming::address primary_ns_address(prefix
441 // , components::get_component_type<agas::server::primary_namespace>()
442 // , header.primary_ns_ptr);
443 // agas_client.bind_local(primary_ns_gid, primary_ns_address);
444 
445 // naming::gid_type symbol_ns_gid(
446 // symbol_namespace::get_service_instance(prefix));
447 // naming::address symbol_ns_address(prefix
448 // , components::get_component_type<agas::server::symbol_namespace>()
449 // , header.symbol_ns_ptr);
450 // agas_client.bind_local(symbol_ns_gid, symbol_ns_address);
451 
452  naming::address locality_addr(hpx::get_locality(),
453  server::locality_namespace::get_component_type(),
454  agas_client.locality_ns_->ptr());
455  naming::address primary_addr(hpx::get_locality(),
456  server::primary_namespace::get_component_type(),
457  agas_client.primary_ns_.ptr());
458  naming::address component_addr(hpx::get_locality(),
459  server::component_namespace::get_component_type(),
460  agas_client.component_ns_->ptr());
461  naming::address symbol_addr(hpx::get_locality(),
462  server::symbol_namespace::get_component_type(),
463  agas_client.symbol_ns_.ptr());
464 
465  // assign cores to the new locality
466  std::uint32_t first_core = rt.assign_cores(header.hostname,
467  header.cores_needed);
468 
469  big_boot_barrier & bbb = get_big_boot_barrier();
470 
471  // register all ids
472  detail::assigned_id_sequence assigned_ids(header.typenames);
473 
474  notification_header hdr (prefix, bbb.here(), locality_addr, primary_addr
475  , component_addr, symbol_addr, rt.get_config().get_num_localities()
476  , first_core, bbb.get_endpoints(), assigned_ids);
477 
478  parcelset::locality dest;
479  parcelset::locality here = bbb.here();
480  for (parcelset::endpoints_type::value_type const & loc : header.endpoints)
481  {
482  if(loc.second.type() == here.type())
483  {
484  dest = loc.second;
485  break;
486  }
487  }
488 
489  // collect endpoints from all registering localities
490  bbb.add_locality_endpoints(naming::get_locality_id_from_gid(prefix),
491  header.endpoints);
492 
493  // TODO: Handle cases where localities try to connect to AGAS while it's
494  // shutting down.
495  if (agas_client.get_status() != state_starting)
496  {
497  // We can just send the parcel now, the connecting locality isn't a part
498  // of startup synchronization.
499  get_big_boot_barrier().apply_late(
500  0
501  , naming::get_locality_id_from_gid(prefix)
502  , dest
503  , notify_worker_action()
504  , std::move(hdr));
505  }
506 
507  else
508  {
509  // AGAS is starting up; this locality is participating in startup
510  // synchronization.
511 
512  // delay the final response until the runtime system is up and running
513  util::unique_function_nonser<void()>* thunk =
514  new util::unique_function_nonser<void()>(
515  util::bind(
516  util::one_shot(&big_boot_barrier::apply_notification)
517  , std::ref(get_big_boot_barrier())
518  , 0
519  , naming::get_locality_id_from_gid(prefix)
520  , dest
521  , std::move(hdr)));
522  get_big_boot_barrier().add_thunk(thunk);
523  }
524 }

526 // AGAS callback to client (first round trip response)
527 void notify_worker(notification_header const& header)
528 {
529  // This lock acquires the bbb mutex on creation. When it goes out of scope,
530  // it's dtor calls big_boot_barrier::notify().
531  big_boot_barrier::scoped_lock lock(get_big_boot_barrier());
532 
533  // register all ids with this locality
534  header.ids.register_ids_on_worker_loc();
535 
536  runtime& rt = get_runtime();
537  naming::resolver_client& agas_client = rt.get_agas_client();
538 
539  if (HPX_UNLIKELY(agas_client.get_status() != state_starting))
540  {
541  std::ostringstream strm;
542  strm << "locality " << rt.here() << " has launched early";
543  HPX_THROW_EXCEPTION(internal_server_error,
544  "agas::notify_worker",
545  strm.str());
546  }
547 
548  util::runtime_configuration& cfg = rt.get_config();
549 
550  // set our prefix
551  agas_client.set_local_locality(header.prefix);
552  agas_client.register_console(header.agas_endpoints);
553  cfg.parse("assigned locality",
554  boost::str(boost::format("hpx.locality!=%1%")
555  % naming::get_locality_id_from_gid(header.prefix)));
556 
557  // store the full addresses of the agas servers in our local service
558  agas_client.component_ns_.reset(
559  new detail::hosted_component_namespace(header.component_ns_address));
560  agas_client.locality_ns_.reset(
561  new detail::hosted_locality_namespace(header.locality_ns_address));
562  naming::gid_type const& here = hpx::get_locality();
563 
564  // register runtime support component
565  naming::gid_type runtime_support_gid(header.prefix.get_msb()
566  , rt.get_runtime_support_lva());
567  naming::address const runtime_support_address(here
568  , components::get_component_type<components::server::runtime_support>()
569  , rt.get_runtime_support_lva());
570  agas_client.bind_local(runtime_support_gid, runtime_support_address);
571 
572  runtime_support_gid.set_lsb(std::uint64_t(0));
573  agas_client.bind_local(runtime_support_gid, runtime_support_address);
574 
575  naming::gid_type const memory_gid(header.prefix.get_msb()
576  , rt.get_memory_lva());
577  naming::address const memory_address(here
578  , components::get_component_type<components::server::memory>()
579  , rt.get_memory_lva());
580  agas_client.bind_local(memory_gid, memory_address);
581 
582  // Assign the initial parcel gid range to the parcelport. Note that we can't
583  // get the parcelport through the parcelhandler because it isn't up yet.
584  naming::gid_type parcel_lower, parcel_upper;
585  agas_client.get_id_range(1000, parcel_lower, parcel_upper);
586 
587  rt.get_id_pool().set_range(parcel_lower, parcel_upper);
588 
589  // store number of initial localities
590  cfg.set_num_localities(header.num_localities);
591 
592  // store number of used cores by other localities
593  cfg.set_first_used_core(header.used_cores);
594  rt.assign_cores();
595 
596  // pre-cache all known locality endpoints in local AGAS
597  agas_client.pre_cache_endpoints(header.endpoints);
598 }

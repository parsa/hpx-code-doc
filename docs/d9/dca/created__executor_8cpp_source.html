28 // A parallel executor that returns void for bulk_execute and hpx::future<void>
29 // for bulk_async_execute
30 struct void_parallel_executor
31  : hpx::parallel::execution::parallel_executor
32 {
33  template <typename F, typename Shape, typename ... Ts>
34  std::vector<hpx::future<void> >
35  bulk_async_execute(F && f, Shape const& shape, Ts &&... ts)
36  {
37  std::vector<hpx::future<void> > results;
38  for(auto const& elem: shape)
39  {
40  results.push_back(this->parallel_executor::async_execute(
41  f, elem, ts...));
42  }
43  return results;
44  }
45 
46  template <typename F, typename Shape, typename ... Ts>
47  void bulk_sync_execute(F && f, Shape const& shape, Ts &&... ts)
48  {
49  return hpx::util::unwrap(
50  bulk_async_execute(std::forward<F>(f), shape,
51  std::forward<Ts>(ts)...));
52  }
53 };

122 std::vector<hpx::util::iterator_range<iter> >
123 split(iter first, iter last, int parts)
124 {
125  typedef std::iterator_traits<iter>::difference_type sz_type;
126  sz_type count = std::distance(first, last);
127  sz_type increment = count/parts;
128 
129  std::vector<hpx::util::iterator_range<iter> > results;
130  while(first != last)
131  {
132  iter prev = first;
133  std::advance(
134  first,
135  (std::min)(increment, std::distance(first,last))
136  );
137  results.push_back(hpx::util::make_iterator_range(prev, first));
138  }
139  return results;
140 }

142 // parallel sum using hpx's parallel executor
143 int parallel_sum(iter first, iter last, int num_parts)
144 {

148  split(first, last, num_parts);

166 // parallel sum using void parallel executer
167 int void_parallel_sum(iter first, iter last, int num_parts)
168 {
169  void_parallel_executor exec;
170 
171  std::vector<int> temp(num_parts + 1, 0);
172  std::iota(std::begin(temp), std::end(temp), 0);
173 
174  std::ptrdiff_t section_size = std::distance(first, last) / num_parts;
175 
176  std::vector<hpx::future<void> > f =
177  hpx::parallel::execution::bulk_async_execute(exec,
178  [&](const int& i)
179  {
180  iter b = first + i*section_size; //-V104
181  iter e = first + (std::min)(
182  std::distance(first, last),
183  static_cast<std::ptrdiff_t>((i+1)*section_size) //-V104
184  );
185  temp[i] = std::accumulate(b, e, 0); //-V108
186  },
187  temp);
188 
189  hpx::when_all(f).get();
190 
191  return std::accumulate(std::begin(temp), std::end(temp), 0);
192 }

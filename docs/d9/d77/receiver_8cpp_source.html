85  // when a receive completes, this callback handler is called
86  void receiver::handle_recv(fi_addr_t const& src_addr, std::uint64_t len)
87  {
88  FUNC_START_DEBUG_MSG;
89  static_assert(sizeof(std::uint64_t) == sizeof(std::size_t),
90  "sizeof(std::uint64_t) != sizeof(std::size_t)");
91 
92  // If we recieve a message of 8 bytes, we got a tag and need to handle
93  // the tag completion...
94  if (len <= sizeof(std::uint64_t))
95  {
96  // @TODO: fixme immediate tag retreival
97  // Get the sender that has completed rma operations and signal to it
98  // that it can now cleanup - all remote get operations are done.
99  sender* snd = *reinterpret_cast<sender **>(header_region_->get_address());
100  pre_post_receive();
101  LOG_DEBUG_MSG("Handling sender tag (RMA ack) completion: "
102  << hexpointer(snd));
103  ++acks_received_;
104  snd->handle_message_completion_ack();
105  return;
106  }
107 
108  LOG_DEBUG_MSG("Handling message");
109  rma_receiver* recv = nullptr;
110  if (!receiver::rma_receivers_.pop(recv))
111  {
112  auto f = [this](rma_receiver* recv)
113  {
114  --active_receivers_;
115  if (!receiver::rma_receivers_.push(recv)) {
116  // if the capacity overflowed, just delete this one
117  delete recv;
118  }
119  // Notify one possibly waiting reciever that one receive just finished
120  if (threads::threadmanager_is_at_least(state_running)
121  && hpx::threads::get_self_ptr())
122  {
123  std::unique_lock<mutex_type> l(active_receivers_mtx_);
124  active_receivers_cv_.notify_one(std::move(l));
125  }
126  };
127  // throttle the creation of new receivers. Wait until the active_receivers_
128  // count drops below the maximum. This can not be a busy wait since it could
129  // potentially block all background threads.
130  const long max_receivers =
131  HPX_PARCELPORT_LIBFABRIC_MAX_PREPOSTS;
132  std::size_t k = 0;
133  if (threads::threadmanager_is_at_least(state_running)
134  && hpx::threads::get_self_ptr())
135  {
136  while (active_receivers_ > max_receivers)
137  {
138  std::unique_lock<mutex_type> l(active_receivers_mtx_);
139  active_receivers_cv_.wait(l);
140  }
141  }
142 
143  recv = new rma_receiver(pp_, endpoint_, memory_pool_, std::move(f));
144  }
145  ++active_receivers_;
146 
147  HPX_ASSERT(recv);
148 
149  // We save the received region and swap it with a newly allocated one
150  // so that we can post a recv again as soon as possible.
151  region_type* region = header_region_;
152  header_region_ = memory_pool_->allocate_region(memory_pool_->small_.chunk_size());
153  pre_post_receive();
154 
155  // we dispatch our work to our rma_receiver once it completed the
156  // prior message. The saved region is passed to the rma handler
157  ++messages_handled_;
158  recv->read_message(region, src_addr);
159 
160  FUNC_END_DEBUG_MSG;
161  }

165  {
166  FUNC_START_DEBUG_MSG;
167  void* desc = header_region_->get_desc();
168  LOG_DEBUG_MSG("Pre-Posting receive "
169  << *header_region_
170  << "context " << hexpointer(this));
171 
172  int ret = 0;
173  for (std::size_t k = 0; true; ++k)
174  {
175  // post a receive using 'this' as the context, so that this receiver object
176  // can be used to handle the incoming receive/request
177  ret = fi_recv(
178  endpoint_,
179  header_region_->get_address(),
180  header_region_->get_size(),
181  desc, 0, this);
182 
183  if (ret == -FI_EAGAIN)
184  {
185  LOG_ERROR_MSG("reposting fi_recv\n");
186  hpx::util::detail::yield_k(k,
187  "libfabric::receiver::post_recv");
188  continue;
189  }
190  if (ret!=0)
191  {
192  // TODO: proper error message
193  throw fabric_error(ret, "pp_post_rx");
194  }
195  break;
196  }

63  void slurm_environment::retrieve_number_of_localities(bool debug)
64  {
65  char* tasks_per_node = std::getenv("SLURM_STEP_TASKS_PER_NODE");
66  char *total_num_tasks = std::getenv("SLURM_STEP_NUM_TASKS");
67  char *num_nodes = std::getenv("SLURM_STEP_NUM_NODES");
68 
69  if(total_num_tasks)
70  {
71  num_localities_ = safe_lexical_cast<std::size_t>(total_num_tasks);
72  }
73  else
74  {
75  num_localities_ = 1;
76  }
77 
78  std::size_t task_count = 0;
79  if (tasks_per_node)
80  {
81  if (debug) {
82  std::cerr << "SLURM tasks per node found (SLURM_STEP_TASKS_PER_NODE): "
83  << tasks_per_node << std::endl;
84  }
85 
86  std::vector<std::string> node_tokens;
87  boost::split(node_tokens, tasks_per_node, boost::is_any_of(","));
88 
89  if (node_tokens.size() == 1 &&
90  node_tokens[0].find_first_of('x') == std::string::npos)
91  {
92  num_tasks_ = safe_lexical_cast<std::size_t>(node_tokens[0]);
93  }
94  else
95  {
96  for(std::string & node_token: node_tokens)
97  {
98  std::size_t paren_pos = node_token.find_first_of('(');
99  HPX_ASSERT(paren_pos != std::string::npos);
100  std::size_t num_tasks
101  = safe_lexical_cast<std::size_t>(
102  node_token.substr(0, paren_pos));
103  std::size_t repetition = 1;
104  if(paren_pos != std::string::npos)
105  {
106  HPX_ASSERT(node_token[paren_pos + 1] == 'x');
107  HPX_ASSERT(node_token[node_token.size() - 1] == ')');
108  std::size_t begin = paren_pos + 2;
109  std::size_t end = node_token.size() - 1;
110  repetition =
111  safe_lexical_cast<std::size_t>(
112  node_token.substr(paren_pos + 2, end - begin));
113  }
114 
115  std::size_t next_task_count
116  = task_count + num_tasks * repetition;
117  if(node_num_ >= task_count && node_num_ < next_task_count)
118  {
119  num_tasks_ = num_tasks;
120  }
121 
122  task_count = next_task_count;
123 
124  if(task_count > node_num_)
125  {
126  if (debug) {
127  std::cerr
128  << "SLURM node number outside of available "
129  "list of tasks"
130  << std::endl;
131  }
132  break;
133  }
134  }
135  }
136  }
137  else
138  {
139  num_tasks_ = 1;
140  task_count = 1;
141  }
142 
143  if (task_count != num_localities_ && num_nodes)
144  {
145  num_tasks_
146  = num_localities_
147  / safe_lexical_cast<std::size_t>(num_nodes);
148  }
149  }

152  {

174  result_type operator()(std::vector<std::string> & nodes,
175  param_type const & p) const
176  {
177  typedef param_type::value_type value_type;
178 
179  std::vector<std::string> tmp_nodes;
180 
181  for (value_type const& value : p)
182  {
183  std::string const & prefix = boost::fusion::at_c<0>(value);
184  optional_type const & ranges = boost::fusion::at_c<1>(value);
185  bool push_now = tmp_nodes.empty();
186  if(ranges)
187  {
188  for (vector_type const& range : *ranges)
189  {
190  if(range.size() == 1)
191  {
192  std::string s(prefix);
193  s += range[0];
194  if(push_now)
195  {
196  tmp_nodes.push_back(s);
197  }
198  else
199  {
200  for (std::string& node : tmp_nodes)
201  {
202  node += s;
203  }
204  }
205  }
206  else
207  {
208  using hpx::util::safe_lexical_cast;
209  std::size_t begin = safe_lexical_cast<std::size_t>
210  (range[0]);
211  std::size_t end = safe_lexical_cast<std::size_t>
212  (range[1]);
213  if(begin > end) std::swap(begin, end);
214 
215  std::vector<std::string> vs;
216 
217  for(std::size_t i = begin; i <= end; ++i)
218  {
219  std::string s(prefix);
220  std::size_t dec = 10;
221  // pad with zeros
222  for(std::size_t j = 0; j < range[0].length()-1; ++j)
223  {
224  if(i < dec)
225  {
226  s += "0";
227  }
228  dec *= 10;
229  }
230  s += std::to_string(i);
231  if(push_now)
232  {
233  tmp_nodes.push_back(s);
234  }
235  else
236  {
237  vs.push_back(s);
238  }
239  }
240  if(!push_now)
241  {
242  std::vector<std::string> tmp;
243  std::swap(tmp, tmp_nodes);
244  for (std::string s : tmp)
245  {
246  for (std::string const& s2 : vs)
247  {
248  s += s2;
249  tmp_nodes.push_back(s);
250  }
251  }
252  }
253  }
254  }
255  }
256  else
257  {
258  if(push_now)
259  {
260  tmp_nodes.push_back(prefix);
261  }
262  else
263  {
264  for (std::string& node : tmp_nodes)
265  {
266  node += prefix;
267  }
268  }
269  }
270  }
271  nodes.insert(nodes.end(), tmp_nodes.begin(), tmp_nodes.end());
272  }

275  void slurm_environment::retrieve_nodelist(std::vector<std::string> & nodes,
276  bool debug)
277  {
278  char* slurm_nodelist_env = std::getenv("SLURM_STEP_NODELIST");
279  if (slurm_nodelist_env)
280  {
281  if (debug) {
282  std::cerr << "SLURM nodelist found (SLURM_STEP_NODELIST): "
283  << slurm_nodelist_env << std::endl;
284  }
285 
286  std::string nodelist_str(slurm_nodelist_env);
287  std::string::iterator begin = nodelist_str.begin();
288  std::string::iterator end = nodelist_str.end();
289 
290  namespace qi = boost::spirit::qi;
291  namespace phoenix = boost::phoenix;
292 
293  qi::rule<std::string::iterator, std::string()> prefix;
294  qi::rule<std::string::iterator, std::string()> range_str;
295  qi::rule<std::string::iterator, std::vector<std::string>()> range;
296  qi::rule<std::string::iterator,
297  boost::fusion::vector<
298  std::string,
299  boost::optional<std::vector<std::vector<std::string> > >
300  >()> ranges;
301  qi::rule<std::string::iterator> hostlist;
302  qi::rule<std::string::iterator> nodelist;
303 
304  // grammar definition
305  prefix %=
306  +(qi::print - (qi::char_("[") | qi::char_(",")))
307  ;
308 
309  range_str %=
310  +(qi::print - (qi::char_("]") | qi::char_( ",") | qi::char_("-")))
311  ;
312 
313  range %= range_str >> *('-' >> range_str);
314 
315  ranges %= prefix >> -(qi::lit("[") >> (range % ',') >> qi::lit("]"));
316 
317  hostlist =
318  (+ranges)[
319  phoenix::bind(construct_nodelist(),
320  phoenix::ref(nodes), qi::_1)
321  ];
322 
323  nodelist = hostlist % ',';
324 
325  if (!qi::parse(begin, end, nodelist) || begin != end)
326  {
327  if (debug) {
328  std::cerr << "failed to parse SLURM nodelist (SLURM_STEP_NODELIST): "
329  << slurm_nodelist_env << std::endl;
330  }
331  }
332  }

335  void slurm_environment::retrieve_number_of_threads()
336  {
337  char *slurm_cpus_on_node = std::getenv("SLURM_CPUS_ON_NODE");
338  if(slurm_cpus_on_node)
339  {
340  std::size_t slurm_num_cpus =
341  safe_lexical_cast<std::size_t>(slurm_cpus_on_node);
342  threads::topology const & top = threads::create_topology();
343  std::size_t num_pus = top.get_number_of_pus();
344 
345  // Figure out if we got the number of logical cores (including
346  // hyper-threading) or just the number of physical cores.
347  char *slurm_cpus_per_task = std::getenv("SLURM_CPUS_PER_TASK");
348  if(slurm_num_cpus == num_pus)
349  {
350  if(slurm_cpus_per_task)
351  num_threads_
352  = safe_lexical_cast<std::size_t>(slurm_cpus_per_task);
353  else
354  num_threads_ = num_pus / num_tasks_;
355  }
356  else
357  {
358  std::size_t num_cores = 0;
359  if(slurm_cpus_per_task)
360  {
361  num_cores
362  = safe_lexical_cast<std::size_t>(slurm_cpus_per_task);
363  }
364  else
365  {
366  num_cores = slurm_num_cpus / num_tasks_;
367  }
368  HPX_ASSERT(num_cores <= top.get_number_of_cores());
369 
370  for(std::size_t core = 0; core != num_cores; ++core)
371  {
372  num_threads_ += top.get_number_of_core_pus(core);
373  }
374  }
375  }

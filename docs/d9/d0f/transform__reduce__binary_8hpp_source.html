60  typedef typename hpx::util::decay<Op2>::type op2_type;

84  template <typename ExPolicy, typename InIter1, typename InIter2,
85  typename T_, typename Op1, typename Op2>
86  static T
87  sequential(ExPolicy && policy, InIter1 first1, InIter1 last1,
88  InIter2 first2, T_ init, Op1 && op1, Op2 && op2)
89  {
90  if (first1 == last1)
91  return std::move(init);
92 
93  // check whether we should apply vectorization
94  if (!util::loop_optimization<ExPolicy>(first1, last1))
95  {
96  util::loop2<ExPolicy>(
97  std::false_type(), first1, last1, first2,
98  transform_reduce_binary_partition<Op1, Op2, T>{
99  op1, op2, init
100  });
101 
102  return init;
103  }
104 
105  // loop_step properly advances the iterators
106  auto part_sum = util::loop_step<ExPolicy>(std::true_type(),
107  transform_reduce_binary_indirect<Op2>{op2}, first1, first2);
108 
109  std::pair<InIter1, InIter2> p = util::loop2<ExPolicy>(
110  std::true_type(), first1, last1, first2,
111  transform_reduce_binary_partition<
112  Op1, Op2, decltype(part_sum)
113  >{op1, op2, part_sum});
114 
115  // this is to support vectorization, it will call op1 for each
116  // of the elements of a value-pack
117  auto result =
118  util::detail::accumulate_values<ExPolicy>(
119  [&op1](T const& sum, T const& val) -> T
120  {
121  return hpx::util::invoke(op1, sum, val);
122  },
123  std::move(part_sum),
124  std::move(init));
125 
126  // the vectorization might not cover all of the sequences,
127  // handle the remainder directly
128  if (p.first != last1)
129  {
130  util::loop2<ExPolicy>(
131  std::false_type(), p.first, last1, p.second,
132  transform_reduce_binary_partition<
133  Op1, Op2, decltype(result)
134  >{op1, op2, result});
135  }
136 
137  return util::detail::extract_value<ExPolicy>(result);
138  }

140  template <typename ExPolicy, typename FwdIter1, typename FwdIter2,
141  typename T_, typename Op1, typename Op2>
142  static typename util::detail::algorithm_result<
143  ExPolicy, T
144  >::type
145  parallel(ExPolicy && policy, FwdIter1 first1, FwdIter1 last1,
146  FwdIter2 first2, T_ && init, Op1 && op1, Op2 && op2)
147  {
148  typedef util::detail::algorithm_result<ExPolicy, T> result;
149  typedef hpx::util::zip_iterator<FwdIter1, FwdIter2>
150  zip_iterator;
151  typedef typename std::iterator_traits<FwdIter1>::difference_type
152  difference_type;
153 
154  if (first1 == last1)
155  return result::get(std::forward<T_>(init));
156 
157  difference_type count = std::distance(first1, last1);
158 
159  auto f1 =
160  [op1, op2, policy](
161  zip_iterator part_begin, std::size_t part_size
162  ) mutable -> T
163  {
164  HPX_UNUSED(policy);
165 
166  auto iters = part_begin.get_iterator_tuple();
167  FwdIter1 it1 = hpx::util::get<0>(iters);
168  FwdIter2 it2 = hpx::util::get<1>(iters);
169 
170  FwdIter1 last1 = it1;
171  std::advance(last1, part_size);
172 
173  if (!util::loop_optimization<ExPolicy>(it1, last1))
174  {
175  // loop_step properly advances the iterators
176  auto result = util::loop_step<ExPolicy>(
177  std::false_type(),
178  transform_reduce_binary_indirect<Op2>{op2},
179  it1, it2);
180 
181  util::loop2<ExPolicy>(
182  std::false_type(), it1, last1, it2,
183  transform_reduce_binary_partition<
184  Op1, Op2, decltype(result)
185  >{op1, op2, result});
186 
187  return util::detail::extract_value<ExPolicy>(result);
188  }
189 
190  // loop_step properly advances the iterators
191  auto part_sum = util::loop_step<ExPolicy>(
192  std::true_type(),
193  transform_reduce_binary_indirect<Op2>{op2}, it1, it2);
194 
195  std::pair<FwdIter1, FwdIter2> p =
196  util::loop2<ExPolicy>(
197  std::true_type(), it1, last1, it2,
198  transform_reduce_binary_partition<
199  Op1, Op2, decltype(part_sum)
200  >{op1, op2, part_sum});
201 
202  // this is to support vectorization, it will call op1
203  // for each of the elements of a value-pack
204  auto result = util::detail::accumulate_values<ExPolicy>(
205  [&op1](T const& sum, T const& val) -> T
206  {
207  return hpx::util::invoke(op1, sum, val);
208  },
209  part_sum);
210 
211  // the vectorization might not cover all of the sequences,
212  // handle the remainder directly
213  if (p.first != last1)
214  {
215  util::loop2<ExPolicy>(
216  std::false_type(), p.first, last1, p.second,
217  transform_reduce_binary_partition<
218  Op1, Op2, decltype(result)
219  >{op1, op2, result});
220  }
221 
222  return util::detail::extract_value<ExPolicy>(result);
223  };
224 
225  using hpx::util::make_zip_iterator;
226  return util::partitioner<ExPolicy, T>::call(
227  std::forward<ExPolicy>(policy),
228  make_zip_iterator(first1, first2), count,
229  std::move(f1),
230  [init, op1](std::vector<hpx::future<T> > && results) -> T
231  {
232  T ret = init;
233  for(auto && fut : results)
234  {
235  ret = hpx::util::invoke(op1, ret, fut.get());
236  }
237  return ret;
238  });
239  }
240  };

103  // parser for affinity options
104  template <typename Iterator>
105  struct mappings_parser : qi::grammar<Iterator, mappings_type()>
106  {
107  mappings_parser()
108  : mappings_parser::base_type(start)
109  {
110  using detail::partlit;
111  using detail::spec_type;
112 
113  start = distribution | (mapping % ';');
114 
115  mapping = thread_spec >> '=' >> pu_spec;
116 
117  distribution =
118  partlit("compact") >> qi::attr(compact)
119  | partlit("scatter") >> qi::attr(scatter)
120  | partlit("balanced") >> qi::attr(balanced)
121  ;
122 
123  thread_spec =
124  partlit("thread") >> ':'
125  >> specs >> qi::attr(spec_type::thread)
126  ;
127 
128  pu_spec =
129  socket_spec >> core_spec >> processing_unit_spec
130 // | '~' >> pu_spec
131  ;
132 
133  socket_spec =
134  partlit("socket") >> ':'
135  >> specs >> qi::attr(spec_type::socket)
136  | partlit("numanode") >> ':'
137  >> specs >> qi::attr(spec_type::numanode)
138  | qi::attr(spec_type::unknown)
139  ;
140 
141  core_spec =
142  -qi::lit('.') >> partlit("core") >> ':'
143  >> specs >> qi::attr(spec_type::core)
144  | qi::attr(spec_type::unknown)
145  ;
146 
147  processing_unit_spec =
148  -qi::lit('.') >> partlit("pu") >> ':'
149  >> specs >> qi::attr(spec_type::pu)
150  | qi::attr(spec_type::unknown)
151  ;
152 
153  specs = spec % ',';
154 
155  spec =
156  qi::uint_ >> -(qi::int_)
157  | partlit("all") >> qi::attr(spec_type::all_entities())
158  ;
159 
160  BOOST_SPIRIT_DEBUG_NODES(
161  (start)(mapping)(distribution)(thread_spec)(pu_spec)(specs)
162  (spec)(socket_spec)(core_spec)(processing_unit_spec)
163  );
164  }

167  qi::rule<Iterator, distribution_type()> distribution;
168  qi::rule<Iterator, full_mapping_type()> mapping;
169  qi::rule<Iterator, spec_type()> thread_spec;

173  qi::rule<Iterator, bounds_type()> specs;

202  bounds_type
203  extract_bounds(spec_type const& m, std::size_t default_last, error_code& ec)
204  {
205  bounds_type result;
206 
207  if (m.index_bounds_.empty())
208  return result;
209 
210  bounds_type::const_iterator first = m.index_bounds_.begin();
211  bounds_type::const_iterator last = m.index_bounds_.end();
212 
213  while (first != last)
214  {
215  if (*first == spec_type::all_entities())
216  {
217  // bind all entities
218  result.clear();
219  for (std::size_t i = 0; i != default_last; ++i)
220  result.push_back(i);
221  break; // we will not get more than 'all'
222  }
223 
224  bounds_type::const_iterator second = first;
225  if (++second != last)
226  {
227  if (*second == 0 || *second == spec_type::all_entities())
228  {
229  // one element only
230  if (default_last <= std::size_t(*first))
231  {
232  result.clear();
233  HPX_THROWS_IF(ec, bad_parameter, "extract_bounds",
234  "the resource id given is larger than the number "
235  "of existing resources");
236  return result;
237  }
238  result.push_back(*first);
239  }
240  else if (*second < 0)
241  {
242  // all elements between min and -max
243  if (default_last <= std::size_t(-*second))
244  {
245  result.clear();
246  HPX_THROWS_IF(ec, bad_parameter, "extract_bounds",
247  "the upper limit given is larger than the number "
248  "of existing resources");
249  return result;
250  }
251  for (std::int64_t i = *first; i <= -*second; ++i)
252  result.push_back(i);
253  }
254  else
255  {
256  // just min and max
257  if (default_last <= std::size_t(*second))
258  {
259  result.clear();
260  HPX_THROWS_IF(ec, bad_parameter, "extract_bounds",
261  "the upper limit given is larger than the number "
262  "of existing resources");
263  return result;
264  }
265  result.push_back(*first);
266  result.push_back(*second);
267  }
268  first = second;
269  }
270  else
271  {
272  // one element only
273  if (default_last <= std::size_t(*first))
274  {
275  result.clear();
276  HPX_THROWS_IF(ec, bad_parameter, "extract_bounds",
277  "the resource id given is larger than the number "
278  "of existing resources");
279  return result;
280  }
281  result.push_back(*first);
282  }
283  ++first;
284  }
285 
286  if (&ec != &throws)
287  ec = make_success_code();
288 
289  return result;
290  }

341  std::vector<mask_info>
342  extract_socket_or_numanode_masks(hwloc_topology_info const& t,
343  spec_type const& s, error_code& ec)
344  {
345  switch (s.type_)
346  {
347  case spec_type::socket:
348  // requested top level is a socket
349  {
350  std::size_t num_sockets = t.get_number_of_sockets();
351  return extract_socket_masks(
352  t, extract_bounds(s, num_sockets, ec));
353  }
354 
355  case spec_type::numanode:
356  // requested top level is a NUMA node
357  {
358  std::size_t num_numanodes = t.get_number_of_numa_nodes();
359  return extract_numanode_masks(
360  t, extract_bounds(s, num_numanodes, ec));
361  }
362 
363  case spec_type::unknown:
364  {
365  std::vector<mask_info> masks;
366  masks.push_back(util::make_tuple(
367  std::size_t(-1), extract_machine_mask(t, ec)
368  ));
369  return masks;
370  }
371 
372  default:
373  HPX_THROWS_IF(ec, bad_parameter, "extract_socket_or_numanode_mask",
374  boost::str(boost::format(
375  "unexpected specification type %s"
376  ) % spec_type::type_name(s.type_)));
377  break;
378  }
379 
380  return std::vector<mask_info>();
381  }

383  std::vector<mask_info>
384  extract_core_masks(hwloc_topology_info const& t, spec_type const& s,
385  std::size_t socket, mask_cref_type socket_mask, error_code& ec)
386  {
387  std::vector<mask_info> masks;
388 
389  switch (s.type_)
390  {
391  case spec_type::core:
392  {
393  std::size_t base = 0;
394  std::size_t num_cores = 0;
395 
396  if (socket != std::size_t(-1))
397  {
398  for (std::size_t i = 0; i != socket; ++i)
399  {
400  // The number of NUMA nodes might be zero if there hwloc
401  // doesn't detect a NUMA domain. This might be the case
402  // when there is no NUMA support configured, or when
403  // there are just sockets, but no direct numa domains.
404  // The bind description might relate to sockets, not
405  // NUMA domains.
406  if (t.get_number_of_numa_nodes() == 0)
407  base += t.get_number_of_socket_cores(i);
408  else
409  base += t.get_number_of_numa_node_cores(i);
410  }
411  if (t.get_number_of_numa_nodes() == 0)
412  num_cores = t.get_number_of_socket_cores(socket);
413  else
414  num_cores = t.get_number_of_numa_node_cores(socket);
415  }
416  else
417  {
418  num_cores = t.get_number_of_cores();
419  }
420 
421  bounds_type bounds = extract_bounds(s, num_cores, ec);
422  if (ec) break;
423 
424  for (std::int64_t index : bounds)
425  {
426  mask_type mask =
427  t.init_core_affinity_mask_from_core(
428  static_cast<std::size_t>(index + base));
429  masks.push_back(util::make_tuple(
430  static_cast<std::size_t>(index), mask & socket_mask));
431  }
432  }
433  break;
434 
435  case spec_type::unknown:
436  {
437  mask_type mask = extract_machine_mask(t, ec);
438  masks.push_back(util::make_tuple(
439  std::size_t(-1), mask & socket_mask
440  ));
441  }
442  break;
443 
444  default:
445  HPX_THROWS_IF(ec, bad_parameter, "extract_core_mask",
446  boost::str(boost::format(
447  "unexpected specification type %s"
448  ) % spec_type::type_name(s.type_)));
449  break;
450  }
451 
452  return masks;
453  }

455  std::vector<mask_info>
456  extract_pu_masks(hwloc_topology_info const& t, spec_type const& s,
457  std::size_t socket, std::size_t core, mask_cref_type core_mask,
458  error_code& ec)
459  {
460  std::vector<mask_info> masks;
461 
462  switch (s.type_)
463  {
464  case spec_type::pu:
465  {
466  std::size_t num_pus = 0;
467  std::size_t socket_base = 0;
468  if (std::size_t(-1) != socket)
469  {
470  // core number is relative to socket
471  for (std::size_t i = 0; i != socket; ++i)
472  {
473  if (t.get_number_of_numa_nodes() == 0)
474  socket_base += t.get_number_of_socket_cores(i);
475  else
476  socket_base += t.get_number_of_numa_node_cores(i);
477  }
478  }
479 
480  if (std::size_t(-1) != core)
481  {
482  num_pus = t.get_number_of_core_pus(core);
483  }
484  else
485  {
486  num_pus = t.get_number_of_pus();
487  }
488  bounds_type bounds = extract_bounds(s, num_pus, ec);
489  if (ec) break;
490 
491  std::size_t num_cores = t.get_number_of_cores();
492  for (std::int64_t index : bounds)
493  {
494  std::size_t base_core = socket_base;
495  if (std::size_t(-1) != core)
496  {
497  base_core += core;
498  }
499  else
500  {
501  // find core the given pu belongs to
502  std::size_t base = 0;
503  for (; base_core < num_cores; ++base_core)
504  {
505  std::size_t num_core_pus =
506  t.get_number_of_core_pus(base_core);
507  if (base + num_core_pus > std::size_t(index))
508  break;
509  base += num_core_pus;
510  }
511  }
512 
513  mask_type mask = t.init_thread_affinity_mask(base_core,
514  static_cast<std::size_t>(index));
515  masks.push_back(util::make_tuple(
516  static_cast<std::size_t>(index), mask & core_mask));
517  }
518  }
519  break;
520 
521  case spec_type::unknown:
522  {
523  mask_type mask = extract_machine_mask(t, ec);
524  masks.push_back(util::make_tuple(
525  std::size_t(-1), mask & core_mask
526  ));
527  }
528  break;
529 
530  default:
531  HPX_THROWS_IF(ec, bad_parameter, "extract_pu_mask",
532  boost::str(boost::format(
533  "unexpected specification type %s"
534  ) % spec_type::type_name(s.type_)));
535  break;
536  }
537 
538  return masks;
539  }

567  // for each given core-mask extract all required pu-masks
568  void extract_pu_affinities(hwloc_topology_info const& t,
569  std::vector<spec_type> const& specs, std::size_t socket,
570  std::vector<mask_info> const& core_masks,
571  std::vector<mask_type>& affinities, error_code& ec)
572  {
573  // get the core masks for each of the sockets
574  for (mask_info const& cmi : core_masks)
575  {
576  if (get_index(cmi) == std::size_t(-1))
577  {
578  // all cores
579  if (specs[2].type_ == spec_type::unknown)
580  {
581  // no pu information
582  affinities.push_back(get_mask(cmi));
583  }
584  else
585  {
586  // handle pu information in the absence of core information
587  std::vector<mask_info> pu_masks = extract_pu_masks(
588  t, specs[2], socket, std::size_t(-1),
589  get_mask(cmi), ec);
590  if (ec) break;
591 
592  for (mask_info const& pmi : pu_masks)
593  {
594  affinities.push_back(get_mask(pmi));
595  }
596  }
597  break;
598  }
599  else
600  {
601  // just this core
602  std::vector<mask_info> pu_masks = extract_pu_masks(
603  t, specs[2], socket, get_index(cmi), get_mask(cmi), ec);
604  if (ec) break;
605 
606  for (mask_info const& pmi : pu_masks)
607  {
608  affinities.push_back(get_mask(pmi));
609  }
610  }
611  }
612  }

614  // for each given socket-mask extract all required pu-masks
615  void extract_core_affinities(hwloc_topology_info const& t,
616  std::vector<spec_type> const& specs,
617  std::vector<mask_info> const& socket_masks,
618  std::vector<mask_type>& affinities, error_code& ec)
619  {
620  // get the core masks for each of the sockets
621  for (mask_info const& smi : socket_masks)
622  {
623  if (get_index(smi) == std::size_t(-1))
624  {
625  // all NUMA domains
626  if (specs[1].type_ == spec_type::unknown)
627  {
628  // no core information
629  if (specs[2].type_ == spec_type::unknown)
630  {
631  // no pu information
632  affinities.push_back(get_mask(smi));
633  }
634  else
635  {
636  // handle pu information in the absence of core/socket
637  std::vector<mask_info> pu_masks = extract_pu_masks(
638  t, specs[2], std::size_t(-1), std::size_t(-1),
639  get_mask(smi), ec);
640  if (ec) break;
641 
642  for (mask_info const& pmi : pu_masks)
643  {
644  affinities.push_back(get_mask(pmi));
645  }
646  }
647  }
648  else
649  {
650  // no socket given, assume cores are numbered for whole
651  // machine
652  if (specs[2].type_ == spec_type::unknown)
653  {
654  // no pu information
655  std::vector<mask_info> core_masks = extract_core_masks(
656  t, specs[1], std::size_t(-1), get_mask(smi), ec);
657  if (ec) break;
658 
659  for (mask_info const& cmi : core_masks)
660  {
661  affinities.push_back(get_mask(cmi));
662  }
663  }
664  else
665  {
666  std::vector<mask_info> core_masks = extract_core_masks(
667  t, specs[1], std::size_t(-1), get_mask(smi), ec);
668  if (ec) break;
669 
670  // get the pu masks (i.e. overall affinity masks) for
671  // all of the core masks
672  extract_pu_affinities(t, specs, std::size_t(-1),
673  core_masks, affinities, ec);
674  if (ec) break;
675  }
676  }
677  break;
678  }
679  else
680  {
681  std::vector<mask_info> core_masks = extract_core_masks(
682  t, specs[1], get_index(smi), get_mask(smi), ec);
683  if (ec) break;
684 
685  // get the pu masks (i.e. overall affinity masks) for
686  // all of the core masks
687  extract_pu_affinities(t, specs, get_index(smi), core_masks,
688  affinities, ec);
689  if (ec) break;
690  }
691  }

694  void decode_mappings(hwloc_topology_info const& t, full_mapping_type& m,
695  std::vector<mask_type>& affinities, std::size_t num_threads,
696  error_code& ec)
697  {
698  // The core numbers are interpreted differently depending on whether a
699  // socket/numanode is given or not. If no socket(s) is(are) given, then
700  // the core numbering covers the whole locality, otherwise the core
701  // numbering is relative to the given socket.
702 
703  // generate overall masks for each of the given sockets
704  std::vector<mask_info> socket_masks =
705  extract_socket_or_numanode_masks(t, m.second[0], ec);
706 
707  HPX_ASSERT(!socket_masks.empty());
708 
709  extract_core_affinities(t, m.second, socket_masks, affinities, ec);
710 
711  // special case, all threads share the same options
712  if (affinities.size() == 1 && num_threads > 1)
713  {
714  affinities.resize(num_threads, affinities[0]);
715  }
716  }

50  // sequential unique_copy with projection function
51  template <typename FwdIter, typename OutIter, typename Pred, typename Proj>
52  std::pair<FwdIter, OutIter>
53  sequential_unique_copy(FwdIter first, FwdIter last, OutIter dest,
54  Pred && pred, Proj && proj, std::true_type)
55  {
56  if (first == last)
57  return std::make_pair(std::move(last), std::move(dest));
58 
59  FwdIter base = first;
60 
61  *dest++ = *first;
62 
63  while (++first != last)
64  {
65  if (!hpx::util::invoke(pred,
66  hpx::util::invoke(proj, *base),
67  hpx::util::invoke(proj, *first)))
68  {
69  base = first;
70  *dest++ = *first;
71  }
72  }
73  return std::make_pair(std::move(last), std::move(dest));
74  }

128  typedef hpx::util::zip_iterator<FwdIter1, bool*> zip_iterator;

133  difference_type;

136  return result::get(

138

142

156  auto f1 =
157  [pred, proj, flags, policy]
158  (
159  zip_iterator part_begin, std::size_t part_size
160  ) -> std::size_t
161  {
162  HPX_UNUSED(flags);
163  HPX_UNUSED(policy);
164 
165  FwdIter1 base = get<0>(part_begin.get_iterator_tuple());
166  std::size_t curr = 0;
167 
168  // MSVC complains if pred or proj is captured by ref below
169  util::loop_n<ExPolicy>(
170  ++part_begin, part_size,
171  [base, pred, proj, &curr](zip_iterator it) mutable
172  {
173  using hpx::util::invoke;
174 
175  bool f = invoke(pred,
176  invoke(proj, *base),
177  invoke(proj, get<0>(*it)));
178 
179  if (!(get<1>(*it) = f))
180  {
181  base = get<0>(it.get_iterator_tuple());
182  ++curr;
183  }
184  });
185 
186  return curr;
187  };
188  auto f3 =
189  [dest, flags, policy](
190  zip_iterator part_begin, std::size_t part_size,
191  hpx::shared_future<std::size_t> curr,
192  hpx::shared_future<std::size_t> next
193  ) mutable -> void
194  {
195  HPX_UNUSED(flags);
196  HPX_UNUSED(policy);
197 
198  next.get(); // rethrow exceptions
199 
200  std::advance(dest, curr.get());
201  util::loop_n<ExPolicy>(
202  ++part_begin, part_size,
203  [&dest](zip_iterator it) mutable
204  {
205  if(!get<1>(*it))
206  *dest++ = get<0>(*it);
207  });
208  };

210  return scan_partitioner_type::call(
211  std::forward<ExPolicy>(policy),
212  make_zip_iterator(first, flags.get()),
213  count - 1, init,
214  // step 1 performs first part of scan algorithm
215  std::move(f1),

218  hpx::util::unwrapping(std::plus<std::size_t>()),
219  // step 3 runs final accumulation on each partition

228

38  hpx::partitioned_vector<T> const& v2, bool must_be_equal = true)
39 {
40  typedef typename hpx::partitioned_vector<T>::const_iterator const_iterator;
41 
42  HPX_TEST_EQ(v1.size(), v2.size());
43 
44  const_iterator it1 = v1.begin(), it2 = v2.begin();
45  const_iterator end1 = v1.end(), end2 = v2.end();

215  // Current element is a range (vector) of futures
216  template <std::size_t I, typename Iter>
217  void await_range(Iter next, Iter end)
218  {
219  typedef typename std::iterator_traits<Iter>::value_type
220  future_type;
221  typedef typename traits::future_traits<future_type>::type
222  future_result_type;
223 
224  void (when_each_frame::*f)(Iter, Iter) =
225  &when_each_frame::await_range<I>;
226 
227  for(; next != end; ++next)
228  {
229  typename traits::detail::shared_state_ptr<
230  future_result_type
231  >::type next_future_data =
232  traits::detail::get_shared_state(*next);
233 
234  if (next_future_data.get() != nullptr &&
235  !next_future_data->is_ready())
236  {
237  next_future_data->execute_deferred();
238 
239  // execute_deferred might have made the future ready
240  if (!next_future_data->is_ready())
241  {
242  // Attach a continuation to this future which will
243  // re-evaluate it and continue to the next argument
244  // (if any).
245  boost::intrusive_ptr<when_each_frame> this_(this);
246  next_future_data->set_on_completed(
247  util::deferred_call(
248  f, std::move(this_),
249  std::move(next), std::move(end)));
250  return;
251  }
252  }
253 
254  dispatch::call(std::forward<F>(f_), count_++,
255  std::move(*next),
256  typename traits::is_invocable<
257  F, std::size_t, future_type
258  >::type()
259  );
260 
261  if(count_ == needed_count_)
262  {
263  do_await<I + 1>(std::true_type());
264  return;
265  }
266  }
267 
268  do_await<I + 1>(is_end<I + 1>());
269  }

38 int monitor(double runfor, std::string const& name, std::uint64_t pause)
39 {
40 #if defined(HPX_WINDOWS) && HPX_USE_WINDOWS_PERFORMANCE_COUNTERS != 0
41  hpx::register_shutdown_function(&uninstall_windows_counters);
42 #endif
43 
44  // Resolve the GID of the performance counter using it's symbolic name.
45  hpx::naming::id_type id = hpx::performance_counters::get_counter(name);
46  if (!id)
47  {
48  std::cout << (boost::format(
49  "error: performance counter not found (%s)")
50  % name) << std::endl;
51  return 1;
52  }
53 
54  std::uint32_t const locality_id = hpx::get_locality_id();
55  if (locality_id == hpx::naming::get_locality_id_from_gid(id.get_gid()))
56  {
57  std::cout << (boost::format(
58  "error: cannot query performance counters on its own locality (%s)")
59  % name) << std::endl;
60  return 1;
61  }
62 
63  std::shared_ptr<hpx::promise<void> > stop_flag =
64  std::make_shared<hpx::promise<void> >();
65  hpx::future<void> f = stop_flag->get_future();
66 
67  hpx::register_shutdown_function(
68  hpx::util::bind(&stop_monitor, stop_flag));
69 
70  std::int64_t zero_time = 0;
71 
72  hpx::util::high_resolution_timer t;
73  while (runfor < 0 || t.elapsed() < runfor)
74  {
75  // stop collecting data when the runtime is exiting
76  if (!hpx::is_running() || f.is_ready())
77  return 0;
78 
79  // Query the performance counter.
80  using namespace hpx::performance_counters;
81  counter_value value =
82  stubs::performance_counter::get_value(hpx::launch::sync, id);
83 
84  if (status_is_valid(value.status_))
85  {
86  if (!zero_time)
87  zero_time = value.time_;
88 
89  std::cout << (boost::format(" %s,%d,%d[s],%d\n")
90  % name
91  % value.count_
92  % double((value.time_ - zero_time) * 1e-9)
93  % value.value_);
94 
95 #if defined(HPX_WINDOWS) && HPX_USE_WINDOWS_PERFORMANCE_COUNTERS != 0
96  update_windows_counters(value.value_);
97 #endif
98  }
99 
100  // Schedule a wakeup.
101  hpx::this_thread::suspend(pause);
102  }
103 
104  return hpx::disconnect();
105 }

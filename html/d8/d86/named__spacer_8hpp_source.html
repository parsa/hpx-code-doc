136  static string_type unescape(string_type escaped) {
137  typedef typename string_type::size_type size_type;
138  size_type idx_start = 0;
139  while ( true) {
140  size_type found = escaped.find( HPX_LOG_STR("%%"), idx_start );
141  if ( found != string_type::npos) {
142  escaped.erase( escaped.begin() +
143  static_cast<typename string_type::difference_type>(found));
144  ++idx_start;
145  }
146  else
147  break;
148  }
149  return escaped;
150  }

152  // recomputes the write steps - note taht this takes place after each operation
153  // for instance, the user might have first set the string and
154  // later added the formatters
155  void compute_write_steps() {
156  typedef typename string_type::size_type size_type;
157 
158  typename data::write info(m_data);
159  info->write_steps.clear();
160  string_type remaining = info->format_string;
161  size_type start_search_idx = 0;
162  while ( !remaining.empty() ) {
163  size_type idx = remaining.find('%', start_search_idx);
164  if ( idx != string_type::npos) {
165  // see if just escaped
166  if ( (idx < remaining.size() - 1) && remaining[idx + 1] == '%') {
167  // we found an escaped char
168  start_search_idx = idx + 2;
169  continue;
170  }
171 
172  // up to here, this is a spacer string
173  start_search_idx = 0;
174  string_type spacer = unescape( remaining.substr(0, idx) );
175  remaining = remaining.substr(idx + 1);
176  // find end of formatter name
177  idx = remaining.find('%');
178  format_base_type * fmt = nullptr;
179  if ( idx != string_type::npos) {
180  string_type name = remaining.substr(0, idx);
181  remaining = remaining.substr(idx + 1);
182  fmt = info->name_to_formatter[name];
183  }
184  // note: fmt could be null, in case
185  info->write_steps.push_back( write_step( spacer, fmt) );
186  }
187  else {
188  // last part
189  info->write_steps.push_back(
190  write_step( unescape(remaining), nullptr) );
191  remaining.clear();
192  }
193  }
194  }

93  template <typename Operation>
94  hpx::performance_counters::counter_value
95  arithmetics_counter<Operation>::get_counter_value(bool reset)
96  {
97  std::vector<counter_value> base_values =
98  counters_.get_counter_values(hpx::launch::sync);
99 
100  // apply arithmetic operation
101  double value = detail::init_value<Operation>::call();
102  for (counter_value const& base_value : base_values)
103  {
104  value = Operation()(value, base_value.get_value<double>());
105  }
106 
107  if (base_values[0].scale_inverse_ && base_values[0].scaling_ != 1.0) //-V550
108  {
109  base_values[0].value_ =
110  static_cast<std::int64_t>(value * base_values[0].scaling_);
111  }
112  else {
113  base_values[0].value_ =
114  static_cast<std::int64_t>(value / base_values[0].scaling_);
115  }
116 
117  base_values[0].time_ = static_cast<std::int64_t>(hpx::get_system_uptime());
118  base_values[0].count_ = counters_.get_invocation_count();
119 
120  return base_values[0];
121  }

201  naming::gid_type arithmetics_counter_creator(counter_info const& info,
202  error_code& ec)
203  {
204  switch (info.type_) {
205  case counter_aggregating:
206  {
207  counter_path_elements paths;
208  get_counter_path_elements(info.fullname_, paths, ec);
209  if (ec) return naming::invalid_gid;
210 
211  if (!paths.parameters_.empty()) {
212  // try to interpret the additional parameter as a list of
213  // two performance counter names
214  std::vector<std::string> names;
215  boost::split(names, paths.parameters_, boost::is_any_of(","));
216 
217  if (names.empty())
218  {
219  HPX_THROWS_IF(ec, bad_parameter,
220  "arithmetics_counter_creator",
221  "the parameter specification for an arithmetic counter "
222  "has to expand to at least one counter name: " +
223  paths.parameters_);
224  return naming::invalid_gid;
225  }
226 
227  for (std::string const& name : names)
228  {
229  counter_path_elements paths;
230  if (status_valid_data != get_counter_path_elements(
231  name, paths, ec) || ec)
232  {
233  HPX_THROWS_IF(ec, bad_parameter,
234  "arithmetics_counter_creator",
235  "the given (expanded) counter name is not "
236  "a validly formed performance counter name: " +
237  name);
238  return naming::invalid_gid;
239  }
240  }
241 
242  return create_arithmetics_counter(info, names, ec);
243  }
244  else {
245  HPX_THROWS_IF(ec, bad_parameter,
246  "arithmetics_counter_creator",
247  "the parameter specification for an arithmetic counter "
248  "has to be a comma separated list of performance "
249  "counter names, none is given: " +
250  remove_counter_prefix(info.fullname_));
251  }
252  }
253  break;
254 
255  default:
256  HPX_THROWS_IF(ec, bad_parameter, "arithmetics_counter_creator",
257  "invalid counter type requested");
258  break;
259  }
260  return naming::invalid_gid;
261  }

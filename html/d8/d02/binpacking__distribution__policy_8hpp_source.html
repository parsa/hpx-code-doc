43  inline std::vector<std::size_t>
44  get_items_count(std::size_t count, std::vector<std::uint64_t> const& values)
45  {
46  std::size_t maxcount = 0;
47  std::size_t existing = 0;
48 
49  for (std::uint64_t value: values)
50  {
51  maxcount = (std::max)(maxcount, std::size_t(value));
52  existing += std::size_t(value);
53  }
54 
55  // distribute the number of components to create in a way, so that
56  // the overall number of component instances on all localities is
57  // approximately the same
58  std::size_t num_localities = values.size();
59 
60  // calculate the number of new instances to create on each of the
61  // localities
62  std::vector<std::size_t> to_create(num_localities, 0);
63 
64  bool even_fill = true;
65  while (even_fill)
66  {
67  even_fill = false;
68  for (std::size_t i = 0; i != num_localities; ++i)
69  {
70  if(values[i] + to_create[i] >= maxcount) continue;
71  even_fill = true;
72 
73  ++to_create[i];
74  --count;
75  if (count == 0) break;
76  }
77  }
78  std::size_t i = 0;
79  while (count != 0)
80  {
81  ++to_create[i];
82  i = (i + 1) % num_localities;
83  --count;
84  }
85 
86  return to_create;
87  }

138  inline hpx::id_type const& get_best_locality(
139  hpx::future<std::vector<std::uint64_t> > && f,
140  std::vector<hpx::id_type> const& localities)
141  {
142  std::vector<std::uint64_t> values = f.get();
143 
144  std::size_t best_locality = 0;
145  std::uint64_t min_value =
146  (std::numeric_limits<std::uint64_t>::max)();
147 
148  for (std::size_t i = 0; i != values.size(); ++i)
149  {
150  if (min_value > values[i])
151  {
152  min_value = values[i];
153  best_locality = i;
154  }
155  }
156 
157  return localities[best_locality];
158  }

212  [=](std::vector<hpx::future<std::vector<hpx::id_type> > > && v)
213  mutable -> std::vector<bulk_locality_result>
214  {
215  HPX_ASSERT(localities_.size() == v.size());

217  std::vector<bulk_locality_result> result;
218  result.reserve(v.size());
219 
220  for (std::size_t i = 0; i != v.size(); ++i)
221  {
222  result.emplace_back(
223  std::move(localities_[i]), v[i].get()
224  );
225  }
226  return result;
227  },
228  std::move(objs));

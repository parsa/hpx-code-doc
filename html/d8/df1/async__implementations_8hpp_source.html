46  public:
47  handle_managed_target(hpx::id_type const& id, future<Result>& f)
48  : target_is_managed_(false), id_(id), f_(f)
49  {
50  if (id.get_management_type() == naming::id_type::managed)
51  {
52  unmanaged_id_ = naming::id_type(
53  id.get_gid(), naming::id_type::unmanaged);
54  target_is_managed_ = true;
55  }
56  }
57 
58  ~handle_managed_target()
59  {
60  // keep id alive, if needed - this allows to send the destination
61  // as an unmanaged id
62  if (target_is_managed_)
63  {
64  typedef typename traits::detail::shared_state_ptr_for<
65  future<Result>
66  >::type shared_state_ptr;
67 
68  shared_state_ptr const& state =
69  traits::detail::get_shared_state(f_);
70  if (state)
71  {
72  HPX_ASSERT(id_);
73  HPX_ASSERT(unmanaged_id_);
74  state->set_on_completed(
75  hpx::detail::keep_id_alive(id_));
76  }
77  }
78  }
79 
80  hpx::id_type const& get_id() const
81  {
82  return target_is_managed_ ? unmanaged_id_ : id_;
83  }

87  naming::id_type unmanaged_id_;

137  hpx::future<
138  typename hpx::traits::extract_action<Action>::type::local_result_type
139  >
140  async_impl(launch policy, hpx::id_type const& id, Ts&&... vs)
141  {
142  typedef typename hpx::traits::extract_action<Action>::type action_type;
143  typedef typename action_type::local_result_type result_type;
144  typedef typename action_type::component_type component_type;
145 
146  std::pair<bool, components::pinned_ptr> r;
147 
148  naming::address addr;
149  if (agas::is_local_address_cached(id, addr))
150  {
151  if (traits::component_supports_migration<component_type>::call())
152  {
153  r = traits::action_was_object_migrated<Action>::call(
154  id, addr.address_);
155  if (policy == launch::sync && !r.first)
156  {
157  return hpx::detail::sync_local_invoke<action_type, result_type>::
158  call(id, std::move(addr), std::forward<Ts>(vs)...);
159  }
160  }
161  else if (policy == launch::sync)
162  {
163  return hpx::detail::sync_local_invoke<action_type, result_type>::
164  call(id, std::move(addr), std::forward<Ts>(vs)...);
165  }
166  }
167 
168  future<result_type> f;
169  {
170  handle_managed_target<result_type> hmt(id, f);
171 
172  if (policy == launch::sync || hpx::detail::has_async_policy(policy))
173  {
174  lcos::packaged_action<action_type, result_type> p;
175 
176  f = p.get_future();
177  p.apply(std::move(addr), hmt.get_id(), std::forward<Ts>(vs)...);
178  if (policy == launch::sync)
179  f.wait();
180  }
181  else if (policy == launch::deferred)
182  {
183  lcos::packaged_action<action_type, result_type> p;
184 
185  f = p.get_future();
186  p.apply_deferred(std::move(addr), hmt.get_id(),
187  std::forward<Ts>(vs)...);
188  }
189  else
190  {
191  HPX_THROW_EXCEPTION(bad_parameter,
192  "async_impl", "unknown launch policy");
193  return f;
194  }
195  }
196  return f;
197  }

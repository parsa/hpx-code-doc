83  // The specifcation of std::mismatch(_binary) states that if FwdIter1
84  // and FwdIter2 meet the requirements of RandomAccessIterator and
85  // last1 - first1 != last2 - first2 then no applications of the
86  // predicate p are made.
87  //
88  // We perform this check for any iterator type better than input
89  // iterators. This could turn into a QoI issue.
90  typedef typename std::iterator_traits<FwdIter2>::difference_type
91  difference_type2;
92  difference_type2 count2 = std::distance(first2, last2);
93  if (count1 != count2)
94  {
95  return util::detail::algorithm_result<ExPolicy, T>::get(
96  std::make_pair(first1, first2));
97  }

103 
104  return util::partitioner<ExPolicy, T, void>::
105  call_with_index(
106  std::forward<ExPolicy>(policy),
107  hpx::util::make_zip_iterator(first1, first2), count1, 1,
108  [f, tok](zip_iterator it, std::size_t part_count,
109  std::size_t base_idx) mutable -> void
110  {

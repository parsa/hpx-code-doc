40  template <typename Algo, typename ExPolicy, typename SegIter,

42  static typename util::detail::algorithm_result<ExPolicy, T>::type

44  SegIter first, SegIter last, T && init,
45  Reduce && red_op, std::true_type)
46  {
47  typedef hpx::traits::segmented_iterator_traits<SegIter> traits;
48  typedef typename traits::segment_iterator segment_iterator;
49  typedef typename traits::local_iterator local_iterator_type;
50  typedef util::detail::algorithm_result<ExPolicy, T> result;
51 
52  segment_iterator sit = traits::segment(first);
53  segment_iterator send = traits::segment(last);
54 
55  T overall_result = init;
56 
57  if (sit == send)
58  {
59  // all elements are on the same partition
60  local_iterator_type beg = traits::local(first);
61  local_iterator_type end = traits::local(last);
62  if (beg != end)
63  {
64  overall_result = hpx::util::invoke(red_op, overall_result,
65  dispatch(traits::get_id(sit), algo, policy,
66  std::true_type(), beg, end, red_op));
67  }
68  }
69  else {
70  // handle the remaining part of the first partition
71  local_iterator_type beg = traits::local(first);
72  local_iterator_type end = traits::end(sit);
73  if (beg != end)
74  {
75  overall_result = hpx::util::invoke(red_op, overall_result,
76  dispatch(traits::get_id(sit), algo, policy,
77  std::true_type(), beg, end, red_op)
78  );
79  }
80 
81  // handle all of the full partitions
82  for (++sit; sit != send; ++sit)
83  {
84  beg = traits::begin(sit);
85  end = traits::end(sit);
86  if (beg != end)
87  {
88  overall_result = hpx::util::invoke(red_op, overall_result,
89  dispatch(traits::get_id(sit), algo, policy,

91  );

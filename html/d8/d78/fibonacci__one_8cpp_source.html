33 //[fib_func
34 std::uint64_t fibonacci(std::uint64_t n)
35 {
36  if (n < 2)
37  return n;
38 
39  // We restrict ourselves to execute the Fibonacci function locally.
40  hpx::naming::id_type const locality_id = hpx::find_here();
41 
42  // Run one branch of the Fibonacci calculation on this thread, while the
43  // other branch is scheduled in a separate thread.
44  hpx::future<std::uint64_t> n1 =
45  hpx::async<fibonacci_action>(locality_id, n-1);
46  std::uint64_t n2 = fibonacci(n-2);
47 
48  return n1.get() + n2; // wait for the Future to return its values
49 }

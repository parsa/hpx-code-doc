46  template <typename InIter, typename OutIter, typename T, typename Op,
47  typename Conv = util::projection_identity>
48  OutIter sequential_exclusive_scan(InIter first, InIter last,
49  OutIter dest, T init, Op && op, Conv && conv = Conv())
50  {
51  T temp = init;
52  for (/* */; first != last; (void) ++first, ++dest)
53  {
54  init = hpx::util::invoke(op, init, hpx::util::invoke(conv, *first));
55  *dest = temp;
56  temp = init;
57  }
58  return dest;
59  }

95  template <typename ExPolicy, typename FwdIter1, typename T, typename Op,
96  typename Conv = util::projection_identity>
97  static typename util::detail::algorithm_result<
98  ExPolicy, FwdIter2
99  >::type
100  parallel(ExPolicy && policy, FwdIter1 first, FwdIter1 last,
101  FwdIter2 dest, T const& init, Op && op, Conv && conv = Conv())
102  {
103  typedef util::detail::algorithm_result<ExPolicy, FwdIter2>
104  result;
105  typedef hpx::util::zip_iterator<FwdIter1, FwdIter2> zip_iterator;
106  typedef typename std::iterator_traits<FwdIter1>::difference_type
107  difference_type;
108 
109  if (first == last)
110  return result::get(std::move(dest));
111 
112  difference_type count = std::distance(first, last);
113 
114  FwdIter2 final_dest = dest;
115  std::advance(final_dest, count);
116 
117  // The overall scan algorithm is performed by executing 3
118  // steps. The first calculates the scan results for each
119  // partition. The second accumulates the result from left to
120  // right to be used by the third step--which operates on the
121  // same partitions the first step operated on.
122 
123  using hpx::util::get;
124  using hpx::util::make_zip_iterator;
125 
126  auto f3 =
127  [op, policy](
128  zip_iterator part_begin, std::size_t part_size,
129  hpx::shared_future<T> curr, hpx::shared_future<T> next
130  )
131  {
132  HPX_UNUSED(policy);
133 
134  next.get(); // rethrow exceptions
135 
136  T val = curr.get();
137  FwdIter2 dst = get<1>(part_begin.get_iterator_tuple());
138  *dst++ = val;
139 
140  // MSVC 2015 fails if op is captured by reference
141  util::loop_n<ExPolicy>(
142  dst, part_size - 1,
143  [=, &val](FwdIter2 it)
144  {
145  *it = hpx::util::invoke(op, val, *it);
146  });
147  };
148 
149  return util::scan_partitioner<ExPolicy, FwdIter2, T>::call(
150  std::forward<ExPolicy>(policy),
151  make_zip_iterator(first, dest), count, init,
152  // step 1 performs first part of scan algorithm
153  [op, conv, last](zip_iterator part_begin,
154  std::size_t part_size) -> T
155  {
156  T part_init = hpx::util::invoke(conv, get<0>(*part_begin++));
157 
158  auto iters = part_begin.get_iterator_tuple();
159  if(get<0>(iters) != last)
160  return sequential_exclusive_scan_n(
161  get<0>(iters),
162  part_size - 1,
163  get<1>(iters),
164  part_init, op, conv);
165  else
166  return part_init;
167  },
168  // step 2 propagates the partition results from left
169  // to right
170  hpx::util::unwrapping(op),
171  // step 3 runs final accumulation on each partition
172  std::move(f3),
173  // step 4 use this return value
174  [final_dest](std::vector<hpx::shared_future<T> > &&,
175  std::vector<hpx::future<void> > &&)
176  {
177  return final_dest;
178  });
179 
180  }
181  };

86  static basic_resolver_iterator create(
87  boost::asio::detail::addrinfo_type* address_info,
88  const std::string& host_name, const std::string& service_name)
89  {
90  basic_resolver_iterator iter;
91  if (!address_info)
92  return std::move(iter);
93 
94  std::string actual_host_name = host_name;
95  if (address_info->ai_canonname)
96  actual_host_name = address_info->ai_canonname;
97 
98  iter.values_.reset(new values_type);
99 
100  while (address_info)
101  {
102  if (address_info->ai_family == PF_INET
103  || address_info->ai_family == PF_INET6)
104  {
105  using namespace std; // For memcpy.
106  typename InternetProtocol::endpoint endpoint;
107  endpoint.resize(static_cast<std::size_t>(address_info->ai_addrlen));
108  memcpy(endpoint.data(), address_info->ai_addr,
109  address_info->ai_addrlen);
110  iter.values_->push_back(
111  basic_resolver_entry<InternetProtocol>(endpoint,
112  actual_host_name, service_name));
113  }
114  address_info = address_info->ai_next;
115  }
116 
117  return std::move(iter);
118  }

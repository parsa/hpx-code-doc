114 void LU( int numBlocks)
115 {
116  printf("LU\n");
117  hpx::naming::id_type here = hpx::find_here();
118  vector<vector<block> > blockList;
119  getBlockList(blockList, numBlocks);
120  vector<vector<vector<shared_future<block> > > > dfArray(numBlocks);
121  shared_future<block> *diag_block, *first_col;
122 
123  for(int i = 0; i < numBlocks; i++){
124  dfArray[i].resize(numBlocks);
125  for(int j = 0; j < numBlocks; j++){
126  dfArray[i][j].resize(numBlocks, hpx::make_ready_future(block()));
127  }
128  }
129  //first iteration through matrix, initialized vector of futures
130  dfArray[0][0][0] = async( ProcessDiagonalBlock, blockList[0][0] );
131  diag_block = &dfArray[0][0][0];
132  for(int i = 1; i < numBlocks; i++) {
133  dfArray[0][0][i] = dataflow( unwrapping( &ProcessBlockOnRow ),
134  hpx::make_ready_future( blockList[0][i] ), *diag_block);
135  }
136  for(int i = 1; i < numBlocks; i++) {
137  dfArray[0][i][0] = dataflow( unwrapping( &ProcessBlockOnColumn ),
138  hpx::make_ready_future( blockList[i][0] ), *diag_block);
139  first_col = &dfArray[0][i][0];
140  for(int j = 1; j < numBlocks; j++) {
141  dfArray[0][i][j] = dataflow( unwrapping( &ProcessInnerBlock ),
142  hpx::make_ready_future( blockList[i][j]), dfArray[0][0][j], *first_col );
143  }
144  }
145  //all calculation after initialization. Each iteration,
146  //the number of tasks/blocks spawned is decreased.
147  for(int i = 1; i < numBlocks; i++) {
148  dfArray[i][i][i] = dataflow( unwrapping( &ProcessDiagonalBlock ),
149  dfArray[i-1][i][i]);
150  diag_block = &dfArray[i][i][i];
151  for(int j = i + 1; j < numBlocks; j++){
152  dfArray[i][i][j] = dataflow( unwrapping(&ProcessBlockOnRow),
153  dfArray[i-1][i][j], *diag_block);
154  }
155  for(int j = i + 1; j < numBlocks; j++){
156  dfArray[i][j][i] = dataflow( unwrapping( &ProcessBlockOnColumn ),
157  dfArray[i-1][j][i], *diag_block);
158  first_col = &dfArray[i][j][i];
159  for(int k = i + 1; k < numBlocks; k++) {
160  dfArray[i][j][k] = dataflow( unwrapping( &ProcessInnerBlock ),
161  dfArray[i-1][j][k], dfArray[i][i][k], *first_col );
162  }
163  }
164  }
165  wait_all(dfArray[numBlocks-1][numBlocks-1][numBlocks-1]);
166 }

168 void getBlockList(vector<vector<block> > &blockList, int numBlocks)
169 {
170  int blockSize, start, height;
171  for(int i=0; i < numBlocks; i++)
172  blockList.push_back(vector<block>());
173 
174  height = size/numBlocks;
175  if(size%numBlocks > 0)
176  height += 1;
177  for(int i=0; i < numBlocks; i++) {
178  if(i < size % numBlocks) {
179  blockSize = size/numBlocks+1;
180  start = (size/numBlocks+1)*i;
181  } else {
182  blockSize = size/numBlocks;
183  start = (size/numBlocks+1)*(size%numBlocks)
184  + (size/numBlocks)*(i-size%numBlocks);
185  }
186  blockList[0].push_back( block( blockSize, start, height));
187  }
188  for(int i = 1; i < numBlocks; i++) {
189  height = blockList[0][i].size;
190  for(int j = 0; j < numBlocks; j++) {
191  blockSize = blockList[0][j].size;
192  start = blockList[i-1][j].start + blockList[i-1][0].height * size;
193  blockList[i].push_back( block( blockSize, start, height));
194  }
195  }
196 }

198 block ProcessDiagonalBlock( block B)
199 {
200  for(int i = 0; i < B.size; i++) {
201  for(int j = i+1; j < B.size; j++){
202  A[B.start+j*size+i] /= A[B.start+i*size+i];
203  for(int k = i+1; k < B.size; k++) {
204  A[B.start+j*size+k] -= A[B.start+j*size+i] * A[B.start+i*size+k];
205  }
206  }
207  }

242 void checkResult( vector<double> &originalA )
243 {
244  int errors = 0;
245  double temp2;
246  vector<double> L(size*size, 0);
247  vector<double> U(size*size, 0);
248  for(int i=0;i<size;i++)
249  for(int j=0;j<size;j++)
250  if (i>j)
251  L[i*size+j] = A[i*size+j];
252  else
253  U[i*size+j] = A[i*size+j];
254  for(int i=0;i<size;i++)
255  L[i*size+i] = 1;
256 
257  for(int i=0;i<size;i++) {
258  for(int j=0;j<size;j++){
259  temp2=0;
260  for(int k=0;k<size;k++)
261  temp2+=L[i*size+k]*U[k*size+j];
262  if( (originalA[i*size+j]-temp2) / originalA[i*size+j] > 0.1
263  || (originalA[i*size+j]-temp2) / originalA[i*size+j] < -0.1 ){
264  printf("error:[%d][%d] ", i, j);
265  errors++;
266  }
267  }
268  }
269  HPX_TEST_EQ(errors, 0);
270 }

45  thread_state_ex_enum newstate_ex, thread_priority priority,

47  {

49  HPX_THROW_EXCEPTION(null_thread_id,

51  "null thread id encountered");
52  return thread_result_type(terminated, nullptr);

54 
55  // make sure that the thread has not been suspended and set active again
56  // in the meantime

74  detail::set_thread_state(thrd, newstate, newstate_ex, priority,

81  inline thread_state set_thread_state(
82  thread_id_type const& thrd, thread_state_enum new_state,
83  thread_state_ex_enum new_state_ex, thread_priority priority,
84  std::size_t thread_num, error_code& ec)
85  {
86  if (HPX_UNLIKELY(!thrd)) {
87  HPX_THROWS_IF(ec, null_thread_id, "threads::detail::set_thread_state",
88  "null thread id encountered");
89  return thread_state(unknown, wait_unknown);
90  }
91 
92  // set_state can't be used to force a thread into active state
93  if (new_state == threads::active) {
94  std::ostringstream strm;
95  strm << "invalid new state: " << get_thread_state_name(new_state);
96  HPX_THROWS_IF(ec, bad_parameter, "threads::detail::set_thread_state",
97  strm.str());
98  return thread_state(unknown, wait_unknown);
99  }
100 
101  // we know that the id is actually the pointer to the thread
102  if (!thrd) {
103  if (&ec != &throws)
104  ec = make_success_code();
105  return thread_state(terminated, wait_unknown);
106  // this thread has already been terminated
107  }
108 
109  thread_state previous_state;
110  do {
111  // action depends on the current state
112  previous_state = thrd->get_state();
113  thread_state_enum previous_state_val = previous_state.state();
114 
115  // nothing to do here if the state doesn't change
116  if (new_state == previous_state_val) {
117  LTM_(warning)
118  << "set_thread_state: old thread state is the same as new "
119  "thread state, aborting state change, thread("
120  << thrd.get() << "), description("
121  << thrd->get_description() << "), new state("
122  << get_thread_state_name(new_state) << ")";
123 
124  if (&ec != &throws)
125  ec = make_success_code();
126 
127  return thread_state(new_state, previous_state.state_ex());
128  }
129 
130  // the thread to set the state for is currently running, so we
131  // schedule another thread to execute the pending set_state
132  switch (previous_state_val) {
133  case active:
134  {
135  // schedule a new thread to set the state
136  LTM_(warning)
137  << "set_thread_state: thread is currently active, scheduling "
138  "new thread, thread(" << thrd.get() << "), description("
139  << thrd->get_description() << "), new state("
140  << get_thread_state_name(new_state) << ")";
141 
142  thread_init_data data(
143  util::bind(&set_active_state,
144  thrd, new_state, new_state_ex,
145  priority, previous_state),
146  "set state for active thread", 0, priority);
147 
148  create_work(thrd->get_scheduler_base(), data, pending, ec);
149 
150  if (&ec != &throws)
151  ec = make_success_code();
152 
153  return previous_state; // done
154  }
155  break;
156  case terminated:
157  {
158  LTM_(warning)
159  << "set_thread_state: thread is terminated, aborting state "
160  "change, thread(" << thrd.get() << "), description("
161  << thrd->get_description() << "), new state("
162  << get_thread_state_name(new_state) << ")";
163 
164  if (&ec != &throws)
165  ec = make_success_code();
166 
167  // If the thread has been terminated while this set_state was
168  // pending nothing has to be done anymore.
169  return previous_state;
170  }
171  break;
172  case pending:
173  case pending_boost:
174  if (suspended == new_state) {
175  // we do not allow explicit resetting of a state to suspended
176  // without the thread being executed.
177  std::ostringstream strm;
178  strm << "set_thread_state: invalid new state, can't demote a "
179  "pending thread, "
180  << "thread(" << thrd.get() << "), description("
181  << thrd->get_description() << "), new state("
182  << get_thread_state_name(new_state) << ")";
183 
184  LTM_(fatal) << strm.str();
185 
186  HPX_THROWS_IF(ec, bad_parameter,
187  "threads::detail::set_thread_state",
188  strm.str());
189  return thread_state(unknown, wait_unknown);
190  }
191  break;
192  case suspended:
193  break; // fine, just set the new state
194  case pending_do_not_schedule:
195  default:
196  HPX_ASSERT(false); // should not happen
197  break;
198  }
199 
200  // If the previous state was pending we are supposed to remove the
201  // thread from the queue. But in order to avoid linearly looking
202  // through the queue we defer this to the thread function, which
203  // at some point will ignore this thread by simply skipping it
204  // (if it's not pending anymore).
205 
206  LTM_(info) << "set_thread_state: thread(" << thrd.get() << "), "
207  "description(" << thrd->get_description() << "), "
208  "new state(" << get_thread_state_name(new_state) << "), "
209  "old state(" << get_thread_state_name(previous_state_val)
210  << ")";
211 
212  // So all what we do here is to set the new state.
213  if (thrd->restore_state(new_state, new_state_ex, previous_state))
214  break;
215 
216  // state has changed since we fetched it from the thread, retry
217  LTM_(error)
218  << "set_thread_state: state has been changed since it was fetched, "
219  "retrying, thread(" << thrd.get() << "), "
220  "description(" << thrd->get_description() << "), "
221  "new state(" << get_thread_state_name(new_state) << "), "
222  "old state(" << get_thread_state_name(previous_state_val)
223  << ")";
224  } while (true);
225 
226  if (new_state == pending) {
227  // REVIEW: Passing a specific target thread may interfere with the
228  // round robin queuing.
229  thrd->get_scheduler_base()->schedule_thread(thrd.get(),
230  thread_num, priority);
231  thrd->get_scheduler_base()->do_some_work(thread_num);
232  }
233 
234  if (&ec != &throws)
235  ec = make_success_code();
236 
237  return previous_state;
238  }

243  inline thread_result_type wake_timer_thread(
244  thread_id_type const& thrd, thread_state_enum newstate,
245  thread_state_ex_enum newstate_ex, thread_priority priority,
246  thread_id_type const& timer_id,
247  std::shared_ptr<std::atomic<bool> > const& triggered)
248  {
249  if (HPX_UNLIKELY(!thrd)) {
250  HPX_THROW_EXCEPTION(null_thread_id,
251  "threads::detail::wake_timer_thread",
252  "null thread id encountered (id)");
253  return thread_result_type(terminated, nullptr);
254  }
255  if (HPX_UNLIKELY(!timer_id)) {
256  HPX_THROW_EXCEPTION(null_thread_id,
257  "threads::detail::wake_timer_thread",
258  "null thread id encountered (timer_id)");
259  return thread_result_type(terminated, nullptr);
260  }
261 
262  bool oldvalue = false;
263  if (triggered->compare_exchange_strong(oldvalue, true)) //-V601
264  {
265  // timer has not been canceled yet, trigger the requested set_state
266  detail::set_thread_state(thrd, newstate, newstate_ex, priority);
267  }
268 
269  // then re-activate the thread holding the deadline_timer
270  error_code ec(lightweight); // do not throw
271  detail::set_thread_state(timer_id, pending, wait_timeout,
272  thread_priority_boost, std::size_t(-1), ec);
273 
274  return thread_result_type(terminated, nullptr);
275  }

279  template <typename SchedulingPolicy>
280  thread_result_type at_timer(SchedulingPolicy& scheduler,
281  util::steady_clock::time_point& abs_time,
282  thread_id_type const& thrd, thread_state_enum newstate,
283  thread_state_ex_enum newstate_ex, thread_priority priority)
284  {
285  if (HPX_UNLIKELY(!thrd)) {
286  HPX_THROW_EXCEPTION(null_thread_id,
287  "threads::detail::at_timer",
288  "null thread id encountered");
289  return thread_result_type(terminated, nullptr);
290  }
291 
292  // create a new thread in suspended state, which will execute the
293  // requested set_state when timer fires and will re-awaken this thread,
294  // allowing the deadline_timer to go out of scope gracefully
295  thread_id_type self_id = get_self_id();
296 
297  std::shared_ptr<std::atomic<bool> > triggered(
298  std::make_shared<std::atomic<bool> >(false));
299 
300  thread_init_data data(
301  util::bind(&wake_timer_thread,
302  thrd, newstate, newstate_ex, priority,
303  self_id, triggered),
304  "wake_timer", 0, priority);
305 
306  thread_id_type wake_id = invalid_thread_id;
307  create_thread(&scheduler, data, wake_id, suspended);
308 
309  // create timer firing in correspondence with given time
310  typedef boost::asio::basic_waitable_timer<
311  util::steady_clock> deadline_timer;
312 
313  deadline_timer t (
314  get_thread_pool("timer-pool")->get_io_service(), abs_time);
315 
316  // let the timer invoke the set_state on the new (suspended) thread
317  t.async_wait(util::bind(&detail::set_thread_state,
318  wake_id, pending, wait_timeout, priority,
319  std::size_t(-1), std::ref(throws)));
320 
321  // this waits for the thread to be reactivated when the timer fired
322  // if it returns signaled the timer has been canceled, otherwise
323  // the timer fired and the wake_timer_thread above has been executed
324  thread_state_ex_enum statex =
325  get_self().yield(thread_result_type(suspended, nullptr));
326 
327  if (wait_timeout != statex) //-V601
328  {
329  triggered->store(true);
330 
331  // wake_timer_thread has not been executed yet, cancel timer
332  t.cancel();
333 
334  // cancel wake_timer_thread
335  error_code ec(lightweight); // do not throw
336  detail::set_thread_state(wake_id, pending, wait_abort,
337  priority, std::size_t(-1), ec);
338  }
339 
340  return thread_result_type(terminated, nullptr);
341  }

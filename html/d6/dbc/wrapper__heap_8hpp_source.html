253  naming::gid_type get_gid(util::unique_id_ranges& ids, void* p,
254  components::component_type type)
255  {
256  util::itt::heap_internal_access hia; HPX_UNUSED(hia);
257 
258  HPX_ASSERT(did_alloc(p));
259 
260  scoped_lock l(mtx_);
261  value_type* addr = static_cast<value_type*>(pool_->address());
262 
263  if (!base_gid_) {
264  naming::gid_type base_gid;
265 
266  {
267  // this is the first call to get_gid() for this heap - allocate
268  // a sufficiently large range of global ids
269  util::unlock_guard<scoped_lock> ul(l);
270  base_gid = ids.get_id(step_);
271 
272  // register the global ids and the base address of this heap
273  // with the AGAS
274  if (!applier::bind_range_local(base_gid, step_,
275  naming::address(hpx::get_locality(), type, addr),
276  sizeof(value_type)))
277  {
278  return naming::invalid_gid;
279  }
280  }
281 
282  // if some other thread has already set the base GID for this
283  // heap, we ignore the result
284  if (!base_gid_)
285  {
286  // this is the first thread succeeding in binding the new gid range
287  base_gid_ = base_gid;
288  }
289  else
290  {
291  // unbind the range which is not needed anymore
292  util::unlock_guard<scoped_lock> ul(l);
293  applier::unbind_range_local(base_gid, step_);
294  }
295  }
296 
297  return base_gid_ + static_cast<std::uint64_t>(
298  static_cast<value_type*>(p) - addr);
299  }

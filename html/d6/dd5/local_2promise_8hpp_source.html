36  struct deleter
37  {
38  template <typename SharedState_>
39  void operator()(SharedState_* state)
40  {
41  typedef std::allocator_traits<Allocator> traits;
42  traits::deallocate(alloc_, state, 1);
43  }
44 
45  Allocator& alloc_;
46  };

54  template <typename Allocator>
55  promise_base(std::allocator_arg_t, Allocator const& a)
56  : shared_state_()
57  , future_retrieved_(false)
58  {
59  typedef typename traits::detail::shared_state_allocator<
60  SharedState, Allocator
61  >::type allocator_shared_state_type;
62 
63  typedef typename
64  std::allocator_traits<Allocator>::template
65  rebind_alloc<allocator_shared_state_type>
66  other_allocator;
67  typedef std::allocator_traits<other_allocator> traits;
68  typedef std::unique_ptr<
69  allocator_shared_state_type, deleter<other_allocator>
70  > unique_pointer;
71 
72  other_allocator alloc(a);
73  unique_pointer p (traits::allocate(alloc, 1),
74  deleter<other_allocator>{alloc});
75 
76 #if BOOST_VERSION >= 105600
77  traits::construct(alloc, p.get(), init_no_addref(), alloc);
78  shared_state_.reset(p.release(), false);
79 #else
80  traits::construct(alloc, p.get(), alloc);
81  shared_state_ = boost::intrusive_ptr<shared_state_type>(p.release());
82 #endif
83  }

249  }
250

252  // Postcondition: *this has the shared state (if any) that other had
253  // prior to the call to swap. other has the shared state

255  void swap(promise& other) noexcept
256  {
257  base_type::swap(other);
258  }
259

261  bool valid() const noexcept
262  {
263  return base_type::valid();
264  }
265 
266  // Returns: A future<R> object with the same shared state as *this.
267  // Throws: future_error if *this has no shared state or if get_future

270  // Error conditions:
271  // - future_already_retrieved if get_future has already been called
272  // on a promise with the same shared state as *this.
273  // - no_state if *this has no shared state.
274  future<R> get_future(error_code& ec = throws)
275  {
276  return base_type::get_future(ec);
277  }
278 
279  // Effects: atomically stores the value r in the shared state and makes

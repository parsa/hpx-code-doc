6 // This example demonstrates several things:
7 //
8 // - How to initialize (and terminate) the HPX runtime from a global object
9 // (see the type `manage_global_runtime' below)
10 // - How to register and unregister any (kernel) thread with the HPX runtime
11 // - How to launch an HPX thread from any (registered) kernel thread and
12 // how to wait for the HPX thread to run to completion before continuing.
13 // Any value returned from the HPX thread will be marshaled back to the
14 // calling (kernel) thread.
15 //
16 // This scheme is generally useful if HPX should be initialized from a shared
17 // library and the main executable might not even be aware of this.

66 // This class demonstrates how to initialize a console instance of HPX
67 // (locality 0). In order to create an HPX instance which connects to a running
68 // HPX application two changes have to be made:
69 //
70 // - replace hpx::runtime_mode_console with hpx::runtime_mode_connect
71 // - replace hpx::finalize() with hpx::disconnect()
72 //
73 struct manage_global_runtime
74 {
75  manage_global_runtime()
76  : running_(false), rts_(nullptr)
77  {
78 #if defined(HPX_WINDOWS)
79  hpx::detail::init_winsocket();
80 #endif
81 
82  std::vector<std::string> const cfg = {
83  // make sure hpx_main is always executed
84  "hpx.run_hpx_main!=1",
85  // allow for unknown command line options
86  "hpx.commandline.allow_unknown!=1",
87  // disable HPX' short options
88  "hpx.commandline.aliasing!=0"
89  };
90 
91  using hpx::util::placeholders::_1;
92  using hpx::util::placeholders::_2;
93  hpx::util::function_nonser<int(int, char**)> start_function =
94  hpx::util::bind(&manage_global_runtime::hpx_main, this, _1, _2);
95 
96  if (!hpx::start(start_function, __argc, __argv, cfg, hpx::runtime_mode_console))
97  {
98  // Something went wrong while initializing the runtime.
99  // This early we can't generate any output, just bail out.
100  std::abort();
101  }
102 
103  // Wait for the main HPX thread (hpx_main below) to have started running
104  std::unique_lock<std::mutex> lk(startup_mtx_);
105  while (!running_)
106  startup_cond_.wait(lk);
107  }
108 
109  ~manage_global_runtime()
110  {
111  // notify hpx_main above to tear down the runtime
112  {
113  std::lock_guard<hpx::lcos::local::spinlock> lk(mtx_);
114  rts_ = nullptr; // reset pointer
115  }
116 
117  cond_.notify_one(); // signal exit
118 
119  // wait for the runtime to exit
120  hpx::stop();
121  }
122 
123  // registration of external (to HPX) threads
124  void register_thread(char const* name)
125  {
126  hpx::register_thread(rts_, name);
127  }
128  void unregister_thread()
129  {
130  hpx::unregister_thread(rts_);
131  }
132 
133 protected:
134  // Main HPX thread, does nothing but wait for the application to exit
135  int hpx_main(int argc, char* argv[])
136  {
137  // Store a pointer to the runtime here.
138  rts_ = hpx::get_runtime_ptr();
139 
140  // Signal to constructor that thread has started running.
141  {
142  std::lock_guard<std::mutex> lk(startup_mtx_);
143  running_ = true;
144  }
145 
146  startup_cond_.notify_one();
147 
148  // Here other HPX specific functionality could be invoked...
149 
150  // Now, wait for destructor to be called.
151  {
152  std::unique_lock<hpx::lcos::local::spinlock> lk(mtx_);
153  if (rts_ != nullptr)
154  cond_.wait(lk);
155  }
156 
157  // tell the runtime it's ok to exit
158  return hpx::finalize();
159  }
160 
161 private:

163  hpx::lcos::local::condition_variable_any cond_;
164 
165  std::mutex startup_mtx_;
166  std::condition_variable startup_cond_;
167  bool running_;
168 
169  hpx::runtime* rts_;
170 };

177 struct thread_registration_wrapper

179  thread_registration_wrapper(char const* name)
180  {
181  // Register this thread with HPX, this should be done once for
182  // each external OS-thread intended to invoke HPX functionality.

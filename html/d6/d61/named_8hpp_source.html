123  // recomputes the write steps - note taht this takes place after each operation
124  // for instance, the user might have first set the string and

126  void compute_write_steps() {
127  typename data::write info(m_data);
128  info->write_steps.clear();
129 
130  std::basic_istringstream<char_type> in(info->format_string);
131  string_type word;
132  while ( in >> word) {
133  if ( word[0] == '+')
134  word.erase( word.begin());
135  else if ( word[0] == '-')
136  // ignore this word
137  continue;
138 
139  if ( info->name_to_destination.find(word) !=
140  info->name_to_destination.end())
141  info->write_steps.push_back( info->
142  name_to_destination.find(word)->second);
143  }

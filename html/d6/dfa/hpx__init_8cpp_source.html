338  void handle_list_and_print_options(hpx::runtime& rt,
339  boost::program_options::variables_map& vm,
340  bool print_counters_locally)
341  {
342  if (vm.count("hpx:list-counters")) {
343  // Print the names of all registered performance counters.
344  std::string option(vm["hpx:list-counters"].as<std::string>());
345  if (0 == std::string("minimal").find(option))
346  rt.add_startup_function(&list_counter_names_minimal);
347  else if (0 == std::string("full").find(option))
348  rt.add_startup_function(&list_counter_names_full);
349  else {
350  std::string msg ("Invalid command line option value"
351  "for --hpx:list-counters: ");
352  msg += option;
353  msg += ", allowed values are 'minimal' and 'full'";
354  throw detail::command_line_error(msg.c_str());
355  }
356  }
357  if (vm.count("hpx:list-counter-infos")) {
358  // Print info about all registered performance counters.
359  std::string option(vm["hpx:list-counter-infos"].as<std::string>());
360  if (0 == std::string("minimal").find(option))
361  rt.add_startup_function(&list_counter_infos_minimal);
362  else if (0 == std::string("full").find(option))
363  rt.add_startup_function(&list_counter_infos_full);
364  else {
365  std::string msg ("Invalid command line option value"
366  "for --hpx:list-counter-infos: ");
367  msg += option;
368  msg += ", allowed values are 'minimal' and 'full'";
369  throw detail::command_line_error(msg.c_str());
370  }
371  }
372  if (vm.count("hpx:list-symbolic-names")) {
373  // Print all registered symbolic names.
374  rt.add_startup_function(&list_symbolic_names);
375  }
376  if (vm.count("hpx:list-component-types")) {
377  // Print all registered component types.
378  rt.add_startup_function(&list_component_types);
379  }
380 
381  if (vm.count("hpx:print-counter") || vm.count("hpx:print-counter-reset"))
382  {
383  std::size_t interval = 0;
384  if (vm.count("hpx:print-counter-interval"))
385  interval = vm["hpx:print-counter-interval"].as<std::size_t>();
386 
387  std::vector<std::string> counters;
388  if (vm.count("hpx:print-counter"))
389  {
390  counters = vm["hpx:print-counter"]
391  .as<std::vector<std::string> >();
392  }
393  std::vector<std::string> reset_counters;
394  if (vm.count("hpx:print-counter-reset"))
395  {
396  reset_counters = vm["hpx:print-counter-reset"]
397  .as<std::vector<std::string> >();
398  }
399 
400  std::vector<std::string> counter_shortnames;
401  std::string counter_format("normal");
402  if (vm.count("hpx:print-counter-format")) {
403  counter_format = vm["hpx:print-counter-format"].as<std::string>();
404  if (counter_format == "csv-short"){
405  for (std::size_t i = 0; i != counters.size() ; ++i) {
406  std::vector<std::string> entry;
407  boost::algorithm::split(entry, counters[i],
408  boost::algorithm::is_any_of(","),
409  boost::algorithm::token_compress_on);
410 
411  if (entry.size() != 2)
412  {
413  throw detail::command_line_error(
414  "Invalid format for command line option "
415  "--hpx:print-counter-format=csv-short");
416  }
417 
418  counter_shortnames.push_back(entry[0]);
419  counters[i] = entry[1];
420  }
421  }
422  }
423 
424  bool csv_header = true;
425  if(vm.count("hpx:no-csv-header"))
426  csv_header = false;
427 
428  std::string destination("cout");
429  if (vm.count("hpx:print-counter-destination"))
430  destination = vm["hpx:print-counter-destination"].as<std::string>();
431 
432  // schedule the query function at startup, which will schedule
433  // itself to run after the given interval
434  std::shared_ptr<util::query_counters> qc =
435  std::make_shared<util::query_counters>(
436  std::ref(counters), std::ref(reset_counters), interval,
437  destination, counter_format, counter_shortnames, csv_header,
438  print_counters_locally);
439 
440  // schedule to print counters at shutdown, if requested
441  if (get_config_entry("hpx.print_counter.shutdown", "0") == "1")
442  {
443  // schedule to run at shutdown
444  rt.add_pre_shutdown_function(
445  util::bind(&util::query_counters::evaluate, qc));
446  }
447 
448  // schedule to start all counters
449  rt.add_startup_function(util::bind(&start_counters, qc));
450 
451  // register the query_counters object with the runtime system
452  rt.register_query_counters(qc);
453  }
454  else if (vm.count("hpx:print-counter-interval")) {
455  throw detail::command_line_error("Invalid command line option "
456  "--hpx:print-counter-interval, valid in conjunction with "
457  "--hpx:print-counter only");
458  }
459  else if (vm.count("hpx:print-counter-destination")) {
460  throw detail::command_line_error("Invalid command line option "
461  "--hpx:print-counter-destination, valid in conjunction with "
462  "--hpx:print-counter only");
463  }
464  else if (vm.count("hpx:print-counter-format")) {
465  throw detail::command_line_error("Invalid command line option "
466  "--hpx:print-counter-format, valid in conjunction with "
467  "--hpx:print-counter only");
468  }
469  else if (vm.count("hpx:print-counter-at")) {
470  throw detail::command_line_error("Invalid command line option "
471  "--hpx:print-counter-at, valid in conjunction with "
472  "--hpx:print-counter only");
473  }
474  else if (vm.count("hpx:reset-counters")) {
475  throw detail::command_line_error("Invalid command line option "
476  "--hpx:reset-counters, valid in conjunction with "
477  "--hpx:print-counter only");
478  }
479  }

482  boost::program_options::variables_map& vm, runtime_mode mode,
483  startup_function_type startup, shutdown_function_type shutdown)
484  {

487  std::string config(vm["hpx:app-config"].as<std::string>());
488  rt.get_config().load_application_configuration(config.c_str());
489  }
490 
491  if (!!startup)
492  rt.add_startup_function(std::move(startup));
493 
494  if (!!shutdown)
495  rt.add_shutdown_function(std::move(shutdown));
496 
497  // Add startup function related to listing counter names or counter
498  // infos (on console only).
499  bool print_counters_locally =
500  vm.count("hpx:print-counters-locally") != 0;
501  if (mode == runtime_mode_console || print_counters_locally)
502  handle_list_and_print_options(rt, vm, print_counters_locally);
503 
504  // Dump the configuration before all components have been loaded.
505  if (vm.count("hpx:dump-config-initial")) {
506  std::cout << "Configuration after runtime construction:\n";
507  std::cout << "-----------------------------------------\n";
508  rt.get_config().dump(0, std::cout);
509  std::cout << "-----------------------------------------\n";
510  }
511 
512  // Dump the configuration after all components have been loaded.
513  if (vm.count("hpx:dump-config"))
514  rt.add_startup_function(dump_config(rt));
515  }

557  int run_or_start(bool blocking, std::unique_ptr<hpx::runtime> rt,
558  util::command_line_handling& cfg,
559  startup_function_type startup, shutdown_function_type shutdown)
560  {
561  if (blocking) {
562  return run(*rt, cfg.hpx_main_f_, cfg.vm_, cfg.rtcfg_.mode_,
563  std::move(startup), std::move(shutdown));
564  }
565 
566  // non-blocking version
567  start(*rt, cfg.hpx_main_f_, cfg.vm_, cfg.rtcfg_.mode_, std::move(startup),
568  std::move(shutdown));
569 
570  rt.release(); // pointer to runtime is stored in TLS
571  return 0;

575  HPX_EXPORT int run_or_start(
576  util::function_nonser<
577  int(boost::program_options::variables_map& vm)
578  > const& f,
579  boost::program_options::options_description const& desc_cmdline,
580  int argc, char** argv, std::vector<std::string> && ini_config,
581  startup_function_type startup, shutdown_function_type shutdown,
582  hpx::runtime_mode mode, bool blocking)
583  {
584 #if !defined(HPX_HAVE_DISABLED_SIGNAL_EXCEPTION_HANDLERS)
585  set_error_handlers();
586 #endif
587 
588 #if defined(HPX_NATIVE_MIC) || defined(__bgq__) || defined(__bgqion__)
589  unsetenv("LANG");
590  unsetenv("LC_CTYPE");
591  unsetenv("LC_NUMERIC");
592  unsetenv("LC_TIME");
593  unsetenv("LC_COLLATE");
594  unsetenv("LC_MONETARY");
595  unsetenv("LC_MESSAGES");
596  unsetenv("LC_PAPER");
597  unsetenv("LC_NAME");
598  unsetenv("LC_ADDRESS");
599  unsetenv("LC_TELEPHONE");
600  unsetenv("LC_MEASUREMENT");
601  unsetenv("LC_IDENTIFICATION");
602  unsetenv("LC_ALL");
603 #endif
604 
605  int result = 0;
606  try {
607  // make sure the runtime system is not active yet
608  if (get_runtime_ptr() != nullptr)
609  {
610  std::cerr << "hpx::init: can't initialize runtime system "
611  "more than once! Exiting...\n";
612  return -1;
613  }
614 
615  // Construct resource partitioner if this has not been done yet
616  // and get a handle to it
617  // (if the command-line parsing has not yet been done, do it now)
618  auto& rp = hpx::resource::detail::create_partitioner(f,
619  desc_cmdline, argc, argv, std::move(ini_config),
620  resource::mode_default, mode, false);
621 
622  // check whether HPX should be exited at this point
623  // (if the program options contain --hpx:help or --hpx:version)
624  if (rp.terminate_after_parse())
625  {
626  if (result > 0)
627  result = 0;
628  return result;
629  }
630 
631  // Setup all internal parameters of the resource_partitioner
632  rp.configure_pools();
633 
634  // initialize logging
635  util::command_line_handling& cms = rp.get_command_line_switches();
636  util::detail::init_logging(
637  cms.rtcfg_, cms.rtcfg_.mode_ == runtime_mode_console);
638 
639  util::apex_wrapper_init apex(argc, argv);
640 
641  // Initialize and start the HPX runtime.
642  LPROGRESS_ << "run_local: create runtime";
643 
644  // Build and configure this runtime instance.
645  typedef hpx::runtime_impl runtime_type;
646  std::unique_ptr<hpx::runtime> rt(
647  new runtime_type(cms.rtcfg_));
648 
649  result = run_or_start(blocking, std::move(rt),
650  cms, std::move(startup), std::move(shutdown));
651  }
652  catch (detail::command_line_error const& e) {
653  std::cerr << "{env}: " << hpx::detail::get_execution_environment();
654  std::cerr << "hpx::init: std::exception caught: " << e.what() << "\n";
655  return -1;
656  }
657  return result;
658  }

663  {
664  if (!config.empty()) {

45  // each available core.

49  template <typename ExPolicy, typename FwdIter, typename F1, typename F2>
50  static R call(ExPolicy && policy, FwdIter first,
51  std::size_t count, F1 && f1, F2 && f2)
52  {
53  typedef typename
54  hpx::util::decay<ExPolicy>::type::executor_parameters_type
55  parameters_type;
56  typedef executor_parameter_traits<parameters_type>
57  parameters_traits;
58 
59  // inform parameter traits
60  scoped_executor_parameters<parameters_type> scoped_param(
61  policy.parameters());
62 
63  std::vector<hpx::future<Result> > inititems;
64  std::list<std::exception_ptr> errors;
65 
66  try {
67  // estimate a chunk size based on number of cores used
68  typedef typename parameters_traits::has_variable_chunk_size
69  has_variable_chunk_size;
70 
71  auto shapes =
72  get_bulk_iteration_shape(policy, inititems, f1,
73  first, count, 1, has_variable_chunk_size());
74 
75  std::vector<hpx::future<Result> > workitems =
76  execution::bulk_async_execute(
77  policy.executor(),
78  partitioner_iteration<Result, F1>{std::forward<F1>(f1)},
79  std::move(shapes));
80 
81  // add the newly created workitems to the list
82  inititems.reserve(inititems.size() + workitems.size());
83  std::move(workitems.begin(), workitems.end(),
84  std::back_inserter(inititems));
85  }
86  catch (...) {
87  handle_local_exceptions<ExPolicy>::call(
88  std::current_exception(), errors);
89  }
90 
91  // wait for all tasks to finish
92  hpx::wait_all(inititems);
93 
94  // always rethrow if 'errors' is not empty or inititems has
95  // exceptional future
96  handle_local_exceptions<ExPolicy>::call(inititems, errors);
97 
98  try {
99  return f2(std::move(inititems));
100  }
101  catch (...) {
102  // rethrow either bad_alloc or exception_list
103  handle_local_exceptions<ExPolicy>::call(
104  std::current_exception());
105  }
106  }

112  FwdIter first, std::size_t count, F1 && f1, F2 && f2,
113  std::vector<std::size_t> const& chunk_sizes, Data && data)
114  {
115  HPX_ASSERT(hpx::util::size(data) >= hpx::util::size(chunk_sizes));
116 
117  typedef typename
118  hpx::util::decay<ExPolicy>::type::executor_parameters_type
119  parameters_type;
120 
121  typedef typename hpx::util::decay<Data>::type data_type;
122 
123  // inform parameter traits
124  scoped_executor_parameters<parameters_type> scoped_param(
125  policy.parameters());
126 
127  typename data_type::const_iterator data_it = hpx::util::begin(data);
128  typename std::vector<std::size_t>::const_iterator chunk_size_it =
129  hpx::util::begin(chunk_sizes);
130 
131  typedef typename hpx::util::tuple<
132  typename data_type::value_type, FwdIter, std::size_t
133  > tuple_type;
134 
135  std::vector<hpx::future<Result> > workitems;
136  std::list<std::exception_ptr> errors;
137 
138  try {
139  // schedule every chunk on a separate thread
140  std::vector<tuple_type> shape;
141  shape.reserve(chunk_sizes.size());
142 
143  while(count != 0)
144  {
145  std::size_t chunk = (std::min)(count, *chunk_size_it);
146  HPX_ASSERT(chunk != 0);
147 
148  shape.push_back(hpx::util::make_tuple(
149  *data_it, first, chunk));
150 
151  count -= chunk;
152  std::advance(first, chunk);
153 
154  ++data_it;
155  ++chunk_size_it;
156  }
157 
158  HPX_ASSERT(chunk_size_it == chunk_sizes.end());

161  policy.executor(),
162  partitioner_iteration<Result, F1>{std::forward<F1>(f1)},
163  std::move(shape));
164  }
165  catch (...) {
166  handle_local_exceptions<ExPolicy>::call(
167  std::current_exception(), errors);
168  }
169

171  hpx::wait_all(workitems);
172

174  // exceptional future

176

178  return f2(std::move(workitems));
179  }
180  catch (...) {

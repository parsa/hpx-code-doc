24  // this class looks like a vector, but can be initialized from a pointer and size,
25  // it is used by the verbs parcelport to pass an rdma memory chunk with received
26  // data into the decode parcel buffer routines.

173  inline void resize(std::size_t s) {
174  LOG_TRACE_MSG("pinned_memory_vector "
175  << "size " << hexuint32(m_size_)
176  << "array " << hexpointer(m_array_)
177  << "region " << hexpointer(m_region_)

179  << "resizing from " << m_size_ << " to " << s);
180 
181  if (m_region_) {
182  if (s > m_region_->get_size()) {
183  LOG_ERROR_MSG(
184  "Resizing from " << m_region_->get_size() << " to " << s);
185  throw std::runtime_error(
186  "pinned_memory_vector should never be resized once an "
187  "allocation has been assigned");
188  }
189  m_size_ = s;
190  }
191  else {
192  m_region_ = m_alloc_->allocate_region(s);
193  m_array_ = static_cast<T*>(m_region_->get_address());
194  m_size_ = s;

206  throw std::runtime_error(

208  "allocation has been assigned");
209  }
210  m_region_ = m_alloc_->allocate_region(s);
211  m_array_ = static_cast<T*>(m_region_->get_address());
212  }

296  template <
297  typename Action
298  , typename ReduceOp
299  , typename ...Ts
300  >
301  typename reduce_result<Action>::type
302  reduce_impl(
303  Action const & act
304  , std::vector<hpx::id_type> const & ids
305  , ReduceOp && reduce_op
306  , std::size_t global_idx
307  , Ts const&... vs
308  )
309  {
310  typedef
311  typename reduce_result<Action>::type
312  result_type;
313 
314  if(ids.empty()) return result_type();
315 
316  std::size_t const local_fanout = HPX_REDUCE_FANOUT;
317  std::size_t local_size = (std::min)(ids.size(), local_fanout);
318  std::size_t fanout = util::calculate_fanout(ids.size(), local_fanout);
319 
320  std::vector<hpx::future<result_type> > reduce_futures;
321  reduce_futures.reserve(local_size + (ids.size()/fanout) + 1);
322  for(std::size_t i = 0; i != local_size; ++i)
323  {
324  reduce_invoke(
325  act
326  , reduce_futures
327  , ids[i]
328  , global_idx + i
329  , vs...
330  );
331  }
332 
333  if(ids.size() > local_fanout)
334  {
335  std::size_t applied = local_fanout;
336  std::vector<hpx::id_type>::const_iterator it =
337  ids.begin() + local_fanout;
338 
339  typedef
340  typename detail::make_reduce_action<Action>::
341  template reduce_invoker<ReduceOp>::type
342  reduce_impl_action;
343 
344  while(it != ids.end())
345  {
346  HPX_ASSERT(ids.size() >= applied);
347 
348  std::size_t next_fan = (std::min)(fanout, ids.size() - applied);
349  std::vector<hpx::id_type> ids_next(it, it + next_fan);
350 
351  hpx::id_type id(ids_next[0]);
352  reduce_futures.push_back(
353  hpx::detail::async_colocated<reduce_impl_action>(
354  id
355  , act
356  , std::move(ids_next)
357  , reduce_op
358  , global_idx + applied
359  , vs...
360  )
361  );
362 
363  applied += next_fan;
364  it += next_fan;
365  }
366  }
367 
368  return hpx::when_all(reduce_futures).
369  then(perform_reduction<result_type, ReduceOp>(reduce_op)).
370  get();
371  }

422  {
423  // yields control if needed
424  wait(ec);
425  if (ec) return nullptr;
426 
427  // No locking is required. Once a future has been made ready, which
428  // is a postcondition of wait, either:
429  //
430  // - there is only one writer (future), or
431  // - there are multiple readers only (shared_future, lock hurts
432  // concurrency)
433 
434  if (state_ == empty) {
435  // the value has already been moved out of this future
436  HPX_THROWS_IF(ec, no_state,

522  // make sure continuation invocation does not recurse deeper than
523  // allowed
524  void handle_on_completed(completed_callback_type && on_completed)
525  {
526  // We need to run the completion on a new thread if we are on a
527  // non HPX thread.
528  bool recurse_asynchronously = hpx::threads::get_self_ptr() == nullptr;
529 #if defined(HPX_HAVE_THREADS_GET_STACK_POINTER)
530  recurse_asynchronously =
531  !this_thread::has_sufficient_stack_space();
532 #else
533  handle_continuation_recursion_count cnt;
534  recurse_asynchronously = recurse_asynchronously ||
535  cnt.count_ > HPX_CONTINUATION_MAX_RECURSION_DEPTH;
536 #endif
537  if (!recurse_asynchronously)
538  {
539  // directly execute continuation on this thread
540  std::exception_ptr ptr;
541  if (!run_on_completed(std::move(on_completed), ptr))
542  {
543  error_code ec(lightweight);
544  set_exception(hpx::detail::access_exception(ec));
545  }
546  }
547  else
548  {
549  // re-spawn continuation on a new thread
550  boost::intrusive_ptr<future_data> this_(this);
551 
552  error_code ec(lightweight);
553  std::exception_ptr ptr;
554  if (!run_on_completed_on_new_thread(
555  util::deferred_call(&future_data::run_on_completed,
556  std::move(this_), std::move(on_completed),
557  std::ref(ptr)),
558  ec))
559  {
560  // thread creation went wrong
561  if (ec) {
562  set_exception(hpx::detail::access_exception(ec));
563  return;
564  }
565

151  void coalescing_message_handler::put_parcel(
152  parcelset::locality const& dest, parcelset::parcel p,
153  write_handler_type f)
154  {
155  std::unique_lock<mutex_type> l(mtx_);
156  ++num_parcels_;
157 
158  // get time since last parcel
159  std::int64_t parcel_time = util::high_resolution_clock::now();
160  std::int64_t time_since_last_parcel = parcel_time - last_parcel_time_;
161  last_parcel_time_ = parcel_time;
162 
163  // collect data for time between parcels histogram
164  if (time_between_parcels_)
165  (*time_between_parcels_)(time_since_last_parcel);
166 
167  std::chrono::microseconds interval(interval_);
168 
169  // just send parcel if the coalescing was stopped or the buffer is
170  // empty and time since last parcel is larger than coalescing interval.
171  if (stopped_ ||
172  (buffer_.empty() &&
173  std::chrono::nanoseconds(time_since_last_parcel) > interval
174  ))
175  {
176  ++num_messages_;
177  l.unlock();
178 
179  // this instance should not buffer parcels anymore
180  pp_->put_parcel(dest, std::move(p), std::move(f));
181  return;
182  }
183 
184  detail::message_buffer::message_buffer_append_state s =
185  buffer_.append(dest, std::move(p), std::move(f));
186 
187  switch(s) {
188  case detail::message_buffer::first_message:
189  // start deadline timer to flush buffer
190  l.unlock();
191  timer_.start(interval);
192  break;
193 
194  case detail::message_buffer::normal:
195  // start deadline timer to flush buffer
196  l.unlock();
197  timer_.start(interval);
198  break;
199 
200  case detail::message_buffer::buffer_now_full:
201  flush_locked(l,
202  parcelset::policies::message_handler::flush_mode_buffer_full,
203  false, true);
204  break;
205 
206  default:
207  l.unlock();
208  HPX_THROW_EXCEPTION(bad_parameter,
209  "coalescing_message_handler::put_parcel",
210  "unexpected return value from message_buffer::append");
211  return;
212  }
213  }

245  bool coalescing_message_handler::flush_locked(
246  std::unique_lock<mutex_type>& l,
247  parcelset::policies::message_handler::flush_mode mode,
248  bool stop_buffering, bool cancel_timer)
249  {
250  HPX_ASSERT(l.owns_lock());
251 
252  // proceed with background work only if explicitly allowed
253  if (!allow_background_flush_ &&
254  mode == parcelset::policies::message_handler::flush_mode_background_work)
255  {
256  return false;
257  }
258 
259  if (!stopped_ && stop_buffering)
260  {
261  stopped_ = true;
262  {
263  hpx::util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
264  timer_.stop(); // interrupt timer
265  }
266  }
267  else if (cancel_timer)
268  {
269  hpx::util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
270  timer_.stop(); // interrupt timer
271  }
272 
273  if (buffer_.empty())
274  return false;
275 
276  detail::message_buffer buff (num_coalesced_parcels_);
277  std::swap(buff, buffer_);
278 
279  ++num_messages_;
280  l.unlock();
281 
282  HPX_ASSERT(nullptr != pp_);
283  buff(pp_); // 'invoke' the buffer
284 
285  return true;
286  }

289  std::int64_t
290  coalescing_message_handler::get_average_time_between_parcels(bool reset)
291  {
292  std::lock_guard<mutex_type> l(mtx_);
293  std::int64_t now = util::high_resolution_clock::now();
294  if (num_parcels_ == 0)
295  {
296  if (reset) started_at_ = now;
297  return 0;
298  }
299 
300  std::int64_t num_parcels = num_parcels_ - reset_time_num_parcels_;
301  if (num_parcels == 0)
302  {
303  if (reset) started_at_ = now;
304  return 0;
305  }
306 
307  HPX_ASSERT(now >= started_at_);
308  std::int64_t value = (now - started_at_) / num_parcels;
309 
310  if (reset)
311  {
312  started_at_ = now;
313  reset_time_num_parcels_ = num_parcels_;
314  }
315 
316  return value;
317  }

329  coalescing_message_handler::get_parcels_per_message_count(bool reset)

351  reset_num_parcels_per_message_messages_ = num_messages_;

84  using namespace H5;
85 
86  // Turn off auto-printing on failure.

88 
89  // Try to open the file.

91 
92  // Try to open the specified dataset.
93  DataSet dataset = file.openDataSet(name);
94  DataSpace dataspace = dataset.getSpace();
95 
96  // Verify number of dimensions
97  HPX_ASSERT(dataspace.getSimpleExtentNdims() == 1);
98 
99  // Get the size of each dimension in the dataspace.
100  hsize_t dims[1];
101  dataspace.getSimpleExtentDims(dims, nullptr);
102  if (end == std::size_t(-1))
103  end = dims[0];
104 
105  // Read the minimum and maximum values.
106  detail::read_values(dataset, dataspace, start, 1, &minval);
107  detail::read_values(dataset, dataspace, end - 1, 1, &maxval);
108 
109  // Read the delta value.
110  detail::read_values(dataset, dataspace, start + 1, 1, &delta);
111  delta -= minval;
112 
113  // Return size of dataset.
114  return dims[0];
115  }
116  catch (H5::Exception e) {
117  std::string msg = e.getDetailMsg().c_str();
118  HPX_THROW_EXCEPTION(hpx::no_success,
119  "sheneos::extract_data_range", msg);
120  }
121 
122  // This return statement keeps the compiler from whining.
123  return 0;
124  }

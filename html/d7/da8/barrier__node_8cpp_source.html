73  hpx::future<void> barrier_node::wait(bool async)
74  {
75  if (num_ < cut_off_)
76  {
77  if (rank_ != 0)
78  {
79  HPX_ASSERT(children_.size() == 1);
80  hpx::lcos::base_lco::set_event_action action;
81  return hpx::async(action, children_[0]);
82  }
83  else
84  {
85  if (async)
86  {
87  boost::intrusive_ptr<barrier_node> this_(this);
88  return hpx::async(&barrier_node::set_event, this_);
89  }
90  set_event();
91  return hpx::make_ready_future();
92  }
93  }
94 
95  future<void> result;
96 
97  if (rank_ == 0)
98  {
99  // The root process calls the gather action on its children
100  // once all those return, we know that everyone entered the
101  // barrier
102  std::vector<hpx::future<void> > futures;
103  futures.reserve(children_.size());
104  for(hpx::id_type& id : children_)
105  {
106  barrier_node::gather_action action;
107  futures.push_back(hpx::async(action, id));
108  }
109  result = hpx::when_all(futures);
110  }
111  else
112  {
113  // Non-root flags that it entered the barrier...
114  gather_promise_.set_value();
115 
116  // The broadcast_promise is set once the root knows that everyone
117  // entered the barrier
118  result = broadcast_promise_.get_future();
119  }
120 
121  if (async)
122  {
123  boost::intrusive_ptr<barrier_node> this_(this);
124  return do_wait(this_, std::move(result));
125  }
126  return do_wait(this, std::move(result));
127  }

178  hpx::future<void> barrier_node::gather()
179  {
180  // We recursively gather the information that everyone entered the
181  // barrier. The recursion is started from the root node.
182  HPX_ASSERT(rank_ != 0);
183  std::vector<hpx::future<void> > futures;
184  futures.reserve(children_.size());
185  for(hpx::id_type& id : children_)
186  {
187  barrier_node::gather_action action;
188  futures.push_back(hpx::async(action, id));
189  }
190 
191  // Once we know that all our children entered the barrier, we flag ourself
192  return hpx::when_all(futures).then(hpx::launch::sync,
193  [this](hpx::future<void> f)
194  {
195  // Trigger possible errors...
196  f.get();
197  return gather_promise_.get_future();
198  });
199  }

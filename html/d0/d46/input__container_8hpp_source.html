136  {

138 
139  if (chunks_ == nullptr ||

141  filter_)
142  {
143  // fall back to serialization_chunk-less archive
144  this->input_container::load_binary(address, count);
145  }
146  else {
147  HPX_ASSERT(current_chunk_ != std::size_t(-1));
148  HPX_ASSERT(get_chunk_type(current_chunk_) == chunk_type_pointer);
149 
150  if (get_chunk_size(current_chunk_) != count)
151  {
152  HPX_THROW_EXCEPTION(serialization_error
153  , "input_container::load_binary_chunk"
154  , "archive data bstream data chunk size mismatch");
155  return;
156  }
157 
158  // unfortunately we can't implement a zero copy policy on
159  // the receiving end
160  // as the memory was already allocated by the serialization code
161  std::memcpy(address, get_chunk_data(current_chunk_).pos_, count);
162  ++current_chunk_;

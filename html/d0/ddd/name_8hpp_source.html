290  // Note: we deliberately don't register this lock with the lock
291  // tracking to avoid false positives. We know that gid_types need
292  // to be locked while suspension.
293  void lock()
294  {
295  HPX_ITT_SYNC_PREPARE(this);
296 
297  for (std::size_t k = 0; !acquire_lock(); ++k)
298  {
299  lcos::local::spinlock::yield(k);
300  }
301 
302  util::register_lock(this);
303 
304  HPX_ITT_SYNC_ACQUIRED(this);
305  }
306 
307  bool try_lock()
308  {
309  HPX_ITT_SYNC_PREPARE(this);
310 
311  if (acquire_lock())
312  {
313  HPX_ITT_SYNC_ACQUIRED(this);
314  util::register_lock(this);
315  return true;
316  }
317 
318  HPX_ITT_SYNC_CANCEL(this);
319  return false;
320  }
321 
322  void unlock()
323  {
324  HPX_ITT_SYNC_RELEASING(this);
325 
326  relinquish_lock();
327  util::unregister_lock(this);
328 
329  HPX_ITT_SYNC_RELEASED(this);
330  }

349  // returns whether lock has been acquired
350  bool acquire_lock()
351  {
352  internal_mutex_type::scoped_lock l(this);
353  bool was_locked = (id_msb_ & is_locked_mask) ? true : false;
354  if (!was_locked)
355  {
356  id_msb_ |= is_locked_mask;
357  return true;
358  }
359  return false;
360  }
361 
362  void relinquish_lock()
363  {
364  util::ignore_lock(this);
365  internal_mutex_type::scoped_lock l(this);
366  util::reset_ignored(this);
367 
368  id_msb_ &= ~is_locked_mask;
369  }
370 
371  // this is used for assertions only, no need to acquire the lock
372  bool is_locked() const
373  {
374  return (id_msb_ & is_locked_mask) ? true : false;
375  }

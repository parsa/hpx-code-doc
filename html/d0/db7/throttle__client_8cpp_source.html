30 int hpx_main(variables_map& vm)
31 {
32  try {
33  std::cout << ( boost::format("prefix: %d")
34  % hpx::naming::get_locality_id_from_id(hpx::find_here()))
35  << std::endl;
36 
37  // Try to connect to existing throttle instance, create a new one if
38  // this fails.
39  char const* throttle_component_name = "/throttle/0";
40  hpx::naming::id_type gid =
41  hpx::agas::resolve_name(hpx::launch::sync, throttle_component_name);
42  throttle::throttle t;
43  if (!t.get_id()) {
44  std::vector<hpx::naming::id_type> localities =
45  hpx::find_remote_localities();
46 
47  // create throttle on the console, register the instance with AGAS
48  // and add an additional reference count to keep it alive
49  if (!localities.empty()) {
50  // use AGAS client to get the component type as we do not
51  // register any factories
52  t.create(localities[0]);
53  hpx::agas::register_name(hpx::launch::sync,
54  throttle_component_name, t.get_id());
55  }
56  else {
57  std::cerr << "Can't find throttle component." << std::endl;
58  }
59  }
60 
61  // handle commands
62  if (t.get_id()) {
63  if (vm.count("suspend")) {
64  t.suspend(vm["suspend"].as<int>());
65  }
66  else if (vm.count("resume")) {
67  t.resume(vm["resume"].as<int>());
68  }
69  else if (vm.count("release")) {
70  // unregister from AGAS, remove additional reference count which
71  // will allow for the throttle instance to be released
72  hpx::agas::unregister_name(hpx::launch::sync,
73  throttle_component_name);
74  }
75  }
76  }
77  catch (hpx::exception const& e) {
78  std::cerr << "throttle_client: caught exception: " << e.what() << std::endl;
79  }
80 
81  hpx::disconnect();
82  return 0;
83 }

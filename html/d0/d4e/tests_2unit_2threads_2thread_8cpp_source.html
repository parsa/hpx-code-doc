33  std::chrono::time_point<Clock, Duration> const& xt2)

40 template <typename Clock, typename Duration, typename Rep, typename Period>
41 inline bool in_range(
42  std::chrono::time_point<Clock, Duration> const& xt,
43  std::chrono::duration<Rep, Period> const& d)
44 {
45  std::chrono::time_point<Clock, Duration> const now = Clock::now();
46  std::chrono::time_point<Clock, Duration> const mint = now - d;
47  return time_cmp(xt, mint) >= 0 && time_cmp(xt, now) <= 0;
48 }

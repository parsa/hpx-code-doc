515  template <typename UserAllocator>
516  bool rdma_chunk_pool<UserAllocator>::release_memory()
517  {
521 
522  // ret is the return value: it will be set to true when we actually call
523  // UserAllocator::free(..)
524  bool ret = false;
525 
526  // This is a current & previous iterator pair over the memory block list
527  details::PODptr<size_type> ptr = list;
528  details::PODptr<size_type> prev;
529 
530  // This is a current & previous iterator pair over the free memory chunk list
531  // Note that "prev_free" in this case does NOT point to the previous memory
532  // chunk in the free list, but rather the last free memory chunk before the
533  // current block.
534  void * free_p = this->first;
535  void * prev_free_p = 0;
536 
537  const size_type partition_size = alloc_size();
538 
539  // Search through all the all the allocated memory blocks
540  while (ptr.valid())
541  {
542  // At this point:
543  // ptr points to a valid memory block
544  // free_p points to either:
545  // 0 if there are no more free chunks
546  // the first free chunk in this or some next memory block
547  // prev_free_p points to either:
548  // the last free chunk in some previous memory block
549  // 0 if there is no such free chunk
550  // prev is either:
551  // the PODptr whose next() is ptr
552  // !valid() if there is no such PODptr
553 
554  // If there are no more free memory chunks, then every remaining
555  // block is allocated out to its fullest capacity, and we can't
556  // release any more memory
557  if (free_p == 0)
558  break;
559 
560  // We have to check all the chunks. If they are *all* free (i.e., present
561  // in the free list), then we can free the block.
562  bool all_chunks_free = true;
563 
564  // Iterate 'i' through all chunks in the memory block
565  // if free starts in the memory block, be careful to keep it there
566  void * saved_free = free_p;
567  for (char * i = ptr.begin(); i != ptr.end(); i += partition_size)
568  {
569  // If this chunk is not free
570  if (i != free_p)
571  {
572  // We won't be able to free this block
573  all_chunks_free = false;
574 
575  // free_p might have travelled outside ptr
576  free_p = saved_free;
577  // Abort searching the chunks; we won't be able to free this
578  // block because a chunk is not free.
579  break;
580  }
581 
582  // We do not increment prev_free_p because we are in the same block
583  free_p = nextof(free_p);
584  }
585 
586  // post: if the memory block has any chunks, free_p points to one of them
587  // otherwise, our assertions above are still valid
588 
589  const details::PODptr<size_type> next = ptr.next();
590 
591  if (!all_chunks_free)
592  {
593  if (is_from(free_p, ptr.begin(), ptr.element_size()))
594  {
595  std::less<void *> lt;
596  void * const end = ptr.end();
597  do
598  {
599  prev_free_p = free_p;
600  free_p = nextof(free_p);
601  } while (free_p && lt(free_p, end));
602  }
603  // This invariant is now restored:
604  // free_p points to the first free chunk in some next memory block, or
605  // 0 if there is no such chunk.
606  // prev_free_p points to the last free chunk in this memory block.
607 
608  // We are just about to advance ptr. Maintain the invariant:
609  // prev is the PODptr whose next() is ptr, or !valid()
610  // if there is no such PODptr
611  prev = ptr;
612  }
613  else
614  {
615  // All chunks from this block are free
616 
617  // Remove block from list
618  if (prev.valid())
619  prev.next(next);
620  else
621  list = next;
622 
623  // Remove all entries in the free list from this block
624  if (prev_free_p != 0)
625  nextof(prev_free_p) = free_p;
626  else
627  this->first = free_p;
628 
629  // delete the storage, and release memory region
630  char *base_ptr = find_POD(ptr.begin()).begin();
631  verbs_memory_region_ptr region = region_map[base_ptr];
632  region_map.erase(base_ptr);
633  (UserAllocator::free)(region);
634 
635  ret = true;
636  }
637 
638  // Increment ptr
639  ptr = next;
640  }
641 
642  next_size = start_size;
643  return ret;
644  }

683  void *rdma_chunk_pool<UserAllocator>::malloc_need_resize()
684  {
685  size_type partition_size = alloc_size();
688  size_type POD_size = static_cast<size_type>(next_size * partition_size +
689  boost::integer::static_lcm<sizeof(size_type), sizeof(void *)>::value

693  if (ptr == 0)

696  {
697  next_size >>= 1;
698  partition_size = alloc_size();
699  POD_size = static_cast<size_type>(next_size * partition_size +
700  boost::integer::static_lcm<sizeof(size_type), sizeof(void *)>::value
701  + sizeof(size_type));
702  ptr = (UserAllocator::malloc)(pd_, POD_size);
703  }
704  if (ptr == 0)
705  return 0;
706  }
707  const details::PODptr<size_type> node(ptr->get_base_address(), POD_size);
708  region_map[ptr->get_base_address()] = ptr;
709 
710  BOOST_USING_STD_MIN();
711  if(!max_size)
712  next_size <<= 1;
713  else if( next_size*partition_size/requested_size < max_size)
714  next_size = min BOOST_PREVENT_MACRO_SUBSTITUTION(next_size << 1,

719 
720  // insert it into the list,
721  node.next(list);
722  list = node;
723

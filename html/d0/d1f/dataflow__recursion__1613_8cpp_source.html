21 // One way to force recursion is to make all futures depend on the next and
22 // make the last of the futures ready, triggering a chain of continuations.
23 void force_recursion_test1()
24 {
25  hpx::lcos::local::promise<void> first_promise;
26 
27  std::vector<hpx::shared_future<void> > results;
28  results.reserve(NUM_FUTURES+1);
29 
30  std::atomic<std::size_t> executed_dataflow(0);
31 
32  results.push_back(first_promise.get_future());
33  for (std::size_t i = 0; i != NUM_FUTURES; ++i)
34  {
35  results.push_back(
36  hpx::dataflow(
37  hpx::launch::sync,
38  [&](hpx::shared_future<void> &&)
39  {
40  ++executed_dataflow;
41  },
42  results.back()
43  )
44  );
45  }
46 
47  // make futures ready in backwards sequence
48  hpx::apply([&first_promise]() { first_promise.set_value(); });
49 
50  hpx::wait_all(results);
51  HPX_TEST_EQ(executed_dataflow.load(), NUM_FUTURES);
52 }

54 // Another way to force recursion is to make all futures ready from the start
55 // (continuations will be executed at the point where they are being attached
56 // to the future), and attach the next continuation from inside a continuation.
57 // This will trigger a chain of continuations as well.
58 void make_ready_continue(
59  std::size_t i,
60  std::vector<hpx::lcos::local::promise<void> >& promises,
61  std::vector<hpx::shared_future<void> > & futures,
62  std::atomic<std::size_t> & executed_continuations)
63 {
64  if (i >= NUM_FUTURES)
65  return;
66 
67  ++executed_continuations;
68  promises[i].set_value();
69  futures[i].then(
70  hpx::util::bind(
71  &make_ready_continue, i+1, std::ref(promises), std::ref(futures),
72  std::ref(executed_continuations)
73  )
74  );
75 }

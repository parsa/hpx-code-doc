39  // Request an initial resource allocation
40  std::size_t resource_manager::initial_allocation(
41  detail::manage_executor* proxy, error_code& ec)
42  {
43  if (nullptr == proxy) {
44  HPX_THROWS_IF(ec, bad_parameter,
45  "resource_manager::init_allocation",
46  "manage_executor pointer is a nullptr");
47  return std::size_t(-1);
48  }
49 
50  // ask executor for its policies
51  error_code ec1(lightweight);
52  std::size_t min_punits =
53  proxy->get_policy_element(detail::min_concurrency, ec1);
54  if (ec1) min_punits = 1;
55  std::size_t max_punits =
56  proxy->get_policy_element(detail::max_concurrency, ec1);
57  if (ec1) max_punits = get_os_thread_count();
58 
59  // lock the resource manager from this point on
60  std::lock_guard<mutex_type> l(mtx_);
61 
62  // allocate initial resources for the given executor
63  std::vector<std::pair<std::size_t, std::size_t> > cores =
64  allocate_virt_cores(proxy, min_punits, max_punits, ec);
65  if (ec) return std::size_t(-1);
66 
67  // attach the given proxy to this resource manager
68  std::size_t cookie = ++next_cookie_;
69  proxies_.insert(proxies_map_type::value_type(
70  cookie, proxy_data(proxy, std::move(cores))));
71 
72  if (&ec != &throws)
73  ec = make_success_code();
74  return cookie;
75  }

77  // Find 'desired' amount of processing units which have the given use count
78  // (use count is the number of schedulers associated with a given processing
79  // unit).
80  //
81  // the resource manager is locked while executing this function
82  std::size_t resource_manager::reserve_processing_units(
83  std::size_t use_count, std::size_t desired,
84  std::vector<punit_status>& available_punits)
85  {
86  std::size_t available = 0;
87  for (std::size_t i = 0; i != punits_.size(); ++i)
88  {
89  if (use_count == punits_[i].use_count_)

92  if (++available == desired)
93  break;
94  }
95  }
96  return available;

118  // Instructs a scheduler proxy to free up a fixed number of resources
119  // This is only a temporary release of resources.
120  // The use count on the global core is decremented and the scheduler
121  // proxy remembers the core as temporarily released
122  // release_cores_to_min - scheduler should release all cores above its minimum
123  // release_borrowed_cores - scheduler should release all its borrowed cores
124  bool resource_manager::release_scheduler_resources(
125  allocation_data_map_type::iterator it, std::size_t number_to_free,
126  std::vector<punit_status>& available_punits)
127  {
128  static_allocation_data& st = (*it).second;
129  std::size_t borrowed_cores;
130  std::size_t owned_cores;
131 
132  proxies_map_type::iterator iter;
133  iter = proxies_.find((*it).first);
134  // FIXME: handle iter == proxies.end()
135 
136  proxy_data& p = (*iter).second;
137 
138  if (number_to_free == release_borrowed_cores)
139  {
140  // We should only get one request to release borrowed cores - there
141  // should be no cores already stolen at this time.
142  HPX_ASSERT(st.num_cores_stolen_ == 0);
143 
144  number_to_free = borrowed_cores = st.num_borrowed_cores_;
145  }
146  else if (number_to_free == release_cores_to_min)
147  {
148  HPX_ASSERT(st.num_borrowed_cores_ == 0 ||
149  st.num_cores_stolen_ >= st.num_borrowed_cores_);
150  HPX_ASSERT(st.num_owned_cores_ >= st.min_proxy_cores_);
151 
152  // Number to stolen includes all borrowed cores, if any, and
153  // possibly some owned cores.
154  number_to_free = st.num_owned_cores_ - st.min_proxy_cores_ -
155  (st.num_cores_stolen_ - st.num_borrowed_cores_);
156  borrowed_cores = 0;
157  }
158  else
159  {
160  // If we're asked to release a specific number of cores, borrowed
161  // cores should already have been released, and we should
162  // not encounter any borrowed cores during our search.
163  HPX_ASSERT(st.num_cores_stolen_ == st.num_borrowed_cores_);
164  HPX_ASSERT(st.num_owned_cores_ >= st.min_proxy_cores_);
165  HPX_ASSERT(number_to_free > 0);
166 
167  if (number_to_free > st.num_owned_cores_ - st.min_proxy_cores_)
168  return false;
169 
170  borrowed_cores = 0;
171  }
172 
173  // owned_cores number of cores can be released only
174  HPX_ASSERT(number_to_free >= borrowed_cores &&
175  number_to_free <= release_cores_to_min);
176  owned_cores = number_to_free - borrowed_cores;
177 
178  if (number_to_free > 0)
179  {
180  for (coreids_type& coreids : p.core_ids_)
181  {
182  // continue if this core is not used by the current executor
183  if (coreids.second == std::size_t(-1))
184  continue;
185 
186  // check all cores
187  if (punits_[coreids.first].use_count_ > 0 || owned_cores > 0)
188  {
189  // The proxy remembers this processor as gone ..
190  // TODO
191 
192  // remove this core form the current scheduler
193  error_code ec(lightweight);
194  p.proxy_->remove_processing_unit(coreids.second, ec);
195 
196  // this virtual core is not running on the current
197  // scheduler anymore
198  coreids.second = std::size_t(-1);
199 
200  // increase number of stolen cores
201  ++(*it).second.num_cores_stolen_;
202 
203  // reduce use_count as core is released
204  --punits_[coreids.first].use_count_;
205 
206  if (punits_[coreids.first].use_count_ > 0)
207  {
208  --owned_cores;
209  }
210 
211  if (--number_to_free == 0)
212  {
213  return true;
214  }
215  }
216  }
217  }
218 
219  // The scheduler proxy does not have any cores available to free.
220  return false;
221  }

223  // Instructs existing schedulers to release cores. Then tries to reserve
224  // available cores for the new scheduler
225  //
226  // The parameter number_to_free can be one of the two special values:
227  // release_cores_to_min: used to release cores until they are at min, or
228  // release_borrowed_cores: only release borrowed cores.
229  //
230  std::size_t resource_manager::release_cores_on_existing_schedulers(
231  std::size_t request, std::size_t number_to_free,
232  std::vector<punit_status>& available_punits,
233  std::size_t new_allocation)
234  {
235  HPX_ASSERT(!proxies_static_allocation_data_.empty());
236 
237  // Ask previously allocated schedulers to release surplus cores,
238  // until either the request is satisfied, or we're out of
239  // schedulers.
240  bool released_cores = false;
241 
242  for (allocation_data_map_type::iterator it =
243  proxies_static_allocation_data_.begin();
244  it != proxies_static_allocation_data_.end(); ++it)
245  {
246  // skip new allocation
247  if ((*it).first == new_allocation)
248  continue;
249 
250  // check each scheduler
251  if (release_scheduler_resources(it, number_to_free, available_punits))
252  {
253  released_cores = true;
254  }
255  }
256 
257  std::size_t available = 0;
258  if (released_cores)
259  {
260  // reserve resources if available
261  available = reserve_processing_units(0, request, available_punits);
262  }
263 
264  return available;
265  }

267  // Tries to redistribute cores allocated to all schedulers proportional
268  // to each schedulers maximum processing units
269  // and reserve any freed cores for the new scheduler.
270  std::size_t resource_manager::redistribute_cores_among_all(
271  std::size_t reserved, std::size_t min_punits, std::size_t max_punits,
272  std::vector<punit_status>& available_punits,
273  std::size_t new_allocation)
274  {
275  std::size_t available = 0;
276  HPX_ASSERT(!proxies_static_allocation_data_.empty() &&
277  proxies_static_allocation_data_.find(new_allocation) !=
278  proxies_static_allocation_data_.end());
279 
280  // Try to proportionally allocate cores to all schedulers w/o
281  // over-subscription. The proportions used will be max_punits for each
282  // scheduler, except that no existing scheduler will be forced to
283  // increase the current allocation.
284  if (proxies_static_allocation_data_.size() > 1)
285  {
286  std::size_t total_minimum = min_punits;
287 
288  // sum of cores that have been previously reserved and cores that
289  // were reserved during this allocation attempt.
290  std::size_t num_schedulers = 1; // includes the current scheduler
291 
292  // total_allocated is the number of cores allocated to new
293  // scheduler so far plus the number of 'owned' cores allocated to
294  // all existing schedulers.
295  std::size_t total_allocated = reserved;
296 
297  // Let total_allocated be the number of cores we have allocated to
298  // the new scheduler so far, plus the number of 'owned' cores
299  // allocated to all existing schedulers.
300  // Let s0,...sn-1 be the currently allocated schedulers with 'desired'
301  // des[0],...,des[n-1] and 'minimum' min[0],...,min[n-1].
302  //
303  // The new scheduler requesting an allocation is sn with desired
304  // des[n] and minimum min[n].
305  for (auto & data : proxies_static_allocation_data_)
306  {
307  // skip new allocation
308  if (data.first == new_allocation)
309  continue;
310 
311  static_allocation_data st = data.second;
312 
313  // Only take into account existing schedulers that have > Min.
314  // We work with the number of 'owned' cores here instead of the
315  // number of 'allocated' cores (which includes borrowed cores).
316  // The borrowed cores should already have been released, but
317  // they are accounted for in the total allocated count, until
318  // the release is confirmed.
319  if (st.num_owned_cores_ > st.min_proxy_cores_)
320  {
321  ++num_schedulers;
322  total_minimum += st.min_proxy_cores_;
323  total_allocated += st.num_owned_cores_;
324  }
325  }
326 
327  if (num_schedulers > 1 && total_minimum <= total_allocated)
328  {
329  // We have found schedulers with cores greater than min.
330  // Moreover, the sum of all cores already allocated to
331  // existing schedulers can at least satisfy all minimums
332  // (including the min requirement of the current scheduler).
333  double total_desired = 0.0;
334 
335 #if defined(HPX_DEBUG)
336  // epsilon allows forgiveness of reasonable round-off errors
337  HPX_CONSTEXPR_OR_CONST double epsilon = 1e-7;
338 #endif
339  allocation_data_map_type scaled_static_allocation_data;
340  scaled_static_allocation_data[new_allocation] =
341  proxies_static_allocation_data_[new_allocation];
342 
343  for (auto const& data : proxies_static_allocation_data_)
344  {
345  // skip new allocation
346  if (data.first == new_allocation)
347  continue;
348 
349  static_allocation_data st = data.second;
350  if (st.num_owned_cores_ > st.min_proxy_cores_)
351  {
352  HPX_ASSERT(st.adjusted_desired_ == double(st.max_proxy_cores_));
353  scaled_static_allocation_data.insert(
354  allocation_data_map_type::value_type(data.first, st));
355 
356  total_desired += st.adjusted_desired_;
357  }
358  }
359 
360  double scaling = 0.0;
361  while (true)
362  {
363  // We're trying to pick a scaling factor r such that
364  // r * (Sum { desired[j] | j = 0,...,n }) == totalAllocated.
365  scaling = double(total_allocated) / total_desired;
366 
367  // multiply the scaling factor by each schedulers 'desired'.
368  for (auto & data : proxies_static_allocation_data_)
369  {
370  static_allocation_data& st = data.second;
371  st.scaled_allocation_ = st.adjusted_desired_ * scaling;
372  }
373 
374  // Convert the floating point scaled allocations into
375  // integer allocations, using the algorithm below.
376  roundup_scaled_allocations(scaled_static_allocation_data,
377  total_allocated);
378 
379  bool re_calculate = false;
380  for (auto & data : scaled_static_allocation_data)
381  {
382  // skip new allocation
383  if (data.first == new_allocation)
384  continue;
385 
386  // Keep recursing until previous allocations do not
387  // increase (excluding the current scheduler).
388  static_allocation_data& st = data.second;
389  if (st.allocation_ > st.num_owned_cores_)
390  {
391  double modifier = static_cast<double>(
392  st.num_owned_cores_) / st.allocation_;
393 
394  // Reduce adjusted_desired by multiplying it with
395  // 'modifier', to try to bias allocation to the
396  // original size or less.
397  total_desired -= st.adjusted_desired_ * (1.0 - modifier);
398  st.adjusted_desired_ *= modifier;
399 
400  re_calculate = true;
401  }
402  }
403 
404  if (re_calculate)
405  {
406 #if defined(HPX_DEBUG)
407  double sum_desired = 0.0;
408  for (auto const& data : scaled_static_allocation_data)
409  {
410  sum_desired += data.second.adjusted_desired_;
411  }
412  HPX_ASSERT(
413  total_desired >= sum_desired - epsilon &&
414  total_desired <= sum_desired + epsilon);
415 #endif
416  continue;
417  }
418 
419  for (auto & data : proxies_static_allocation_data_)
420  {
421  // Keep recursing until all allocations are no greater
422  // than desired (including the current scheduler).
423  static_allocation_data& st = data.second;
424  if (st.allocation_ > st.max_proxy_cores_)
425  {
426  double modifier = st.max_proxy_cores_ /
427  static_cast<double>(st.allocation_);
428 
429  // Reduce adjusted_desired by multiplying with it
430  // 'modifier', to try to bias allocation to desired
431  // or less.
432  total_desired -= st.adjusted_desired_ * (1.0 - modifier);
433  st.adjusted_desired_ *= modifier;
434 
435  re_calculate = true;
436  }
437  }
438 
439  if (re_calculate)
440  {
441 #if defined(HPX_DEBUG)
442  double sum_desired = 0.0;
443  for (auto const& data : scaled_static_allocation_data)
444  {
445  sum_desired += data.second.adjusted_desired_;
446  }
447  HPX_ASSERT(
448  total_desired >= sum_desired - epsilon &&
449  total_desired <= sum_desired + epsilon);
450 #endif
451  continue;
452  }
453 
454  for (auto & data : proxies_static_allocation_data_)
455  {
456  // Keep recursing until all allocations are at least
457  // minimum (including the current scheduler).
458  static_allocation_data& st = data.second;
459  if (st.min_proxy_cores_ > st.allocation_)
460  {
461  double new_desired = st.min_proxy_cores_ / scaling;
462 
463  // Bias desired to get allocation closer to min.
464  total_desired += new_desired - st.adjusted_desired_;
465  st.adjusted_desired_ = new_desired;
466 
467  re_calculate = true;
468  }
469  }
470 
471  if (re_calculate)
472  {
473 #if defined(HPX_DEBUG)
474  double sum_desired = 0.0;
475  for (auto const& data : scaled_static_allocation_data)
476  {
477  sum_desired += data.second.adjusted_desired_;
478  }
479  HPX_ASSERT(
480  total_desired >= sum_desired - epsilon &&
481  total_desired <= sum_desired + epsilon);
482 #endif
483  continue;
484  }
485 
486 #if defined(HPX_DEBUG)
487  hpx::error_code ec(lightweight);
488  for (auto const& data : scaled_static_allocation_data)
489  {
490  // skip new allocation
491  if (data.first == new_allocation)
492  continue;
493 
494  HPX_ASSERT(
495  data.second.proxy_->get_policy_element(
496  hpx::threads::detail::min_concurrency, ec) <=
497  data.second.allocation_);
498  }
499 
500  static_allocation_data const& st =
501  scaled_static_allocation_data[new_allocation];
502  HPX_ASSERT(st.min_proxy_cores_ <= st.allocation_);
503 #endif
504  break;
505  } // end of while(true)
506 
507  //
508  static_allocation_data const& st =
509  scaled_static_allocation_data[new_allocation];
510  if (st.allocation_ > total_allocated)
511  {
512  allocation_data_map_type::iterator end =
513  scaled_static_allocation_data.end();
514 
515  for (allocation_data_map_type::iterator it =
516  scaled_static_allocation_data.begin();
517  it != end; ++it)
518  {
519  // skip new allocation
520  if ((*it).first == new_allocation)
521  continue;
522 
523  static_allocation_data const& st = (*it).second;
524  std::size_t reduce_by = st.num_owned_cores_ - st.allocation_;
525  if (reduce_by > 0)
526  {
527  release_scheduler_resources(
528  it, reduce_by, available_punits);
529  }
530  }
531 
532  // Reserve out of the cores we just freed.
533  available = reserve_processing_units(0,
534  st.allocation_ - reserved, available_punits);
535  }
536 
537  scaled_static_allocation_data.clear();
538  }
539  }
540  return available;
541  }

558  void resource_manager::roundup_scaled_allocations(
559  allocation_data_map_type& scaled_static_allocation_data,
560  std::size_t total_allocated)
561  {
562  HPX_ASSERT(!scaled_static_allocation_data.empty());
563 
564  // epsilon allows forgiveness of reasonable round-off errors
565  double epsilon = 1e-07;
566  double fraction = 0.0;
567 
568  for (auto & data : scaled_static_allocation_data)
569  {
570  static_allocation_data& st = data.second;
571  st.allocation_ = static_cast<std::size_t>(st.scaled_allocation_);
572  st.scaled_allocation_ -= st.allocation_;
573  }
574 
575  // Sort by scaled_allocation
576  typedef std::pair<double, allocation_data_map_type::iterator> item;
577  typedef std::vector<item> helper_type;
578 
579  helper_type d;
580  d.reserve(scaled_static_allocation_data.size());
581 
582  for (allocation_data_map_type::iterator it =
583  scaled_static_allocation_data.begin();
584  it != scaled_static_allocation_data.end(); ++it)
585  {
586  d.push_back(std::make_pair(it->second.scaled_allocation_, it));
587  }
588 
589  std::sort(
590  d.begin(), d.end(),
591  [](item const& rhs, item const& lhs)
592  {
593  return rhs.first < lhs.first;
594  });
595 
596  // Round up those with the largest truncation, stealing the fraction
597  // from those with the least.
598  helper_type::reverse_iterator rit = d.rbegin();
599  for (helper_type::iterator it = d.begin(); it != d.end(); ++it)
600  {
601  while (fraction > epsilon && rit != d.rend())
602  {
603  if (rit->second->second.scaled_allocation_ > epsilon)
604  {
605  do
606  {
607  fraction -= rit->second->second.scaled_allocation_;
608  rit->second->second.scaled_allocation_ = 0.0;
609  ++rit;
610  }
611  while (fraction > epsilon && rit != d.rend());
612  HPX_ASSERT(it <= rit.base());
613  }
614  else
615  {
616  ++rit;
617  HPX_ASSERT(it <= rit.base());
618  }
619  }
620 
621  if (it <= rit.base())
622  {
623  if (it->second->second.scaled_allocation_ > epsilon)
624  {
625  fraction += (1.0 - it->second->second.scaled_allocation_);
626  it->second->second.scaled_allocation_ = 0.0;
627  it->second->second.allocation_ += 1;
628  }
629  }
630  else
631  {
632  break;
633  }
634  }
635 
636  HPX_ASSERT(fraction <= epsilon && fraction >= -epsilon);
637 
638 #if defined(HPX_DEBUG)
639  std::size_t sum_allocation = 0;
640  for (auto & data : d)
641  {
642  sum_allocation += data.second->second.allocation_;
643  }
644  HPX_ASSERT(sum_allocation == total_allocated);
645 #endif
646  }

648  // store all information required for static allocation in
649  // proxies_static_allocation_data_
650  // also store new proxy scheduler
651  std::size_t resource_manager::preprocess_static_allocation(
652  std::size_t min_punits, std::size_t max_punits)
653  {
654  proxies_map_type::iterator it;
655  proxies_static_allocation_data_.clear();
656 
657  for (auto & proxy : proxies_)
658  {
659  proxy_data& p = proxy.second;
660  static_allocation_data st;
661  st.proxy_ = p.proxy_;
662 
663  // ask executor for its policies
664  error_code ec1(lightweight);
665  st.min_proxy_cores_ =
666  p.proxy_->get_policy_element(detail::min_concurrency, ec1);
667  if (ec1) st.min_proxy_cores_ = 1;
668 
669  st.max_proxy_cores_ =
670  p.proxy_->get_policy_element(detail::max_concurrency, ec1);
671  if (ec1) st.max_proxy_cores_ = get_os_thread_count();
672 
673  st.num_borrowed_cores_ = 0;
674  st.num_owned_cores_ = 0;
675  st.adjusted_desired_ = static_cast<double>(st.max_proxy_cores_);
676  st.num_cores_stolen_ = 0;
677 
678  for (coreids_type coreids : p.core_ids_)
679  {
680  if (punits_[coreids.first].use_count_ > 1)
681  st.num_borrowed_cores_++;
682  if (punits_[coreids.first].use_count_ == 1)
683  st.num_owned_cores_++;
684  }
685 
686  proxies_static_allocation_data_.insert(
687  std::make_pair(proxy.first , st));
688  }
689 
690  std::size_t cookie = next_cookie_ + 1;
691 
692  static_allocation_data st;
693  st.min_proxy_cores_ = min_punits;
694  st.max_proxy_cores_ = max_punits;

705  resource_manager::allocate_virt_cores(

707  std::size_t max_punits, error_code& ec)

711  // array of available processing units
712  std::vector<punit_status> available_punits(
713  punits_.size(), punit_status::unassigned);
714 
715  // find all available processing units with zero use count
716  std::size_t reserved = reserve_processing_units(0, max_punits,
717  available_punits);
718  if (reserved < max_punits)
719  {
720  // insufficient available cores found, try to share
721  // processing units
722  std::size_t cookie =
723  preprocess_static_allocation(min_punits, max_punits);
724 
725  std::size_t num_to_release = max_punits - reserved;
726  reserved += release_cores_on_existing_schedulers(
727  num_to_release, num_to_release, available_punits, cookie);
728 
729  if (reserved < max_punits)
730  {
731  reserved += redistribute_cores_among_all(
732  reserved, min_punits, max_punits, available_punits,
733  cookie);
734 
735  if (reserved < min_punits)
736  {
737  num_to_release = min_punits - reserved;
738  reserved += release_cores_on_existing_schedulers(
739  num_to_release, num_to_release, available_punits, cookie);
740 
741  if (reserved < min_punits)
742  {
743  reserved += reserve_at_higher_use_count(
744  min_punits - reserved, available_punits);
745  }
746  }
747  }
748 
749  HPX_ASSERT(reserved >= min_punits);
750  }
751 
752  // processing units found, inform scheduler
753  std::size_t punit = 0;
754  for (std::size_t i = 0; i != available_punits.size(); ++i)
755  {
756  // allocate only as many processing units as necessary
757  if (available_punits[i] == punit_status::reserved)
758  {
759  proxy->add_processing_unit(punit, i, ec);
760  if (ec) break;
761 
762  core_ids.push_back(std::make_pair(i, punit));
763 
764  // update use count for reserved processing units
765  ++punits_[i].use_count_;
766 
767  // no need to reserve more cores than requested
768  if (++punit == max_punits)
769  break;
770  }
771  }
772  HPX_ASSERT(punit <= max_punits);
773

775  {

777  for (std::size_t j = 0; j != available_punits.size(); ++j)
778  {
779  if (available_punits[j] == punit_status::reserved && j != max_punits)
780  {
781  proxy->remove_processing_unit(j, ec);
782  --punits_[j].use_count_;
783  }
784  }
785  return std::vector<coreids_type>();
786  }
787 
788  if (&ec != &throws)
789  ec = make_success_code();
790  return core_ids;

20 def make_component(raw, type):
21  """
22  Transliterate characters from an element returned by platform.uname() to
23  match the regex ``^[a-z0-9\-.]+$``. Returns ``"unknown-%s" % type`` if
24  the transliterated string doesn't match the regex pattern.
25  """
26  from string import lower
27 
28  comp = compile(r'\s|_').sub('-', lower(raw))
29 
30  if compile(r'^[a-z0-9\-.]+$').match(comp):
31  return comp
32  else:
33  return "unknown-%s" % type

35 def make_compiler_component(driver):

38  suite and version. Returns ``"unknown-compiler"`` if the compiler cannot be
39  identified.
40  """

42

44

46  from sys import getwindowsversion
47  # on windows, running cl.exe with no args returns what we want
48  windows = 1

52  proc = process("%s --version" % driver)
53

56 
57  if (windows):
58  compiler = compile(r'Version ([0-9.]+)').match(raw)
59

62  if compile(r'^[a-z0-9\-.]+$').match(compiler):
63  return compiler
64  else:
65  return "msvc"
66 
67  # handle GNU GCC and Intel
68  compiler = compile(r'^(icc|icpc|gcc|g[+][+])[^ ]* [(][^)]+[)] ([0-9.]+)').match(raw)
69 
70  if (compiler):
71  unescaped = compiler.expand(r'\1-\2')
72  compiler = compile(r'[+]').sub("x", unescaped)
73 
74  if compile(r'^[a-z0-9\-.]+$').match(compiler):
75  return compiler
76  else:
77  unescaped = compile(r'^(icc|icpc|gcc|g[+][+])').match(raw).expand(r'\1')
78  return compile(r'[+]').sub("x", unescaped)

392  void good()
393  {
394  hpx::error_code ec;
395  for (data_good* t = data; !t->fullname_.empty(); ++t)
396  {
397  using namespace hpx::performance_counters;
398 
399  std::string fullname;
400  HPX_TEST(status_valid_data == get_counter_name(t->path_, fullname, ec));
401  HPX_TEST(ec.value() == hpx::success);
402  HPX_TEST(fullname == t->fullname_);
403 
404  std::string type_name;
405  HPX_TEST(status_valid_data ==
406  get_counter_type_name(t->path_, type_name, ec));
407  HPX_TEST(ec.value() == hpx::success);
408  HPX_TEST(type_name == t->typename_);
409 
410  counter_path_elements p;
411 
412  HPX_TEST(status_valid_data ==
413  get_counter_path_elements(t->fullname_, p, ec));
414  HPX_TEST(ec.value() == hpx::success);
415  HPX_TEST(p.objectname_ == t->path_.objectname_);
416  HPX_TEST(p.parentinstancename_ == t->path_.parentinstancename_);
417  HPX_TEST(p.instancename_ == t->path_.instancename_);
418  HPX_TEST(p.subinstancename_ == t->path_.subinstancename_);
419  HPX_TEST(p.instanceindex_ == t->path_.instanceindex_);
420  HPX_TEST(p.subinstanceindex_ == t->path_.subinstanceindex_);
421  HPX_TEST(p.countername_ == t->path_.countername_);
422 
423  fullname.erase();
424  HPX_TEST(status_valid_data == get_counter_name(p, fullname, ec));
425  HPX_TEST(ec.value() == hpx::success);
426  HPX_TEST(fullname == t->fullname_);
427 
428  counter_type_path_elements tp1, tp2;
429 
430  HPX_TEST(status_valid_data ==
431  get_counter_type_path_elements(t->fullname_, tp1, ec));
432  HPX_TEST(ec.value() == hpx::success);
433  HPX_TEST(tp1.objectname_ == t->path_.objectname_);
434  HPX_TEST(tp1.countername_ == t->path_.countername_);
435 
436  type_name.erase();
437  HPX_TEST(status_valid_data == get_counter_type_name(tp1, type_name, ec));
438  HPX_TEST(ec.value() == hpx::success);
439  HPX_TEST(type_name == t->typename_);
440 
441  type_name.erase();
442  HPX_TEST(status_valid_data ==

449  HPX_TEST(type_name == t->typename_ + '@' + t->path_.parameters_);

451

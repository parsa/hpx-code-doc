26 // The purpose of this example is to execute a HPX-thread printing "Hello world"
27 // once on each OS-thread on each of the connected localities.
28 //
29 // The function hello_world_foreman_action is executed once on each locality.
30 // It schedules a HPX-thread (encapsulating hello_world_worker) once for each
31 // OS-thread on that locality. The code make sure that the HPX-thread gets
32 // really executed by the requested OS-thread. While the HPX-thread is scheduled
33 // to run on a particular OS-thread, we may have to retry as the HPX-thread may
34 // end up being 'stolen' by another OS-thread.
35 
37 //[hello_world_worker
38 std::size_t hello_world_worker(std::size_t desired)
39 {
40  // Returns the OS-thread number of the worker that is running this
41  // HPX-thread.
42  std::size_t current = hpx::get_worker_thread_num();
43  if (current == desired)
44  {
45  // The HPX-thread has been run on the desired OS-thread.
46  char const* msg = "hello world from OS-thread %1% on locality %2%";
47 
48  hpx::cout << (boost::format(msg) % desired % hpx::get_locality_id())
49  << std::endl << hpx::flush;
50 
51  return desired;
52  }
53 
54  // This HPX-thread has been run by the wrong OS-thread, make the foreman
55  // try again by rescheduling it.
56  return std::size_t(-1);
57 }

66 //[hello_world_foreman
67 void hello_world_foreman()
68 {
69  // Get the number of worker OS-threads in use by this locality.
70  std::size_t const os_threads = hpx::get_os_thread_count();
71 
72  // Find the global name of the current locality.
73  hpx::naming::id_type const here = hpx::find_here();
74 
75  // Populate a set with the OS-thread numbers of all OS-threads on this
76  // locality. When the hello world message has been printed on a particular
77  // OS-thread, we will remove it from the set.
78  std::set<std::size_t> attendance;
79  for (std::size_t os_thread = 0; os_thread < os_threads; ++os_thread)
80  attendance.insert(os_thread);
81 
82  // As long as there are still elements in the set, we must keep scheduling
83  // HPX-threads. Because HPX features work-stealing task schedulers, we have
84  // no way of enforcing which worker OS-thread will actually execute
85  // each HPX-thread.
86  while (!attendance.empty())
87  {
88  // Each iteration, we create a task for each element in the set of
89  // OS-threads that have not said "Hello world". Each of these tasks
90  // is encapsulated in a future.
91  std::vector<hpx::lcos::future<std::size_t> > futures;
92  futures.reserve(attendance.size());
93 
94  for (std::size_t worker : attendance)
95  {
96  // Asynchronously start a new task. The task is encapsulated in a
97  // future, which we can query to determine if the task has
98  // completed.
99  typedef hello_world_worker_action action_type;
100  futures.push_back(hpx::async<action_type>(here, worker));
101  }
102 
103  // Wait for all of the futures to finish. The callback version of the
104  // hpx::lcos::wait_each function takes two arguments: a vector of futures,
105  // and a binary callback. The callback takes two arguments; the first
106  // is the index of the future in the vector, and the second is the
107  // return value of the future. hpx::lcos::wait_each doesn't return until
108  // all the futures in the vector have returned.
109  hpx::lcos::local::spinlock mtx;
110  hpx::lcos::wait_each(
111  hpx::util::unwrapping([&](std::size_t t) {
112  if (std::size_t(-1) != t)
113  {
114  std::lock_guard<hpx::lcos::local::spinlock> lk(mtx);
115  attendance.erase(t);
116  }
117  }),
118  futures);
119  }
120 }

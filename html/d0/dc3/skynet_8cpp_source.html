6 // This benchmark provides an equivalent for the benchmarks published at
7 // https://github.com/atemerev/skynet. It is called the Skynet 1M concurrency
8 // micro benchmark.
9 //
10 // It creates an actor (goroutine, whatever), which spawns 10 new actors, each
11 // of them spawns 10 more actors, etc. until one million actors are created on
12 // the final level. Then, each of them returns back its ordinal number (from 0
13 // to 999999), which are summed on the previous level and sent back upstream,
14 // until reaching the root actor. (The answer should be 499999500000).
15 
16 // This code implements two versions of the skynet micro benchmark: a 'normal'
17 // and a futurized one.

27 std::int64_t skynet(std::int64_t num, std::int64_t size, std::int64_t div)
28 {
29  if (size != 1)
30  {
31  size /= div;
32 
33  std::vector<hpx::future<std::int64_t> > results;
34  results.reserve(div);
35 
36  for (std::int64_t i = 0; i != div; ++i)
37  {
38  std::int64_t sub_num = num + i * size;
39  results.push_back(hpx::async(skynet, sub_num, size, div));
40  }
41 
42  hpx::wait_all(results);
43 
44  std::int64_t sum = 0;
45  for (auto & f : results)
46  sum += f.get();
47  return sum;
48  }
49  return num;
50 }

67  template <typename Scheduler>
68  this_thread_executor<Scheduler>::~this_thread_executor()
69  {
70  // if we're still starting up, give this executor a chance of executing
71  // its tasks
72  while (scheduler_.get_state(0) < state_running)
73  this_thread::suspend();
74 
75  // Inform the resource manager that this executor is about to be
76  // destroyed. This will cause it to invoke remove_processing_unit below
77  // for each of the currently allocated virtual cores.
78  resource_manager& rm = resource_manager::get();
79  rm.stop_executor(cookie_);
80 
81  // wait for executor to finish executing
82  shutdown_sem_.wait(1);
83 
84  // detach this executor from resource manager
85  rm.detach(cookie_); // this releases proxy (manage_thread_executor)
86 
87 #if defined(HPX_DEBUG)
88  // all resources should have been stopped at this point (or have never
89  // been initialized)
90  hpx::state s = scheduler_.get_state(0);
91  HPX_ASSERT(s == state_initialized || s == state_stopped);
92 
93  // all scheduled tasks should have completed executing
94  HPX_ASSERT(tasks_completed_ == tasks_scheduled_);
95 #endif
96  }

157  // than time abs_time. This call never blocks, and may violate
158  // bounds on the executor's queue size.
159  template <typename Scheduler>
160  void this_thread_executor<Scheduler>::add_at(
161  util::steady_clock::time_point const& abs_time,
162  closure_type && f, util::thread_description const& desc,
163  threads::thread_stacksize stacksize, error_code& ec)
164  {
165  HPX_ASSERT(std::size_t(-1) != thread_num_);
166 
167  // if the scheduler was stopped, we need to restart it
168  state expected = state_stopped;
169  scheduler_.get_state(0).compare_exchange_strong(expected, state_starting);
170 
171  // create a new suspended thread
172  thread_init_data data(util::bind(
173  util::one_shot(&this_thread_executor::thread_function_nullary),
174  this, std::move(f)), desc);
175  data.stacksize = threads::get_stack_size(stacksize);
176 
177  threads::thread_id_type id = threads::invalid_thread_id;
178  threads::detail::create_thread( //-V601
179  &scheduler_, data, id, suspended, true, ec);

181  HPX_ASSERT(invalid_thread_id != id); // would throw otherwise
182 
183  // update statistics
184  ++tasks_scheduled_;
185 
186  // now schedule new thread for execution
187  threads::detail::set_thread_state_timed(scheduler_, abs_time, id, ec);
188  if (ec) {
189  --tasks_scheduled_;
190  return;
191  }
192 
193  // execute scheduler directly, if necessary
194  run();
195 
196  if (&ec != &throws)
197  ec = make_success_code();

250  template <typename Scheduler>
251  void this_thread_executor<Scheduler>::suspend_back_into_calling_context()
252  {
253  // Give invoking context a chance to catch up with its tasks, but only
254  // if this scheduler is currently in running state (this scheduler is
255  // always in stopping state as it has to exit as early as possible).
256  std::atomic<hpx::state>& state = scheduler_.get_state(0);
257  hpx::state expected = state_stopping;
258  if (state.compare_exchange_strong(expected, state_suspended))
259  {
260  {
261  threads::detail::reset_tss_helper reset_on_exit(
262  parent_thread_num_);
263  on_self_reset on_exit(self_);
264 
265  this_thread::suspend();
266  }
267 
268  // reset state to running if current state is still suspended
269  expected = state_suspended;
270  state.compare_exchange_strong(expected, state_stopping);
271  }
272  else
273  {
274  HPX_ASSERT(expected != state_suspended);
275  }
276  }

299  // execute all work
300  template <typename Scheduler>
301  void this_thread_executor<Scheduler>::run()
302  {
303  // We want to exit this scheduling loop as early as possible, thus
304  // we use 'state_stopping' instead of 'state_running'.
305 
306  // Set the state to 'state_stopping' only if it's still in
307  // 'state_starting' state, otherwise our destructor is currently being
308  // executed, which means we need to still execute all threads.
309  std::atomic<hpx::state>& state = scheduler_.get_state(0);
310  hpx::state expected = state_starting;
311  if (state.compare_exchange_strong(expected, state_stopping))
312  {
313 // {
314 // std::unique_lock<mutex_type> l(mtx_);
315 // resource::get_partitioner().get_affinity_data().add_punit(
316 // 0, thread_num_);
317 // scheduler_.on_start_thread(0);
318 // }
319 
320  self_ = threads::get_self_ptr();
321 
322  this_thread_on_run_exit on_exit(shutdown_sem_, self_);
323 
324  // manage the thread num
325  HPX_ASSERT(orig_thread_num_ != std::size_t(-1));
326 
327  threads::detail::reset_tss_helper reset_on_exit(orig_thread_num_);
328  parent_thread_num_ = reset_on_exit.previous_thread_num();
329 
330  // FIXME: turn these values into performance counters
331  std::int64_t executed_threads = 0, executed_thread_phases = 0;
332  std::uint64_t overall_times = 0, thread_times = 0;
333  std::int64_t idle_loop_count = 0, busy_loop_count = 0;
334  std::uint8_t task_active = 0;
335 
336  threads::detail::scheduling_counters counters(
337  executed_threads, executed_thread_phases,
338  overall_times, thread_times, idle_loop_count, busy_loop_count,
339  task_active);
340 
341  threads::detail::scheduling_callbacks callbacks(
342  threads::detail::scheduling_callbacks::callback_type(),
343  util::bind( //-V107
344  &this_thread_executor::suspend_back_into_calling_context,
345  this));
346 
347  scheduler_.set_scheduler_mode(policies::fast_idle_mode);
348  threads::detail::scheduling_loop(0, scheduler_, counters, callbacks);
349 
350  // the scheduling_loop is allowed to exit only if no more HPX
351  // threads exist
352  HPX_ASSERT(!scheduler_.get_thread_count(
353  unknown, thread_priority_default, 0) ||
354  state >= state_terminating);
355  }
356  }

291  template <typename DistPolicy, typename ...Ts>
292  static type call(DistPolicy const& policy, std::size_t count,
293  Ts&&... vs)
294  {
295  typedef typename DistPolicy::bulk_locality_result
296  bulk_locality_result;
297 
298  hpx::future<std::vector<bulk_locality_result> > f =
299  policy.template bulk_create<Component>(count,
300  std::forward<Ts>(vs)...);
301 
302  return f.then(launch::sync,
303  [count](hpx::future<std::vector<bulk_locality_result> >
304  && f) -> std::vector<hpx::id_type>
305  {
306  std::vector<hpx::id_type> result;
307  result.reserve(count);
308 
309  for (bulk_locality_result& r: f.get())
310  {
311  std::move(r.second.begin(), r.second.end(),
312  std::back_inserter(result));
313  }
314  return result;
315  });
316  }
317  };
318  }

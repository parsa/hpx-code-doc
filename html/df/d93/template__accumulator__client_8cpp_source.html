26 template <typename T>
27 void run_template_accumulator(char const* type)
28 {
29  typedef typename examples::server::template_accumulator<T> accumulator_type;
30  typedef typename accumulator_type::argument_type argument_type;
31 
32  // Find the localities connected to this application.
33  std::vector<hpx::id_type> localities = hpx::find_all_localities();
34 
35  // Create an accumulator component either on this locality (if the
36  // example is executed on one locality only) or on any of the remote
37  // localities (otherwise).
38  examples::template_accumulator<T> accu(
39  hpx::components::new_<accumulator_type>(localities.back()));
40 
41  // Print out the available commands.
42  std::cout << std::endl
43  << "Running accumulator accepting argument type: "
44  << type << std::endl;
45  std::cout << help << std::endl;
46 
47  // Enter the interpreter loop.
48  std::string line;
49  while (std::getline(std::cin, line))
50  {
51  boost::algorithm::trim(line);
52 
53  std::vector<std::string> cmd;
54  boost::algorithm::split(cmd, line,
55  boost::algorithm::is_any_of(" \t\n"),
56  boost::algorithm::token_compress_on);
57 
58  if (!cmd.empty() && !cmd[0].empty())
59  {
60  // try to interpret the entered command
61  if (cmd[0] == "reset") {
62  accu.reset();
63  }
64  else if (cmd[0] == "add") {
65  if (cmd.size() == 2) {
66  try {
67  double val = boost::lexical_cast<double>(cmd[1]);
68  accu.add(argument_type(val));
69  }
70  catch (boost::bad_lexical_cast const&) {
71  std::cout << "error: invalid argument for add: '"
72  << cmd[1] << "'" << std::endl;
73  }
74  }
75  else {
76  std::cout << "error: invalid command '"
77  << line << "'" << std::endl
78  << help << std::endl;
79  }
80  }
81  else if (cmd[0] == "query") {
82  std::cout << accu.query() << std::endl;
83  }
84  else if (cmd[0] == "help") {
85  std::cout << help << std::endl;
86  }
87  else if (cmd[0] == "quit") {
88  break;
89  }
90  else {
91  std::cout << "error: invalid command '"
92  << line << "'" << std::endl
93  << help << std::endl;
94  }
95  }
96 
97  std:: cout << "> ";
98  }
99 }

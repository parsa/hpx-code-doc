23  const char * uint128::toString (unsigned int radix) const throw () {
24  if (!*this) return "0";
25  if (radix < 2 || radix > 37) return "(invalid radix)";
26 
27  static char sz [256];
28  std::memset (sz, 0, 256);

30  uint128 r;
31  uint128 ii = *this;
32  int i = 255;

34  while (!!ii && i) {
35  ii = ii.div (radix, r);

42  uint128::uint128 (const char * sz) throw ()
43  : lo (0u), hi (0u) {
44 
45  if (!sz) return;
46  if (!sz [0]) return;
47 
48  unsigned int radix = 10;
49  std::size_t i = 0;
50  bool minus = false;
51 
52  if (sz [i] == '-') {
53  ++i;
54  minus = true;
55  };
56 
57  if (sz [i] == '0') {
58  radix = 8;
59  ++i;
60  if (sz [i] == 'x') {
61  radix = 16;
62  ++i;
63  };
64  };
65 
66  std::size_t len = strlen (sz);
67  for (; i < len; ++i) {
68  unsigned int n = 0;
69  if (sz [i] >= '0' && sz [i] <= (std::min)(('0' + (int)radix), (int)'9'))
70  n = sz [i] - '0';
71  else if (sz [i] >= 'a' && sz [i] <= 'a' + (int) radix - 10)
72  n = sz [i] - 'a' + 10;
73  else if (sz [i] >= 'A' && sz [i] <= 'A' + (int) radix - 10)
74  n = sz [i] - 'A' + 10;
75  else
76  break;
77 
78  (*this) *= radix;
79  (*this) += n;
80  };
81 
82  if (minus)
83  *this = 0u - *this;
84 
85  return;
86  };

198  if (!ds)
199  return 1u / (unsigned int) ds.lo;
200 
201  uint128 dd = *this;
202 
203  // only remainder
204  if (ds > dd) {
205  remainder = *this;
206  return std::uint64_t(0ull);
207  };

209  uint128 r = std::uint64_t(0ull);
210  uint128 q = std::uint64_t(0ull);
211  // while (dd >= ds) { dd -= ds; q += 1; }; // extreme slow version
212 
213  unsigned int b = 127;
214  while (r < ds) {
215  r <<= 1;
216  if (dd.bit (b--))
217  r.lo |= 1;
218  };
219  ++b;
220 
221  while (true)
222  if (r < ds) {
223  if (!(b--)) break;

225  r <<= 1;
226  if (dd.bit (b))
227  r.lo |= 1;

229  } else {

232  };

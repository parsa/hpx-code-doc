42 // Use a special allocator for the partition data to remove a major contention
43 // point - the constant allocation and deallocation of the data arrays.
44 template <typename T>
45 struct partition_allocator
46 {
47 private:
48  typedef hpx::lcos::local::spinlock mutex_type;
49 
50 public:
51  partition_allocator(std::size_t max_size = std::size_t(-1))
52  : max_size_(max_size)
53  {}
54 
55  ~partition_allocator()
56  {
57  std::lock_guard<mutex_type> l(mtx_);
58  while (!heap_.empty())
59  {
60  T* p = heap_.top();
61  heap_.pop();
62  delete [] p;
63  }
64  }
65 
66  T* allocate(std::size_t n)
67  {
68  std::lock_guard<mutex_type> l(mtx_);
69  if (heap_.empty())
70  {
71  return new T[n];
72  }
73 
74  T* next = heap_.top();
75  heap_.pop();
76  return next;
77  }
78 
79  void deallocate(T* p)
80  {
81  std::lock_guard<mutex_type> l(mtx_);
82  if (max_size_ == static_cast<std::size_t>(-1) || heap_.size() < max_size_)
83  heap_.push(p);
84  else
85  delete [] p;
86  }
87 
88 private:
89  mutex_type mtx_;
90  std::size_t max_size_;
91  std::stack<T*> heap_;
92 };

214 // This is the server side representation of the data. We expose this as a HPX
215 // component which allows for it to be created and accessed remotely through
216 // a global address (hpx::id_type).
217 struct partition_server
218  : hpx::components::component_base<partition_server>
219 {
220  enum partition_type
221  {
222  left_partition, middle_partition, right_partition
223  };
224 
225  // construct new instances
226  partition_server() {}
227 
228  partition_server(partition_data const& data)
229  : data_(data)
230  {}
231 
232  partition_server(std::size_t size, double initial_value)
233  : data_(size, initial_value)
234  {}
235 
236  // Access data. The parameter specifies what part of the data should be
237  // accessed. As long as the result is used locally, no data is copied,
238  // however as soon as the result is requested from another locality only
239  // the minimally required amount of data will go over the wire.
240  partition_data get_data(partition_type t) const
241  {
242  switch (t)
243  {
244  case left_partition:
245  return partition_data(data_, data_.size()-1);
246 
247  case middle_partition:
248  break;
249 
250  case right_partition:
251  return partition_data(data_, 0);
252 
253  default:
254  HPX_ASSERT(false);
255  break;
256  }
257  return data_;
258  }
259 
260  // Every member function which has to be invoked remotely needs to be
261  // wrapped into a component action. The macro below defines a new type
262  // 'get_data_action' which represents the (possibly remote) member function
263  // partition::get_data().
264  HPX_DEFINE_COMPONENT_DIRECT_ACTION(partition_server, get_data, get_data_action);
265 
266 private:
267  partition_data data_;
268 };

479 // The partitioned operator, it invokes the heat operator above on all elements
480 // of a partition.
481 partition stepper_server::heat_part(partition const& left,
482  partition const& middle, partition const& right)
483 {
484  hpx::shared_future<partition_data> middle_data =
485  middle.get_data(partition_server::middle_partition);
486 
487  hpx::future<partition_data> next_middle = middle_data.then(
488  hpx::util::unwrapping(
489  [middle](partition_data const& m) -> partition_data
490  {
491  HPX_UNUSED(middle);
492 
493  // All local operations are performed once the middle data of
494  // the previous time step becomes available.
495  std::size_t size = m.size();
496  partition_data next(size);
497  for (std::size_t i = 1; i != size-1; ++i)
498  next[i] = heat(m[i-1], m[i], m[i+1]);
499  return next;
500  }
501  )
502  );
503 
504  return hpx::dataflow(
505  hpx::launch::async,
506  hpx::util::unwrapping(
507  [left, middle, right](partition_data next, partition_data const& l,
508  partition_data const& m, partition_data const& r) -> partition
509  {
510  HPX_UNUSED(left);
511  HPX_UNUSED(right);
512 
513  // Calculate the missing boundary elements once the
514  // corresponding data has become available.
515  std::size_t size = m.size();
516  next[0] = heat(l[size-1], m[0], m[1]);
517  next[size-1] = heat(m[size-2], m[size-1], r[0]);
518 
519  // The new partition_data will be allocated on the same locality
520  // as 'middle'.
521  return partition(middle.get_id(), next);
522  }
523  ),
524  std::move(next_middle),
525  left.get_data(partition_server::left_partition),
526  middle_data,
527  right.get_data(partition_server::right_partition)
528  );
529 }

532 // This is the implementation of the time step loop
533 //
534 // Do all the work on 'np' partitions, 'nx' data points each, for 'nt'
535 // time steps, limit depth of dependency tree to 'nd'.
536 stepper_server::space stepper_server::do_work(std::size_t local_np,
537  std::size_t nx, std::size_t nt, std::uint64_t nd)
538 {
539  // U[t][i] is the state of position i at time t.
540  for (space& s: U_)
541  s.resize(local_np);
542 
543  // Initial conditions: f(0, i) = i
544  hpx::id_type here = hpx::find_here();
545  for (std::size_t i = 0; i != local_np; ++i)
546  U_[0][i] = partition(here, nx, double(i));
547 
548  // send initial values to neighbors
549  if (nt != 0)
550  {
551  send_left(0, U_[0][0]);
552  send_right(0, U_[0][local_np-1]);
553  }
554 
555  // limit depth of dependency tree
556  hpx::lcos::local::sliding_semaphore sem(nd);
557 
558  for (std::size_t t = 0; t != nt; ++t)
559  {
560  space const& current = U_[t % 2];
561  space& next = U_[(t + 1) % 2];
562 
563  // handle special case (one partition per locality) in a special way
564  if (local_np == 1)
565  {
566  next[0] = hpx::dataflow(
567  hpx::launch::async, &stepper_server::heat_part,
568  receive_left(t), current[0], receive_right(t)
569  );
570 
571  // send to left and right if not last time step
572  if (t != nt-1)
573  {
574  send_left(t + 1, next[0]);
575  send_right(t + 1, next[0]);
576  }
577  }
578  else
579  {
580  next[0] = hpx::dataflow(
581  hpx::launch::async, &stepper_server::heat_part,
582  receive_left(t), current[0], current[1]
583  );
584 
585  // send to left if not last time step
586  if (t != nt-1) send_left(t + 1, next[0]);
587 
588  for (std::size_t i = 1; i != local_np-1; ++i)
589  {
590  next[i] = hpx::dataflow(
591  hpx::launch::async, &stepper_server::heat_part,
592  current[i-1], current[i], current[i+1]
593  );
594  }
595 
596  next[local_np-1] = hpx::dataflow(
597  hpx::launch::async, &stepper_server::heat_part,
598  current[local_np-2], current[local_np-1], receive_right(t)
599  );
600 
601  // send to right if not last time step
602  if (t != nt-1) send_right(t + 1, next[local_np-1]);
603  }
604 
605  // every nd time steps, attach additional continuation which will
606  // trigger the semaphore once computation has reached this point
607  if ((t % nd) == 0)
608  {
609  next[0].then(
610  [&sem, t](partition &&)
611  {
612  // inform semaphore about new lower limit
613  sem.signal(t);
614  });
615  }
616 
617  // suspend if the tree has become too deep, the continuation above
618  // will resume this thread once the computation has caught up
619  sem.wait(t);
620  }
621 
622  return U_[nt % 2];
623 }

628 void do_all_work(std::uint64_t nt, std::uint64_t nx, std::uint64_t np,
629  std::uint64_t nd)
630 {
631  std::vector<hpx::id_type> localities = hpx::find_all_localities();
632  std::size_t nl = localities.size(); // Number of localities
633 
634  if (np < nl)
635  {
636  std::cout << "The number of partitions should not be smaller than "
637  "the number of localities" << std::endl;
638  return;
639  }
640 
641  // Create the local stepper instance, register it
642  stepper step(nl);
643 
644  // Measure execution time.
645  std::uint64_t t = hpx::util::high_resolution_clock::now();
646 
647  // Perform all work and wait for it to finish
648  hpx::future<stepper_server::space> result = step.do_work(np/nl, nx, nt, nd);
649 
650  // Gather results from all localities
651  if (0 == hpx::get_locality_id())
652  {
653  std::uint64_t const num_worker_threads = hpx::get_num_worker_threads();
654 
655  hpx::future<std::vector<stepper_server::space> > overall_result =
656  hpx::lcos::gather_here(gather_basename, std::move(result), nl);
657 
658  std::vector<stepper_server::space> solution = overall_result.get();
659  for (std::size_t i = 0; i != nl; ++i)
660  {
661  stepper_server::space const& s = solution[i];
662  for (std::size_t i = 0; i != s.size(); ++i)
663  {
664  s[i].get_data(partition_server::middle_partition).wait();
665  }
666  }
667 
668  std::uint64_t elapsed = hpx::util::high_resolution_clock::now() - t;
669 
670  // Print the solution at time-step 'nt'.
671  if (print_results)
672  {
673  for (std::size_t i = 0; i != nl; ++i)
674  {
675  stepper_server::space const& s = solution[i];
676  for (std::size_t j = 0; j != s.size(); ++j)
677  {
678  std::cout << "U[" << i*(s.size()) + j << "] = "
679  << s[j].get_data(partition_server::middle_partition).get()
680  << std::endl;
681  }
682  }
683  }
684 
685  print_time_results(std::uint32_t(nl), num_worker_threads, elapsed,
686  nx, np, nt, header);
687  }
688  else
689  {
690  hpx::lcos::gather_there(gather_basename, std::move(result)).wait();
691  }
692 }

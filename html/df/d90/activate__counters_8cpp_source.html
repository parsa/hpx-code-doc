64  void activate_counters::find_counters()
65  {
66  std::vector<std::string> names;
67  std::swap(names, names_);
68 
69  names_.reserve(names.size());
70  if (ids_.empty())
71  {
72  using util::placeholders::_1;
73  using util::placeholders::_2;
74 
75  performance_counters::discover_counter_func func(
76  util::bind(&activate_counters::find_counter, this, _1, _2));
77 
78  ids_.reserve(names.size());
79  uoms_.reserve(names.size());
80  for (std::string& name : names)
81  {
82  // do INI expansion on counter name
83  util::expand(name);
84 
85  // find matching counter type
86  performance_counters::discover_counter_type(name, func,
87  performance_counters::discover_counters_full);
88  }
89  }
90 
91  HPX_ASSERT(ids_.size() == names_.size());
92  HPX_ASSERT(ids_.size() == uoms_.size());
93  }

96  void activate_counters::start_counters(error_code& ec)
97  {
98  using performance_counters::stubs::performance_counter;
99 
100  // add counter prefix, if necessary
101  for (std::string& name : names_)
102  performance_counters::ensure_counter_prefix(name);
103 
104  find_counters();
105 
106  // Query the performance counters.
107  std::vector<future<bool> > started;
108 
109  started.reserve(ids_.size());
110  for (std::size_t i = 0; i != ids_.size(); ++i)
111  started.push_back(performance_counter::start(launch::async, ids_[i]));
112 
113  // wait for all counters to be started
114  wait_all(started);
115 
116  for (future<bool>& f : started)
117  {
118  if (f.has_exception())
119  {
120  if (&ec == &hpx::throws)
121  {
122  f.get();
123  }
124  else
125  {
126  ec = make_error_code(f.get_exception_ptr());
127  }
128  return;
129  }

48 int hpx_main(boost::program_options::variables_map &vm)
49 {
50  namespace process = hpx::components::process;
51  namespace fs = boost::filesystem;
52 
53  // find where the HPX core libraries are located
54  fs::path base_dir = hpx::util::find_prefix();
55  base_dir /= "bin";
56 
57  fs::path exe = base_dir / "launched_process_test" HPX_EXECUTABLE_EXTENSION;
58 
59  std::string launch_target;
60  if (vm.count("launch"))
61  {
62  launch_target = vm["launch"].as<std::string>();
63  std::cout << "using launch: " << launch_target << std::endl;
64  exe = launch_target;
65  }
66  else
67  {
68  std::cout << "using launch (default): " << exe << std::endl;
69  }
70 
71  // set up command line for launched executable
72  std::vector<std::string> args;
73  args.push_back(exe.string());
74  args.push_back("--exit_code=42");
75  args.push_back("--component=test_server");
76  args.push_back("--set_message=accessed");
77  args.push_back("--hpx:ignore-batch-env");
78 
79  // set up environment for launched executable
80  std::vector<std::string> env = get_environment(); // current environment
81 
82  // Pass along the console parcelport address
83  env.push_back("HPX_AGAS_SERVER_ADDRESS=" +
84  hpx::get_config_entry("hpx.agas.address", HPX_INITIAL_IP_ADDRESS));
85  env.push_back("HPX_AGAS_SERVER_PORT=" +
86  hpx::get_config_entry("hpx.agas.port",
87  std::to_string(HPX_INITIAL_IP_PORT)));
88 
89  // Pass along the parcelport address which should be used by the launched
90  // executable
91 
92  // The launched executable will run on the same host as this test
93  int port = 42; // each launched HPX locality needs to be assigned a
94  // unique port
95 
96  env.push_back("HPX_PARCEL_SERVER_ADDRESS=" +
97  hpx::get_config_entry("hpx.agas.address", HPX_INITIAL_IP_ADDRESS));
98  env.push_back("HPX_PARCEL_SERVER_PORT=" +
99  std::to_string(HPX_CONNECTING_IP_PORT - port));
100 
101  // Instruct new locality to connect back on startup using the given name.
102  env.push_back("HPX_ON_STARTUP_WAIT_ON_LATCH=launch_process");
103 
104  // launch test executable
105  process::child c = process::execute(
106  hpx::find_here(),
107  process::run_exe(exe.string()),
108  process::set_args(args),
109  process::set_env(env),
110  process::start_in_dir(base_dir.string()),
111  process::throw_on_error(),
112  process::wait_on_latch("launch_process") // same as above!
113  );
114 
115  {
116  // now create an instance of the test_server component
117  hpx::components::client<launch_process::test_server> t =
118  hpx::new_<launch_process::test_server>(hpx::find_here());
119 
120  hpx::future<std::string> f =
121  hpx::async(launch_process_get_message_action(), t);
122  HPX_TEST_EQ(f.get(), std::string("initialized"));
123 
124  // register the component instance with AGAS
125  t.register_as("test_server"); // same as --component=<> above
126 
127  // wait for the HPX locality to be up and running
128  c.wait();
129  HPX_TEST(c);
130 
131  // the launched executable should have connected back as a new locality
132  HPX_TEST_EQ(hpx::find_all_localities().size(), std::size_t(2));
133 
134  // wait for it to exit, we know it returns 42 (see --exit_code=<> above)
135  int exit_code = c.wait_for_exit(hpx::launch::sync);
136  HPX_TEST_EQ(exit_code, 42);
137 
138  // make sure the launched process has set the message in the component
139  // this should be the same as --set_message=<> above
140  f = hpx::async(launch_process_get_message_action(), t);
141  HPX_TEST_EQ(f.get(), std::string("accessed"));
142 
143  } // release the component
144 
145  // the new locality should have disconnected now
146  HPX_TEST_EQ(hpx::find_all_localities().size(), std::size_t(1));
147 
148  return hpx::finalize();
149 }

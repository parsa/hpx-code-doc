233 // visit_predicate (determines which directories are visited) --------------//

235  typedef bool(*pred_type)(const path&);
236 
237  bool visit_predicate( const path & pth )
238  {
239  string local( boost::inspect::relative_to( pth, search_root_path() ) );
240  string leaf( pth.leaf().string() );
241  if (leaf[0] == '.') // ignore hidden by convention directories such as
242  return false; // .htaccess, .git, .svn, .bzr, .DS_Store, etc.
243 
244  return
245  // so we can inspect a GIT checkout
246  leaf != ".git"
247  // don't look at binaries
248  && leaf != "bin"
249  && leaf != "bin.v2"
250  && leaf != "build"
251  // no point in checking doxygen xml output
252  && local.find("doc/xml") != 0
253  && local.find("doc\\xml") != 0
254  // ignore if tag file present
255  && !boost::filesystem::exists(pth / hpx_no_inspect)
256  ;
257  }

261  string library_from_content( const string & content )
262  {
263  const string unknown_library ( "unknown" );
264  const string lib_root ( "www.boost.org/libs/" );
265  string::size_type pos( content.find( lib_root ) );
266 
267  string lib = unknown_library;
268 
269  if ( pos != string::npos ) {
270 
271  pos += lib_root.length();
272 
273  const char delims[] = " " // space and...
274  "/\n\r\t";
275 
276  string::size_type n = content.find_first_of( string(delims), pos );
277  if (n != string::npos)
278  lib = string(content, pos, n - pos);
279 
280  }
281 
282  return lib;
283  }

287  bool find_signature( const path & file_path,
288  const boost::inspect::string_set & signatures )
289  {
290  string name( file_path.leaf().string() );
291  if ( signatures.find( name ) == signatures.end() )
292  {
293  string::size_type pos( name.rfind( '.' ) );
294  if ( pos == string::npos
295  || signatures.find( name.substr( pos ) )
296  == signatures.end() ) return false;
297  }
298  return true;
299  }

317  void check( const string & lib,
318  const path & pth, const string & content, const inspector_list & insp_list )
319  {
320  // invoke each inspector
321  for ( inspector_list::const_iterator itr = insp_list.begin();
322  itr != insp_list.end(); ++itr )

325  if ( find_signature( pth, itr->inspector->signatures() ) )

328  }

334  template< class DirectoryIterator >
335  void visit_all( const string & lib,
336  const path & dir_path, const inspector_list & insps )
337  {
338  static DirectoryIterator end_itr;
339  ++directory_count;
340 
341  for ( DirectoryIterator itr( dir_path ); itr != end_itr; ++itr )
342  {
343  if ( fs::is_directory( *itr ) )
344  {
345  if ( visit_predicate( *itr ) )
346  {
347  string cur_lib( boost::inspect::impute_library( *itr ) );
348  check( cur_lib, *itr, "", insps );
349  visit_all<DirectoryIterator>( cur_lib, *itr, insps );
350  }
351  }
352  else if (itr->path().leaf().string()[0] != '.') // ignore if hidden
353  {
354  ++file_count;
355  string content;
356  load_content( *itr, content );
357  if (content.find(hpx_no_inspect) == string::npos)
358  check( lib.empty() ? library_from_content( content ) : lib,
359  *itr, content, insps );
360  }
361  }
362  }

98  template <typename Handler, typename ParcelPostprocess>
99  void async_write(Handler && handler,
100  ParcelPostprocess && parcel_postprocess)
101  {
102 #if defined(HPX_TRACK_STATE_OF_OUTGOING_TCP_CONNECTION)
103  HPX_ASSERT(state_ == state_send_pending);
104 #endif
105  HPX_ASSERT(!buffer_.data_.empty());
106  HPX_ASSERT(!handler_);
107  HPX_ASSERT(!postprocess_handler_);
108 
109  handler_ = std::forward<Handler>(handler);
110  postprocess_handler_ = std::forward<ParcelPostprocess>(parcel_postprocess);
111  HPX_ASSERT(handler_);
112  HPX_ASSERT(postprocess_handler_);
113 
114 #if defined(HPX_TRACK_STATE_OF_OUTGOING_TCP_CONNECTION)
115  state_ = state_async_write;
116 #endif
117  buffer_.data_point_.time_ = timer_.elapsed_nanoseconds();
119 
120  // Write the serialized data to the socket. We use "gather-write"
121  // to send both the header and the data in a single write operation.
122  std::vector<boost::asio::const_buffer> buffers;
123  buffers.push_back(boost::asio::buffer(&buffer_.size_,
124  sizeof(buffer_.size_)));
125  buffers.push_back(boost::asio::buffer(&buffer_.data_size_,
126  sizeof(buffer_.data_size_)));
127 
128  // add chunk description
129  buffers.push_back(boost::asio::buffer(&buffer_.num_chunks_,
130  sizeof(buffer_.num_chunks_)));
131 
132  std::vector<parcel_buffer_type::transmission_chunk_type>& chunks =
133  buffer_.transmission_chunks_;
134  if (!chunks.empty()) {
135  buffers.push_back(
136  boost::asio::buffer(chunks.data(), chunks.size() *
137  sizeof(parcel_buffer_type::transmission_chunk_type)));
138 
139  // add main buffer holding data which was serialized normally
140  buffers.push_back(boost::asio::buffer(buffer_.data_));
141 
142  // now add chunks themselves, those hold zero-copy serialized chunks
143  for (serialization::serialization_chunk& c : buffer_.chunks_)
144  {
145  if (c.type_ == serialization::chunk_type_pointer)
146  buffers.push_back(boost::asio::buffer(c.data_.cpos_, c.size_));
147  }
148  }
149  else {
150  // add main buffer holding data which was serialized normally
151  buffers.push_back(boost::asio::buffer(buffer_.data_));
152  }
153 
154  // this additional wrapping of the handler into a bind object is
155  // needed to keep this parcelport_connection object alive for the whole
156  // write operation
157  void (sender::*f)(boost::system::error_code const&, std::size_t)
158  = &sender::handle_write;

160  using util::placeholders::_1;
161  using util::placeholders::_2;
162  boost::asio::async_write(socket_, buffers,
163  util::bind(f, shared_from_this(), _1, _2));

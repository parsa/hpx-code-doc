338  // Iterators whose dereference operators reference the same value for
339  // all iterators into the same sequence (like many input iterators)
340  // need help with their postfix ++: the referenced value must be read
341  // and stored away before the increment occurs so that *a++ yields the
342  // originally referenced element and not the next one.
343  template <typename Iterator>
344  class postfix_increment_proxy
345  {
346  typedef
347  typename std::iterator_traits<Iterator>::value_type value_type;
348 
349  public:
350  HPX_HOST_DEVICE explicit postfix_increment_proxy(Iterator const& x)
351  : stored_value(*x)
352  {
353  }
354 
355  // Returning a mutable reference allows nonsense like (*r++).mutate(),
356  // but it imposes fewer assumptions about the behavior of the
357  // value_type. In particular, recall that (*r).mutate() is legal if
358  // operator* returns by value.
359  HPX_HOST_DEVICE value_type& operator*() const
360  {
361  return this->stored_value;
362  }
363 
364  private:
365  mutable typename std::remove_const<value_type>::type stored_value;
366  };

369  // -- we also need to store a copy of the old iterator so that it can
370  // be written into.
371  template <typename Iterator>
372  class writable_postfix_increment_proxy

387  // and the conversion operator below is used to ensure readability.
388  HPX_HOST_DEVICE

435  // Because the C++98 input iterator requirements say that *r++ has
436  // type T (value_type), implementations of some standard algorithms
437  // like lexicographical_compare may use constructions like:
438  //
439  // *r++ < *s++
440  //
441  // If *r++ returns a proxy (as required if r is writable but not
442  // multi-pass), this sort of expression will fail unless the proxy
443  // supports the operator<. Since there are any number of such
444  // operations, we're not going to try to support them. Therefore,
445  // even if r++ returns a proxy, *r++ will only return a proxy if *r
446  // also returns a proxy.

448  typename Enable = void>
449  struct postfix_increment_result
450  {
451  typedef Iterator type;
452  };

20 int hpx_main(boost::program_options::variables_map& vm)
21 {
22  // extract the counter name (pattern) from command line
23  std::string name;
24  if (vm.count("counter"))
25  {
26  name = vm["counter"].as<std::string>();
27  }
28  else
29  {
30  name = "/threads{locality#*/worker-thread#*}/count/cumulative";
31  }
32 
33  // create the performance counter set from the given pattern (name)
34  using namespace hpx::performance_counters;
35  performance_counter_set set(name);
36 
37  // retrieve the counter information for all attached counters
38  std::vector<counter_info> infos = set.get_counter_infos();
39 
40  // retrieve the current counter values
41  std::vector<double> values = set.get_values<double>(hpx::launch::sync);
42 
43  // print the values for all 'raw' (non-histogram) counters
44  for (std::size_t i = 0, j = 0; i != infos.size(); ++i)
45  {
46  if (infos[i].type_ != counter_raw)
47  continue;
48 
49  if (infos[i].unit_of_measure_.empty())
50  {
51  hpx::cout << infos[i].fullname_ << ":" << values[j] << std::endl;
52  }
53  else
54  {
55  hpx::cout
56  << infos[i].fullname_ << ":" << values[j] << "["
57  << infos[i].unit_of_measure_ << "]" << std::endl;
58  }
59  ++j;
60  }
61 
62  return hpx::finalize();
63 }

6 // The purpose of this example is to demonstrate how it is possible to
7 // schedule an IO task onto one of the IO-threads in HPX (which are OS-threads)
8 // and how to synchronize the result of this IO task with a waiting HPX thread.

20 void do_async_io(char const* string_to_write,
21  std::shared_ptr<hpx::lcos::local::promise<int> > p)

30 // This function will be executed by an HPX thread
31 hpx::lcos::future<int> async_io(char const* string_to_write)
32 {
33  std::shared_ptr<hpx::lcos::local::promise<int> > p =
34  std::make_shared<hpx::lcos::local::promise<int> >();
35 
36  // Get a reference to one of the IO specific HPX io_service objects ...
37  hpx::util::io_service_pool* pool =
38  hpx::get_runtime().get_thread_pool("io_pool");
39 
40  // ... and schedule the handler to run on one of its OS-threads.
41  pool->get_io_service().post(
42  hpx::util::bind(&do_async_io, string_to_write, p));
43 
44  return p->get_future();
45 }

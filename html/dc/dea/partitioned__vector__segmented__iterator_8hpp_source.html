135  template <typename T, typename Data>
136  struct local_vector_value_proxy
137  {
138  local_vector_value_proxy(local_vector_iterator<T, Data> const& it)
139  : it_(it)
140  {}
141 
142  operator T() const
143  {
144  if (!it_.get_data())
145  {
146  return it_.get_partition().get_value(launch::sync,
147  it_.get_local_index());
148  }
149  return *(it_.get_data()->begin() + it_.get_local_index());
150  }
151 
152  template <typename T_>
153  local_vector_value_proxy& operator=(T_ && value)
154  {
155  if (!it_.get_data())
156  {
157  it_.get_partition().set_value(launch::sync,
158  it_.get_local_index(), std::forward<T_>(value));
159  }
160  else
161  {
162  *(it_.get_data()->begin() + it_.get_local_index()) =
163  std::forward<T_>(value);
164  }
165  return *this;
166  }
167 
168  local_vector_iterator<T, Data> const& it_;
169  };

601  template <typename T, typename Data, typename BaseIter>
602  class local_segment_vector_iterator
603  : public hpx::util::iterator_adaptor<
604  local_segment_vector_iterator<T, Data, BaseIter>, BaseIter,
605  Data, std::forward_iterator_tag
606  >
607  {
608  private:
609  typedef hpx::util::iterator_adaptor<
610  local_segment_vector_iterator<T, Data, BaseIter>, BaseIter,
611  Data, std::forward_iterator_tag
612  > base_type;
613  typedef detail::is_requested_locality<BaseIter> predicate;
614 
615  public:
616  local_segment_vector_iterator()
617  {}
618 
619  local_segment_vector_iterator(BaseIter const& end)
620  : base_type(end), predicate_(), end_(end)
621  {}
622 
623  local_segment_vector_iterator(
624  BaseIter const& it, BaseIter const& end,
625  std::uint32_t locality_id)
626  : base_type(it), predicate_(locality_id), end_(end)
627  {
628  satisfy_predicate();
629  }
630 
631  bool is_at_end() const
632  {
633  return !data_ || this->base() == end_;
634  }
635 
636  // increment until predicate is not satisfied any more
637  void unsatisfy_predicate()
638  {
639  while (this->base() != end_ && predicate_(*this->base()))
640  ++(this->base_reference());
641 
642  if (this->base() != end_)
643  data_ = this->base()->local_data_;
644  else
645  data_.reset();
646  }
647 
648  private:
649  friend class hpx::util::iterator_core_access;
650 
651  typename base_type::reference dereference() const
652  {
653  HPX_ASSERT(!is_at_end());
654  return data_->get_data();
655  }
656 
657  void increment()
658  {
659  ++(this->base_reference());
660 
661  if (this->base() != end_)
662  data_ = this->base()->local_data_;
663  else
664  data_.reset();
665  }
666 
667  void satisfy_predicate()
668  {
669  while (this->base() != end_ && !predicate_(*this->base()))
670  ++(this->base_reference());
671 
672  if (this->base() != end_)
673  data_ = this->base()->local_data_;
674  else
675  data_.reset();
676  }
677 
678  private:
679  std::shared_ptr<server::partitioned_vector<T, Data> > data_;
680  predicate predicate_;
681  BaseIter end_;
682  };

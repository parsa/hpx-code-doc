105  // operations
106  void* alloc(std::size_t count = 1)
107  {
108  unique_lock_type guard(mtx_);
109 
110  if (HPX_UNLIKELY(0 == count))
111  {
112  HPX_THROW_EXCEPTION(bad_parameter,
113  name() + "::alloc",
114  "cannot allocate 0 objects");
115  }
116 
117  //std::size_t size = 0;
118  value_type* p = nullptr;
119  {
120  if (!heap_list_.empty())
121  {
122  //size = heap_list_.size();
123  for (iterator it = heap_list_.begin(); it != heap_list_.end(); ++it)
124  {
125  typename list_type::value_type heap = *it;
126  bool allocated = false;
127 
128  {
129  util::unlock_guard<unique_lock_type> ul(guard);
130  allocated = heap->alloc(&p, count);
131  }
132 
133  if (allocated)
134  {
135 #if defined(HPX_DEBUG)
136  // Allocation succeeded, update statistics.
137  alloc_count_ += count;
138  if (alloc_count_ - free_count_ > max_alloc_count_)
139  max_alloc_count_ = alloc_count_- free_count_;
140 #endif
141  return p;
142  }
143 
144 #if defined(HPX_DEBUG)
145  LOSH_(info)
146  << (boost::format(
147  "%1%::alloc: failed to allocate from heap[%2%] "
148  "(heap[%2%] has allocated %3% objects and has "
149  "space for %4% more objects)")
150  % name()
151  % (*it)->heap_count_
152  % (*it)->size()
153  % (*it)->free_size());
154 #endif
155  }
156  }
157  }
158 
159  // Create new heap.
160  bool did_create = false;
161  {
162 #if defined(HPX_DEBUG)
163  heap_list_.push_front(typename list_type::value_type(
164  new heap_type(class_name_.c_str(), heap_count_ + 1, heap_step)));
165 #else
166  heap_list_.push_front(typename list_type::value_type(
167  new heap_type(class_name_.c_str(), 0, heap_step)));
168 #endif
169 
170  iterator itnew = heap_list_.begin();
171  typename list_type::value_type heap = *itnew;
172  bool result = false;
173 
174  {
175  util::unlock_guard<unique_lock_type> ul(guard);
176  result = heap->alloc(&p, count);
177  }
178 
179  if (HPX_UNLIKELY(!result || nullptr == p))
180  {
181  // out of memory
182  HPX_THROW_EXCEPTION(out_of_memory,
183  name() + "::alloc",
184  boost::str(boost::format(
185  "new heap failed to allocate %1% objects")
186  % count));
187  }
188 
189 #if defined(HPX_DEBUG)
190  alloc_count_ += count;
191  ++heap_count_;
192 
193  LOSH_(info)
194  << (boost::format(
195  "%1%::alloc: creating new heap[%2%], size is now %3%")
196  % name()
197  % heap_count_
198  % heap_list_.size());
199 #endif
200  did_create = true;
201  }
202 
203  if (did_create)
204  return p;
205 
206  guard.unlock();
207 
208  // Try again, we just got a new heap, so we should be good.
209  return alloc(count);
210  }

259  void free(void* p, std::size_t count = 1)
260  {
261  unique_lock_type ul(mtx_);
262 
263  if (nullptr == p || !threads::threadmanager_is(state_running))
264  return;
265 
266  // if this is called from outside a HPX thread we need to
267  // re-schedule the request
268  if (reschedule(p, count))
269  return;
270 
271  // Find the heap which allocated this pointer.
272  for (iterator it = heap_list_.begin(); it != heap_list_.end(); ++it)
273  {
274  typename list_type::value_type heap = *it;
275  bool did_allocate = false;
276 
277  {
278  util::unlock_guard<unique_lock_type> ull(ul);
279  did_allocate = heap->did_alloc(p);
280  if (did_allocate)
281  heap->free(p, count);
282  }
283 
284  if (did_allocate)
285  {
286 #if defined(HPX_DEBUG)
287  free_count_ += count;
288 #endif
289  return;
290  }
291  }
292 
293  HPX_THROW_EXCEPTION(bad_parameter,
294  name() + "::free",
295  boost::str(boost::format(
296  "pointer %1% was not allocated by this %2%")
297  % p % name()));
298  }

290  struct parse_formatter {
291  // formatter starts and ends with %
292  bool has_manipulator_name() const {
293  if ( m_manipulator.empty() )
294  return false;
295  if ( m_manipulator.size() > 1)
296  if ( m_manipulator[0] == '%' && (*m_manipulator.rbegin() == '%') )
297  return true;
298 
299  return false;
300  }
301 
302  string_type get_manipulator_name() const {
303  HPX_ASSERT( has_manipulator_name() );
304  // ignore starting and ending %
305  return m_manipulator.substr( 1, m_manipulator.size() - 2);
306  }
307  void clear() { m_manipulator.clear(); }
308 
309  void add(char_type c) {
310  if ( has_manipulator_name() )
311  // was a manipulator until now
312  clear();
313 
314  if ( c == '%') {
315  m_manipulator += c;
316  if ( !has_manipulator_name() )
317  // it could be the start of a formatter
318  m_manipulator = '%';
319  }
320  else if ( m_manipulator.empty())
321  ; // ignore this char - not from a manipulator
322  else if ( m_manipulator[0] == '%')
323  m_manipulator += c;
324  else
325  // manipulator should always start with %
326  HPX_ASSERT(false);
327  }
328  private:
329  string_type m_manipulator;
330  };

333  template<class manipulator, class parser_type>
334  void set_and_configure(manipulator & manip, const string_type & name,
335  parser_type parser) {
336  // need to parse string
337  bool parsing_params = false;
338  string_type params;
339  string_type stripped_str;
340  for ( typename string_type::const_iterator b = name.begin(),
341  e = name.end(); b != e; ++b) {
342  if ( (*b == '(') && !parsing_params) {
343  if ( parser.has_manipulator_name() ) {
344  parsing_params = true;
345  params.clear();
346  }
347  else {
348  stripped_str += *b;
349  parser.add( *b);
350  }
351  }
352  else if ( (*b == ')') && parsing_params) {
353  HPX_ASSERT ( parser.has_manipulator_name() );
354  manip.configure_inner( parser.get_manipulator_name(), params);
355  parser.clear();
356  parsing_params = false;
357  }
358  else {
359  if ( parsing_params)
360  params += *b;
361  else {
362  stripped_str += *b;
363  parser.add( *b);
364  }
365  }
366  }
367  manip.string( stripped_str);
368  }

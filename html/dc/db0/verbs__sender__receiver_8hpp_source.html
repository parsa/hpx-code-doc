45  // Provides the most low level access to send/receive read/write operations
46  // using the verbs rdma API.
47  // For performance monitoring and debugging, counters of send/recv/read/write
48  // operations are maintained.
49  //
50  // The number of preposted receives is tracked to prevent the receive queue
51  // from becoming empty and causing RNR retry errors.
52  // NB. Allowing the preposted receives to drop to zero can be disastrous
53  // if infinite retries are not enabled.

139  uint64_t post_send(verbs_memory_region *region, bool signaled,
140  bool withImmediate, uint32_t immediateData)
141  {
142  // Build scatter/gather element for outbound data.
143  struct ibv_sge send_sge;
144  send_sge.addr = (uint64_t)region->get_address();
145  send_sge.length = region->get_message_length();
146  send_sge.lkey = region->get_local_key();
147 
148  // Build a send work request.
149  struct ibv_send_wr send_wr;
150  memset(&send_wr, 0, sizeof(send_wr));
151  send_wr.next = nullptr;
152  send_wr.sg_list = &send_sge;
153  send_wr.num_sge = 1;
154  if (withImmediate) {
155  send_wr.opcode = IBV_WR_SEND_WITH_IMM;
156  send_wr.imm_data = immediateData;
157  }
158  else {
159  send_wr.opcode = IBV_WR_SEND;
160  }
161  if (signaled) {
162  send_wr.send_flags |= IBV_SEND_SIGNALED;
163  }
164  // use address for wr_id
165  send_wr.wr_id = (uint64_t)region;
166 
167  ++total_posted_send_;
168 
169  LOG_TRACE_MSG("Posted Send wr_id " << hexpointer(send_wr.wr_id)
170  << "with Length " << decnumber(send_sge.length)
171  << hexpointer(send_sge.addr)
172  << "total send posted " << decnumber(total_posted_send_));
173  // Post a send for outbound message.
174  // ++_waitingSendPosted;
175  return post_request(&send_wr);
176  }

290  // Post an RDMA read operation from a remote memory region to the
291  // specified local memory region.
292  inline uint64_t post_read(verbs_memory_region *localregion,
293  uint32_t remoteKey, const void *remoteAddr, std::size_t length)
294  {
295  // Build scatter/gather element for message.
296  struct ibv_sge read_sge;
297  read_sge.addr = (uint64_t)localregion->get_address();
298  read_sge.length = length;
299  read_sge.lkey = localregion->get_local_key();
300 
301  // Build a work request for the read operation
302  struct ibv_send_wr send_wr;
303  memset(&send_wr, 0, sizeof(send_wr));
304  send_wr.next = nullptr;
305  send_wr.sg_list = &read_sge;
306  send_wr.num_sge = 1;
307  send_wr.opcode = IBV_WR_RDMA_READ;
308  // Force completion queue to be posted with result.
309  send_wr.send_flags = IBV_SEND_SIGNALED;
310  send_wr.wr_id = (uint64_t)localregion;
311  send_wr.wr.rdma.remote_addr = (uint64_t)remoteAddr;
312  send_wr.wr.rdma.rkey = remoteKey;
313 
314  ++total_posted_read_;
315 
316  // Post a send to read data.
317  LOG_TRACE_MSG("Posted Read wr_id " << hexpointer(send_wr.wr_id)
318  << " with Length " << decnumber(read_sge.length) << " "
319  << hexpointer(read_sge.addr)
320  << " remote key " << decnumber(send_wr.wr.rdma.rkey)
321  << " remote addr " << hexpointer(send_wr.wr.rdma.remote_addr));
322  return post_request(&send_wr);
323  }

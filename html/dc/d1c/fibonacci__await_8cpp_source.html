29 HPX_NOINLINE std::uint64_t fibonacci_serial(std::uint64_t n)
30 {
31  if (n < 2)
32  return n;
33  return fibonacci_serial(n-1) + fibonacci_serial(n-2);

71 void _fibonacci(std::shared_ptr<_fibonacci_frame> const& frame_)
72 {
73  _fibonacci_frame* frame = frame_.get();
74  int state = frame->state_;
75 
76  switch (state)
77  {
78  case 1:
79  goto L1;
80  case 2:
81  goto L2;
82  }
83 
84  // if (n < 2) return hpx::make_ready_future(n);
85  if (frame->n_ < 2)
86  {
87  if (state == 0)
88  // never paused
89  frame->result_ = hpx::make_ready_future(frame->n_);
90  else
91  frame->result_promise_.set_value(frame->n_);
92  return;
93  }
94 
95  // if (n < threshold) return hpx::make_ready_future(fibonacci_serial(n));
96  if (frame->n_ < threshold)
97  {
98  if (state == 0)
99  // never paused
100  frame->result_ = hpx::make_ready_future(fibonacci_serial(frame->n_));
101  else
102  frame->result_promise_.set_value(fibonacci_serial(frame->n_));
103  return;
104  }
105 
106  // hpx::future<std::uint64_t> lhs = hpx::async(&fibonacci, n-1);
107  frame->lhs_ = hpx::async(&fibonacci, frame->n_-1);
108 
109  // hpx::future<std::uint64_t> rhs = fibonacci(n-2);
110  frame->rhs_ = fibonacci(frame->n_-2);
111 
112  if (!frame->lhs_.is_ready())
113  {
114  frame->state_ = 1;
115  if (!frame->result_.valid())
116  frame->result_ = frame->result_promise_.get_future();
117  frame->lhs_.then(hpx::util::bind(&_fibonacci, frame_));
118  return;
119  }
120 
121 L1:
122  frame->lhs_result_ = frame->lhs_.get();
123 
124  if ( !frame->rhs_.is_ready())
125  {
126  frame->state_ = 2;
127  if (!frame->result_.valid())
128  frame->result_ = frame->result_promise_.get_future();
129  frame->rhs_.then(hpx::util::bind(&_fibonacci, frame_));
130  return;
131  }
132 
133 L2:
134  frame->rhs_result_ = frame->rhs_.get();
135 
136  if (state == 0)
137  // never paused
138  frame->result_ = hpx::make_ready_future(frame->lhs_result_ + frame->rhs_result_);
139  else
140  frame->result_promise_.set_value(frame->lhs_result_ + frame->rhs_result_);
141  return;
142 }

155 int hpx_main(boost::program_options::variables_map& vm)
156 {
157  // extract command line argument, i.e. fib(N)
158  std::uint64_t n = vm["n-value"].as<std::uint64_t>();
159  std::string test = vm["test"].as<std::string>();
160  std::uint64_t max_runs = vm["n-runs"].as<std::uint64_t>();
161 
162  if (max_runs == 0) {
163  std::cerr << "fibonacci_await: wrong command line argument value for "
164  "option 'n-runs', should not be zero" << std::endl;
165  return hpx::finalize(); // Handles HPX shutdown
166  }
167 
168  threshold = vm["threshold"].as<unsigned int>();
169  if (threshold < 2 || threshold > n) {
170  std::cerr << "fibonacci_await: wrong command line argument value for "
171  "option 'threshold', should be in between 2 and n-value"
172  ", value specified: " << threshold << std::endl;
173  return hpx::finalize(); // Handles HPX shutdown
174  }
175 
176  bool executed_one = false;
177  std::uint64_t r = 0;
178 
179  if (test == "all" || test == "0")
180  {
181  // Keep track of the time required to execute.
182  std::uint64_t start = hpx::util::high_resolution_clock::now();
183 
184  for (std::size_t i = 0; i != max_runs; ++i)
185  {
186  // serial execution
187  r = fibonacci_serial(n);
188  }
189 
190 // double d = double(hpx::util::high_resolution_clock::now() - start) / 1.e9;
191  std::uint64_t d = hpx::util::high_resolution_clock::now() - start;
192  char const* fmt = "fibonacci_serial(%1%) == %2%,"
193  "elapsed time:,%3%,[s]\n";
194  std::cout << (boost::format(fmt) % n % r % (d / max_runs));
195 
196  executed_one = true;
197  }
198 
199  if (test == "all" || test == "1")
200  {
201  // Keep track of the time required to execute.
202  std::uint64_t start = hpx::util::high_resolution_clock::now();
203 
204  for (std::size_t i = 0; i != max_runs; ++i)
205  {
206  // Create a future for the whole calculation, execute it locally,
207  // and wait for it.
208  r = fibonacci(n).get();
209  }
210 
211 // double d = double(hpx::util::high_resolution_clock::now() - start) / 1.e9;
212  std::uint64_t d = hpx::util::high_resolution_clock::now() - start;
213  char const* fmt = "fibonacci_await(%1%) == %2%,"
214  "elapsed time:,%3%,[s]\n";
215  std::cout << (boost::format(fmt) % n % r % (d / max_runs));
216 
217  executed_one = true;
218  }
219 
220  if (!executed_one)
221  {
222  std::cerr << "fibonacci_await: wrong command line argument value for "
223  "option 'tests', should be either 'all' or a number between zero "
224  "and 1, value specified: " << test << std::endl;
225  }
226 
227  return hpx::finalize(); // Handles HPX shutdown
228 }

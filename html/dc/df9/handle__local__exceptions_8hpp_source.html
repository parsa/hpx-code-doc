87  static void call(std::vector<hpx::future<T> >& workitems,
88  std::list<std::exception_ptr>& errors, Cleanup && cleanup)
89  {
90  bool has_exception = false;
91  std::exception_ptr bad_alloc_exception;
92  for (hpx::future<T>& f: workitems)
93  {
94  if (f.has_exception())
95  {
96  std::exception_ptr e = f.get_exception_ptr();
97  try {
98  std::rethrow_exception(e);
99  }
100  catch (std::bad_alloc const&) {
101  bad_alloc_exception = e;
102  }
103  catch (...) {
104  errors.push_back(e);
105  }
106  has_exception = true;
107  }
108  }
109 
110  // If at least one partition failed with an exception, call
111  // the cleanup function for all others (the failed partitioned
112  // are assumed to have already run the cleanup).
113  if (has_exception)
114  {
115  for (hpx::future<T>& f: workitems)
116  {
117  if (!f.has_exception())
118  cleanup(f.get());
119  }
120  }
121 
122  if (bad_alloc_exception)
123  std::rethrow_exception(bad_alloc_exception);
124 
125  if (!errors.empty())
126  throw exception_list(std::move(errors));
127  }

57 int app_main(
58  variables_map&
59  )
60 {
62  // Initialize the PRNG seed.
63  if (!seed)
64  seed = std::uint64_t(std::time(nullptr));
65 
67  // Validate command-line arguments.
68  if (0 == tasks)
69  throw std::invalid_argument("count of 0 tasks specified\n");
70 
71  if (min_delay > max_delay)
72  throw std::invalid_argument("minimum delay cannot be larger than "
73  "maximum delay\n");
74 
75  if (min_delay > total_delay)
76  throw std::invalid_argument("minimum delay cannot be larger than"
77  "total delay\n");
78 
79  if (max_delay > total_delay)
80  throw std::invalid_argument("maximum delay cannot be larger than "
81  "total delay\n");
82 
83  if ((min_delay * tasks) > total_delay)
84  throw std::invalid_argument("minimum delay is too small for the "
85  "specified total delay and number of "
86  "tasks\n");
87 
88  if ((max_delay * tasks) < total_delay)
89  throw std::invalid_argument("maximum delay is too small for the "
90  "specified total delay and number of "
91  "tasks\n");
92 
94  // Randomly generate a description of the heterogeneous workload.
95  std::vector<std::uint64_t> payloads;
96  payloads.reserve(tasks);
97 
98  // For random numbers, we use a 64-bit specialization of stdlib's
99  // mersenne twister engine (good uniform distribution up to 311
100  // dimensions, cycle length 2 ^ 19937 - 1)
101  std::mt19937_64 prng(seed);
102 
103  std::uint64_t current_sum = 0;
104 
105  for (std::uint64_t i = 0; i < tasks; ++i)
106  {
107  // Credit to Spencer Ruport for putting this algorithm on
108  // stackoverflow.
109  std::uint64_t const low_calc
110  = (total_delay - current_sum) - (max_delay * (tasks - 1 - i));
111 
112  bool const negative
113  = (total_delay - current_sum) < (max_delay * (tasks - 1 - i));
114 
115  std::uint64_t const low
116  = (negative || (low_calc < min_delay)) ? min_delay : low_calc;
117 
118  std::uint64_t const high_calc
119  = (total_delay - current_sum) - (min_delay * (tasks - 1 - i));
120 
121  std::uint64_t const high
122  = (high_calc > max_delay) ? max_delay : high_calc;
123 
124  // Our range is [low, high].
125  std::uniform_int_distribution<std::uint64_t>
126  dist(low, high);
127 
128  std::uint64_t const payload = dist(prng);
129 
130  if (payload < min_delay)
131  throw std::logic_error("task delay is below minimum");
132 
133  if (payload > max_delay)
134  throw std::logic_error("task delay is above maximum");
135 
136  current_sum += payload;
137  payloads.push_back(payload);
138  }
139 
140  // Randomly shuffle the entire sequence to deal with drift.
141  std::random_shuffle(payloads.begin(), payloads.end(),
142  boost::bind(&shuffler, std::ref(prng), _1));
143 
145  // Validate the payloads.
146  if (payloads.size() != tasks)
147  throw std::logic_error("incorrect number of tasks generated");
148 
149  std::uint64_t const payloads_sum =
150  std::accumulate(payloads.begin(), payloads.end(), 0ULL);
151  if (payloads_sum != total_delay)
152  throw std::logic_error("incorrect total delay generated");
153 
154  for (std::size_t i = 0; i < payloads.size(); ++i)
155  std::cout << payloads[i] << "\n";
156 
157  return 0;
158 }

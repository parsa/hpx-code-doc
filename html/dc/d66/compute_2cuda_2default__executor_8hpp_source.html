40  // generic implementation which simply passes through the shape elements
41  template <typename Shape, typename Enable = void>
42  struct bulk_launch_helper
43  {
44  template <typename F, typename ... Ts>
45  static void call(cuda::target const& target, F && f,
46  Shape const& shape, Ts &&... ts)
47  {
48  std::size_t count = util::size(shape);
49 
50  int threads_per_block =
51  (std::min)(1024, static_cast<int>(count));
52  int num_blocks = static_cast<int>(
53  (count + threads_per_block - 1) / threads_per_block);
54 
55  typedef typename hpx::traits::range_traits<Shape>::value_type
56  value_type;
57  typedef cuda::allocator<value_type> alloc_type;
58 
59  // transfer shape to the GPU
60  compute::vector<value_type, alloc_type> shape_container(
61  util::begin(shape), util::end(shape), alloc_type(target));
62 
63  detail::launch(
64  target, num_blocks, threads_per_block,
65  [] HPX_DEVICE
66  (F f, value_type* p, std::size_t count, Ts&... ts)
67  {
68  std::size_t idx = blockIdx.x * blockDim.x + threadIdx.x;
69  if (idx < count)
70  {
71  hpx::util::invoke_r<void>(f, *(p + idx), ts...);
72  }
73  },
74  std::forward<F>(f), shape_container.data(), count,
75  std::forward<Ts>(ts)...);
76  }
77  };

79  // specialization used by partitioner implementations
80  template <typename Iterator>
81  struct bulk_launch_helper<
82  hpx::util::tuple<Iterator, std::size_t, std::size_t>,
83  typename std::enable_if<
84  hpx::traits::is_iterator<Iterator>::value
85  >::type
86  >
87  {
88  template <typename F, typename Shape, typename ... Ts>
89  static void call(cuda::target const& target, F && f,
90  Shape const& shape, Ts &&... ts)
91  {
92  typedef typename hpx::traits::range_traits<Shape>::value_type
93  value_type;
94 
95  for (auto const& s: shape)
96  {
97  auto begin = hpx::util::get<0>(s);
98  std::size_t chunk_size = hpx::util::get<1>(s);
99 
100  // FIXME: make the 1024 to be configurable...
101  int threads_per_block =
102  (std::min)(1024, static_cast<int>(chunk_size));
103  int num_blocks = static_cast<int>(
104  (chunk_size + threads_per_block - 1) / threads_per_block);
105 
106  detail::launch(
107  target, num_blocks, threads_per_block,
108  [begin, chunk_size]
109  HPX_DEVICE (F f, Ts&... ts)
110  {
111  std::size_t idx
112  = blockIdx.x * blockDim.x + threadIdx.x;
113  if(idx < chunk_size)
114  {
115  hpx::util::invoke_r<void>(f,
116  value_type(begin + idx, 1, idx), ts...);
117  }
118  },
119  std::forward<F>(f), std::forward<Ts>(ts)...
120  );
121  }
122  }

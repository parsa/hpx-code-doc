159  std::cout << "os threads : " << os_threads << std::endl;
160  std::cout << "----------------------------------------------\n" << std::endl;
161 
162  std::cout << "* Preparing Benchmark..." << std::endl;
163  std::vector<int> v(vector_size);
164 
165  // initialize data
166  using namespace hpx::parallel;
167  generate(execution::par, std::begin(v), std::end(v), random_fill());
168  std::make_heap(std::begin(v), std::next(std::begin(v), break_pos));
169  if (break_pos < vector_size)
170  v[break_pos] = static_cast<int>((std::numeric_limits<std::size_t>::max)());
171 
172  std::cout << "* Running Benchmark..." << std::endl;
173  double time_std =

175  double time_seq =
176  run_is_heap_until_benchmark_seq(test_count, v);
177  double time_par =

179  double time_par_unseq =
180  run_is_heap_until_benchmark_par_unseq(test_count, v);
181 
182  std::cout << "\n-------------- Benchmark Result --------------" << std::endl;

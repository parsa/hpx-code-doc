59  template <typename ExPolicy, typename RandIter, typename F, typename Proj>
60  hpx::future<RandIter>
61  operator()(ExPolicy && policy, RandIter first, RandIter last,
62  std::size_t size, F f, Proj proj, std::size_t chunks)
63  {
64  if (chunks < 2)
65  {
66  return execution::async_execute(
67  policy.executor(),
68  [first, last, f, proj]() -> RandIter
69  {
70  return std::stable_partition(
71  first, last,
72  util::invoke_projected<F, Proj>(f, proj));
73  });
74  }
75 
76  std::size_t mid_point = size / 2;
77  chunks /= 2;
78 
79  RandIter mid = first;
80  std::advance(mid, mid_point);
81 
82  hpx::future<RandIter> left = execution::async_execute(
83  policy.executor(), *this, policy, first, mid,
84  mid_point, f, proj, chunks);
85  hpx::future<RandIter> right = execution::async_execute(
86  policy.executor(), *this, policy, mid, last,
87  size - mid_point, f, proj, chunks);
88 
89  return
90  dataflow(
91  policy.executor(),
92  [mid](
93  hpx::future<RandIter> && left,
94  hpx::future<RandIter> && right
95  ) -> RandIter
96  {
97  if (left.has_exception() || right.has_exception())
98  {
99  std::list<std::exception_ptr> errors;
100  if(left.has_exception())
101  hpx::parallel::util::detail::
102  handle_local_exceptions<ExPolicy>::call(
103  left.get_exception_ptr(), errors);
104  if(right.has_exception())
105  hpx::parallel::util::detail::
106  handle_local_exceptions<ExPolicy>::call(
107  right.get_exception_ptr(), errors);
108 
109  if (!errors.empty())
110  {
111  throw exception_list(std::move(errors));
112  }
113  }
114  RandIter first = left.get();
115  RandIter last = right.get();
116 
117  std::rotate(first, mid, last);
118 
119  // for some library implementations std::rotate
120  // does not return the new middle point
121  std::advance(first, std::distance(mid, last));
122  return first;
123  },
124  std::move(left), std::move(right));
125  }
126  };

307  // sequential partition with projection function for bidirectional iterator.
308  template <typename BidirIter, typename Pred, typename Proj,
309  HPX_CONCEPT_REQUIRES_(
310  hpx::traits::is_bidirectional_iterator<BidirIter>::value)>
311  BidirIter
312  sequential_partition(BidirIter first, BidirIter last,
313  Pred && pred, Proj && proj)
314  {
315  using hpx::util::invoke;
316 
317  while (true)
318  {
319  while (first != last && invoke(pred, invoke(proj, *first)))
320  ++first;
321  if (first == last)
322  break;
323 
324  while (first != --last && !invoke(pred, invoke(proj, *last)))
325  ;
326  if (first == last)
327  break;
328 
329  std::iter_swap(first++, last);
330  }
331 
332  return first;
333  }

335  // sequential partition with projection function for forward iterator.
336  template <typename FwdIter, typename Pred, typename Proj,
337  HPX_CONCEPT_REQUIRES_(
338  hpx::traits::is_forward_iterator<FwdIter>::value &&
339  !hpx::traits::is_bidirectional_iterator<FwdIter>::value)>
340  FwdIter
341  sequential_partition(FwdIter first, FwdIter last,
342  Pred && pred, Proj && proj)
343  {
344  using hpx::util::invoke;
345 
346  while (first != last && invoke(pred, invoke(proj, *first)))
347  ++first;
348 
349  if (first == last)
350  return first;
351 
352  for (FwdIter it = std::next(first); it != last; ++it)
353  {
354  if (invoke(pred, invoke(proj, *it)))
355  std::iter_swap(first++, it);
356  }
357 
358  return first;

363  template <typename FwdIter>
364  struct block
365  {
366  FwdIter first;
367  FwdIter last;
368  // Maybe 'std::int64_t' is enough to avoid overflow.
369  std::int64_t block_no;
370 
371  block() = default;
372  block(FwdIter first, FwdIter last, std::int64_t block_no = -1)
373  : first(first), last(last), block_no(block_no)
374  {}
375 
376  bool empty() const { return first == last; }
377 
378  // The blocks are sorted by their positions.
379  // The block_no implies block's position.
380  // If the block is on leftside of boundary, block_no is negative number.
381  // Otherwise, block_no is positive number.
382  // The farther from the boundary, the larger its absolute value.
383  // The example of sorted blocks below (the number means block_no):
384  // -3 -2 -1 1 2 3 4
385  bool operator<(block<FwdIter> const& other) const
386  {
387  if ((this->block_no < 0 && other.block_no < 0) ||
388  (this->block_no > 0 && other.block_no > 0))
389  return this->block_no > other.block_no;
390  else
391  return this->block_no < other.block_no;
392  }
393  };

398  // block manager for random access iterator.
399  template <typename RandIter>
400  class block_manager<RandIter,
401  typename std::enable_if<
402  hpx::traits::is_random_access_iterator<RandIter>::value
403  >::type>
404  {
405  public:
406  block_manager(RandIter first, RandIter last, std::size_t block_size)
407  : first_(first), left_(0), right_(std::distance(first, last)),
408  block_size_(block_size)
409  {}
410  block_manager(const block_manager&) = delete;
411  block_manager& operator=(const block_manager&) = delete;
412 
413  // Get block from the end of leftside of boundary.
414  block<RandIter> get_left_block()
415  {
416  std::lock_guard<decltype(mutex_)> lk(mutex_);
417 
418  if (left_ >= right_)
419  return { first_, first_ };
420 
421  std::size_t begin_index = left_;
422  std::size_t end_index = (std::min)(left_ + block_size_, right_);
423 
424  left_ += end_index - begin_index;
425 
426  RandIter begin_iter = std::next(first_, begin_index);
427  RandIter end_iter = std::next(first_, end_index);
428 
429  boundary_ = end_iter;
430 
431  return { begin_iter, end_iter, left_block_no_-- };
432  }
433 
434  // Get block from the end of rightside of boundary.
435  block<RandIter> get_right_block()
436  {
437  std::lock_guard<decltype(mutex_)> lk(mutex_);
438 
439  if (left_ >= right_)
440  return { first_, first_ };
441 
442  std::size_t begin_index = (std::max)(right_ - block_size_, left_);
443  std::size_t end_index = right_;
444 
445  right_ -= end_index - begin_index;
446 
447  RandIter begin_iter = std::next(first_, begin_index);
448  RandIter end_iter = std::next(first_, end_index);
449 
450  boundary_ = begin_iter;
451 
452  return { begin_iter, end_iter, right_block_no_++ };
453  }
454 
455  RandIter boundary() { return boundary_; }
456 
457  private:
458  RandIter first_, boundary_;
459  std::size_t left_, right_;
460  std::size_t block_size_;
461  std::int64_t left_block_no_{ -1 }, right_block_no_{ 1 };
462  hpx::lcos::local::spinlock mutex_;
463  };

465  // block manager for forward access iterator.
466  template <typename FwdIter>
467  class block_manager<FwdIter,
468  typename std::enable_if<
469  hpx::traits::is_forward_iterator<FwdIter>::value &&
470  !hpx::traits::is_random_access_iterator<FwdIter>::value
471  >::type>
472  {
473  public:
474  // In constructor, prepare all blocks for fast acquirements of blocks.
475  block_manager(FwdIter first, FwdIter last, std::size_t block_size)
476  : boundary_(first), blocks_(
477  (std::distance(first, last) + block_size - 1) / block_size)
478  {
479  left_ = 0;
480  right_ = blocks_.size();
481 
482  if (blocks_.size() == 1)
483  {
484  blocks_.front() = { first, last };
485  return;
486  }
487 
488  FwdIter next = std::next(first, block_size);
489 
490  blocks_.front() = { first, next };
491 
492  for (std::size_t i = 1; i < blocks_.size() - 1; ++i)
493  {
494  first = next;
495  next = std::next(first, block_size);
496  blocks_[i] = { first, next };
497  }
498 
499  blocks_.back() = { next, last };
500  }

503  block_manager& operator=(const block_manager&) = delete;

506  block<FwdIter> get_left_block()
507  {
508  std::lock_guard<decltype(mutex_)> lk(mutex_);
509 
510  if (left_ >= right_)
511  return { boundary_, boundary_ };
512

523 
524  if (left_ >= right_)
525  return { boundary_, boundary_ };
526

528  blocks_[right_].block_no = right_block_no_++;
529 
530  return std::move(blocks_[right_]);
531  }
532 
533  FwdIter boundary() { return boundary_; }
534

536  FwdIter boundary_;
537  std::vector<block<FwdIter> > blocks_;
538  std::size_t left_, right_;
539  std::int64_t left_block_no_{ -1 }, right_block_no_{ 1 };
540  hpx::lcos::local::spinlock mutex_;

543  // std::swap_ranges doens't support overlapped ranges in standard.
544  // But, actually general implementations of std::swap_ranges are useful
545  // in specific cases.
546  // The problem is that standard doesn't guarantee that implementation.
547  // The swap_ranges_forward is the general implementation of
548  // std::swap_ranges for guaranteeing utilizations in specific cases.
549  // If dest is previous to first, the range [first, last) can be
550  // successfully moved to the range [dest, dest+distance(first, last)).
551  template <class FwdIter1, class FwdIter2>
552  static FwdIter2
553  swap_ranges_forward(FwdIter1 first, FwdIter1 last, FwdIter2 dest)
554  {
555  while (first != last)
556  std::iter_swap(first++, dest++);
557 
558  return dest;
559  }

561  // The function which performs sub-partitioning.
562  template <typename FwdIter, typename Pred, typename Proj>
563  static block<FwdIter>
564  partition_thread(block_manager<FwdIter>& block_manager,
565  Pred pred, Proj proj)
566  {
567  using hpx::util::invoke;
568 
569  block<FwdIter> left_block, right_block;
570 
571  left_block = block_manager.get_left_block();
572  right_block = block_manager.get_right_block();
573 
574  while (true)
575  {
576  while ( (!left_block.empty() ||
577  !(left_block = block_manager.get_left_block()).empty()) &&
578  invoke(pred, invoke(proj, *left_block.first)))
579  {
580  ++left_block.first;
581  }
582 
583  while ( (!right_block.empty() ||
584  !(right_block = block_manager.get_right_block()).empty()) &&
585  !invoke(pred, invoke(proj, *right_block.first)))
586  {
587  ++right_block.first;
588  }
589 
590  if (left_block.empty())
591  return right_block;
592  if (right_block.empty())
593  return left_block;
594 
595  std::iter_swap(left_block.first++, right_block.first++);
596  }
597  }

599  // The function which collapses remaining blocks.
600  // Performs sequential sub-partitioning to remaining blocks for
601  // reducing the number and size of remaining blocks.
602  template <typename FwdIter, typename Pred, typename Proj>
603  static void
604  collapse_remaining_blocks(std::vector<block<FwdIter>>& remaining_blocks,
605  Pred& pred, Proj& proj)
606  {
607  if (remaining_blocks.empty())
608  return;
609 
610  auto left_iter = std::begin(remaining_blocks);
611  auto right_iter = std::end(remaining_blocks) - 1;
612 
613  if (left_iter->block_no > 0 || right_iter->block_no < 0)
614  return;
615 
616  while (true)
617  {
618  using hpx::util::invoke;
619 
620  while (true)
621  {
622  if (left_iter->empty())
623  {
624  ++left_iter;
625  if (left_iter == std::end(remaining_blocks) ||
626  left_iter->block_no > 0)
627  break;
628  }
629  if (!invoke(pred, invoke(proj, *left_iter->first)))
630  break;
631  ++left_iter->first;
632  }
633 
634  while (true)
635  {
636  if (right_iter->empty())
637  {
638  if (right_iter == std::begin(remaining_blocks) ||
639  (--right_iter)->block_no < 0)
640  break;
641  }
642  if (invoke(pred, invoke(proj, *right_iter->first)))
643  break;
644  ++right_iter->first;
645  }
646 
647  if (left_iter == std::end(remaining_blocks) ||
648  left_iter->block_no > 0)
649  break;
650  if (right_iter->empty() ||
651  right_iter->block_no < 0)
652  break;
653 
654  std::iter_swap(left_iter->first++, right_iter->first++);
655  }
656 
657  if (left_iter < right_iter ||
658  (!right_iter->empty() && left_iter == right_iter))
659  {
660  remaining_blocks.erase(
661  right_iter->empty() ? right_iter : right_iter + 1,
662  std::end(remaining_blocks));
663 
664  remaining_blocks.erase(
665  std::begin(remaining_blocks), left_iter);
666  }
667  else
668  {
669  remaining_blocks.clear();
670  }
671  }

673  // The function which merges remaining blocks that are placed
674  // leftside of boundary. Requires bidirectional iterator.
675  // Move remaining blocks to the adjacent left of the boundary.
676  // In the end, all remaining blocks are merged into one block which
677  // is adjacent to the left of boundary.
678  template <typename BidirIter,
679  HPX_CONCEPT_REQUIRES_(
680  hpx::traits::is_bidirectional_iterator<BidirIter>::value)>
681  static block<BidirIter>
682  merge_leftside_remaining_blocks(
683  std::vector<block<BidirIter>>& remaining_blocks,
684  BidirIter boundary, BidirIter first)
685  {
686  HPX_ASSERT(!remaining_blocks.empty());
687  HPX_UNUSED(first);
688 
689  auto boundary_rbegin = std::reverse_iterator<BidirIter>(boundary);
690  for (auto it = remaining_blocks.rbegin();
691  it != remaining_blocks.rend();
692  ++it)
693  {
694  auto rbegin = std::reverse_iterator<BidirIter>(it->last);
695  auto rend = std::reverse_iterator<BidirIter>(it->first);
696 
697  if (boundary_rbegin == rbegin)
698  {
699  boundary_rbegin = rend;
700  continue;
701  }
702 
703  boundary_rbegin =
704  swap_ranges_forward(rbegin, rend, boundary_rbegin);
705  }
706 
707  return { boundary_rbegin.base(), boundary };
708  }

710  // The function which merges remaining blocks that are placed
711  // leftside of boundary. Requires forward iterator.
712  // Move remaining blocks to the adjacent left of the boundary.
713  // In the end, all remaining blocks are merged into one block which
714  // is adjacent to the left of boundary.
715  template <typename FwdIter,
716  HPX_CONCEPT_REQUIRES_(
717  hpx::traits::is_forward_iterator<FwdIter>::value &&
718  !hpx::traits::is_bidirectional_iterator<FwdIter>::value)>
719  static block<FwdIter>
720  merge_leftside_remaining_blocks(
721  std::vector<block<FwdIter>>& remaining_blocks,
722  FwdIter boundary, FwdIter first)
723  {
724  HPX_ASSERT(!remaining_blocks.empty());
725 
726  std::vector<FwdIter> dest_iters(remaining_blocks.size());
727  std::vector<std::size_t> dest_iter_indexes(remaining_blocks.size());
728  std::vector<std::size_t> remaining_block_indexes(
729  remaining_blocks.size());
730  std::vector<std::size_t> counts(remaining_blocks.size());
731  std::size_t count_sum = 0u;
732 
733  for (std::size_t i = 0; i < counts.size(); ++i)
734  {
735  counts[i] = std::distance(
736  remaining_blocks[i].first,
737  remaining_blocks[i].last);
738  count_sum += counts[i];
739  }
740 
741  remaining_block_indexes[0] = std::distance(first,
742  remaining_blocks[0].first);
743  for (std::size_t i = 1; i < remaining_block_indexes.size(); ++i)
744  {
745  remaining_block_indexes[i] =
746  remaining_block_indexes[i - 1] +
747  counts[i - 1] +
748  std::distance(remaining_blocks[i - 1].last,
749  remaining_blocks[i].first);
750  }
751 
752  std::size_t boundary_end_index = std::distance(first, boundary);
753  std::size_t boundary_begin_index = boundary_end_index - count_sum;
754 
755  dest_iters[0] = std::next(first, boundary_begin_index);
756  dest_iter_indexes[0] = boundary_begin_index;
757 
758  for (std::size_t i = 0; i < dest_iters.size() - 1; ++i)
759  {
760  dest_iters[i + 1] = std::next(dest_iters[i], counts[i]);
761  dest_iter_indexes[i + 1] = dest_iter_indexes[i] + counts[i];
762  }

193  void stabilize_left(anchor_pair& lrs)
194  {
195  // Get the right node of the leftmost pointer held by lrs and its ABA
196  // tag (tagged_ptr).
197  node_pointer prev = lrs.get_left_ptr()->right.load();
198 
199  if (anchor_ != lrs)
200  return;
201 
202  // Get the left node of prev and its tag (again, a tuple represented by
203  // a tagged_ptr).
204  node_pointer prevnext = prev.get_ptr()->left.load();
205 
206  // Check if prevnext is equal to r.
207  if (prevnext.get_ptr() != lrs.get_left_ptr())
208  {
209  if (anchor_ != lrs)
210  return;
211 
212  // Attempt the CAS, incrementing the tag to protect from the ABA
213  // problem.
214  if (!prev.get_ptr()->left.compare_exchange_strong(prevnext,
215  node_pointer(lrs.get_left_ptr(), prevnext.get_tag() + 1)))
216  return;
217  }
218  // Try to update the anchor, modifying the status and ABA tag.
219  anchor_.cas(lrs, anchor_pair(lrs.get_left_ptr(), lrs.get_right_ptr(),
220  stable, lrs.get_right_tag() + 1));
221  }

228

291  // Thread-safe and non-blocking (may block if node needs to be allocated
292  // from the operating system). Returns false if the freelist is not able to
293  // allocate a new deque node.
294  // Complexity: O(Processes)
295  bool push_left(T const& data)
296  {
297  // Allocate the new node which we will be inserting.
298  node* n = alloc_node(nullptr, nullptr, data);
299 
300  if (n == nullptr)
301  return false;
302 
303  // Loop until we insert successfully.
304  while (true)
305  {
306  // Load the anchor.
307  anchor_pair lrs = anchor_.lrs();
308 
309  // Check if the deque is empty.
310  // FIXME: Should we check both pointers here?
311  if (lrs.get_left_ptr() == nullptr)
312  {
313  // If the deque is empty, we simply install a new anchor which
314  // points to the new node as both its leftmost and rightmost
315  // element.
316  if (anchor_.cas(lrs, anchor_pair(n, n,
317  lrs.get_left_tag(), lrs.get_right_tag() + 1)))
318  return true;
319  }
320 
321  // Check if the deque is stable.
322  else if (lrs.get_left_tag() == stable)
323  {
324  // Make the right pointer on our new node refer to the current
325  // leftmost node.
326  n->right.store(node_pointer(lrs.get_left_ptr()));
327 
328  // Now we want to make the anchor point to our new node as the
329  // leftmost node. We change the state to lpush as the deque
330  // will become unstable if this operation succeeds.
331  anchor_pair new_anchor(n, lrs.get_right_ptr(),
332  lpush, lrs.get_right_tag() + 1);
333 
334  if (anchor_.cas(lrs, new_anchor))
335  {
336  stabilize_left(new_anchor);
337  return true;
338  }
339  }
340 
341  // The deque must be unstable, so we have to stabilize it before
342  // we can continue.
343  else // lrs.s() != stable
344  stabilize(lrs);
345  }
346  }

405  // Thread-safe and non-blocking. Returns false if the deque is empty.
406  // Complexity: O(Processes)
407  bool pop_left(T& r)
408  {
409  // Loop until we either pop an element or learn that the deque is empty.
410  while (true)
411  {
412  // Load the anchor.
413  anchor_pair lrs = anchor_.lrs();
414 
415  // Check if the deque is empty.
416  // FIXME: Should we check both pointers here?
417  if (lrs.get_left_ptr() == nullptr)
418  return false;
419 
420  // Check if the deque has 1 element.
421  if (lrs.get_left_ptr() == lrs.get_right_ptr())
422  {
423  // Try to set both anchor pointer
424  if (anchor_.cas(lrs, anchor_pair(nullptr, nullptr,
425  lrs.get_left_tag(), lrs.get_right_tag() + 1)))
426  {
427  // Set the result, deallocate the popped node, and return.
428  r = lrs.get_left_ptr()->data;
429  dealloc_node(lrs.get_left_ptr());
430  return true;
431  }
432  }
433 
434  // Check if the deque is stable.
435  else if (lrs.get_left_tag() == stable)
436  {
437  // Make sure the anchor hasn't changed since we loaded it.
438  if (anchor_ != lrs)

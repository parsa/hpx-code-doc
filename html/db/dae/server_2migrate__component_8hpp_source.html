25  //
26  // Migrate a given component instance to the specified target locality (the
27  // component instance is called 'object' below).
28  //
29  // Migration of an object will involve at least two localities, but can
30  // touch on up to 4 localities:
31  //
32  // A) The locality on which the migration operation is triggered. This is
33  // the locality where `hpx::components::migrate()` is invoked.
34  // B) The locality where the object to be migrated is currently located.
35  // C) The locality where the object should be moved (migrated) to.
36  // D) The locality which hosts the AGAS instance responsible for resolving
37  // the global address of the object to be migrated.
38  //
39  // The localities B and C will be different, while the localities A and
40  // D could be the same as either of the others.
41  //
42  // Object migration is performed in several steps:
43  //
44  // 1) The migration is triggered on locality A by invoking the `migrate()`
45  // function. This invokes the action `perform_migrate_component_action`
46  // on the locality where the object to migrate is currently located
47  // (locality B).
48  //
49  // The action `perform_migrate_component_action` executes the following
50  // steps:
51  // a) It will delay the start of the migration operation until no more
52  // actions (threads) are pending or currently running for the object
53  // to be migrated (the object is unpinned).
54  // b) It marks the object which is about to be migrated as 'was
55  // migrated'. This information is stored in the AGAS client side
56  // representation on locality B. It is used to forward all incoming
57  // parcels to the object's new locality. It is also used to force any
58  // locally triggered actions for the object to go through the parcel
59  // layer. For this, the object to be migrated is removed from the
60  // local AGAS cache.
61  // c) It will trigger the actual migration operation (see step 2).
62  //
63  // In order to keep track of any pending and currently running actions
64  // (threads) for the object to migrate, any thread which is being
65  // scheduled will pin the object. The object will be unpinned only once
66  // the scheduled thread has executed to completion. Any last unpinning
67  // of the object will release any possibly pending migration operations
68  // (see step 1a).
69  //
70  // 2) The migration is triggered by invoking the action
71  // `trigger_migration_component_action` on the locality which is
72  // responsible for managing the address resolution for the object which
73  // has to be migrated (locality D).
74  //
75  // The action `trigger_migration_component_action` performs 3 steps:
76  // a) Invoke `agas::begin_migration`, which marks the global id in AGAS,
77  // deferring all address resolution requests until end_migration is
78  // called. Note that the future returned by agas::begin_migration
79  // will become ready only after all currently pending operations on
80  // the target object have finished executing.
81  // b) Invoke the actual migration operation (see step 3)
82  // c) Invoke `agas::end_migration`, which un-marks the global id and
83  // releases all pending address resolution requests. Those requests
84  // now return the new object location (on locality C).
85  //
86  // 3) The actual migration (`migrate_component_action`) is executed on the
87  // locality where the object is currently located (locality B). This
88  // involves several steps as well:
89  // a) Retrieve the (shared-) pointer to the object, this pins the
90  // object. The object is unpinned by the deleter associated with the
91  // shared pointer.
92  // b) Invoke the action `runtime_support::migrate_component` on the
93  // locality where the object has to be moved to. This passes
94  // along the shared pointer to the object and recreates the object
95  // on the target locality and updates the association of the object's
96  // global id with the new local virtual address in AGAS.
97  // c) Mark the old object (through the original shared pointer) as
98  // migrated which will delete it once the shared pointer goes out of
99  // scope.
100  //
101  // The entry in the AGAS client side representation on locality B which
102  // marks the object as 'was migrated' will be left untouched (for now).
103  // This is necessary to allow for all parcels which where still resolved
104  // to the old locality will be properly forwarded to the new location
105  // of the object. Eventually this entry will have to be cleaned up no
106  // later than when the object is destroyed.
107  //
109 
111  // This is step 3 of the migration process
112  //
113  // This will be executed on the locality where the object lives which is
114  // to be migrated

131  std::shared_ptr<Component> const& ptr,
132  id_type const& to_migrate, DistPolicy const& policy)
133  {
134  using components::stubs::runtime_support;
135 
136  std::uint32_t pin_count = ptr->pin_count();
137 
138  if (pin_count == ~0x0u)
139  {
140  return hpx::make_exceptional_future<id_type>(
141  HPX_GET_EXCEPTION(invalid_status,
142  "hpx::components::server::migrate_component",
143  "attempting to migrate an instance of a component "
144  "which was already migrated"));
145  }
146 
147  if (pin_count > 1)
148  {
149  return hpx::make_exceptional_future<id_type>(
150  HPX_GET_EXCEPTION(invalid_status,
151  "hpx::components::server::migrate_component",

164  template <typename Component, typename DistPolicy>
165  future<id_type> migrate_component(
166  id_type const& to_migrate, naming::address const& addr,
167  DistPolicy const& policy)
168  {
169  // 'migration' to same locality as before is a no-op
170  if (policy.get_next_target() == hpx::find_here())
171  {
172  return make_ready_future(to_migrate);
173  }
174 
175  if (!traits::component_supports_migration<Component>::call())
176  {
177  return hpx::make_exceptional_future<hpx::id_type>(
178  HPX_GET_EXCEPTION(invalid_status,
179  "hpx::components::server::migrate_component",
180  "attempting to migrate an instance of a component which "
181  "does not support migration"));
182  }
183 
184  // retrieve pointer to object (must be local)
185  std::shared_ptr<Component> ptr =
186  hpx::detail::get_ptr_for_migration<Component>(addr, to_migrate);
187 
188  // perform actual migration by sending data over to target locality
189  return detail::migrate_component_postproc<Component>(
190  ptr, to_migrate, policy);
191  }

204  //
205  // This is executed on the locality responsible for managing the address

208  future<id_type> trigger_migrate_component(
209  id_type const& to_migrate, DistPolicy const& policy)
210  {
211  if (!traits::component_supports_migration<Component>::call())
212  {
213  return hpx::make_exceptional_future<id_type>(
214  HPX_GET_EXCEPTION(invalid_status,
215  "hpx::components::server::trigger_migrate_component",
216  "attempting to migrate an instance of a component which "
217  "does not support migration"));
218  }
219 
220  if (naming::get_locality_id_from_id(to_migrate) != get_locality_id())
221  {
222  return hpx::make_exceptional_future<id_type>(
223  HPX_GET_EXCEPTION(invalid_status,
224  "hpx::components::server::trigger_migrate_component",
225  "this function has to be executed on the locality "
226  "responsible for managing the address of the given object"));
227  }
228

230  .then(
231  [=](future<std::pair<id_type, naming::address> > && f)

234  // rethrow errors
235  std::pair<id_type, naming::address> r = f.get();

237  // perform actual object migration
238  typedef migrate_component_action<
239  Component, DistPolicy
240  > action_type;

243  })
244  .then(
245  [to_migrate](future<id_type> && f) -> id_type
246  {
247  agas::end_migration(to_migrate).get();
248  return f.get();
249  });

261  // This is step 1 of the migration process
262  //
263  // This is executed on the locality where the object to migrate is
264  // currently located.
265  template <typename Component, typename DistPolicy>
266  future<id_type> perform_migrate_component(
267  id_type const& to_migrate, DistPolicy const& policy)
268  {
269  if (!traits::component_supports_migration<Component>::call())
270  {
271  return hpx::make_exceptional_future<id_type>(
272  HPX_GET_EXCEPTION(invalid_status,
273  "hpx::components::server::perform_migrate_component",
274  "attempting to migrate an instance of a component which "
275  "does not support migration"));
276  }
277 
278  // retrieve pointer to object (must be local)
279  return hpx::get_ptr<Component>(to_migrate)
280  .then(
281  [=](future<std::shared_ptr<Component> > && f) -> future<id_type>
282  {
283  future<void> trigger_migration;
284 
285  {
286  std::shared_ptr<Component> ptr = f.get();
287 
288  // Delay the start of the migration operation until no
289  // more actions (threads) are pending or currently
290  // running for the given object (until the object is
291  // unpinned).
292  trigger_migration = ptr->mark_as_migrated(to_migrate);
293 
294  // Unpin the object, will trigger migration if this is
295  // the only pin-count.
296  }
297 
298  // Once the migration is possible (object is not pinned
299  // anymore trigger the necessary actions)
300  return trigger_migration
301  .then(
302  launch::async, // run on separate thread
303  [=](future<void> && f) -> future<id_type>
304  {
305  f.get(); // rethrow exceptions
306 
307  // now trigger 2nd step of migration
308  typedef trigger_migrate_component_action<
309  Component, DistPolicy
310  > action_type;
311 
312  return async<action_type>(
313  naming::get_locality_from_id(to_migrate),
314  to_migrate, policy);
315  });
316  });
317  }

6 // The purpose of this example is to demonstrate how the partitioned_vector
7 // data structure can be used in an SPMD style way, i.e. how to execute code
8 // on each of the localities the partinioned_vector is located. Each locality
9 // touches only the elements located on this locality. The access to those
10 // elements is done directly, with best possible performance, working directly
11 // on the std::vector's the data is stored in.

34 //
35 // Define a view for a partitioned vector which exposes the part of the vector
36 // which is located on the current locality.
37 //
38 // This view does not own the data and relies on the partitioned_vector to be
39 // available during the full lifetime of the view.
40 //
41 template <typename T>
42 struct partitioned_vector_view
43 {
44 private:
45  typedef typename hpx::partitioned_vector<T>::iterator global_iterator;
46  typedef typename hpx::partitioned_vector<T>::const_iterator
47  const_global_iterator;
48 
49  typedef hpx::traits::segmented_iterator_traits<global_iterator> traits;
50  typedef hpx::traits::segmented_iterator_traits<const_global_iterator>
51  const_traits;
52 
53  typedef typename traits::local_segment_iterator local_segment_iterator;
54 
55 public:
56  typedef typename traits::local_raw_iterator iterator;
57  typedef typename const_traits::local_raw_iterator const_iterator;
58  typedef T value_type;
59 
60 public:
61  explicit partitioned_vector_view(hpx::partitioned_vector<T>& data)
62  : segment_iterator_(data.segment_begin(hpx::get_locality_id()))
63  {
64 #if defined(HPX_DEBUG)
65  // this view assumes that there is exactly one segment per locality
66  typedef typename traits::local_segment_iterator local_segment_iterator;
67  local_segment_iterator sit = segment_iterator_;
68  HPX_ASSERT(++sit == data.segment_end(hpx::get_locality_id()));
69 #endif
70  }

72  iterator begin()
73  {
74  return traits::begin(segment_iterator_);
75  }
76  iterator end()
77  {
78  return traits::end(segment_iterator_);
79  }

81  const_iterator begin() const
82  {
83  return const_traits::begin(segment_iterator_);
84  }
85  const_iterator end() const
86  {
87  return const_traits::end(segment_iterator_);
88  }
89  const_iterator cbegin() const

94  {

99  {
100  return (*segment_iterator_)[index];
101  }

103  {

117 int hpx_main(boost::program_options::variables_map& vm)
118 {
119  unsigned int seed = (unsigned int)std::time(nullptr);
120  if (vm.count("seed"))
121  seed = vm["seed"].as<unsigned int>();
122 
123  std::cout << "using seed: " << seed << std::endl;
124  std::srand(seed);
125 
126  unsigned int size = 10000;
127  if (vm.count("size"))
128  size = vm["size"].as<unsigned int>();
129 
130  char const* const example_vector_name =
131  "partitioned_vector_spmd_foreach_example";
132  char const* const example_latch_name =
133  "latch_spmd_foreach_example";
134 
135  {
136  // create vector on one locality, connect to it from all others
137  hpx::partitioned_vector<int> v;
138  hpx::lcos::latch l;
139 
140  if (0 == hpx::get_locality_id())
141  {
142  std::vector<hpx::id_type> localities = hpx::find_all_localities();
143 
144  v = hpx::partitioned_vector<int>(size,
145  hpx::container_layout(localities));
146  v.register_as(example_vector_name);
147 
148  l = hpx::lcos::latch(localities.size());
149  l.register_as(example_latch_name);
150  }
151  else
152  {
153  hpx::future<void> f1 = v.connect_to(example_vector_name);
154  l.connect_to(example_latch_name);
155  f1.get();
156  }
157 
158  std::mt19937 gen(std::rand());
159  std::uniform_int_distribution<> dist;
160 
161  // fill the vector with random numbers
162  partitioned_vector_view<int> view(v);
163  hpx::parallel::generate(
164  hpx::parallel::execution::par,
165  view.begin(), view.end(),
166  [&]()
167  {
168  return dist(gen);
169  });
170 
171  // square all numbers in the array
172  hpx::parallel::for_each(
173  hpx::parallel::execution::par,
174  view.begin(), view.end(),
175  [](int& val)
176  {
177  val *= val;
178  });
179 
180  // do the same using a plain loop
181  std::size_t maxnum = view.size();
182  for (std::size_t i = 0; i != maxnum; ++i)
183  view[i] *= 2;
184 
185  // Wait for all localities to reach this point.
186  l.count_down_and_wait();
187  }
188 
189  return hpx::finalize();
190 }

92  void barrier::release()
93  {
94  if (node_)
95  {
96  if (hpx::get_runtime_ptr() != nullptr &&
97  hpx::threads::threadmanager_is(state_running) &&
98  !hpx::is_stopped_or_shutting_down())
99  {
100  hpx::future<void> f;
101  if ((*node_)->num_ >= (*node_)->cut_off_ || (*node_)->rank_ == 0)
102  f = hpx::unregister_with_basename(
103  (*node_)->base_name_, (*node_)->rank_);
104 
105  // we need to wait on everyone to have its name unregistered,
106  // and hold on to our node long enough...
107  boost::intrusive_ptr<wrapping_type> node = node_;
108  hpx::when_all(f, wait(hpx::launch::async)).then(
109  hpx::launch::sync,
110  [node](hpx::future<void> f)
111  {
112  HPX_UNUSED(node);
113  f.get();
114  }
115  ).get();
116  }
117  node_.reset();
118  }
119  }

483  template <typename ConnectionHandler_>
484  typename std::enable_if<
485  connection_handler_traits<
486  ConnectionHandler_
487  >::send_immediate_parcels::value,
488  void
489  >::type
490  send_immediate_impl(
491  parcelport_impl &this_, locality const&dest_,
492  write_handler_type *fs, parcel *ps, std::size_t num_parcels)
493  {
494  std::uint64_t addr;
495  error_code ec;
496  // First try to get a connection ...
497  connection *sender = this_.connection_handler().get_connection(dest_, addr);
498 
499  // If we couldn't get one ... enqueue the parcel and move on
500  std::size_t encoded_parcels = 0;
501  std::vector<parcel> parcels;
502  std::vector<write_handler_type> handlers;
503  if (sender != nullptr)
504  {
505  if (fs == nullptr)
506  {
507  HPX_ASSERT(ps == nullptr);
508  HPX_ASSERT(num_parcels == 0u);
509 
510  if(!dequeue_parcels(dest_, parcels, handlers))
511  {
512  // Give this connection back to the connection handler as
513  // we couldn't dequeue parcels.
514  this_.connection_handler().reclaim_connection(sender);
515 
516  return;
517  }
518 
519  ps = parcels.data();
520  fs = handlers.data();
521  num_parcels = parcels.size();
522  HPX_ASSERT(parcels.size() == handlers.size());
523  }
524 
525  auto encoded_buffer = sender->get_new_buffer();
526  // encode the parcels
527  encoded_parcels = encode_parcels(this_, ps, num_parcels,
528  encoded_buffer,
529  this_.archive_flags_,
530  this_.get_max_outbound_message_size());
531 
532  typedef detail::call_for_each handler_type;
533 
534  if (sender->parcelport_->async_write(
535  handler_type(
536  handler_type::handlers_type(
537  std::make_move_iterator(fs),
538  std::make_move_iterator(fs + encoded_parcels)),
539  handler_type::parcels_type(
540  std::make_move_iterator(ps),
541  std::make_move_iterator(ps + encoded_parcels))
542  ),
543  sender, addr,
544  encoded_buffer))
545  {
546  // we don't propagate errors for now
547  }
548  }
549  if (num_parcels != encoded_parcels && fs != nullptr)
550  {
551  std::vector<parcel> overflow_parcels(
552  std::make_move_iterator(ps + encoded_parcels),
553  std::make_move_iterator(ps + num_parcels));
554  std::vector<write_handler_type> overflow_handlers(
555  std::make_move_iterator(fs + encoded_parcels),
556  std::make_move_iterator(fs + num_parcels));
557  enqueue_parcels(dest_, std::move(overflow_parcels),
558  std::move(overflow_handlers));
559  }
560  }

578  std::shared_ptr<connection> get_connection(
579  locality const& l, bool force, error_code& ec)
580  {
581  // Request new connection from connection cache.
582  std::shared_ptr<connection> sender_connection;
583 
584  if (connection_handler_traits<ConnectionHandler>::
585  send_immediate_parcels::value)

591  if (!connection_cache_.get_or_reserve(l, sender_connection))

595  // to the cache.

603  if (!sender_connection)

613  void enqueue_parcel(locality const& locality_id,
614  parcel&& p, write_handler_type&& f)
615  {
616  typedef pending_parcels_map::mapped_type mapped_type;
617 
618  std::unique_lock<lcos::local::spinlock> l(mtx_);
619  // We ignore the lock here. It might happen that while enqueuing,
620  // we need to acquire a lock. This should not cause any problems
621  // (famous last words)
622  util::ignore_while_checking<
623  std::unique_lock<lcos::local::spinlock>
624  > il(&l);
625 
626  mapped_type& e = pending_parcels_[locality_id];
627  util::get<0>(e).push_back(std::move(p));
628  util::get<1>(e).push_back(std::move(f));
629 
630  parcel_destinations_.insert(locality_id);
631  ++num_parcel_destinations_;
632  }
633 
634  void enqueue_parcels(locality const& locality_id,
635  std::vector<parcel>&& parcels,
636  std::vector<write_handler_type>&& handlers)
637  {
638  typedef pending_parcels_map::mapped_type mapped_type;
639 
640  std::unique_lock<lcos::local::spinlock> l(mtx_);
641  // We ignore the lock here. It might happen that while enqueuing,
642  // we need to acquire a lock. This should not cause any problems
643  // (famous last words)
644  util::ignore_while_checking<
645  std::unique_lock<lcos::local::spinlock>
646  > il(&l);
647 
648  HPX_ASSERT(parcels.size() == handlers.size());
649 
650  mapped_type& e = pending_parcels_[locality_id];
651  if (util::get<0>(e).empty())
652  {
653  HPX_ASSERT(util::get<1>(e).empty());
654  std::swap(util::get<0>(e), parcels);
655  std::swap(util::get<1>(e), handlers);
656  }
657  else
658  {
659  HPX_ASSERT(util::get<0>(e).size() == util::get<1>(e).size());
660  std::size_t new_size = util::get<0>(e).size() + parcels.size();
661  util::get<0>(e).reserve(new_size);
662 
663  std::move(parcels.begin(), parcels.end(),
664  std::back_inserter(util::get<0>(e)));
665  util::get<1>(e).reserve(new_size);
666  std::move(handlers.begin(), handlers.end(),
667  std::back_inserter(util::get<1>(e)));
668  }
669 
670  parcel_destinations_.insert(locality_id);
671  ++num_parcel_destinations_;
672  }

674  bool dequeue_parcels(locality const& locality_id,
675  std::vector<parcel>& parcels,
676  std::vector<write_handler_type>& handlers)
677  {
678  typedef pending_parcels_map::iterator iterator;
679 
680  {
681  std::unique_lock<lcos::local::spinlock> l(mtx_, std::try_to_lock);
682 
683  if (!l) return false;
684 
685  iterator it = pending_parcels_.find(locality_id);
686 
687  // do nothing if parcels have already been picked up by
688  // another thread
689  if (it != pending_parcels_.end() && !util::get<0>(it->second).empty())
690  {
691  HPX_ASSERT(it->first == locality_id);
692  HPX_ASSERT(handlers.size() == 0);
693  HPX_ASSERT(handlers.size() == parcels.size());
694  std::swap(parcels, util::get<0>(it->second));
695  HPX_ASSERT(util::get<0>(it->second).size() == 0);
696  std::swap(handlers, util::get<1>(it->second));
697  HPX_ASSERT(handlers.size() == parcels.size());
698 
699  HPX_ASSERT(!handlers.empty());
700  }
701  else
702  {
703  HPX_ASSERT(util::get<1>(it->second).empty());
704  return false;
705  }
706 
707  parcel_destinations_.erase(locality_id);
708 
709  HPX_ASSERT(0 != num_parcel_destinations_.load());
710  --num_parcel_destinations_;
711 
712  return true;
713  }
714  }

716  protected:
717  bool dequeue_parcel(locality& dest, parcel& p, write_handler_type& handler)
718  {
719  {
720  std::unique_lock<lcos::local::spinlock> l(mtx_, std::try_to_lock);

781  void get_connection_and_send_parcels(
782  locality const& locality_id, bool background = false)
783  {
784

787  {

789  *this, locality_id, nullptr, nullptr, 0);

791  }
792

794  // need to force a new connection to avoid deadlocks.
795  bool force_connection = true;
796

799  get_connection(locality_id, force_connection, ec);
800 
801  if (!sender_connection)

804  // at this point. As soon as a connection becomes
805  // available it checks for pending parcels and sends
806  // those out.
807  return;
808  }

811  std::vector<parcel> parcels;
812  std::vector<write_handler_type> handlers;
813 
814  if(!dequeue_parcels(locality_id, parcels, handlers))
815  {
816  // Give this connection back to the cache as we couldn't dequeue
817  // parcels.
818  connection_cache_.reclaim(locality_id, sender_connection);
819 
820  return;
821  }

824  send_pending_parcels(
825  locality_id,

827  std::move(handlers));

830  // HPX thread the chance to put a subsequent parcel which
831  // leads to a more effective parcel buffering
832 // if (hpx::threads::get_self_ptr())
833 // hpx::this_thread::yield();

839  locality const& locality_id,

843  --operations_in_flight_;
844 
845 #if defined(HPX_TRACK_STATE_OF_OUTGOING_TCP_CONNECTION)

847 #endif

849  {

853  }

865  return;

873  void send_pending_parcels(
874  parcelset::locality const & parcel_locality_id,
875  std::shared_ptr<connection> sender_connection,
876  std::vector<parcel>&& parcels,
877  std::vector<write_handler_type>&& handlers)
878  {
879 #if defined(HPX_TRACK_STATE_OF_OUTGOING_TCP_CONNECTION)
880  sender_connection->set_state(connection::state_send_pending);
881 #endif
882 
883 #if defined(HPX_DEBUG)

885 // HPX_ASSERT(parcel_locality_id == sender_connection->destination());
886  sender_connection->verify(parcel_locality_id);
887 #endif

889  std::size_t num_parcels = encode_parcels(*this, &parcels[0],
890  parcels.size(), sender_connection->buffer_,
891  archive_flags_,
892  this->get_max_outbound_message_size());
893

895  using hpx::util::placeholders::_1;
896  using hpx::util::placeholders::_2;
897  using hpx::util::placeholders::_3;
898  if (num_parcels == parcels.size())
899  {
900  ++operations_in_flight_;
901  // send all of the parcels
902  sender_connection->async_write(
903  call_for_each(std::move(handlers), std::move(parcels)),
904  util::bind(&parcelport_impl::send_pending_parcels_trampoline,
905  this, _1, _2, _3));
906  }
907  else
908  {
909  ++operations_in_flight_;
910  HPX_ASSERT(num_parcels < parcels.size());
911

913  handled_handlers.reserve(num_parcels);
914 
915  std::move(handlers.begin(), handlers.begin()+num_parcels,
916  std::back_inserter(handled_handlers));
917

919  handled_parcels.reserve(num_parcels);
920 
921  std::move(parcels.begin(), parcels.begin()+num_parcels,
922  std::back_inserter(handled_parcels));
923 
924  // send only part of the parcels
925  sender_connection->async_write(
926  call_for_each(
927  std::move(handled_handlers), std::move(handled_parcels)),
928  util::bind(&parcelport_impl::send_pending_parcels_trampoline,
929  this, _1, _2, _3));
930 
931  // give back unhandled parcels
932  parcels.erase(parcels.begin(), parcels.begin()+num_parcels);
933  handlers.erase(handlers.begin(), handlers.begin()+num_parcels);
934 
935  enqueue_parcels(parcel_locality_id, std::move(parcels),
936  std::move(handlers));
937  }
938 
939  if(threads::get_self_ptr())
940  {
941  // We suspend our thread, which will make progress on the network
942  hpx::this_thread::suspend(hpx::threads::pending_boost,
943  "parcelport_impl::send_pending_parcels");
944  }
945  }

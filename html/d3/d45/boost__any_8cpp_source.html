233  original_ptr, "address in pre-swapped original");
234  HPX_TEST_NEQ_MSG(
235  original_ptr,
236  any_cast<small_object>(&swapped),
237  "comparing address in swapped against original");
238  HPX_TEST_EQ_MSG(swap_result, &original, "address of swap result");

43 double benchmark_serialization(std::size_t data_size, std::size_t iterations,
44  bool continuation, bool zerocopy)
45 {
46  hpx::naming::id_type const here = hpx::find_here();
47  hpx::naming::address addr(hpx::get_locality(),
48  hpx::components::component_invalid,
49  reinterpret_cast<std::uint64_t>(&test_function));
50 
51  // compose archive flags
52 #ifdef BOOST_BIG_ENDIAN
53  std::string endian_out =
54  hpx::get_config_entry("hpx.parcel.endian_out", "big");
55 #else
56  std::string endian_out =
57  hpx::get_config_entry("hpx.parcel.endian_out", "little");
58 #endif
59 
60  unsigned int out_archive_flags = 0U;
61  if (endian_out == "little")
62  out_archive_flags |= hpx::serialization::endian_little;
63  else if (endian_out == "big")
64  out_archive_flags |= hpx::serialization::endian_big;
65  else {
66  HPX_ASSERT(endian_out =="little" || endian_out == "big");
67  }
68 
69  std::string array_optimization =
70  hpx::get_config_entry("hpx.parcel.array_optimization", "1");
71 
72  if (boost::lexical_cast<int>(array_optimization) == 0)
73  {
74  out_archive_flags |= hpx::serialization::disable_array_optimization;
75  out_archive_flags |= hpx::serialization::disable_data_chunking;
76  }
77  else
78  {
79  std::string zero_copy_optimization =
80  hpx::get_config_entry("hpx.parcel.zero_copy_optimization", "1");
81  if (!zerocopy || boost::lexical_cast<int>(zero_copy_optimization) == 0)
82  {
83  out_archive_flags |= hpx::serialization::disable_data_chunking;
84  }
85  }
86 
87  // create argument for action
88  std::vector<double> data;
89  data.resize(data_size);
90 
91  hpx::serialization::serialize_buffer<double> buffer(data.data(), data.size(),
92  hpx::serialization::serialize_buffer<double>::reference);
93 
94  // create a parcel with/without continuation
95  hpx::parcelset::parcel outp;
96  hpx::naming::gid_type dest = here.get_gid();
97  if (continuation) {
98  outp = hpx::parcelset::parcel(hpx::parcelset::detail::create_parcel::call(
99  std::true_type(),
100  std::move(dest), std::move(addr),
101  hpx::actions::typed_continuation<int>(here),
102  test_action(), hpx::threads::thread_priority_normal, buffer
103  ));
104  }
105  else {
106  outp = hpx::parcelset::parcel(hpx::parcelset::detail::create_parcel::call(
107  std::false_type(),
108  std::move(dest), std::move(addr),
109  test_action(), hpx::threads::thread_priority_normal, buffer));
110  }
111 
112  outp.set_source_id(here);
113 
114  std::vector<hpx::serialization::serialization_chunk>* chunks = nullptr;
115  if (zerocopy)
116  chunks = new std::vector<hpx::serialization::serialization_chunk>();
117 
118  //std::uint32_t dest_locality_id = outp.destination_locality_id();
119  hpx::util::high_resolution_timer t;
120 
121  for (std::size_t i = 0; i != iterations; ++i)
122  {
123  std::size_t arg_size = get_archive_size(outp, out_archive_flags, chunks);
124  std::vector<char> out_buffer;
125 
126  out_buffer.resize(arg_size + HPX_PARCEL_SERIALIZATION_OVERHEAD);
127 
128  {
129  // create an output archive and serialize the parcel
130  hpx::serialization::output_archive archive(
131  out_buffer, out_archive_flags, chunks);
132  archive << outp;

137 
138  {
139  // create an input archive and deserialize the parcel
140  hpx::serialization::input_archive archive(
141  out_buffer, arg_size, chunks);
142 
143  archive >> inp;
144  }
145 
146  if (chunks)
147  chunks->clear();
148  }
149 
150  return t.elapsed();
151 }

158 int hpx_main(boost::program_options::variables_map& vm)
159 {
160  bool print_header = vm.count("no-header") == 0;
161  bool continuation = vm.count("continuation") != 0;
162  bool zerocopy = vm.count("zerocopy") != 0;
163 
164  std::vector<hpx::future<double> > timings;
165  for (std::size_t i = 0; i != concurrency; ++i)
166  {
167  timings.push_back(hpx::async(
168  &benchmark_serialization, data_size, iterations,
169  continuation, zerocopy));
170  }
171 
172  double overall_time = 0;
173  for (std::size_t i = 0; i != concurrency; ++i)
174  overall_time += timings[i].get();
175 
176  if (print_header)
177  hpx::cout << "datasize,testcount,average_time[s]\n" << hpx::flush;
178 
179  hpx::cout << (boost::format("%d,%d,%f\n") %
180  data_size % iterations % (overall_time / concurrency)) << hpx::flush;
181 
182  return hpx::finalize();
183 }

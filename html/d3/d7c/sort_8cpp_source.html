23 // this function times a sort and outputs the time for cDash to plot it
24 void sort_benchmark()
25 {
26  try {
27  using namespace hpx::parallel;
28  // Fill vector with random values
29  std::vector<double> c(HPX_SORT_TEST_SIZE << 4);
30  rnd_fill<double>(c, (std::numeric_limits<double>::min)(),
31  (std::numeric_limits<double>::max)(), double(std::rand()));
32 
33  hpx::util::high_resolution_timer t;
34  // sort, blocking when seq, par, par_vec
35  hpx::parallel::sort(execution::par, c.begin(), c.end());
36  double elapsed = t.elapsed();
37 
38  bool is_sorted = (verify(c, std::less<double>(), elapsed, true)!=0);
39  HPX_TEST(is_sorted);
40  if (is_sorted) {
41  std::cout << "<DartMeasurement name=\"SortDoublesTime\" \n"
42  << "type=\"numeric/double\">" << elapsed << "</DartMeasurement> \n";
43  }
44  }
45  catch (...) {
46  HPX_TEST(false);
47  }
48 }

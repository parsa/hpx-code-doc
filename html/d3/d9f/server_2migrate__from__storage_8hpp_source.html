28  // Migrate given component from the specified storage component
29  //
30  // Object migration is performed from the storage in several steps:
31  //
32  // 1) The migration is triggered by invoking the
33  // trigger_migrate_from_storage_here_action on the locality which is
34  // responsible for managing the address resolution for the object which
35  // has to be migrated.
36  // 2) The trigger_migrate_from_storage_here_action performs 3 steps:
37  // a) Invoke agas::begin_migration, which marks the global id in AGAS,
38  // deferring all address resolution requests until end_migration is
39  // called.
40  // b) Invoke the actual migration operation (see step 3)
41  // c) Invoke end_migration, which un-marks the global id and releases
42  // all pending address resolution requests. Those requests now return
43  // the new object location.
44  // 3) The actual migration (component_storage::migrate_from_here_action)
45  // is executed on the storage facility where the object is currently
46  // stored. This involves several steps as well:
47  // a) Retrieve the byte stream representing the object from the storage
48  // b) Deserialize the byte stream to re-create the object. The newly
49  // recreated object is pinned immediately. The object is unpinned by
50  // the deleter associated with the shared pointer.
51  // c) Invoke the action runtime_support::migrate_component on the
52  // locality where the object has to be moved to. This passes
53  // along the shared pointer to the object and recreates the object
54  // on the target locality and updates the association of the object's
55  // global id with the new local virtual address in AGAS.
56  // d) Mark the old object (through the original shared pointer) as
57  // migrated which will delete it once the shared pointer goes out of
58  // scope.
59  //
60  namespace detail

63  template <typename Component>
64  future<naming::id_type> migrate_from_storage_here_id(
65  naming::id_type const& target_locality,
66  std::shared_ptr<Component> const& ptr,
67  naming::id_type const& to_resurrect)
68  {
69  // and resurrect it on the specified locality
70  using hpx::components::runtime_support;
71  return runtime_support::migrate_component_async<Component>(
72  target_locality, ptr, to_resurrect)
73  .then(util::bind(
74  &detail::migrate_component_cleanup<Component>,
75  util::placeholders::_1, ptr, to_resurrect));

88  // convert the extracted data into a living component instance
89  template <typename Component>
90  future<naming::id_type> migrate_from_storage_here(
91  future<std::vector<char> > && f,
92  naming::id_type const& to_resurrect,
93  naming::address const& addr,
94  naming::id_type const& target_locality)
95  {
96  // recreate the object
97  std::shared_ptr<Component> ptr;
98 
99  {
100  std::vector<char> data = f.get();
101  serialization::input_archive archive(data, data.size(), nullptr);
102  archive >> ptr;
103  }
104 
105  // make sure the migration code works properly
106  ptr->pin();
107 
108  // if target locality is not specified, use the address of the last
109  // locality where the object was living before
110  if (target_locality == naming::invalid_id)
111  {
112  return migrate_from_storage_here_address<Component>(addr, ptr,
113  to_resurrect);
114  }
115 
116  // otherwise directly refer to the locality where the object should
117  // be resurrected
118  return migrate_from_storage_here_id(target_locality, ptr,
119  to_resurrect);
120  }

124  // This is executed on the locality responsible for managing the address
125  // resolution for the given object.
126  template <typename Component>
127  future<naming::id_type> trigger_migrate_from_storage_here(
128  naming::id_type const& to_resurrect,
129  naming::id_type const& target_locality)
130  {
131  if (!traits::component_supports_migration<Component>::call())
132  {
133  HPX_THROW_EXCEPTION(invalid_status,
134  "hpx::components::server::trigger_migrate_from_storage_here",
135  "attempting to migrate an instance of a component which "
136  "does not support migration");
137  return make_ready_future(naming::invalid_id);
138  }
139 
140  if (naming::get_locality_id_from_id(to_resurrect) != get_locality_id())
141  {
142  HPX_THROW_EXCEPTION(invalid_status,
143  "hpx::components::server::trigger_migrate_from_storage_here",
144  "this function has to be executed on the locality responsible "
145  "for managing the address of the given object");
146  return make_ready_future(naming::invalid_id);
147  }
148 
149  return agas::begin_migration(to_resurrect)
150  .then(
151  [to_resurrect, target_locality](
152  future<std::pair<naming::id_type, naming::address> > && f)
153  -> future<naming::id_type>
154  {
155  // rethrow errors
156  std::pair<naming::id_type, naming::address> r = f.get();
157 
158  // retrieve the data from the given storage
159  typedef typename server::component_storage::migrate_from_here_action
160  action_type;
161  return async<action_type>(r.first, to_resurrect.get_gid())
162  .then(util::bind(
163  &detail::migrate_from_storage_here<Component>,
164  util::placeholders::_1, to_resurrect,
165  r.second, target_locality));
166  })
167  .then(
168  [to_resurrect](future<naming::id_type> && f) -> naming::id_type
169  {
170  agas::end_migration(to_resurrect).get();
171  return f.get();
172  });
173  }

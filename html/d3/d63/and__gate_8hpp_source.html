121  template <typename OuterLock>
122  bool set(std::size_t which, OuterLock & outer_lock, error_code& ec = throws)
123  {
124  std::unique_lock<mutex_type> l(mtx_);
125  if (which >= received_segments_.size())
126  {
127  // out of bounds index
128  l.unlock();
129  HPX_THROWS_IF(ec, bad_parameter, "base_and_gate<>::set",
130  "index is out of range for this base_and_gate");
131  return false;
132  }
133  if (received_segments_.test(which))
134  {
135  // segment already filled, logic error
136  l.unlock();
137  HPX_THROWS_IF(ec, bad_parameter, "base_and_gate<>::set",
138  "input with the given index has already been triggered");
139  return false;
140  }
141 
142  if (&ec != &throws)
143  ec = make_success_code();
144 
145  // set the corresponding bit
146  received_segments_.set(which);
147 
148  if (received_segments_.count() == received_segments_.size())
149  {
150  // we have received the last missing segment
151  promise<void> p;
152  std::swap(p, promise_);
153  received_segments_.reset(); // reset data store
154  {
155  // Unlock the lock to avoid locking problems
156  // when triggering the promise
157  l.unlock();
158  outer_lock.unlock();
159  p.set_value(); // fire event
160  return true;
161  }
162  }
163 
164  return false;
165  }

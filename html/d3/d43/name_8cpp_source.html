42 //
43 // Here is how our distributed garbage collection works
44 //
45 // Each id_type instance - while always referring to some (possibly remote)
46 // entity - can either be 'managed' or 'unmanaged'. If an id_type instance is
47 // 'unmanaged' it does not perform any garbage collection. Otherwise (if it's
48 // 'managed'), all of its copies are globally tracked which allows to
49 // automatically delete the entity a particular id_type instance is referring
50 // to after the last reference to it goes out of scope.
51 //
52 // An id_type instance is essentially a shared_ptr<> maintaining two reference
53 // counts: a local reference count and a global one. The local reference count
54 // is incremented whenever the id_type instance is copied locally, and decremented
55 // whenever one of the local copies goes out of scope. At the point when the last
56 // local copy goes out of scope, it returns its current share of the global
57 // reference count back to AGAS. The share of the global reference count owned
58 // by all copies of an id_type instance on a single locality is called its
59 // credit. Credits are issued in chunks which allows to create a global copy
60 // of an id_type instance (like passing it to another locality) without needing
61 // to talk to AGAS to request a global reference count increment. The referenced
62 // entity is freed when the global reference count falls to zero.
63 //
64 // Any newly created object assumes an initial credit. This credit is not
65 // accounted for by AGAS as long as no global increment or decrement requests
66 // are received. It is important to understand that there is no way to distinguish
67 // whether an object has already been deleted (and therefore no entry exists in
68 // the table storing the global reference count for this object) or whether the
69 // object is still alive but no increment/decrement requests have been received
70 // by AGAS yet. While this is a pure optimization to avoid storing global
71 // reference counts for all objects, it has implications for the implemented
72 // garbage collection algorithms at large.
73 //
74 // As long as an id_type instance is not sent to another locality (a locality
75 // different from the initial locality creating the referenced entity), all
76 // lifetime management for this entity can be handled purely local without
77 // even talking to AGAS.
78 //
79 // Sending an id_type instance to another locality (which includes using an
80 // id_type as the destination for an action) splits the current credit into
81 // two parts. One part stays with the id_type on the sending locality, the
82 // other part is sent along to the destination locality where it turns into the
83 // global credit associated with the remote copy of the id_type. As stated
84 // above, this allows to avoid talking to AGAS for incrementing the global
85 // reference count as long as there is sufficient global credit left in order
86 // to be split.
87 //
88 // The current share of the global credit associated with an id_type instance
89 // is encoded in the bits 88..92 of the underlying gid_type (encoded as the
90 // logarithm to the base 2 of the credit value). Bit 94 is a flag which is set
91 // whenever the credit is valid. Bit 95 encodes whether the given id_type
92 // has been split at any time. This information is needed to be able to decide
93 // whether a garbage collection can be assumed to be a purely local operation.
94 // Bit 93 is used by the locking scheme for gid_types.
95 //
96 // Credit splitting is performed without any additional AGAS traffic as long as
97 // sufficient credit is available. If the credit of the id_type to be split is
98 // exhausted (reaches the value '1') it has to be replenished. This operation
99 // is performed synchronously. This is done to ensure that AGAS has accounted
100 // for the requested credit increase.
101 //
102 // Note that both the id_type instance staying behind and the one sent along
103 // are replenished before sending out the parcel at the sending locality.
104 //

119  void decrement_refcnt(detail::id_type_impl* p)
120  {
121  // do nothing if it's too late in the game
122  if (!get_runtime_ptr())
123  {
124  delete p; // delete local gid representation in any case
125  return;
126  }
127 
128  // Talk to AGAS only if this gid was split at some time in the past,
129  // i.e. if a reference actually left the original locality.
130  // Alternatively we need to go this way if the id has never been
131  // resolved, which means we don't know anything about the component
132  // type.
133  naming::address addr;
134  if ((gid_was_split(*p) ||
135  !naming::get_agas_client().resolve_cached(*p, addr)))
136  {
137  // guard for wait_abort and other shutdown issues
138  try {
139  // decrement global reference count for the given gid,
140  std::int64_t credits = detail::get_credit_from_gid(*p);
141  HPX_ASSERT(0 != credits);
142 
143  if (get_runtime_ptr())
144  {
145  // Fire-and-forget semantics.
146  error_code ec(lightweight);
147  agas::decref(*p, credits, ec);
148  }
149  }
150  catch (hpx::exception const& e) {
151  LTM_(error)
152  << "Unhandled exception while executing decrement_refcnt:"
153  << e.what();
154  }
155  }
156  else {
157  // If the gid was not split at any point in time we can assume
158  // that the referenced object is fully local.
159  HPX_ASSERT(addr.type_ != components::component_invalid);
160 
161  // Third parameter is the count of how many components to destroy.
162  // FIXME: The address should still be in the cache, but it could
163  // be evicted. It would be nice to have a way to pass the address
164  // directly to free_component_sync.
165  try {
166  using components::stubs::runtime_support;
167  agas::gva g (addr.locality_, addr.type_, 1, addr.address_);
168  runtime_support::free_component_sync(g, *p);
169  }
170  catch (hpx::exception const& e) {
171  // This request might come in too late and the thread manager
172  // was already stopped. We ignore the request if that's the
173  // case.
174  if (e.get_error() != invalid_status) {
175  throw; // rethrow if not invalid_status
176  }
177  else if (!threads::threadmanager_is(hpx::state_stopping)) {
178  throw; // rethrow if not stopping
179  }
180  }
181  }
182  delete p; // delete local gid representation in any case
183  }

228  {

266  gid_type postprocess_incref(gid_type &gid)
267  {
268  typedef std::unique_lock<gid_type::mutex_type> scoped_lock;
269  scoped_lock l(gid.get_mutex());
270 
271  gid_type new_gid = gid; // strips lock-bit
272  HPX_ASSERT(new_gid != invalid_gid);
273 
274  // The old gid should have been marked as been split below
275  HPX_ASSERT(gid_was_split(gid));
276 
277  // Fill the new gid with our new credit and mark it as being split
278  naming::detail::set_credit_for_gid(new_gid,
279  static_cast<std::int64_t>(HPX_GLOBALCREDIT_INITIAL));
280  set_credit_split_mask_for_gid(new_gid);
281 
282  // Another concurrent split operation might have happened
283  // concurrently, we need to add the new split credits to the old
284  // and account for overflow.
285 
286  // Get the current credit for our gid. If no other concurrent
287  // split has happened since we invoked incref below, the credit
288  // of this gid is equal to 2, otherwise it is larger.
289  std::int64_t src_credit = get_credit_from_gid(gid);
290  HPX_ASSERT(src_credit >= 2);
291 
292  std::int64_t split_credit =
293  static_cast<std::int64_t>(HPX_GLOBALCREDIT_INITIAL) - 2;
294  std::int64_t new_credit = src_credit + split_credit;
295  std::int64_t overflow_credit = new_credit -
296  static_cast<std::int64_t>(HPX_GLOBALCREDIT_INITIAL);
297 
298  new_credit
299  = (std::min)(
300  static_cast<std::int64_t>(HPX_GLOBALCREDIT_INITIAL),
301  new_credit);
302  naming::detail::set_credit_for_gid(gid, new_credit);
303 
304  // Account for a possible overflow ...
305  if(overflow_credit > 0)
306  {
307  HPX_ASSERT(overflow_credit <= HPX_GLOBALCREDIT_INITIAL-1);
308  l.unlock();
309 
310  // Note that this operation may be asynchronous
311  agas::decref(new_gid, overflow_credit);
312  }
313 
314  return new_gid;
315  }

317  hpx::future<gid_type> split_gid_if_needed_locked(
318  std::unique_lock<gid_type::mutex_type> &l, gid_type& gid)
319  {
320  HPX_ASSERT_OWNS_LOCK(l);
321 
322  if (naming::detail::has_credits(gid))
323  {
324  // The splitting is happening in two parts:
325  // First get the current credit and split it:
326  // Case 1: credit == 1 ==> we need to request new credit from
327  // AGAS. This is happening asynchronously.
328  // Case 2: credit != 1 ==> Just fill with new credit
329  //
330  // Scenario that might happen:
331  // An id_type which needs to be split is being split concurrently
332  // while we unlock the lock to ask for more credit:
333  // This might lead to an overflow in the credit mask and
334  // needs to be accounted for by sending a decref with the
335  // excessive credit.
336  //
337  // An early decref can't happen as the id_type with the new
338  // credit is guaranteed to arrive only after we incremented the
339  // credit successfully in agas.
340  HPX_ASSERT(get_log2credit_from_gid(gid) > 0);
341  std::int16_t src_log2credits = get_log2credit_from_gid(gid);
342 
343  // Credit exhaustion - we need to get more.
344  if(src_log2credits == 1)
345  {
346  // mark gid as being split
347  set_credit_split_mask_for_gid(gid);
348 
349  l.unlock();
350 
351  // We add HPX_GLOBALCREDIT_INITIAL credits for the new gid
352  // and HPX_GLOBALCREDIT_INITIAL - 2 for the old one.
353  std::int64_t new_credit = 2 *
354  (static_cast<std::int64_t>(HPX_GLOBALCREDIT_INITIAL) - 1);
355 
356  naming::gid_type new_gid = gid; // strips lock-bit
357  HPX_ASSERT(new_gid != invalid_gid);
358  return agas::incref(new_gid, new_credit)
359  .then(
360  hpx::util::bind(postprocess_incref, std::ref(gid))
361  );
362  }
363 
364  HPX_ASSERT(src_log2credits > 1);
365 
366  naming::gid_type new_gid = split_credits_for_gid_locked(l, gid);
367 
368  HPX_ASSERT(detail::has_credits(gid));
369  HPX_ASSERT(detail::has_credits(new_gid));
370 
371  return hpx::make_ready_future(new_gid);
372  }
373 
374  naming::gid_type new_gid = gid; // strips lock-bit
375  return hpx::make_ready_future(new_gid);
376  }

435  HPX_EXPORT std::int64_t replenish_credits_locked(
436  std::unique_lock<gid_type::mutex_type>& l, gid_type& gid)
437  {
438  std::int64_t added_credit = 0;
439 
440  HPX_ASSERT(0 == get_credit_from_gid(gid));
441 
442  added_credit = naming::detail::fill_credit_for_gid(gid);
443  naming::detail::set_credit_split_mask_for_gid(gid);
444 
445  gid_type unlocked_gid = gid; // strips lock-bit
446 
447  std::int64_t result = 0;
448  {
449  hpx::util::scoped_unlock<std::unique_lock<gid_type::mutex_type> >
450  ul(l);
451  result = agas::incref(launch::sync, unlocked_gid, added_credit);
452  }
453 
454  return result;
455  }

42 template <typename T>
43 inline std::size_t partition(T* data, std::size_t begin, std::size_t end)
44 {
45  T* first = data + begin;
46  T* last = first + (end - begin);
47 
48  T* middle = std::partition(
49  first, last, std::bind2nd(std::less<T>(), *first));
50 
51  return middle - data;
52 }

55 template <typename T>
56 struct quicksort_serial
57 {
58  static std::size_t sort_count;
59 
60  static void call(T* data, std::size_t begin, std::size_t end)
61  {
62  if (begin != end)
63  {
64  std::size_t middle_idx = partition(data, begin, end);
65 
66  ++sort_count;
67 
68  // always spawn the larger part in a new thread
69  if (2 * middle_idx < end - begin)
70  {
71  call(data, (std::max)(begin + 1, middle_idx), end);
72  call(data, begin, middle_idx);
73  }
74 
75  else
76  {
77  call(data, begin, middle_idx);
78  call(data, (std::max)(begin + 1, middle_idx), end);
79  }
80  }
81  }
82 };

104  std::size_t end)
105 {
106  if (begin != end)
107  {

111  std::size_t middle_idx = partition(data.get_ptr(), begin, end);
112 
113  ++sort_count;
114 
115  // always spawn the larger part in a new thread
116  if (2 * middle_idx < end - begin)

123  }
124 
125  else

127  future<void> n = async<quicksort_int_action>(prefix, prefix, d,
128  begin, middle_idx);

130  call(prefix, d, (std::max)(begin + 1, middle_idx), end);

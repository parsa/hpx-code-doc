29 /*
30  * Based on "Spinlocks and Read-Write Locks" by Dr. Steven Fuerst:
31  * http://locklessinc.com/articles/locks/
32  *
33  * Dr. Fuerst further credits:
34  * There exists a form of the ticket lock that is designed for read-write
35  * locks. An example written in assembly was posted to the Linux kernel mailing
36  * list in 2002 by David Howells from RedHat. This was a highly optimized
37  * version of a read-write ticket lock developed at IBM in the early 90's by
38  * Joseph Seigh. Note that a similar (but not identical) algorithm was published
39  * by John Mellor-Crummey and Michael Scott in their landmark paper "Scalable
40  * Reader-Writer Synchronization for Shared-Memory Multiprocessors".
41  *
42  * The following is an explanation of this code. First, the underlying lock
43  * structure.
44  *
45  * struct {
46  * uint16_t writers; Now serving for writers
47  * uint16_t readers; Now serving for readers
48  * uint16_t next; Next available ticket number
49  * uint16_t __notused; Padding
50  * }
51  *
52  * First, imagine a store's 'take a number' ticket algorithm. A customer takes
53  * a unique ticket number and customers are served in ticket order. In the data
54  * structure, 'writers' is the next writer to be served, 'readers' is the next
55  * reader to be served, and 'next' is the next available ticket number.
56  *
57  * Next, consider exclusive (write) locks. The 'now serving' number for writers
58  * is 'writers'. To lock, 'take a number' and wait until that number is being
59  * served; more specifically, atomically copy and increment the current value of
60  * 'next', and then wait until 'writers' equals that copied number.
61  *
62  * Shared (read) locks are similar. Like writers, readers atomically get the
63  * next number available. However, instead of waiting for 'writers' to equal
64  * their number, they wait for 'readers' to equal their number.
65  *
66  * This has the effect of queuing lock requests in the order they arrive
67  * (incidentally avoiding starvation).
68  *
69  * Each lock/unlock pair requires incrementing both 'readers' and 'writers'.
70  * In the case of a reader, the 'readers' increment happens when the reader
71  * acquires the lock (to allow read-lock sharing), and the 'writers' increment
72  * happens when the reader releases the lock. In the case of a writer, both
73  * 'readers' and 'writers' are incremented when the writer releases the lock.
74  *
75  * For example, consider the following read (R) and write (W) lock requests:
76  *
77  * writers readers next
78  * 0 0 0
79  * R: ticket 0, readers match OK 0 1 1
80  * R: ticket 1, readers match OK 0 2 2
81  * R: ticket 2, readers match OK 0 3 3
82  * W: ticket 3, writers no match block 0 3 4
83  * R: ticket 2, unlock 1 3 4
84  * R: ticket 0, unlock 2 3 4
85  * R: ticket 1, unlock 3 3 4
86  * W: ticket 3, writers match OK 3 3 4
87  *
88  * Note the writer blocks until 'writers' equals its ticket number and it does
89  * not matter if readers unlock in order or not.
90  *
91  * Readers or writers entering the system after the write lock is queued block,
92  * and the next ticket holder (reader or writer) will unblock when the writer
93  * unlocks. An example, continuing from the last line of the above example:
94  *
95  * writers readers next
96  * W: ticket 3, writers match OK 3 3 4
97  * R: ticket 4, readers no match block 3 3 5
98  * R: ticket 5, readers no match block 3 3 6
99  * W: ticket 6, writers no match block 3 3 7
100  * W: ticket 3, unlock 4 4 7
101  * R: ticket 4, readers match OK 4 5 7
102  * R: ticket 5, readers match OK 4 6 7
103  *
104  * The 'next' field is a 2-byte value so the available ticket number wraps at
105  * 64K requests. If a thread's lock request is not granted until the 'next'
106  * field cycles and the same ticket is taken by another thread, we could grant
107  * a lock to two separate threads at the same time, and bad things happen: two
108  * writer threads or a reader thread and a writer thread would run in parallel,
109  * and lock waiters could be skipped if the unlocks race. This is unlikely, it
110  * only happens if a lock request is blocked by 64K other requests. The fix is
111  * to grow the lock structure fields, but the largest atomic instruction we have
112  * is 8 bytes, the structure has no room to grow.
113  */
114 // from http://locklessinc.com/articles/locks/

134  typedef union {
135  uint64_t u;
136  struct {
137  uint32_t wr; // Writers and readers
138  } i;
139  struct {
140  uint16_t writers; // Now serving for writers
141  uint16_t readers; // Now serving for readers
142  uint16_t next; // Next available ticket number
143  uint16_t notused; // Padding to 8 bytes
144  } s;
145  } readwrite_ticket;
146 
147  readwrite_ticket ticket;
148  bool readlock_;
149 
150  public:
151  readers_writer_mutex() : ticket{0}, readlock_(false) {}

153  //
154  // acquire lock for a unique writer
155  //
156  void lock()
157  {
158  RWL_DEBUG_MSG("lock wr " << std::hex << this
159  << " r " << ticket.s.readers
160  << " w " << ticket.s.writers
161  << " n " << ticket.s.next
162  << " v " << ticket.s.next);
163 
164  // memory ordering barrier
165  rwl_barrier_();
166 
167  uint16_t val = atomic_xadd(&ticket.s.next, 1);
168  while (val != ticket.s.writers) {
169  hpx::util::detail::yield_k(4, nullptr);
170  }
171  readlock_ = false;
172 
173  // memory ordering rwl_barrier_
174  rwl_barrier_();
175  }
176 
177  //
178  // unlock writer
179  //
180  void unlock()
181  {
182  // readlock incremented readers when it took the lock
183  if (readlock_) {
184  atomic_inc(&ticket.s.writers);
185  RWL_DEBUG_MSG("unlock rd " << std::hex << this
186  << " r " << ticket.s.readers
187  << " w " << ticket.s.writers
188  << " n " << ticket.s.next);
189  }
190  else {
191  // only one writer can enter unlock at a time, do not need atomics
192  readwrite_ticket new_ticket = ticket;
193  ++new_ticket.s.writers;
194  ++new_ticket.s.readers;
195  RWL_DEBUG_MSG("unlock wr " << std::hex << this
196  << " r " << new_ticket.s.readers
197  << " w " << new_ticket.s.writers
198  << " n " << new_ticket.s.next);
199  ticket.i.wr = new_ticket.i.wr;
200  }
201  }
202 
203  //
204  // try to obtain unique writer lock
205  //
206  bool try_lock()
207  {
208  readwrite_ticket new_ticket, old_ticket;
209  new_ticket = old_ticket = ticket;
210 
211  /*
212  * This write lock can only be granted if the lock was last granted to
213  * a writer and there are no readers or writers blocked on the lock,
214  * that is, if this thread's ticket would be the next ticket granted.
215  * Do the cheap test to see if this can possibly succeed (and confirm
216  * the lock is in the correct state to grant this write lock).
217  */
218  if (old_ticket.s.writers != old_ticket.s.next)
219  return false;
220 
221  // The replacement lock value is a result of allocating a new ticket.
222  ++new_ticket.s.next;
223 
224  bool granted =
225  (cmpxchg(&ticket.u, old_ticket.u, new_ticket.u) ? true : false);
226  if (granted) {
227  readlock_ = false;
228  }
229  return granted;
230  }

232  //
233  // obtain a reader lock, many readers may have the lock simultaneously
234  //
235  void lock_shared()
236  {
237  RWL_DEBUG_MSG("lock rd " << std::hex << this
238  << " r " << ticket.s.readers
239  << " w " << ticket.s.writers
240  << " n " << ticket.s.next
241  << " v " << ticket.s.next);
242 
243  // memory ordering rwl_barrier_
244  rwl_barrier_();
245 
246  uint16_t val = atomic_xadd(&ticket.s.next, 1);
247  while (val != ticket.s.readers) {
248  hpx::util::detail::yield_k(0, nullptr);
249  }
250  readlock_ = true;
251 
252  // only one writer can lock, so no need for atomic increment
253  atomic_inc(&ticket.s.readers);
254 
255  // memory ordering rwl_barrier_
256  rwl_barrier_();
257  }

259  //
260  // unlock one reader
261  //
262  void unlock_shared()
263  {
264  atomic_inc(&ticket.s.writers);
265  }
266 
267  //
268  // try to obtain a reader lock
269  //
270  bool try_lock_shared()
271  {
272  readwrite_ticket new_ticket, old_ticket;
273  new_ticket = old_ticket = ticket;
274  //
275  /*
276  * This read lock can only be granted if the lock was last granted to
277  * a reader and there are no readers or writers blocked on the lock,
278  * that is, if this thread's ticket would be the next ticket granted.
279  * Do the cheap test to see if this can possibly succeed (and confirm
280  * the lock is in the correct state to grant this read lock).
281  */
282  if (old_ticket.s.readers != new_ticket.s.next)
283  return false;
284  //
285  /*
286  * The replacement lock value is a result of allocating a new ticket and
287  * incrementing the reader value to match it.
288  */
289  new_ticket.s.readers = new_ticket.s.next = old_ticket.s.next + 1;
290  bool granted =
291  (cmpxchg(&ticket.u, old_ticket.u, new_ticket.u) ? true : false);
292  if (granted) {
293  readlock_ = true;
294  }
295  return granted;
296  }
297 
298  // return true if a reader or writer has the lock
299  bool owns_lock()
300  {
301  return ((ticket.s.writers != ticket.s.next)
302  || (ticket.s.readers != ticket.s.next));
303  }
304  };

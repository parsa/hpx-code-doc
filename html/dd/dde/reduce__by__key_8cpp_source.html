101 template<typename ExPolicy, typename Tkey, typename Tval, typename Op, typename HelperOp>
102 void test_reduce_by_key1(ExPolicy && policy, Tkey, Tval, bool benchmark, const Op &op,
103  const HelperOp &ho)
104  {
105  static_assert(
106  hpx::parallel::execution::is_execution_policy<ExPolicy>::value,
107  "hpx::parallel::execution::is_execution_policy<ExPolicy>::value");
108  msg(typeid(ExPolicy).name(), typeid(Tval).name(), typeid(Op).name(), sync);
109  std::cout << "\n";
110 
111  Tval rnd_min = -256;
112  Tval rnd_max = 256;
113 
114  // vector of values, and keys
115  std::vector<Tval> values, o_values;
116  std::vector<Tkey> keys, o_keys;
117  values.reserve(HPX_REDUCE_BY_KEY_TEST_SIZE);
118  keys.reserve(HPX_REDUCE_BY_KEY_TEST_SIZE);
119 
120  std::vector<Tval> check_values;
121 
122  // use the default random engine and an uniform distribution for values
123  std::mt19937 eng(static_cast<unsigned int>(std::rand()));
124  std::uniform_real_distribution<double> distr(rnd_min, rnd_max);
125 
126  // use the default random engine and an uniform distribution for keys
127  std::mt19937 engk(static_cast<unsigned int>(std::rand()));
128  std::uniform_real_distribution<double> distrk(0, 256);
129 
130  // generate test data
131  int keysize = 0;
132  Tkey key = 0, helperkey = 0, lastkey = 0;
133  for (/* */; keysize < HPX_REDUCE_BY_KEY_TEST_SIZE;)
134  {
135  do {
136  key = static_cast<Tkey>(distrk(engk));
137  } while (ho(key) == lastkey);
138  helperkey = ho(key);
139  lastkey = helperkey;
140  //
141  int numkeys = static_cast<Tkey>(distrk(engk)) + 1;
142  //
143  Tval sum = 0;
144  for (int i = 0; i < numkeys && keysize < HPX_REDUCE_BY_KEY_TEST_SIZE; ++i) {
145  Tval value = static_cast<Tval>(distr(eng));
146  keys.push_back(key);
147  values.push_back(value);
148  sum += value;
149  keysize++;
150  }
151  check_values.push_back(sum);
152  }
153  o_values = values;
154  o_keys = keys;
155 
156  hpx::util::high_resolution_timer t;
157  // reduce_by_key, blocking when seq, par, par_vec
158  auto result = hpx::parallel::reduce_by_key(
159  std::forward<ExPolicy>(policy),
160  keys.begin(), keys.end(),
161  values.begin(),
162  keys.begin(),
163  values.begin(),
164  op);
165  double elapsed = t.elapsed();
166 
167  bool is_equal = std::equal(values.begin(), result.second, check_values.begin(),
168  almost_equal());
169  HPX_TEST(is_equal);
170  if (is_equal) {
171  if (benchmark) {
172  std::cout << "<DartMeasurement name=\"ReduceByKeyTime\" \n"
173  << "type=\"numeric/double\">" << elapsed << "</DartMeasurement> \n";
174  }
175  }
176  else {
177 // debug::output("keys ", o_keys);
178 // debug::output("values ", o_values);
179 // debug::output("key range", keys.begin(), result.first);
180 // debug::output("val range", values.begin(), result.second);
181 // debug::output("expected ", check_values);
182 // throw std::string("Problem");
183 #if defined(EXTRA_DEBUG)
184  for (std::size_t i = 0; i != check_values.size(); ++i)
185  {
186  if (!almost_equal()(values[i],check_values[i]))
187  {
188  std::cout
189  << i << ": "
190  << values[i] << " != " << check_values[i]
191  << "\n";
192  }
193  }
194 #endif
195  }

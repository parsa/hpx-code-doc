38  // provide our own implementation of std::uninitialized_copy as some
39  // versions of MSVC horribly fail at compiling it for some types T
40  template <typename InIter1, typename InIter2>
41  InIter2 std_uninitialized_copy(InIter1 first, InIter1 last, InIter2 d_first)
42  {
43  typedef typename std::iterator_traits<InIter2>::value_type
44  value_type;
45 
46  InIter2 current = d_first;
47  try {
48  for (/* */; first != last; (void) ++first, ++current)
49  {
50  ::new (std::addressof(*current)) value_type(*first);
51  }
52  return current;
53  }
54  catch (...) {
55  for (/* */; d_first != current; ++d_first)
56  {
57  (*d_first).~value_type();
58  }
59  throw;
60  }
61  }

99  typedef typename std::iterator_traits<FwdIter2>::value_type
100  value_type;

103  return util::partitioner_with_cleanup<
104  ExPolicy, FwdIter2, partition_result_type
105  >::call(
106  std::forward<ExPolicy>(policy),
107  hpx::util::make_zip_iterator(first, dest), count,
108  [tok](zip_iterator t, std::size_t part_size)
109  mutable -> partition_result_type
110  {
111  using hpx::util::get;
112  auto iters = t.get_iterator_tuple();
113  FwdIter2 dest = get<1>(iters);
114  return std::make_pair(dest,
115  sequential_uninitialized_copy_n(
116  get<0>(iters), part_size,
117  dest, tok));
118  },
119  // finalize, called once if no error occurred
120  [dest, count](
121  std::vector<hpx::future<partition_result_type> > &&)
122  mutable -> FwdIter2
123  {
124  std::advance(dest, count);
125  return dest;
126  },
127  // cleanup function, called for each partition which
128  // didn't fail, but only if at least one failed
129  [](partition_result_type && r) -> void
130  {
131  while (r.first != r.second)
132  {
133  (*r.first).~value_type();
134  ++r.first;
135  }

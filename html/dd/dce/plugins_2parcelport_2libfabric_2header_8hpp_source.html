64  template <int SIZE>
65  struct header
66  {
67  static constexpr unsigned int header_block_size = sizeof(detail::header_block);
68  static constexpr unsigned int data_size_ = SIZE - header_block_size;
69  //
70  static const unsigned int chunk_flag = 0x01; // chunks piggybacked
71  static const unsigned int message_flag = 0x02; // message pigybacked
72  static const unsigned int normal_flag = 0x04; // normal chunks present
73  static const unsigned int zerocopy_flag = 0x08; // zerocopy chunks present
74 
75  typedef serialization::serialization_chunk chunktype;
76 
77  private:
78  //
79  // this is the actual header content
80  //
81  detail::header_block message_header;
82  std::array<char, data_size_> data_;
83  // the data block is laid out as follows for each optional item
84  // message_header - always present header_block_size
85  // chunk data : sizeof(chunktype) * numchunks : when chunks piggybacked
86  // or : sizeof(chunk_header) : when chunks not piggybacked
87  // rma_info : sizeof(rma_info) : when we have anything to be rma'd
88  // message_info : sizeof(message_info) : only when message pigybacked
89  // or : sizeof(message_chunk) : when message+chunk both not piggybacked
90  // .....
91  // message : buffer.size_ : only when message piggybacked
92 
93  public:
94  //
95  template <typename Buffer>
96  header(Buffer const & buffer, void* tag)
97  {
98  const std::vector<chunktype>& chunks = buffer.chunks_;
99  //
100  message_header.flags = 0;
101  message_header.num_chunks = chunks.size();
102  message_header.flags |= buffer.num_chunks_.first ? zerocopy_flag : 0;
103  message_header.flags |= buffer.num_chunks_.second ? normal_flag : 0;
104 
105  // space occupied by chunk data
106  size_t chunkbytes = chunks.size() * sizeof(chunktype);
107 
108  // can we send the chunk info inside the header
109  // (NB. we add +1 chunk just in case of a non piggybacked message chunk)
110  if ((chunkbytes+sizeof(chunktype)) <= data_size_) {
111  message_header.flags |= chunk_flag;
112  // copy chunk data directly into the header
113  std::memcpy(&data_[chunk_data_offset()], chunks.data(), chunkbytes);
114  }
115  else
116  {
117  LOG_DEBUG_MSG("Too many chunks for header "
118  << decnumber(chunks.size())
119  << "requires bytes " << decnumber(chunkbytes));
120  message_header.flags &= ~chunk_flag;
121  message_header.flags |= zerocopy_flag;
122  // send just rma-get information, address and rma key will be added later
123  detail::chunk_header *ch =
124  reinterpret_cast<detail::chunk_header*>(&data_[chunk_data_offset()]);
125  ch->num_rma_chunks = buffer.num_chunks_.first;
126  ch->chunk_rma =
127  serialization::create_pointer_chunk(nullptr, chunkbytes, 0);
128  // reset chunkbytes size to size of rma hunk header
129  chunkbytes = sizeof(detail::chunk_header);
130  }
131 
132  // can we send main message inside the header
133  if (buffer.data_.size() <= (data_size_ - chunkbytes)) {
134  message_header.flags |= message_flag;
135  detail::message_info *info = message_info_ptr();
136  info->message_size = buffer.size_;
137  }
138  else {
139  message_header.flags &= ~message_flag;
140  message_header.flags |= zerocopy_flag;
141  if ((message_header.flags & chunk_flag) != 0) {
142  // if chunks are piggybacked, just add one rma chunk for the message
143  message_header.num_chunks += 1;
144  chunktype message =
145  serialization::create_pointer_chunk(nullptr, buffer.size_, 0);
146  std::memcpy(&data_[chunkbytes], &message, sizeof(chunktype));
147  }
148  else {
149  // the message isn't piggybacked and neither is the chunk data
150  // so we must add rma-get information for the message
151  detail::message_chunk *mc = reinterpret_cast<detail::message_chunk*>
152  (&data_[message_info_offset()]);
153  LOG_DEBUG_MSG("Setting chunk free message size to "
154  << decnumber(buffer.size_)
155  << "offset " << decnumber(message_info_offset()));
156  mc->message_rma =
157  serialization::create_pointer_chunk(nullptr, buffer.size_, 0);
158  }
159  }
160 
161  // set the rma tag
162  if ((message_header.flags & zerocopy_flag) != 0) {
163  auto ptr = rma_info_ptr();
164  ptr->tag = reinterpret_cast<uint64_t>(tag);
165  }
166 
167  LOG_DEBUG_MSG("Header : " << *this);
168  }

98 int hpx_main(
99  variables_map& vm
100  )
101 {
102  if (vm.count("no-header"))
103  header = false;
104 
106  // Initialize the PRNG seed.
107  if (!seed)
108  seed = std::uint64_t(std::time(nullptr));
109 
110  {
111 
113  // Validate command-line arguments.
114  if (0 == tasks)
115  throw std::invalid_argument("count of 0 tasks specified\n");
116 
117  if (min_delay > max_delay)
118  throw std::invalid_argument("minimum delay cannot be larger than "
119  "maximum delay\n");
120 
121  if (min_delay > total_delay)
122  throw std::invalid_argument("minimum delay cannot be larger than"
123  "total delay\n");
124 
125  if (max_delay > total_delay)
126  throw std::invalid_argument("maximum delay cannot be larger than "
127  "total delay\n");
128 
129  if ((min_delay * tasks) > total_delay)
130  throw std::invalid_argument("minimum delay is too small for the "
131  "specified total delay and number of "
132  "tasks\n");
133 
134  if ((max_delay * tasks) < total_delay)
135  throw std::invalid_argument("maximum delay is too small for the "
136  "specified total delay and number of "
137  "tasks\n");
138 
140  // Randomly generate a description of the heterogeneous workload.
141  std::vector<std::uint64_t> payloads;
142  payloads.reserve(tasks);
143 
144  // For random numbers, we use a 64-bit specialization of stdlib's
145  // mersenne twister engine (good uniform distribution up to 311
146  // dimensions, cycle length 2 ^ 19937 - 1)
147  std::mt19937_64 prng(seed);
148 
149  std::uint64_t current_sum = 0;
150 
151  for (std::uint64_t i = 0; i < tasks; ++i)
152  {
153  // Credit to Spencer Ruport for putting this algorithm on
154  // stackoverflow.
155  std::uint64_t const low_calc
156  = (total_delay - current_sum) - (max_delay * (tasks - 1 - i));
157 
158  bool const negative
159  = (total_delay - current_sum) < (max_delay * (tasks - 1 - i));
160 
161  std::uint64_t const low
162  = (negative || (low_calc < min_delay)) ? min_delay : low_calc;
163 
164  std::uint64_t const high_calc
165  = (total_delay - current_sum) - (min_delay * (tasks - 1 - i));
166 
167  std::uint64_t const high
168  = (high_calc > max_delay) ? max_delay : high_calc;
169 
170  // Our range is [low, high].
171  std::uniform_int_distribution<std::uint64_t>
172  dist(low, high);
173 
174  std::uint64_t const payload = dist(prng);
175 
176  if (payload < min_delay)
177  throw std::logic_error("task delay is below minimum");
178 
179  if (payload > max_delay)
180  throw std::logic_error("task delay is above maximum");
181 
182  current_sum += payload;
183  payloads.push_back(payload);
184  }
185 
186  // Randomly shuffle the entire sequence to deal with drift.
187  using hpx::util::placeholders::_1;
188  std::random_shuffle(payloads.begin(), payloads.end(),
189  hpx::util::bind(&shuffler, std::ref(prng), _1));
190 
192  // Validate the payloads.
193  if (payloads.size() != tasks)
194  throw std::logic_error("incorrect number of tasks generated");
195 
196  std::uint64_t const payloads_sum =
197  std::accumulate(payloads.begin(), payloads.end(), 0ULL);
198  if (payloads_sum != total_delay)
199  throw std::logic_error("incorrect total delay generated");
200 
202  // Start the clock.
203  high_resolution_timer t;
204 
206  // Queue the tasks in a serial loop.
207  for (std::uint64_t i = 0; i < tasks; ++i)
208  register_work(hpx::util::bind(&worker_timed, payloads[i] * 1000));
209 
211  // Wait for the work to finish.
212  do {
213  // Reschedule hpx_main until all other HPX-threads have finished. We
214  // should be resumed after most of the null HPX-threads have been
215  // executed. If we haven't, we just reschedule ourselves again.
216  suspend();
217  } while (get_thread_count(hpx::threads::thread_priority_normal) > 1);
218 
220  // Print the results.
221  print_results(get_os_thread_count(), t.elapsed());
222  }
223 
224  finalize();
225  return 0;
226 }

71 bool test_migrate_component_to_storage(hpx::id_type const& source,
72  hpx::components::component_storage storage, hpx::id_type::management_type t)
73 {
74  hpx::id_type oldid;
75 
76  {
77  // create component on given locality
78  test_client t1(source);
79  HPX_TEST_NEQ(hpx::naming::invalid_id, t1.get_id());
80 
81  // the new object should live on the source locality
82  HPX_TEST_EQ(t1.call(), source);
83 
84  // remember the original id for later resurrection
85  oldid = hpx::id_type(t1.get_id().get_gid(), t);
86 
87  try {
88  // migrate of t1 to the target storage
89  test_client t2(hpx::components::migrate_to_storage(t1, storage));
90  HPX_TEST_EQ(hpx::naming::invalid_id, t2.get_id());
91  }
92  catch (hpx::exception const&) {
93  return false;
94  }
95 
96  HPX_TEST_EQ(storage.size(hpx::launch::sync), std::size_t(1));
97 
98  // make sure all local references go out of scope if t == unmanaged
99  }
100 
101  HPX_TEST_EQ(storage.size(hpx::launch::sync), std::size_t(1));
102 
103  {
104  test_client t1(hpx::components::migrate_from_storage<test_server>(oldid));
105 
106  // the id of the newly resurrected object should be the same as the old id
107  HPX_TEST_EQ(oldid, t1.get_id());
108 
109  // the new object should live on the (original) source locality

113  }

178  // the new object should live on the source locality

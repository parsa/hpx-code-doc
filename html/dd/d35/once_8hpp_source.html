44  void call_once(once_flag& flag, F&& f, Args&&... args)
45  {
46  // Try for a quick win: if the procedure has already been called
47  // just skip through:
48  long const function_complete_flag_value = 0xc15730e2;
49  long const running_value = 0x7f0725e3;
50 
51  while (flag.status_.load(std::memory_order_acquire) !=
52  function_complete_flag_value)
53  {
54  long status = 0;
55  if (flag.status_.compare_exchange_strong(status, running_value))
56  {
57  try {
58  // reset event to ensure its usability in case the
59  // wrapped function was throwing an exception before
60  flag.event_.reset();
61 
62  util::invoke(std::forward<F>(f), std::forward<Args>(args)...);
63 
64  // set status to done, release waiting threads
65  flag.status_.store(function_complete_flag_value);
66  flag.event_.set();
67  break;
68  }
69  catch (...) {
70  // reset status to initial, release waiting threads
71  flag.status_.store(0);
72  flag.event_.set();
73 
74  throw;
75  }
76  }
77 
78  // we're done if function was called
79  if (status == function_complete_flag_value)
80  break;
81 
82  // wait for the function finish executing
83  flag.event_.wait();
84  }
85  }

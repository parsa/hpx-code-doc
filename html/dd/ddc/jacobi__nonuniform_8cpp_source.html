39  void jacobi_kernel_nonuniform(
40  crs_matrix<double> const & A
41  , std::vector<double> & dst
42  , std::vector<double> const & src
43  , std::vector<double> const & b
44  , std::size_t row
45  )
46  {
47  double result = b[row];
48  double div = 1.0;
49  const std::size_t begin = A.row_begin(row);
50  const std::size_t end = A.row_end(row);
51 
52  for(std::size_t j = begin; j < end; ++j)
53  {
54  if(row == j) div = div/A.values[j];
55  else result -= A.values[j] * src[A.indices[j]];
56  }
57  dst[row] = result * div;
58  }
59 }

58 int checktick()
59 {
60  static const std::size_t M = 20;
61  int minDelta, Delta;
62  double t1, t2, timesfound[M];
63 
64  // Collect a sequence of M unique time values from the system.
65  for (std::size_t i = 0; i < M; i++) {
66  t1 = mysecond();
67  while( ((t2=mysecond()) - t1) < 1.0E-6 )
68  ;
69  timesfound[i] = t1 = t2;
70  }
71 
72  // Determine the minimum difference between these M values.
73  // This result will be our estimate (in microseconds) for the
74  // clock granularity.
75  minDelta = 1000000;
76  for (std::size_t i = 1; i < M; i++) {
77  Delta = (int)( 1.0E6 * (timesfound[i]-timesfound[i-1]));
78  minDelta = (std::min)(minDelta, (std::max)(Delta,0));
79  }
80 
81  return(minDelta);
82 }

85 void check_results(std::size_t iterations,
86  Vector const & a_res, Vector const & b_res, Vector const & c_res)
87 {
88  std::vector<STREAM_TYPE> a(a_res.size());
89  std::vector<STREAM_TYPE> b(b_res.size());
90  std::vector<STREAM_TYPE> c(c_res.size());
91 
92  hpx::parallel::copy(hpx::parallel::execution::par,
93  a_res.begin(), a_res.end(), a.begin());
94  hpx::parallel::copy(hpx::parallel::execution::par,
95  b_res.begin(), b_res.end(), b.begin());
96  hpx::parallel::copy(hpx::parallel::execution::par,
97  c_res.begin(), c_res.end(), c.begin());
98 
99  STREAM_TYPE aj,bj,cj,scalar;
100  STREAM_TYPE aSumErr,bSumErr,cSumErr;
101  STREAM_TYPE aAvgErr,bAvgErr,cAvgErr;
102  double epsilon;
103  int ierr,err;
104 
105  /* reproduce initialization */
106  aj = 1.0;
107  bj = 2.0;
108  cj = 0.0;
109  /* a[] is modified during timing check */
110  aj = 2.0E0 * aj;
111  /* now execute timing loop */
112  scalar = 3.0;
113  for (std::size_t k=0; k<iterations; k++)
114  {
115  cj = aj;
116  bj = scalar*cj;
117  cj = aj+bj;
118  aj = bj+scalar*cj;
119  }
120 
121  /* accumulate deltas between observed and expected results */
122  aSumErr = 0.0;
123  bSumErr = 0.0;
124  cSumErr = 0.0;
125  for (std::size_t j=0; j<a.size(); j++) {
126  aSumErr += std::abs(a[j] - aj);
127  bSumErr += std::abs(b[j] - bj);
128  cSumErr += std::abs(c[j] - cj);
129  // if (j == 417) printf("Index 417: c[j]: %f, cj: %f\n",c[j],cj); // MCCALPIN
130  }
131  aAvgErr = aSumErr / (STREAM_TYPE) a.size();
132  bAvgErr = bSumErr / (STREAM_TYPE) a.size();
133  cAvgErr = cSumErr / (STREAM_TYPE) a.size();
134 
135  if (sizeof(STREAM_TYPE) == 4) {
136  epsilon = 1.e-6;
137  }
138  else if (sizeof(STREAM_TYPE) == 8) {
139  epsilon = 1.e-13;
140  }
141  else {
142  printf("WEIRD: sizeof(STREAM_TYPE) = %zu\n", sizeof(STREAM_TYPE));
143  epsilon = 1.e-6;
144  }
145 
146  err = 0;
147  if (std::abs(aAvgErr/aj) > epsilon) {
148  err++;
149  printf ("Failed Validation on array a[], AvgRelAbsErr > epsilon (%e)\n",
150  epsilon);
151  printf (" Expected Value: %e, AvgAbsErr: %e, AvgRelAbsErr: %e\n",
152  aj,aAvgErr,std::abs(aAvgErr)/aj);
153  ierr = 0;
154  for (std::size_t j=0; j<a.size(); j++) {
155  if (std::abs(a[j]/aj-1.0) > epsilon) {
156  ierr++;
157 #ifdef VERBOSE
158  if (ierr < 10) {
159  printf(" array a: index: %ld, expected: %e, "
160  "observed: %e, relative error: %e\n",
161  j,aj,a[j],std::abs((aj-a[j])/aAvgErr));
162  }
163 #endif
164  }
165  }
166  printf(" For array a[], %d errors were found.\n",ierr);
167  }
168  if (std::abs(bAvgErr/bj) > epsilon) {
169  err++;
170  printf ("Failed Validation on array b[], AvgRelAbsErr > epsilon (%e)\n",
171  epsilon);
172  printf (" Expected Value: %e, AvgAbsErr: %e, AvgRelAbsErr: %e\n",
173  bj,bAvgErr,std::abs(bAvgErr)/bj);
174  printf (" AvgRelAbsErr > Epsilon (%e)\n",epsilon);
175  ierr = 0;
176  for (std::size_t j=0; j<a.size(); j++) {
177  if (std::abs(b[j]/bj-1.0) > epsilon) {
178  ierr++;
179 #ifdef VERBOSE
180  if (ierr < 10) {
181  printf(" array b: index: %ld, expected: %e, "
182  "observed: %e, relative error: %e\n",
183  j,bj,b[j],std::abs((bj-b[j])/bAvgErr));
184  }
185 #endif
186  }
187  }
188  printf(" For array b[], %d errors were found.\n",ierr);
189  }
190  if (std::abs(cAvgErr/cj) > epsilon) {
191  err++;
192  printf ("Failed Validation on array c[], AvgRelAbsErr > epsilon (%e)\n",
193  epsilon);
194  printf (" Expected Value: %e, AvgAbsErr: %e, AvgRelAbsErr: %e\n",
195  cj,cAvgErr,std::abs(cAvgErr)/cj);
196  printf (" AvgRelAbsErr > Epsilon (%e)\n",epsilon);
197  ierr = 0;
198  for (std::size_t j=0; j<a.size(); j++) {
199  if (std::abs(c[j]/cj-1.0) > epsilon) {
200  ierr++;
201 #ifdef VERBOSE
202  if (ierr < 10) {
203  printf(" array c: index: %ld, expected: %e, "
204  "observed: %e, relative error: %e\n",
205  j,cj,c[j],std::abs((cj-c[j])/cAvgErr));
206  }
207 #endif
208  }
209  }
210  printf(" For array c[], %d errors were found.\n",ierr);
211  }
212  if (err == 0) {
213  printf ("Solution Validates: avg error less than %e on all three arrays\n",
214  epsilon);
215  }
216 #ifdef VERBOSE
217  printf ("Results Validation Verbose Results: \n");
218  printf (" Expected a(1), b(1), c(1): %f %f %f \n",aj,bj,cj);
219  printf (" Observed a(1), b(1), c(1): %f %f %f \n",a[1],b[1],c[1]);
220  printf (" Rel Errors on a, b, c: %e %e %e \n",std::abs(aAvgErr/aj),
221  std::abs(bAvgErr/bj),std::abs(cAvgErr/cj));
222 #endif

227 struct multiply_step
228 {
229  multiply_step(T factor) : factor_(factor) {}

231  // FIXME : call operator of multiply_step is momentarily defined with
232  // a generic parameter to allow the host_side invoke_result<>
233  // (used in invoke()) to get the return type

237  {
238  return val * factor_;

241  T factor_;

277 template <typename Allocator, typename Executor, typename Target, typename... Targets>
278 std::vector<std::vector<double> >
279 run_benchmark(
280  std::size_t iterations, std::size_t size, Target target, Targets... targets)
281 {
282  // Creating our allocator ...
283  Allocator alloc(target);
284 
285  // Allocate our data
286  typedef hpx::compute::vector<STREAM_TYPE, Allocator> vector_type;
287 
288  vector_type a(size, alloc);
289  vector_type b(size, alloc);
290  vector_type c(size, alloc);
291 
292  // Creating our executor ....
293  Executor exec(target, targets...);
294 
295  // Creating the policy used in the parallel algorithms
296  auto policy = hpx::parallel::execution::par.on(exec);
297 
298  // Initialize arrays
299  hpx::parallel::fill(policy, a.begin(), a.end(), 1.0);
300  hpx::parallel::fill(policy, b.begin(), b.end(), 2.0);
301  hpx::parallel::fill(policy, c.begin(), c.end(), 0.0);
302 
303  // Check clock ticks ...
304  double t = mysecond();
305  hpx::parallel::transform(
306  policy, a.begin(), a.end(), a.begin(),
307  multiply_step<STREAM_TYPE>(2.0));
308  t = 1.0E6 * (mysecond() - t);
309 
310  // Get initial value for system clock.
311  int quantum = checktick();
312  if(quantum >= 1)
313  {
314  std::cout
315  << "Your clock granularity/precision appears to be " << quantum
316  << " microseconds.\n"
317  ;
318  }
319  else
320  {
321  std::cout
322  << "Your clock granularity appears to be less than one microsecond.\n"
323  ;
324  quantum = 1;
325  }
326 
327  std::cout
328  << "Each test below will take on the order"
329  << " of " << (int) t << " microseconds.\n"
330  << " (= " << (int) (t/quantum) << " clock ticks)\n"
331  << "Increase the size of the arrays if this shows that\n"
332  << "you are not getting at least 20 clock ticks per test.\n"
333  << "-------------------------------------------------------------\n"
334  ;
335 
336  std::cout
337  << "WARNING -- The above is only a rough guideline.\n"
338  << "For best results, please be sure you know the\n"
339  << "precision of your system timer.\n"
340  << "-------------------------------------------------------------\n"
341  ;
342 
344  // Main Loop
345  std::vector<std::vector<double> > timing(4, std::vector<double>(iterations));
346 
347  double scalar = 3.0;
348  for(std::size_t iteration = 0; iteration != iterations; ++iteration)
349  {
350  // Copy
351  timing[0][iteration] = mysecond();
352  hpx::parallel::copy(policy, a.begin(), a.end(), c.begin());
353  timing[0][iteration] = mysecond() - timing[0][iteration];
354 
355  // Scale
356  timing[1][iteration] = mysecond();
357  hpx::parallel::transform(policy,
358  c.begin(), c.end(), b.begin(),
359  multiply_step<STREAM_TYPE>(scalar)
360  );
361  timing[1][iteration] = mysecond() - timing[1][iteration];
362 
363  // Add
364  timing[2][iteration] = mysecond();
365  hpx::parallel::transform(policy,
366  a.begin(), a.end(), b.begin(), b.end(), c.begin(),
367  add_step<STREAM_TYPE>()
368  );
369  timing[2][iteration] = mysecond() - timing[2][iteration];
370 
371  // Triad
372  timing[3][iteration] = mysecond();
373  hpx::parallel::transform(policy,
374  b.begin(), b.end(), c.begin(), c.end(), a.begin(),
375  triad_step<STREAM_TYPE>(scalar)
376  );
377  timing[3][iteration] = mysecond() - timing[3][iteration];
378  }
379 
380  // Check Results ...
381  check_results(iterations, a, b, c);
382 
383  std::cout
384  << "-------------------------------------------------------------\n"
385  ;
386 
387  return timing;
388 }

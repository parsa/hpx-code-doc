249  bool parcelhandler::do_background_work(std::size_t num_thread,
250  bool stop_buffering)
251  {
252  bool did_some_work = false;
253 
254 #if defined(HPX_HAVE_NETWORKING)
255  // flush all parcel buffers
256  if(0 == num_thread)
257  {
258  std::unique_lock<mutex_type> l(handlers_mtx_, std::try_to_lock);
259 
260  if(l.owns_lock())
261  {
262  using parcelset::policies::message_handler;
263  message_handler::flush_mode mode =
264  message_handler::flush_mode_background_work;
265 
266  message_handler_map::iterator end = handlers_.end();
267  for (message_handler_map::iterator it = handlers_.begin();
268  it != end; ++it)
269  {
270  if ((*it).second)
271  {
272  std::shared_ptr<policies::message_handler> p((*it).second);
273  util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
274  did_some_work =
275  p->flush(mode, stop_buffering) || did_some_work;
276  }
277  }
278  }
279  }
280 
281  // make sure all pending parcels are being handled
282  for (pports_type::value_type& pp : pports_)
283  {
284  if(pp.first > 0)
285  {
286  did_some_work = pp.second->do_background_work(num_thread) ||
287  did_some_work;
288  }
289  }
290 #endif
291  return did_some_work;
292  }

437  void parcelhandler::put_parcel(parcel p, write_handler_type f)
438  {
439 #if defined(HPX_HAVE_NETWORKING)
440  HPX_ASSERT(resolver_);
441 
442  naming::gid_type const& gid = p.destination();
443  naming::address& addr = p.addr();
444 
445  // During bootstrap this is handled separately (see
446  // addressing_service::resolve_locality.
447 
448  // if this isn't an HPX thread, the stack space check will return false
449  if (!this_thread::has_sufficient_stack_space() &&
450  hpx::threads::threadmanager_is(hpx::state::state_running))
451  {
452 // naming::gid_type locality =
453 // naming::get_locality_from_gid(id.get_gid());
454 // if (!resolver_->has_resolved_locality(locality))
455  {
456  // reschedule request as an HPX thread to avoid hangs
457  void (parcelhandler::*put_parcel_ptr) (
458  parcel p, write_handler_type f
459  ) = &parcelhandler::put_parcel;
460 
461  threads::register_thread_nullary(
462  util::deferred_call(put_parcel_ptr, this,
463  std::move(p), std::move(f)),
464  "parcelhandler::put_parcel", threads::pending, true,
465  threads::thread_priority_boost, std::size_t(-1),
466  threads::thread_stacksize_medium);
467  return;
468  }
469  }
470 
471  // properly initialize parcel
472  init_parcel(p);
473 
474  bool resolved_locally = true;
475 
476  if (!addr)
477  {
478  resolved_locally = resolver_->resolve_local(gid, addr);
479  }
480 
481 #if defined(HPX_HAVE_PARCEL_PROFILING)
482  if (!p.parcel_id())
483  {
484  p.parcel_id() = parcelset::parcel::generate_unique_id();
485  }
486 #endif
487 
488  using util::placeholders::_1;
489  using util::placeholders::_2;
490  write_handler_type wrapped_f =
491  util::bind(&detail::parcel_sent_handler, std::move(f), _1, _2);
492 
493  // If we were able to resolve the address(es) locally we send the
494  // parcel directly to the destination.
495  if (resolved_locally)
496  {
497  // dispatch to the message handler which is associated with the
498  // encapsulated action
499  typedef std::pair<std::shared_ptr<parcelport>, locality> destination_pair;
500  destination_pair dest = find_appropriate_destination(addr.locality_);
501 
502  if (load_message_handlers_ && !hpx::is_stopped_or_shutting_down())
503  {
504  policies::message_handler* mh =
505  p.get_message_handler(this, dest.second);
506 
507  if (mh) {
508  mh->put_parcel(dest.second, std::move(p), std::move(wrapped_f));
509  return;
510  }
511  }
512 
513  dest.first->put_parcel(dest.second, std::move(p), std::move(wrapped_f));
514  return;
515  }
516 
517  // At least one of the addresses is locally unknown, route the parcel
518  // to the AGAS managing the destination.
519  ++count_routed_;
520 
521  resolver_->route(std::move(p), std::move(wrapped_f));
522 #else
523  HPX_THROW_EXCEPTION(invalid_status,
524  "Networking was disabled at configuration time. Please "
525  "reconfigure HPX using -DHPX_WITH_NETWORKING=On.",
526  "parcelhandler::put_parcel");
527 #endif
528  }

530  void parcelhandler::put_parcels(std::vector<parcel> parcels,
531  std::vector<write_handler_type> handlers)
532  {
533 #if defined(HPX_HAVE_NETWORKING)
534  HPX_ASSERT(resolver_);
535 
536  if (parcels.size() != handlers.size())
537  {
538  HPX_THROW_EXCEPTION(bad_parameter,
539  "parcelhandler::put_parcels",
540  "mismatched number of parcels and handlers");
541  return;
542  }
543 
544  // if this isn't an HPX thread, the stack space check will return false
545  if (!this_thread::has_sufficient_stack_space() &&
546  hpx::threads::threadmanager_is(hpx::state::state_running))
547  {
548 // naming::gid_type locality = naming::get_locality_from_gid(
549 // parcels[0].destination());
550 // if (!resolver_->has_resolved_locality(locality))
551  {
552  // reschedule request as an HPX thread to avoid hangs
553  void (parcelhandler::*put_parcels_ptr) (
554  std::vector<parcel>, std::vector<write_handler_type>
555  ) = &parcelhandler::put_parcels;
556 
557  threads::register_thread_nullary(
558  util::deferred_call(put_parcels_ptr, this,
559  std::move(parcels), std::move(handlers)),
560  "parcelhandler::put_parcels", threads::pending, true,
561  threads::thread_priority_boost, std::size_t(-1),
562  threads::thread_stacksize_medium);
563  return;
564  }
565  }
566 
567  // partition parcels depending on whether their destination can be
568  // resolved locally
569  std::size_t num_parcels = parcels.size();
570 
571  std::vector<parcel> resolved_parcels;
572  resolved_parcels.reserve(num_parcels);
573  std::vector<write_handler_type> resolved_handlers;
574  resolved_handlers.reserve(num_parcels);
575 
576  typedef std::pair<std::shared_ptr<parcelport>, locality>
577  destination_pair;
578 
579  destination_pair resolved_dest;
580 
581  std::vector<parcel> nonresolved_parcels;
582  nonresolved_parcels.reserve(num_parcels);
583  std::vector<write_handler_type> nonresolved_handlers;
584  nonresolved_handlers.reserve(num_parcels);
585 
586  for (std::size_t i = 0; i != num_parcels; ++i)
587  {
588  parcel& p = parcels[i];
589 
590  // make sure all parcels go to the same locality
591  if (parcels[0].destination_locality() !=
592  p.destination_locality())
593  {
594  HPX_THROW_EXCEPTION(bad_parameter,
595  "parcelhandler::put_parcels",
596  "mismatched destinations, all parcels are expected to "
597  "target the same locality");
598  return;
599  }
600 
601  // properly initialize parcel
602  init_parcel(p);
603 #if defined(HPX_HAVE_PARCEL_PROFILING)
604  if (!p.parcel_id())
605  {
606  p.parcel_id() = parcelset::parcel::generate_unique_id();
607  }
608 #endif
609 
610  bool resolved_locally = true;
611  naming::address& addr = p.addr();
612 
613  if (!addr)
614  {
615  resolved_locally = resolver_->resolve_local(
616  p.destination(), addr);
617  }
618 
619  using util::placeholders::_1;
620  using util::placeholders::_2;
621  write_handler_type f = util::bind(&detail::parcel_sent_handler,
622  std::move(handlers[i]), _1, _2);
623 
624  // If we were able to resolve the address(es) locally we would send
625  // the parcel directly to the destination.
626  if (resolved_locally)
627  {
628  // dispatch to the message handler which is associated with the
629  // encapsulated action
630  destination_pair dest = find_appropriate_destination(
631  addr.locality_);
632 
633  if (load_message_handlers_)
634  {
635  policies::message_handler* mh = p.get_message_handler(
636  this, dest.second);
637 
638  if (mh) {
639  mh->put_parcel(dest.second, std::move(p), std::move(f));
640  continue;
641  }
642  }
643 
644  resolved_parcels.push_back(std::move(p));
645  resolved_handlers.push_back(std::move(f));
646  if (!resolved_dest.second)
647  {
648  resolved_dest = dest;
649  }
650  else
651  {
652  HPX_ASSERT(resolved_dest == dest);
653  }
654  }
655  else
656  {
657  nonresolved_parcels.push_back(std::move(p));
658  nonresolved_handlers.push_back(std::move(f));
659  }
660  }
661 
662  // handle parcel which have been locally resolved
663  if (!resolved_parcels.empty())
664  {
665  HPX_ASSERT(!!resolved_dest.first && !!resolved_dest.second);
666  resolved_dest.first->put_parcels(resolved_dest.second,
667  std::move(resolved_parcels),
668  std::move(resolved_handlers));
669  }
670 
671  // At least one of the addresses is locally unknown, route the
672  // parcel to the AGAS managing the destination.
673  for (std::size_t i = 0; i != nonresolved_parcels.size(); ++i)
674  {
675  ++count_routed_;
676  resolver_->route(std::move(nonresolved_parcels[i]),
677  std::move(nonresolved_handlers[i]));
678  }
679 #else
680  HPX_THROW_EXCEPTION(invalid_status,
681  "Networking was disabled at configuration time. Please "
682  "reconfigure HPX using -DHPX_WITH_NETWORKING=On.",
683  "parcelhandler::put_parcels");
684 #endif
685  }

729  policies::message_handler* parcelhandler::get_message_handler(
730  char const* action, char const* message_handler_type,
731  std::size_t num_messages, std::size_t interval,
732  locality const& loc, error_code& ec)
733  {
734 #if defined(HPX_HAVE_NETWORKING)
735  std::unique_lock<mutex_type> l(handlers_mtx_);
736  handler_key_type key(loc, action);
737  message_handler_map::iterator it = handlers_.find(key);
738 
739  if (it == handlers_.end()) {
740  std::shared_ptr<policies::message_handler> p;
741 
742  {
743  // Just ignore the handlers_mtx_ while checking. We need to hold
744  // the lock here to avoid multiple registrations that happens
745  // right now in the parcel coalescing plugin
746  hpx::util::ignore_while_checking<std::unique_lock<mutex_type>> il(&l);
747  p.reset(hpx::create_message_handler(message_handler_type,
748  action, find_parcelport(loc.type()), num_messages, interval, ec));
749  }
750 
751  it = handlers_.find(key);
752  if (it != handlers_.end()) {
753  // if some other thread has created the entry in the meantime
754  l.unlock();
755  if (&ec != &throws) {
756  if ((*it).second.get())
757  ec = make_success_code();
758  else
759  ec = make_error_code(bad_parameter, lightweight);
760  }
761  return (*it).second.get();
762  }
763 
764  if (ec || !p.get()) {
765  // insert an empty entry into the map to avoid trying to
766  // create this handler again
767  p.reset();
768  std::pair<message_handler_map::iterator, bool> r =
769  handlers_.insert(message_handler_map::value_type(key, p));
770 
771  l.unlock();
772  if (!r.second) {
773  HPX_THROWS_IF(ec, internal_server_error,
774  "parcelhandler::get_message_handler",
775  "could not store empty message handler");
776  return nullptr;
777  }
778  return nullptr; // no message handler available
779  }
780 
781  std::pair<message_handler_map::iterator, bool> r =
782  handlers_.insert(message_handler_map::value_type(key, p));
783 
784  l.unlock();
785  if (!r.second) {
786  HPX_THROWS_IF(ec, internal_server_error,
787  "parcelhandler::get_message_handler",
788  "could not store newly created message handler");
789  return nullptr;
790  }
791  it = r.first;
792  }
793  else if (!(*it).second.get()) {
794  l.unlock();
795  if (&ec != &throws)
796  {
797  ec = make_error_code(bad_parameter, lightweight);
798  }
799  else
800  {
801  HPX_THROW_EXCEPTION(bad_parameter,
802  "parcelhandler::get_message_handler",
803  "couldn't find an appropriate message handler");
804  }
805  return nullptr; // no message handler available
806  }
807 
808  if (&ec != &throws)
809  ec = make_success_code();
810 
811  return (*it).second.get();
812 #else
813  HPX_THROW_EXCEPTION(invalid_status,
814  "Networking was disabled at configuration time. Please "
815  "reconfigure HPX using -DHPX_WITH_NETWORKING=On.",
816  "parcelhandler::get_message_handler");
817  return nullptr;
818 #endif
819  }

12 // This application will just sit and wait for being terminated from the
13 // console window for the specified amount of time. This is useful for testing
14 // the heartbeat tool which connects and disconnects to a running application.
15 int hpx_main(boost::program_options::variables_map& vm)
16 {
17  double const runfor = vm["runfor"].as<double>();
18 
19  hpx::cout << "Heartbeat Console, waiting for";
20  if (runfor > 0)
21  hpx::cout << " " << runfor << "[s].\n" << hpx::flush;
22  else
23  hpx::cout << "ever.\n" << hpx::flush;
24 
25  hpx::util::high_resolution_timer t;
26  while (runfor <= 0 || t.elapsed() < runfor)
27  {
28  hpx::this_thread::suspend(1000);
29  hpx::cout << "." << hpx::flush;
30  }
31 
32  hpx::cout << "\n" << hpx::flush;

34 }

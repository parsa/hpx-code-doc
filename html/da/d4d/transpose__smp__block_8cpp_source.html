187 void transpose(sub_block A, sub_block B, std::uint64_t block_order,
188  std::uint64_t tile_size)
189 {
190  if(tile_size < block_order)
191  {
192  for(std::uint64_t i = 0; i < block_order; i += tile_size)
193  {
194  for(std::uint64_t j = 0; j < block_order; j += tile_size)
195  {
196  std::uint64_t i_max = (std::min)(block_order, i + tile_size);
197  std::uint64_t j_max = (std::min)(block_order, j + tile_size);
198 
199  for(std::uint64_t it = i; it < i_max; ++it)
200  {
201  for(std::uint64_t jt = j; jt < j_max; ++jt)
202  {
203  B[it + block_order * jt] = A[jt + block_order * it];
204  }
205  }
206  }
207  }
208  }
209  else
210  {
211  for(std::uint64_t i = 0; i < block_order; ++i)
212  {
213  for(std::uint64_t j = 0; j < block_order; ++j)
214  {
215  B[i + block_order * j] = A[j + block_order * i];
216  }
217  }
218  }
219 }

232  double errsq =
233  transform_reduce(par, std::begin(range), std::end(range), 0.0,
234  [](double lhs, double rhs) { return lhs + rhs; },
235  [&](std::uint64_t b) -> double
236  {
237  double errsq = 0.0;
238  for(std::uint64_t i = 0; i < order; ++i)
239  {
240  double col_val = COL_SHIFT * i;
241  for(std::uint64_t j = 0; j < block_order; ++j)
242  {
243  double diff = trans[b][i * block_order + j] -
244  (col_val + ROW_SHIFT * (b * block_order + j));
245  errsq += diff * diff;
246  }
247  }
248  return errsq;
249  }
250  );
251 
252  if(verbose)
253  std::cout << " Squared sum of differences: " << errsq << "\n";
254

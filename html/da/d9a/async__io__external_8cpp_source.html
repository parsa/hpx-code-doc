19 struct registration_wrapper
20 {
21  registration_wrapper(hpx::runtime* rt, char const* name)
22  : rt_(rt)
23  {
24  // Register this thread with HPX, this should be done once for
25  // each external OS-thread intended to invoke HPX functionality.
26  // Calling this function more than once will silently fail (will
27  // return false).
28  hpx::register_thread(rt_, name);
29  }
30  ~registration_wrapper()
31  {
32  // Unregister the thread from HPX, this should be done once in the
33  // end before the external thread exists.
34  hpx::unregister_thread(rt_);
35  }
36 
37  hpx::runtime* rt_;
38 };

49 // this function will be executed by a dedicated OS thread
50 void do_async_io(char const* string_to_write,
51  std::shared_ptr<hpx::lcos::local::promise<int> > p,
52  hpx::runtime* rt)
53 {
54  // register this thread in order to be able to call HPX functionality
55  registration_wrapper wrap(rt, "external-io");
56 
57  // This IO operation will possibly block the IO thread in the
58  // kernel.
59  std::cout << "OS-thread: " << string_to_write << std::endl;
60 
61  // Create an HPX thread to guarantee that the promise::set_value
62  // function can be invoked safely.
63  hpx::threads::register_thread(hpx::util::bind(&set_value, p, 0));
64 }

66 // This function will be executed by an HPX thread
67 int io(char const* string_to_write)
68 {
69  std::shared_ptr<hpx::lcos::local::promise<int> > p =
70  std::make_shared<hpx::lcos::local::promise<int> >();
71 
72  // Create a new external OS-thread and schedule the handler to
73  // run on one of its OS-threads.
74  hpx::compat::thread external_os_thread(
75  hpx::util::bind(&do_async_io, string_to_write, p, hpx::get_runtime_ptr()));
76 
77  int result = p->get_future().get();
78 
79  // wait for the external thread to exit
80  external_os_thread.join();
81 
82  return result;
83 }

29 std::size_t thread_affinity_worker(std::size_t desired)
30 {
31  // Returns the OS-thread number of the worker that is running this
32  // PX-thread.
33  std::size_t current = hpx::get_worker_thread_num();
34  if (current == desired)
35  {
36 #if defined(HPX_HAVE_HWLOC) && !defined(__APPLE__)
37  bool numa_sensitive = hpx::is_scheduler_numa_sensitive();
38 
39  // extract the desired affinity mask
40  hpx::runtime & rt = hpx::get_runtime();
41  hpx::threads::topology const& t = rt.get_topology();
42  hpx::threads::mask_type desired_mask = t.get_thread_affinity_mask(
43  hpx::resource::get_partitioner().get_pu_num(current));
44 
45  std::size_t logical_idx = hpx::threads::find_first(desired_mask);
46 
47  std::size_t idx = 0;
48 
49  hwloc_topology_t topo;
50  hwloc_topology_init(&topo);
51  hwloc_topology_load(topo);
52 
53  int const pu_depth = hwloc_get_type_or_below_depth(topo, HWLOC_OBJ_PU);
54  hwloc_obj_t const pu_obj = hwloc_get_obj_by_depth(topo, pu_depth, logical_idx);
55  idx = pu_obj->os_index;
56 
57 
58  // retrieve the current affinity mask
59  hwloc_cpuset_t cpuset = hwloc_bitmap_alloc();
60  hwloc_bitmap_zero(cpuset);
61  if (0 == hwloc_get_cpubind(topo, cpuset, HWLOC_CPUBIND_THREAD)) {
62  // sadly get_cpubind is not implemented for Windows based systems
63  hwloc_cpuset_t cpuset_cmp = hwloc_bitmap_alloc();
64  hwloc_bitmap_zero(cpuset_cmp);
65  hwloc_bitmap_only(cpuset_cmp, unsigned(idx));
66  HPX_TEST(hwloc_bitmap_compare(cpuset, cpuset_cmp) == 0);
67  hwloc_bitmap_free(cpuset_cmp);
68  }
69  else
70  {
71  HPX_TEST(false && "hwloc_get_cpubind(topo, cpuset, \
72  HWLOC_CPUBIND_THREAD) failed!");
73  }
74 
75  hwloc_bitmap_free(cpuset);
76  hwloc_topology_destroy(topo);
77 #endif
78  return desired;
79  }
80 
81  // This PX-thread has been run by the wrong OS-thread, make the foreman
82  // try again by rescheduling it.
83  return std::size_t(-1);
84 }

94 void thread_affinity_foreman()
95 {
96  // Get the number of worker OS-threads in use by this locality.
97  std::size_t const os_threads = hpx::get_os_thread_count();
98 
99  // Find the global name of the current locality.
100  hpx::naming::id_type const here = hpx::find_here();
101 
102  // Populate a set with the OS-thread numbers of all OS-threads on this
103  // locality. When the hello world message has been printed on a particular
104  // OS-thread, we will remove it from the set.
105  std::set<std::size_t> attendance;
106  for (std::size_t os_thread = 0; os_thread < os_threads; ++os_thread)
107  attendance.insert(os_thread);
108 
109  // As long as there are still elements in the set, we must keep scheduling
110  // PX-threads. Because HPX features work-stealing task schedulers, we have
111  // no way of enforcing which worker OS-thread will actually execute
112  // each PX-thread.
113  while (!attendance.empty())
114  {
115  // Each iteration, we create a task for each element in the set of
116  // OS-threads that have not said "Hello world". Each of these tasks
117  // is encapsulated in a future.
118  std::vector<hpx::lcos::future<std::size_t> > futures;
119  futures.reserve(attendance.size());
120 
121  for (std::size_t worker : attendance)
122  {
123  // Asynchronously start a new task. The task is encapsulated in a
124  // future, which we can query to determine if the task has
125  // completed.
126  typedef thread_affinity_worker_action action_type;
127  futures.push_back(hpx::async<action_type>(here, worker));
128  }
129 
130  // Wait for all of the futures to finish. The callback version of the
131  // hpx::lcos::wait function takes two arguments: a vector of futures,
132  // and a binary callback. The callback takes two arguments; the first
133  // is the index of the future in the vector, and the second is the
134  // return value of the future. hpx::lcos::wait doesn't return until
135  // all the futures in the vector have returned.
136  using hpx::util::placeholders::_1;
137  hpx::lcos::wait_each(hpx::util::unwrapping(
138  hpx::util::bind(&check_in, std::ref(attendance), _1)), futures);
139  }
140 }

144 int hpx_main(boost::program_options::variables_map& /*vm*/)
146 {
147  {
148  // Get a list of all available localities.
149  std::vector<hpx::naming::id_type> localities =
150  hpx::find_all_localities();
151 
152  // Reserve storage space for futures, one for each locality.
153  std::vector<hpx::lcos::future<void> > futures;
154  futures.reserve(localities.size());
155 
156  for (hpx::naming::id_type const& node : localities)
157  {
158  // Asynchronously start a new task. The task is encapsulated in a
159  // future, which we can query to determine if the task has
160  // completed.
161  typedef thread_affinity_foreman_action action_type;
162  futures.push_back(hpx::async<action_type>(node));
163  }
164 
165  // The non-callback version of hpx::lcos::wait takes a single parameter,
166  // a future of vectors to wait on. hpx::lcos::wait only returns when
167  // all of the futures have finished.
168  hpx::wait_all(futures);
169  }
170 
171  // Initiate shutdown of the runtime system.
172  hpx::finalize();
173  return hpx::util::report_errors();
174 }

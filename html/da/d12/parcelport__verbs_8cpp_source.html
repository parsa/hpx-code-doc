109  // --------------------------------------------------------------------
110  // Simple atomic counter we use for tags
111  // When a parcel is sent to a remote locality, it may need to pull zero copy
112  // chunks from us. We keep the chunks until the remote locality sends a zero byte
113  // message with the tag we gave them and then we know it is safe to release the
114  // memory back to the pool.
115  // The tags can have a short lifetime, but must be unique, so we encode the ip
116  // address with a counter to generate tags per destination.
117  // The tag is sent in immediate data so must be 32bits only : Note that the tag
118  // only has a lifetime of the unprocessed parcel, so it can be reused as soon as
119  // the parcel has been completed and therefore a 16bit count is sufficient as we
120  // only keep a few parcels per locality in flight at a time
121  // --------------------------------------------------------------------
122  struct tag_provider {
123  tag_provider() : next_tag_(1) {}
124 
125  uint32_t next(uint32_t ip_addr)
126  {
127  // @TODO track wrap around and collisions (how?)
128  return (next_tag_++ & 0x0000FFFF) + (ip_addr & 0xFFFF0000);
129  }
130 
131  // using 16 bits currently.
132  std::atomic<uint32_t> next_tag_;
133  };

596  std::uint32_t dest_ip = client->get_remote_address()->sin_addr.s_addr;
597  if (client->is_client_endpoint()) {

599  << ipaddress(_ibv_ip) << "to "
600  << ipaddress(dest_ip) << "( " << ipaddress(_ibv_ip) << ")");
601  }
602  else {
603  LOG_DEVEL_MSG("Connection established from "
604  << ipaddress(dest_ip) << "to "
605  << ipaddress(_ibv_ip) << "( " << ipaddress(_ibv_ip) << ")");
606  }
607  //
608  auto present = ip_endpoint_map.is_in_map(dest_ip);
609  if (!present.second) {
610  ip_endpoint_map.insert(std::make_pair(dest_ip, client));
611  LOG_DEVEL_MSG("handle_verbs_connection OK adding to ip_qp_map "
612  << ipaddress(dest_ip));
613  }
614  else {
615  throw std::runtime_error("verbs parcelport "
616  ": should not be receiving a connection more than once");
617  }
618  }

620  // --------------------------------------------------------------------
621  // When a send completes take one of two actions ...
622  // if there are no zero copy chunks, we consider the parcel sending complete
623  // so release memory and trigger write_handler.
624  // If there are zero copy chunks to be RDMA GET from the remote end, then
625  // we hold onto data until they have completed.
626  // --------------------------------------------------------------------
627  void handle_send_completion(uint64_t wr_id)
628  {
629  bool found_wr_id = false;
630  active_send_iterator current_send;
631  {
632  // we must be very careful here.
633  // if the lock is not obtained and this thread is waiting, then
634  // zero copy GETs might complete and another thread might receive
635  // a zero copy complete message then delete the current send data
636  // whilst we are still waiting
637 #if HPX_PARCELPORT_VERBS_USE_CONCURRENT_MAPS
638  auto is_present = SendCompletionMap.is_in_map(wr_id);
639  if (is_present.second) {
640  found_wr_id = true;
641  current_send = is_present.first->second;
642  SendCompletionMap.erase(is_present.first);
643  LOG_DEBUG_MSG("erasing " << hexpointer(wr_id)
644  << "from SendCompletionMap : size before erase "
645  << SendCompletionMap.size());
646  }
647 #else
648  unique_lock lock(SendCompletionMap_mutex);
649  auto it = SendCompletionMap.find(wr_id);
650  if (it!=SendCompletionMap.end()) {
651  found_wr_id = true;
652  current_send = it->second;
653  SendCompletionMap.erase(it);
654  LOG_DEBUG_MSG("erasing " << hexpointer(wr_id)
655  << "from SendCompletionMap : size before erase "
656  << SendCompletionMap.size());
657  }
658 #endif
659  else {
660 #if HPX_PARCELPORT_VERBS_IMM_UNSUPPORTED
661  lock.unlock();
662  handle_tag_send_completion(wr_id);
663  return;
664 #else
665  LOG_ERROR_MSG("FATAL : SendCompletionMap did not find "
666  << hexpointer(wr_id));
667  std::terminate();
668 #endif
669  }
670  }
671  if (found_wr_id) {
672  // if the send had no zero_copy regions, then it has completed
673  if (!current_send->has_zero_copy) {
674  LOG_DEBUG_MSG("Deleting send data "
675  << hexpointer(&(*current_send)) << "normal");
676  delete_send_data(current_send);
677  }
678  // if another thread signals to say zero-copy complete, delete the data
679  else if (current_send->delete_flag.test_and_set(
680  std::memory_order_acquire))
681  {
682  LOG_DEBUG_MSG("Deleting send data "
683  << hexpointer(&(*current_send)) << "after race detection");
684  delete_send_data(current_send);
685  }
686  }
687  else {
688  throw std::runtime_error("RDMA Send completed with unmatched Id");
689  }
690  }

692  // --------------------------------------------------------------------
693  // When a recv completes, take one of two actions ...
694  // if there are no zero copy chunks, we consider the parcel receive complete
695  // so release memory and trigger write_handler.
696  // If there are zero copy chunks to be RDMA GET'ed from the remote end,
697  // then we hold onto data until they have completed.
698  // --------------------------------------------------------------------
699  void handle_recv_completion(uint64_t wr_id, verbs_endpoint* client)
700  {
701  util::high_resolution_timer timer;
702 
703  // store details about this parcel so that all memory buffers can be kept
704  // until all recv operations have completed.
705  active_recv_iterator current_recv;
706  {
707  scoped_lock lock(active_recv_mutex);
708  active_recvs.emplace_back();
709  current_recv = std::prev(active_recvs.end());
710  LOG_DEBUG_MSG("Active recv after insert size "
711  << hexnumber(active_recvs.size()));
712  }
713  parcel_recv_data &recv_data = *current_recv;
714  // get the header of the new message/parcel
715  recv_data.header_region = (verbs_memory_region *)wr_id;
716  header_type *h = (header_type*)recv_data.header_region->get_address();
717  recv_data.message_region = nullptr;
718  // zero copy chunks we have to GET from the source locality
719  if (h->piggy_back()) {
720  recv_data.rdma_count = h->num_chunks().first;
721  }
722  else {
723  recv_data.rdma_count = 1 + h->num_chunks().first;
724  }
725  // each parcel has a unique tag which we use to organize
726  // zero-copy data if we need any
727  recv_data.tag = h->tag();
728  //
729  LOG_DEBUG_MSG( "received IBV_WC_RECV " <<
730  "buffsize " << decnumber(h->size())
731  << "chunks zerocopy( " << decnumber(h->num_chunks().first) << ") "
732  << ", chunk_flag " << decnumber(h->header_length())
733  << ", normal( " << decnumber(h->num_chunks().second) << ") "
734  << " chunkdata " << decnumber((h->chunk_data()!=nullptr))
735  << " piggyback " << decnumber((h->piggy_back()!=nullptr))
736  << " tag " << hexuint32(h->tag())
737  << " total receives " << decnumber(handled_receives)
738  );
739 
740  // setting this flag to false - if more data is needed -
741  // disables final parcel receive call
742  bool parcel_complete = true;
743 
744  // if message was not piggybacked
745  char *piggy_back = h->piggy_back();
746  char *chunk_data = h->chunk_data();
747  if (chunk_data) {
748  // all the info about chunks we need is stored inside the header
749  recv_data.chunks.resize(h->num_chunks().first + h->num_chunks().second);
750  size_t chunkbytes =
751  recv_data.chunks.size() * sizeof(serialization::serialization_chunk);
752  std::memcpy(recv_data.chunks.data(), chunk_data, chunkbytes);
753  LOG_DEBUG_MSG("Copied chunk data from header : size "
754  << decnumber(chunkbytes));
755 
756  // setup info for zero-copy rdma get chunks (if there are any)
757  if (recv_data.rdma_count>0) {
758  parcel_complete = false;
759  int index = 0;
760  LOG_EXCLUSIVE(
761  for (const serialization::serialization_chunk &c : recv_data.chunks)
762  {
763  LOG_DEBUG_MSG("recv : chunk : size " << hexnumber(c.size_)
764  << " type " << decnumber((uint64_t)c.type_)
765  << " rkey " << decnumber(c.rkey_)
766  << " cpos " << hexpointer(c.data_.cpos_)
767  << " pos " << hexpointer(c.data_.pos_)
768  << " index " << decnumber(c.data_.index_));
769  })
770  for (serialization::serialization_chunk &c : recv_data.chunks) {
771  if (c.type_ == serialization::chunk_type_pointer) {
772  verbs_memory_region *get_region =
773  chunk_pool_->allocate_region(c.size_);
774 
775  LOG_DEBUG_MSG("RDMA Get address " << hexpointer(c.data_.cpos_)
776  << " rkey " << decnumber(c.rkey_)
777  << " size " << hexnumber(c.size_)
778  << " tag " << hexuint32(recv_data.tag)
779  << " local address " << get_region->get_address()
780  << " length " << c.size_);
781  recv_data.zero_copy_regions.push_back(get_region);
782  LOG_DEBUG_MSG("Zero copy regions size is (create) "
783  << decnumber(recv_data.zero_copy_regions.size()));
784  // put region into map before posting read in case it
785  // completes whilst this thread is suspended
786  {
787 #if HPX_PARCELPORT_VERBS_USE_CONCURRENT_MAPS
788  ReadCompletionMap.insert(
789  std::make_pair((uint64_t)get_region, current_recv));
790 #else
791  scoped_lock lock(ReadCompletionMap_mutex);
792  ReadCompletionMap.insert(
793  std::make_pair((uint64_t)get_region, current_recv));
794 #endif
795  }
796  // overwrite the serialization data to account for the
797  // local pointers instead of remote ones
798  const void *remoteAddr = c.data_.cpos_;
799  recv_data.chunks[index] =
800  hpx::serialization::create_pointer_chunk(
801  get_region->get_address(), c.size_, c.rkey_);
802  ++total_reads;
803  // post the rdma read/get
804  client->post_read(get_region, c.rkey_, remoteAddr, c.size_);
805  }
806  index++;
807  }
808  }
809  }
810  else {
811  // no chunk information was sent in this message, so we must GET it
812  parcel_complete = false;
813  LOG_ERROR_MSG("@TODO implement RDMA GET of mass chunk information "
814  "when header too small");
815  std::terminate();
816  throw std::runtime_error("@TODO implement RDMA GET of mass chunk "
817  "information when header too small");
818  }
819 
820  LOG_DEBUG_MSG("piggy_back is " << hexpointer(piggy_back)
821  << " chunk data is " << hexpointer(h->chunk_data()));
822  // if the main serialization chunk is piggybacked in second SGE
823  if (piggy_back) {
824  if (parcel_complete) {
825  rcv_data_type wrapped_pointer(piggy_back, h->size(),
826  util::bind(&parcelport::delete_recv_data, this, current_recv),
827  chunk_pool_.get(), nullptr);
828  rcv_buffer_type buffer(std::move(wrapped_pointer), chunk_pool_.get());
829  LOG_DEBUG_MSG("calling parcel decode for complete NORMAL parcel");
830  parcelset::decode_message_with_chunks<parcelport, rcv_buffer_type>
831  (*this, std::move(buffer), 0, recv_data.chunks);
832  LOG_DEBUG_MSG("parcel decode called for complete NORMAL parcel");
833  }
834  }
835  else {
836  std::size_t size = h->get_message_rdma_size();
837  verbs_memory_region *get_region = chunk_pool_->allocate_region(size);
838  get_region->set_message_length(size);
839  recv_data.zero_copy_regions.push_back(get_region);
840  // put region into map before posting read in case it completes
841  // whilst this thread is suspended during map insertion
842  {
843 
844 #if HPX_PARCELPORT_VERBS_USE_CONCURRENT_MAPS
845  ReadCompletionMap.insert(
846  std::make_pair((uint64_t)get_region, current_recv));
847 #else
848  scoped_lock lock(ReadCompletionMap_mutex);
849  ReadCompletionMap.insert(
850  std::make_pair((uint64_t)get_region, current_recv));
851 #endif
852  }
853  const void *remoteAddr = h->get_message_rdma_addr();
854  LOG_DEBUG_MSG("@TODO Pushing back an extra chunk description");
855  recv_data.chunks.push_back(
856  hpx::serialization::create_pointer_chunk(get_region->get_address(),
857  size, h->get_message_rdma_key()));
858  ++total_reads;
859  client->post_read(get_region, h->get_message_rdma_key(), remoteAddr,
860  h->get_message_rdma_size());
861  }

868  // data.time_ = timer.elapsed_nanoseconds() - data.time_;

936  {
937  bool found_wr_id;
938  active_recv_iterator current_recv;
939  {
940 #if HPX_PARCELPORT_VERBS_USE_CONCURRENT_MAPS
941  auto is_present = ReadCompletionMap.is_in_map(wr_id);
942  if (is_present.second) {

945  LOG_DEBUG_MSG("erasing " << hexpointer(wr_id)
946  << "from ReadCompletionMap : size before erase "
947  << ReadCompletionMap.size());
948  ReadCompletionMap.erase(is_present.first);
949  }
950 #else
951  scoped_lock lock(ReadCompletionMap_mutex);
952  auto it = ReadCompletionMap.find(wr_id);
953  if (it!=ReadCompletionMap.end()) {
954  found_wr_id = true;
955  current_recv = it->second;
956  LOG_DEBUG_MSG("erasing " << hexpointer(wr_id)
957  << "from ReadCompletionMap : size before erase "
958  << ReadCompletionMap.size());
959  ReadCompletionMap.erase(it);
960  }
961 #endif
962  else {
963  LOG_ERROR_MSG("Fatal error as wr_id is not in completion map");
964  std::terminate();
965  }
966  }
967  if (found_wr_id) {
968  parcel_recv_data &recv_data = *current_recv;
969  LOG_DEBUG_MSG("RDMA Get tag " << hexuint32(recv_data.tag)
970  << " has count of " << decnumber(recv_data.rdma_count));
971  if (--recv_data.rdma_count > 0) {
972  // we can't do anything until all zero copy chunks are here
973  return;
974  }
975  //
976 #if HPX_PARCELPORT_VERBS_IMM_UNSUPPORTED
977  LOG_DEBUG_MSG("RDMA Get tag " << hexuint32(recv_data.tag)
978  << " has completed : posting 4 byte ack to origin");
979  // allocate space for a single uint32_t
980  verbs_memory_region *tag_region = chunk_pool_->allocate_region(4);
981  uint32_t *tag_memory = (uint32_t*)(tag_region->get_address());
982  *tag_memory = recv_data.tag;
983  tag_region->set_message_length(4);
984  client->post_send(tag_region, true, false, 0);
985 #else
986  LOG_DEBUG_MSG("RDMA Get tag " << hexuint32(recv_data.tag)
987  << " has completed : posting zero byte ack to origin");
988  // convert uint32 to uint64 so we can use it as a fake message region
989  // (wr_id only for 0 byte send)
990  uint64_t fake_region = recv_data.tag;
991  client->post_send_x0((verbs_memory_region*)fake_region,
992  false, true, recv_data.tag);
993 #endif
994  //
995  LOG_DEBUG_MSG("Zero copy regions size is (completion) "
996  << decnumber(recv_data.zero_copy_regions.size()));
997 
998  header_type *h = (header_type*)recv_data.header_region->get_address();
999  LOG_DEBUG_MSG( "rdma get completion "
1000  << "buffsize " << decnumber(h->size())
1001  << "chunks zerocopy( " << decnumber(h->num_chunks().first) << ") "
1002  << ", chunk_flag " << decnumber(h->header_length())
1003  << ", normal( " << decnumber(h->num_chunks().second) << ") "
1004  << " chunkdata " << decnumber((h->chunk_data()!=nullptr))
1005  << " piggyback " << decnumber((h->piggy_back()!=nullptr))
1006  << " tag " << hexuint32(h->tag())
1007  );

1011  if (message) {
1012  message_length = h->size();
1013  }
1014  else {

1016  recv_data.zero_copy_regions.back();
1017  recv_data.zero_copy_regions.resize(
1018  recv_data.zero_copy_regions.size()-1);
1019  message = static_cast<char *>(message_region->get_address());

1021  LOG_DEBUG_MSG("No piggy_back message, RDMA GET : "
1022  << hexpointer(message_region)
1023  << hexpointer(recv_data.message_region)
1024  << " length " << decnumber(message_length));
1025  }
1026 
1027  LOG_DEBUG_MSG("Creating a release buffer callback for tag "
1028  << hexuint32(recv_data.tag));
1029  rcv_data_type wrapped_pointer(message, message_length,
1030  util::bind(&parcelport::delete_recv_data, this, current_recv),
1031  chunk_pool_.get(), nullptr);
1032  rcv_buffer_type buffer(std::move(wrapped_pointer), chunk_pool_.get());
1033  LOG_DEBUG_MSG("calling parcel decode for complete ZEROCOPY parcel");
1034 
1035  LOG_EXCLUSIVE(
1036  for (serialization::serialization_chunk &c : recv_data.chunks) {
1037  LOG_DEBUG_MSG("get : chunk : size " << hexnumber(c.size_)

1040  << " cpos " << hexpointer(c.data_.cpos_)
1041  << " pos " << hexpointer(c.data_.pos_)
1042  << " index " << decnumber(c.data_.index_));

1046  //buffer.data_.resize(static_cast<std::size_t>(h->size()));
1047  //buffer.data_size_ = h->size();
1048  buffer.chunks_.resize(recv_data.chunks.size());
1049  decode_message_with_chunks(*this, std::move(buffer), 0, recv_data.chunks);
1050  LOG_DEBUG_MSG("parcel decode called for ZEROCOPY complete parcel");

1057  // --------------------------------------------------------------------
1058  // Every (signalled) rdma operation triggers a completion event when
1059  // it completes, the controller calls this callback function and we
1060  // must clean up all temporary memory etc and signal hpx when sends
1061  // or receives finish.
1062  // --------------------------------------------------------------------
1063  int handle_verbs_completion(const struct ibv_wc completion,
1064  verbs_endpoint *client)
1065  {
1066  uint64_t wr_id = completion.wr_id;
1067  LOG_DEVEL_MSG("Handle verbs completion " << hexpointer(wr_id)
1068  << decnumber(completion.qp_num)
1069  << sockaddress(client->get_remote_address())
1070  << verbs_completion_queue::wc_opcode_str(completion.opcode));
1071 
1072  // get the header of the new message/parcel
1073 
1074  LOG_EXCLUSIVE(
1075  if (0 && completion.opcode==IBV_WC_RECV) {
1076  auto region = (verbs_memory_region *)wr_id;
1077  std::stringstream temp;
1078  temp << "Completion Buffer "
1079  << verbs_completion_queue::wc_opcode_str(completion.opcode) << ":\n";
1080  if (completion.byte_len<1024) {
1081  int P = completion.byte_len;
1082  uintptr_t *ptr = reinterpret_cast<uintptr_t*>(region->get_address());
1083  for(int j=0; j<P/8; ++j) {
1084  temp << dec4(j) << " " << hexuint64(ptr[j]) << "\n";
1085  }
1086  }
1087  temp << std::endl;
1088  std::cout << temp.str().c_str();
1089  }
1090  )
1091 
1092  ++completions_handled;
1093 
1094  if (completion.opcode==IBV_WC_SEND) {
1095  LOG_DEBUG_MSG("Handle general completion " << hexpointer(wr_id)
1096  << verbs_completion_queue::wc_opcode_str(completion.opcode));
1097  handle_send_completion(wr_id);
1098  return 0;
1099  }
1100 
1101  //
1102  // When an Rdma Get operation completes, either add it to an ongoing parcel
1103  // receive, or if it is the last one, trigger decode message
1104  //
1105  else if (completion.opcode==IBV_WC_RDMA_READ) {
1106  handle_rdma_get_completion(wr_id, client);
1107  return 0;
1108  }
1109 
1110  //
1111  // A zero byte receive indicates we are being informed that remote
1112  // GET operations are complete we can release any data we were holding onto
1113  // and signal a send as finished.
1114  // On hardware not supporting immediate data, a 4 byte tag message is used.
1115  //
1116  else if (completion.opcode==IBV_WC_RECV && completion.byte_len<=4) {
1117  handle_tag_recv_completion(wr_id, completion.imm_data, client);
1118  return 0;
1119  }
1120 
1121  //
1122  // When an unmatched receive completes, it is a new parcel, if everything
1123  // fits into the header, call decode message, otherwise, queue all the
1124  // Rdma Get operations necessary to complete the message
1125  //
1126  else if (completion.opcode==IBV_WC_RECV) {
1127  LOG_DEBUG_MSG("Entering receive (completion handler) section with "
1128  "received size " << decnumber(completion.byte_len));
1129  handle_recv_completion(wr_id, client);
1130  return 0;
1131  }
1132 
1133  throw std::runtime_error("Unexpected opcode in handle_verbs_completion "
1134  + verbs_sender_receiver::
1135  ibv_wc_opcode_string((ibv_wr_opcode)(completion.opcode)));
1136  return 0;
1137  }

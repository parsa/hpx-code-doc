60  // Creation function for action invocation counter
61  naming::gid_type action_invocation_counter_creator(counter_info const& info,
62  hpx::actions::detail::invocation_count_registry& registry,
63  error_code& ec)
64  {
65  switch (info.type_) {
66  case counter_raw:
67  {
68  counter_path_elements paths;
69  get_counter_path_elements(info.fullname_, paths, ec);
70  if (ec) return naming::invalid_gid;
71 
72  if (paths.parentinstance_is_basename_) {
73  HPX_THROWS_IF(ec, bad_parameter,
74  "action_invocation_counter_creator",
75  "invalid action invocation counter name (instance name "
76  "must not be a valid base counter name)");
77  return naming::invalid_gid;
78  }
79 
80  if (paths.parameters_.empty()) {
81  HPX_THROWS_IF(ec, bad_parameter,
82  "action_invocation_counter_creator",
83  "invalid action invocation counter parameter: must "
84  "specify an action type");
85  return naming::invalid_gid;
86  }
87 
88  // ask registry
89  hpx::util::function_nonser<std::int64_t(bool)> f =
90  registry.get_invocation_counter(paths.parameters_);
91 
92  return detail::create_raw_counter(info, std::move(f), ec);
93  }
94  break;
95 
96  default:
97  HPX_THROWS_IF(ec, bad_parameter,
98  "action_invocation_counter_creator",
99  "invalid counter type requested");
100  return naming::invalid_gid;
101  }
102  }

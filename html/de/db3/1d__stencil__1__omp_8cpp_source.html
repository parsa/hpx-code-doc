6 // This is the first in a series of examples demonstrating the development of a
7 // fully distributed solver for a simple 1D heat distribution problem.
8 //
9 // This example provides a serial base line implementation. No parallelization
10 // is performed.
11 //
12 // The only difference to 1d_stencil_1 is that this example uses OpenMP for
13 // parallelizing the inner loop.

46 struct stepper
47 {
48  // Our partition type
49  typedef double partition;
50 
51  // Our data for one time step
52  typedef std::vector<partition> space;
53 
54  // Our operator
55  static double heat(double left, double middle, double right)
56  {
57  return middle + (k*dt/(dx*dx)) * (left - 2*middle + right);
58  }
59 
60  // do all the work on 'nx' data points for 'nt' time steps
61  space do_work(std::size_t nx, std::size_t nt)
62  {
63  // U[t][i] is the state of position i at time t.
64  std::vector<space> U(2);
65  for (space& s : U)
66  s.resize(nx);
67 
68  // Initial conditions: f(0, i) = i
69  for (std::size_t i = 0; i != nx; ++i)
70  U[0][i] = double(i);
71 
72  // Actual time step loop
73  for (std::size_t t = 0; t != nt; ++t)
74  {
75  space const& current = U[t % 2];
76  space& next = U[(t + 1) % 2];
77 
78  next[0] = heat(current[nx-1], current[0], current[1]);
79 
80  // Visual Studio requires OMP loop variables to be signed :/
81  # pragma omp parallel for
82  for (std::int64_t i = 1; i < std::int64_t(nx-1); ++i)
83  next[i] = heat(current[i-1], current[i], current[i+1]);
84 
85  next[nx-1] = heat(current[nx-2], current[nx-1], current[0]);
86  }
87 
88  // Return the solution at time-step 'nt'.
89  return U[nt % 2];
90  }
91 };

39  inline void
40  partition3d::init_dimension(std::string const& datafilename, int d,
41  dimension const& dim, char const* name, boost::scoped_array<double>& values)
42  {
43  // Store all parameters.
44  dim_[d] = dim;
45 
46  // Account for necessary overlap on the right hand side of the data
47  // interval (the interpolation algorithm we're using does not go beyond
48  // the left hand side).
49  std::size_t count = dim.count_;
50  if (dim.offset_ + dim.count_ < dim.size_-2) {
51  dim_[d].count_ += 2;
52  ++count;
53  }
54 
55  // Read the full data range.
56  values.reset(new double[dim_[d].count_]);
57  extract_data(datafilename, name, values.get(), dim.offset_,
58  dim_[d].count_);
59 
60  // Extract range (without ghost-zones).
61  min_value_[d] = values[0];
62  max_value_[d] = values[count-1];
63  delta_[d] = values[1] - values[0];
64  }

115  inline std::size_t
116  partition3d::get_index(dimension::type d, double value) const
117  {
118  if (value < min_value_[d] || value > max_value_[d])
119  {
120  HPX_THROW_EXCEPTION(hpx::bad_parameter,
121  "sheneos::partition3d::get_index",
122  "argument out of range");
123  return 0;
124  }
125 
126  std::size_t index = static_cast<std::size_t>(
127  (value - min_value_[d]) / delta_[d]);
128 
129  // Either the index has to be inside bounds or the requested value
130  // corresponds to the right end edge of the managed data range.
131  HPX_ASSERT(index < dim_[d].count_ ||
132  (index == dim_[d].count_ && value == max_value_[d]));
133 
134  return index;
135  }

151  inline double partition3d::tl_interpolate(double* values,
152  std::size_t idx_x, std::size_t idx_y, std::size_t idx_z,
153  double delta_ye, double delta_logtemp, double delta_logrho) const
154  {
155  double value000 = values[index(idx_x, idx_y, idx_z, dim_)];
156  double value001 = values[index(idx_x, idx_y, idx_z+1, dim_)];
157  double value010 = values[index(idx_x, idx_y+1, idx_z, dim_)];
158  double value011 = values[index(idx_x, idx_y+1, idx_z+1, dim_)];
159  double value100 = values[index(idx_x+1, idx_y, idx_z, dim_)];
160  double value101 = values[index(idx_x+1, idx_y, idx_z+1, dim_)];
161  double value110 = values[index(idx_x+1, idx_y+1, idx_z, dim_)];
162  double value111 = values[index(idx_x+1, idx_y+1, idx_z+1, dim_)];
163 
164  double comp_delta_ye = 1. - delta_ye;
165  double comp_delta_logtemp = 1. - delta_logtemp;
166  double comp_delta_logrho = 1. - delta_logrho;
167 
168  return value000 * comp_delta_ye * comp_delta_logtemp * comp_delta_logrho +
169  value001 * comp_delta_ye * comp_delta_logtemp * delta_logrho +
170  value010 * comp_delta_ye * delta_logtemp * comp_delta_logrho +
171  value011 * comp_delta_ye * delta_logtemp * delta_logrho +
172  value100 * delta_ye * comp_delta_logtemp * comp_delta_logrho +
173  value101 * delta_ye * comp_delta_logtemp * delta_logrho +
174  value110 * delta_ye * delta_logtemp * comp_delta_logrho +
175  value111 * delta_ye * delta_logtemp * delta_logrho;
176  }

179  std::vector<double> partition3d::interpolate(double ye, double temp,
180  double rho, std::uint32_t eosvalues) const
181  {
182  double logrho = std::log10(rho);
183  double logtemp = std::log10(temp);
184 
185  std::size_t idx_ye = get_index(dimension::ye, ye);
186  std::size_t idx_logtemp = get_index(dimension::temp, logtemp);
187  std::size_t idx_logrho = get_index(dimension::rho, logrho);
188 
189  double delta_ye = (ye - ye_values_[idx_ye]) / delta_[dimension::ye];
190  double delta_logtemp = (logtemp - logtemp_values_[idx_logtemp])
191  / delta_[dimension::temp];
192  double delta_logrho = (logrho - logrho_values_[idx_logrho])
193  / delta_[dimension::rho];
194 
195  std::vector<double> results;
196  results.reserve(19);
197 
198  // Calculate all required values.
199  if (eosvalues & logpress) {
200  double value = tl_interpolate(logpress_values_.get(),
201  idx_ye, idx_logtemp, idx_logrho,
202  delta_ye, delta_logtemp, delta_logrho);
203  results.push_back(std::pow(10., value));
204  }
205  if (eosvalues & logenergy) {
206  double value = tl_interpolate(logenergy_values_.get(),
207  idx_ye, idx_logtemp, idx_logrho,
208  delta_ye, delta_logtemp, delta_logrho);
209  results.push_back(std::pow(10., value) - energy_shift_);
210  }
211  if (eosvalues & entropy) {
212  results.push_back(tl_interpolate(entropy_values_.get(),
213  idx_ye, idx_logtemp, idx_logrho,
214  delta_ye, delta_logtemp, delta_logrho));
215  }
216  if (eosvalues & munu) {
217  results.push_back(tl_interpolate(munu_values_.get(),
218  idx_ye, idx_logtemp, idx_logrho,
219  delta_ye, delta_logtemp, delta_logrho));
220  }
221  if (eosvalues & cs2) {
222  results.push_back(tl_interpolate(cs2_values_.get(),
223  idx_ye, idx_logtemp, idx_logrho,
224  delta_ye, delta_logtemp, delta_logrho));
225  }
226  if (eosvalues & dedt) {
227  results.push_back(tl_interpolate(dedt_values_.get(),
228  idx_ye, idx_logtemp, idx_logrho,
229  delta_ye, delta_logtemp, delta_logrho));
230  }
231  if (eosvalues & dpdrhoe) {
232  results.push_back(tl_interpolate(dpdrhoe_values_.get(),
233  idx_ye, idx_logtemp, idx_logrho,
234  delta_ye, delta_logtemp, delta_logrho));
235  }
236  if (eosvalues & dpderho) {
237  results.push_back(tl_interpolate(dpderho_values_.get(),
238  idx_ye, idx_logtemp, idx_logrho,
239  delta_ye, delta_logtemp, delta_logrho));
240  }
241 
242  return results;
243  }

262  double partition3d::interpolate_one(double ye, double temp,
263  double rho, std::uint32_t eosvalue) const
264  {
265  double logrho = std::log10(rho);
266  double logtemp = std::log10(temp);
267 
268  std::size_t idx_ye = get_index(dimension::ye, ye);
269  std::size_t idx_logtemp = get_index(dimension::temp, logtemp);
270  std::size_t idx_logrho = get_index(dimension::rho, logrho);
271 
272  double delta_ye = (ye - ye_values_[idx_ye]) / delta_[dimension::ye];
273  double delta_logtemp = (logtemp - logtemp_values_[idx_logtemp])
274  / delta_[dimension::temp];
275  double delta_logrho = (logrho - logrho_values_[idx_logrho])
276  / delta_[dimension::rho];
277 
278  if (detail::more_than_one_value_requested(eosvalue)) {
279  HPX_THROW_EXCEPTION(hpx::bad_parameter,
280  "partition3d::interpolate_one",
281  "requested to interpolate more than one physical value: " +
282  std::to_string(eosvalue));
283  }
284 
285  // Calculate all required values.
286  switch (eosvalue) {
287  case logpress:
288  {
289  double value = tl_interpolate(logpress_values_.get(),
290  idx_ye, idx_logtemp, idx_logrho,
291  delta_ye, delta_logtemp, delta_logrho);
292  return std::pow(10., value);
293  }
294  case logenergy:
295  {
296  double value = tl_interpolate(logenergy_values_.get(),
297  idx_ye, idx_logtemp, idx_logrho,
298  delta_ye, delta_logtemp, delta_logrho);
299  return std::pow(10., value) - energy_shift_;
300  }
301  case entropy:
302  return tl_interpolate(entropy_values_.get(),
303  idx_ye, idx_logtemp, idx_logrho,
304  delta_ye, delta_logtemp, delta_logrho);
305 
306  case munu:
307  return tl_interpolate(munu_values_.get(),
308  idx_ye, idx_logtemp, idx_logrho,
309  delta_ye, delta_logtemp, delta_logrho);
310 
311  case cs2:
312  return tl_interpolate(cs2_values_.get(),
313  idx_ye, idx_logtemp, idx_logrho,
314  delta_ye, delta_logtemp, delta_logrho);
315 
316  case dedt:
317  return tl_interpolate(dedt_values_.get(),
318  idx_ye, idx_logtemp, idx_logrho,
319  delta_ye, delta_logtemp, delta_logrho);
320 
321  case dpdrhoe:
322  return tl_interpolate(dpdrhoe_values_.get(),
323  idx_ye, idx_logtemp, idx_logrho,
324  delta_ye, delta_logtemp, delta_logrho);
325 
326  case dpderho:
327  return tl_interpolate(dpderho_values_.get(),
328  idx_ye, idx_logtemp, idx_logrho,
329  delta_ye, delta_logtemp, delta_logrho);
330 
331  default:
332  break;
333  }

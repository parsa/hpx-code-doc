359  // delete an existing instance of a component
360  void runtime_support::free_component(
361  agas::gva const& g, naming::gid_type const& gid, std::uint64_t count)
362  {
363  // Special case: component_memory_block.
364  if (g.type == components::component_memory_block) {
365  for (std::size_t i = 0; i != count; ++i)
366  {
367  naming::gid_type target = gid + i;
368 
369  // make sure this component is located here
370  if (get_locality() != g.prefix)
371  {
372  // FIXME: should the component be re-bound ?
373  std::ostringstream strm;
374  strm << "global id " << target << " is not bound to any "
375  "local component instance";
376 
377  // FIXME: If this throws then we leak the rest of count.
378  // What should we do instead?
379  HPX_THROW_EXCEPTION(hpx::unknown_component_address,
380  "runtime_support::free_component",
381  strm.str());
382  return;
383  }
384 
385  // free the memory block
386  components::server::memory_block::destroy(
387  reinterpret_cast<components::server::memory_block*>(
388  g.lva(target, gid)));
389 
390  LRT_(info) << "successfully destroyed memory block " << target;
391  }
392 
393  return;
394  }
395  else if (naming::refers_to_virtual_memory(gid))
396  {
397  // simply delete the memory
398  delete [] reinterpret_cast<std::uint8_t*>(gid.get_lsb());
399  return;
400  }
401 
402  // locate the factory for the requested component type
403  std::shared_ptr<component_factory_base> factory;
404 
405  {
406  std::unique_lock<component_map_mutex_type> l(cm_mtx_);
407  component_map_type::const_iterator it = components_.find(g.type);
408  if (it == components_.end()) {
409  // we don't know anything about this component
410  std::ostringstream strm;
411 
412  naming::resolver_client& client = naming::get_agas_client();
413  error_code ec(lightweight);
414  strm << "attempt to destroy component "
415  << gid
416  << " of invalid/unknown type: "
417  << components::get_component_type_name(g.type)
418  << " ("
419  << client.get_component_type_name(g.type, ec)
420  << ")" << std::endl;
421 
422  strm << "list of registered components: \n";
423  component_map_type::iterator end = components_.end();
424  for (component_map_type::iterator cit = components_.begin();
425  cit!= end; ++cit)
426  {
427  strm << " "
428  << components::get_component_type_name((*cit).first)
429  << " ("
430  << client.get_component_type_name((*cit).first, ec)
431  << ")" << std::endl;
432  }
433 
434  l.unlock();
435  HPX_THROW_EXCEPTION(hpx::bad_component_type,
436  "runtime_support::free_component",
437  strm.str());
438  return;
439  }
440 
441  factory = (*it).second.first;
442  }
443 
444  // we might end up with the same address, so cache the already deleted
445  // ones here.
446 #if defined(HPX_DEBUG)
447  std::vector<naming::address> freed_components;
448  freed_components.reserve(std::size_t(count));
449 #endif
450 
451  for (std::size_t i = 0; i != count; ++i)
452  {
453  naming::gid_type target(gid + i);
454  naming::address addr(g.prefix, g.type, g.lva(target, gid));
455 
456 #if defined(HPX_DEBUG)
457  bool found = false;
458  for (naming::address const& a : freed_components)
459  {
460  if(a == addr)
461  {
462  found = true;
463  break;
464  }
465  }
466  HPX_ASSERT(!found);
467 #endif
468  // FIXME: If this throws then we leak the rest of count.
469  // What should we do instead?
470 
471  // destroy the component instance
472  factory->destroy(target, addr);
473 
474  LRT_(info) << "successfully destroyed component " << (gid + i)
475  << " of type: " << components::get_component_type_name(g.type);
476 
477 #if defined(HPX_DEBUG)
478  freed_components.push_back(std::move(addr));
479 #endif
480  }

562  void runtime_support::dijkstra_make_black()
563  {
564  // Rule 1: A machine sending a message makes itself black.
565  std::lock_guard<dijkstra_mtx_type> l(dijkstra_mtx_);
566  dijkstra_color_ = true;

569 #if defined(HPX_USE_FAST_DIJKSTRA_TERMINATION_DETECTION)
570  // This new code does not work, currently, as the return actions generated
571  // by the futures used by hpx::reduce make the sender black. This causes
572  // an infinite loop while waiting for the Dijkstra termination detection
573  // to return.

575  // invoked during termination detection
576  bool runtime_support::dijkstra_termination()
577  {
578  applier::applier& appl = hpx::applier::get_applier();
579  naming::resolver_client& agas_client = appl.get_agas_client();
580 
581  agas_client.start_shutdown();
582 
583  // First wait for this locality to become passive. We do this by
584  // periodically checking the number of still running threads.
585  //
586  // Rule 0: When active, machine nr.i + 1 keeps the token; when passive,
587  // it hands over the token to machine nr.i.
588  threads::threadmanager& tm = appl.get_thread_manager();
589 
590  for (std::size_t k = 0;
591  tm.get_thread_count() > std::int64_t(1 + hpx::get_os_thread_count());
592  ++k)
593  {
594  // avoid timed suspension, don't boost priority
595  util::detail::yield_k(k % 32,
596  "runtime_support::dijkstra_termination",
597  hpx::threads::pending);
598  }
599 
600  // Now this locality has become passive, thus we can send the token
601  // to the next locality.
602  //
603  // Rule 2: When machine nr.i + 1 propagates the probe, it hands over a
604  // black token to machine nr.i if it is black itself, whereas while
605  // being white it leaves the color of the token unchanged.
606  std::lock_guard<dijkstra_mtx_type> l(dijkstra_mtx_);
607  bool dijkstra_token = dijkstra_color_;
608 
609  // Rule 5: Upon transmission of the token to machine nr.i, machine
610  // nr.i + 1 becomes white.
611  dijkstra_color_ = false;
612 
613  // The reduce-function (logical_or) will make sure black will be
614  // propagated.
615  return dijkstra_token;
616  }

642  }

644  std::size_t count = 0; // keep track of number of trials
645 
646  {
647  // Note: we protect the entire loop here since the stopping condition
648  // depends on the shared variable "dijkstra_color_"
649  // Proper unlocking for possible remote actions needs to be taken care of
650  typedef std::unique_lock<dijkstra_mtx_type> dijkstra_scoped_lock;
651  dijkstra_scoped_lock l(dijkstra_mtx_);
652  do {
653  // Rule 4: Machine nr.0 initiates a probe by making itself white
654  // and sending a white token to machine nr.N - 1.
655  dijkstra_color_ = false; // start off with white
656 
657  dijkstra_termination_action act;
658  bool termination_aborted = false;
659  {
660  util::unlock_guard<dijkstra_scoped_lock> ul(l);
661  termination_aborted = lcos::reduce(act,
662  locality_ids, std_logical_or_type()).get()
663  }
664 
665  if (termination_aborted)
666  {
667  dijkstra_color_ = true; // unsuccessful termination
668  }
669 
670  // Rule 3: After the completion of an unsuccessful probe, machine
671  // nr.0 initiates a next probe.
672 
673  ++count;
674 
675  } while (dijkstra_color_);
676  }
677 
678  return count;
679  }
680 #else
681  void runtime_support::send_dijkstra_termination_token(
682  std::uint32_t target_locality_id,
683  std::uint32_t initiating_locality_id,
684  std::uint32_t num_localities, bool dijkstra_token)
685  {
686  // First wait for this locality to become passive. We do this by
687  // periodically checking the number of still running threads.
688  //
689  // Rule 0: When active, machine nr.i + 1 keeps the token; when passive,
690  // it hands over the token to machine nr.i.
691  applier::applier& appl = hpx::applier::get_applier();
692  threads::threadmanager& tm = appl.get_thread_manager();
693 
694  for (std::size_t k = 0;
695  tm.get_thread_count() > std::int64_t(1 + hpx::get_os_thread_count());
696  ++k)
697  {
698  // avoid timed suspension, don't boost priority
699  util::detail::yield_k(k % 32,
700  "runtime_support::send_dijkstra_termination_token",
701  hpx::threads::pending);
702  }
703 
704  // Now this locality has become passive, thus we can send the token
705  // to the next locality.
706  //
707  // Rule 2: When machine nr.i + 1 propagates the probe, it hands over a
708  // black token to machine nr.i if it is black itself, whereas while
709  // being white it leaves the color of the token unchanged.
710  {
711  std::lock_guard<dijkstra_mtx_type> l(dijkstra_mtx_);
712  if (dijkstra_color_)
713  dijkstra_token = dijkstra_color_;
714 
715  // Rule 5: Upon transmission of the token to machine nr.i, machine
716  // nr.i + 1 becomes white.
717  dijkstra_color_ = false;
718  }
719 
720  naming::id_type id(naming::get_id_from_locality_id(target_locality_id));
721  apply<dijkstra_termination_action>(id, initiating_locality_id,
722  num_localities, dijkstra_token);
723  }

726  void runtime_support::dijkstra_termination(
727  std::uint32_t initiating_locality_id, std::uint32_t num_localities,
728  bool dijkstra_token)
729  {
730  applier::applier& appl = hpx::applier::get_applier();
731  naming::resolver_client& agas_client = appl.get_agas_client();
732  parcelset::parcelhandler& ph = appl.get_parcel_handler();
733 
734  agas_client.start_shutdown();
735  ph.flush_parcels();
736 
737  std::uint32_t locality_id = get_locality_id();
738 
739  if (initiating_locality_id == locality_id)
740  {
741  // we received the token after a full circle
742  if (dijkstra_token)
743  {
744  std::lock_guard<dijkstra_mtx_type> l(dijkstra_mtx_);
745  dijkstra_color_ = true; // unsuccessful termination
746  }
747 
748  dijkstra_cond_.notify_one();
749  return;
750  }
751 
752  if (0 == locality_id)
753  locality_id = num_localities;
754 
755  send_dijkstra_termination_token(locality_id - 1,
756  initiating_locality_id, num_localities, dijkstra_token);
757  }

759  // Kick off termination detection, this is modeled after Dijkstra's paper:
760  // http://www.cs.mcgill.ca/~lli22/575/termination3.pdf.
761  std::size_t runtime_support::dijkstra_termination_detection(
762  std::vector<naming::id_type> const& locality_ids)
763  {
764  std::uint32_t num_localities =
765  static_cast<std::uint32_t>(locality_ids.size());
766  if (num_localities == 1)
767  {
768  // While no real distributed termination detection has to be
769  // performed, we should still wait for the thread-queues to drain.
770  applier::applier& appl = hpx::applier::get_applier();
771  threads::threadmanager& tm = appl.get_thread_manager();
772 
773  for (std::size_t k = 0;
774  tm.get_thread_count() > std::int64_t(1 + hpx::get_os_thread_count());
775  ++k)
776  {
777  // avoid timed suspension, don't boost priority
778  util::detail::yield_k(k % 32,
779  "runtime_support::dijkstra_termination_detection",
780  hpx::threads::pending);
781  }
782 
783  return 0;
784  }
785 
786  std::uint32_t initiating_locality_id = get_locality_id();
787 
788  // send token to previous node
789  std::uint32_t target_id = initiating_locality_id;
790  if (0 == target_id)
791  target_id = static_cast<std::uint32_t>(num_localities);
792 
793  std::size_t count = 0; // keep track of number of trials
794 
795  {
796  // Note: we protect the entire loop here since the stopping condition
797  // depends on the shared variable "dijkstra_color_"
798  // Proper unlocking for possible remote actions needs to be taken care of
799  typedef std::unique_lock<dijkstra_mtx_type> dijkstra_scoped_lock;
800  dijkstra_scoped_lock l(dijkstra_mtx_);
801  do {
802  // Rule 4: Machine nr.0 initiates a probe by making itself white
803  // and sending a white token to machine nr.N - 1.
804  dijkstra_color_ = false; // start off with white
805 
806  {
807  util::unlock_guard<dijkstra_scoped_lock> ul(l);
808  send_dijkstra_termination_token(target_id - 1,
809  initiating_locality_id, num_localities, false);
810  }
811 
812  // wait for token to come back to us
813  dijkstra_cond_.wait(l);
814 
815  // Rule 3: After the completion of an unsuccessful probe, machine
816  // nr.0 initiates a next probe.
817 
818  ++count;
819 
820  } while (dijkstra_color_);
821  }
822 
823  return count;
824  }

828  void runtime_support::shutdown_all(double timeout)
829  {
830  if (find_here() != hpx::find_root_locality())
831  {
832  HPX_THROW_EXCEPTION(invalid_status,
833  "runtime_support::shutdown_all",
834  "shutdown_all should be invoked on the root locality only");
835  return;
836  }
837 
838  // make sure shutdown_all is invoked only once
839  bool flag = false;
840  if (!shutdown_all_invoked_.compare_exchange_strong(flag, true))
841  {
842  return;
843  }
844 
845  LRT_(info) << "runtime_support::shutdown_all: "
846  "initializing application shutdown";
847 
848  applier::applier& appl = hpx::applier::get_applier();
849  naming::resolver_client& agas_client = appl.get_agas_client();
850 
851  agas_client.start_shutdown();
852

854

856  std::size_t count = dijkstra_termination_detection(locality_ids);
857 
858  LRT_(info) << "runtime_support::shutdown_all: " //-V128
859  "passed first termination detection (count: "
860  << count << ").";
861 
862  // execute registered shutdown functions on all localities
863  invoke_shutdown_functions(locality_ids, true);
864  invoke_shutdown_functions(locality_ids, false);
865 
866  LRT_(info) << "runtime_support::shutdown_all: "
867  "invoked shutdown functions";
868 
869  // Do a second round of termination detection to synchronize with all
870  // work which was triggered by the invocation of the shutdown
871  // functions.
872  count = dijkstra_termination_detection(locality_ids);
873 
874  LRT_(info) << "runtime_support::shutdown_all: " //-V128
875  "passed second termination detection (count: "
876  << count << ").";
877 
878  // Shut down all localities except the local one, we can't use
879  // broadcast here as we have to handle the back parcel in a special
880  // way.
881  std::reverse(locality_ids.begin(), locality_ids.end());
882  std::uint32_t locality_id = get_locality_id();
883  std::vector<lcos::future<void> > lazy_actions;
884 
885  for (naming::id_type const& id : locality_ids)
886  {
887  if (locality_id != naming::get_locality_id_from_id(id))
888  {
889  using components::stubs::runtime_support;
890  lazy_actions.push_back(runtime_support::shutdown_async(id,
891  timeout));
892  }
893  }
894 
895  // wait for all localities to be stopped
896  wait_all(lazy_actions);
897 
898  LRT_(info) << "runtime_support::shutdown_all: "
899  "all localities have been shut down";
900 
901  // Now make sure this local locality gets shut down as well.
902  // There is no need to respond...
903  stop(timeout, naming::invalid_id, false);
904  }

1085  void runtime_support::stop(double timeout,
1086  naming::id_type const& respond_to, bool remove_from_remote_caches)
1087  {
1088  std::unique_lock<compat::mutex> l(mtx_);
1089  if (!stopped_) {
1090  // push pending logs
1091  components::cleanup_logging();
1092 
1093  HPX_ASSERT(!terminated_);
1094 
1095  applier::applier& appl = hpx::applier::get_applier();
1096  threads::threadmanager& tm = appl.get_thread_manager();
1097  naming::resolver_client& agas_client = appl.get_agas_client();
1098 
1099  util::high_resolution_timer t;
1100  double start_time = t.elapsed();
1101  bool timed_out = false;
1102  error_code ec(lightweight);
1103 
1104  stopped_ = true;
1105 
1106  for (std::size_t k = 0;
1107  tm.get_thread_count() >
1108  std::int64_t(1 + hpx::get_os_thread_count());
1109  ++k)
1110  {
1111  // let thread-manager clean up threads
1112  cleanup_threads(tm, l);
1113 
1114  // obey timeout
1115  if (timeout >= 0.0 && timeout < (t.elapsed() - start_time))
1116  {
1117  // we waited long enough
1118  timed_out = true;
1119  break;
1120  }
1121 
1122  // avoid timed suspension, don't boost priority
1123  util::detail::yield_k(k % 32,
1124  "runtime_support::stop", hpx::threads::pending);
1125  }
1126 
1127  // If it took longer than expected, kill all suspended threads as
1128  // well.
1129  if (timed_out) {
1130  // now we have to wait for all threads to be aborted
1131  start_time = t.elapsed();
1132 
1133  for (std::size_t k = 0;
1134  tm.get_thread_count() >
1135  std::int64_t(1 + hpx::get_os_thread_count());
1136  ++k)
1137  {
1138  // abort all suspended threads
1139  tm.abort_all_suspended_threads();
1140 
1141  // let thread-manager clean up threads
1142  cleanup_threads(tm, l);
1143 
1144  // obey timeout
1145  if (timeout >= 0.0 && timeout < (t.elapsed() - start_time))
1146  {
1147  // we waited long enough
1148  break;
1149  }
1150 
1151  // avoid timed suspension, don't boost priority
1152  util::detail::yield_k(k % 32,
1153  "runtime_support::stop", hpx::threads::pending);
1154  }
1155  }
1156 
1157  // Drop the locality from the partition table.
1158  naming::gid_type here = agas_client.get_local_locality();
1159 
1160  // unregister fixed components
1161  agas_client.unbind_local(appl.get_runtime_support_raw_gid(), ec);
1162  agas_client.unbind_local(appl.get_memory_raw_gid(), ec);
1163 
1164  if (remove_from_remote_caches)
1165  remove_here_from_connection_cache();
1166 
1167  agas_client.unregister_locality(here, ec);
1168 
1169  if (remove_from_remote_caches)
1170  remove_here_from_console_connection_cache();
1171 
1172  if (respond_to) {
1173  // respond synchronously
1174  typedef lcos::base_lco_with_value<void> void_lco_type;
1175  typedef void_lco_type::set_event_action action_type;
1176 
1177  naming::address addr;
1178  if (agas::is_local_address_cached(respond_to, addr)) {
1179  // this should never happen
1180  HPX_ASSERT(false);
1181  }
1182  else {
1183  // apply remotely, parcel is sent synchronously
1184  hpx::applier::detail::apply_r_sync<action_type>(
1185  std::move(addr), respond_to);
1186  }
1187  }
1188 
1189  wait_condition_.notify_all();
1190  stop_condition_.wait(l); // wait for termination
1191  }
1192  }

1772  // path = ... # the path where to find this component module
1773  // enabled = false # optional (default is assumed to be true)

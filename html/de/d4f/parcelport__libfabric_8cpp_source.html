157  // Start the handling of communication.
158  bool parcelport::do_run()
159  {
160  if (!parcelport_enabled_) return false;
161 
162 #ifndef HPX_PARCELPORT_LIBFABRIC_HAVE_PMI
163  auto &as = this->applier_->get_agas_client();
164  libfabric_controller_->initialize_localities(as);
165 #endif
166 
167  FUNC_START_DEBUG_MSG;
168  libfabric_controller_->startup(this);
169 
170  LOG_DEBUG_MSG("Fetching memory pool");
171  chunk_pool_ = &libfabric_controller_->get_memory_pool();
172 
173  for (std::size_t i = 0; i < HPX_PARCELPORT_LIBFABRIC_THROTTLE_SENDS; ++i)
174  {
175  sender *snd =
176  new sender(this,
177  libfabric_controller_->ep_active_,
178  libfabric_controller_->get_domain(),
179  chunk_pool_);
180  snd->postprocess_handler_ = [this](sender* s)
181  {
182  --senders_in_use_;
183  senders_.push(s);
184  trigger_pending_work();
185  };
186  senders_.push(snd);
187  }
188 
189  if (bootstrap_enabled_)
190  {
191  for (std::size_t i = 0; i != io_service_pool_.size(); ++i)
192  {
193  io_service_pool_.get_io_service(int(i)).post(
194  hpx::util::bind(
195  &parcelport::io_service_work, this));
196  }
197  }
198  return true;
199  }

398  // --------------------------------------------------------------------
399  // This is called to poll for completions and handle all incoming messages
400  // as well as complete outgoing messages.
401  //
402  // Since the parcelport can be serviced by hpx threads or by OS threads,
403  // we must use extra care when dealing with mutexes and condition_variables
404  // since we do not want to suspend an OS thread, but we do want to suspend
405  // hpx threads when necessary.
406  //
407  // NB: There is no difference any more between background polling work
408  // on OS or HPX as all has been tested thoroughly
409  // --------------------------------------------------------------------
410  inline bool parcelport::background_work_OS_thread() {
411  LOG_TIMED_INIT(background);
412  bool done = false;
413  do {
414  LOG_TIMED_BLOCK(background, DEVEL, 5.0, {
415  LOG_DEBUG_MSG("number of senders in use "
416  << decnumber(senders_in_use_));
417  });
418  // if an event comes in, we may spend time processing/handling it
419  // and another may arrive during this handling,
420  // so keep checking until none are received
421  // libfabric_controller_->refill_client_receives(false);
422  int numc = libfabric_controller_->poll_endpoints();
423  completions_handled_ += numc;
424  done = (numc==0);
425  } while (!done);
426  return (done!=0);
427  }
428

435  bool parcelport::background_work(std::size_t num_thread) {
436  if (stopped_ || hpx::is_stopped()) {

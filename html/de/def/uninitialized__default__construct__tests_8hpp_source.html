157 template <typename ExPolicy, typename IteratorTag>
158 void test_uninitialized_default_construct_exception(ExPolicy policy, IteratorTag)
159 {
160  static_assert(
161  hpx::parallel::execution::is_execution_policy<ExPolicy>::value,
162  "hpx::parallel::execution::is_execution_policy<ExPolicy>::value");
163 
164  typedef test::count_instances_v<default_constructable> data_type;
165  typedef data_type* base_iterator;
166  typedef test::decorated_iterator<base_iterator, IteratorTag>
167  decorated_iterator;
168 
169  data_type* p = (data_type*)std::malloc(data_size * sizeof(data_type));
170  std::memset(p, 0xcd, data_size * sizeof(data_type));
171 
172  std::atomic<std::size_t> throw_after(std::rand() % data_size); //-V104
173  std::size_t throw_after_ = throw_after.load();
174 
175  data_type::instance_count.store(0);
176  data_type::max_instance_count.store(0);
177 
178  bool caught_exception = false;
179  try {
180  hpx::parallel::uninitialized_default_construct(policy,
181  decorated_iterator(p,
182  [&throw_after]()
183  {
184  if (throw_after-- == 0)
185  throw std::runtime_error("test");
186  }),
187  decorated_iterator(p + data_size));
188  HPX_TEST(false);
189  }
190  catch (hpx::exception_list const& e) {
191  caught_exception = true;
192  test::test_num_exceptions<ExPolicy, IteratorTag>::call(policy, e);
193  }
194  catch (...) {
195  HPX_TEST(false);
196  }
197 
198  HPX_TEST(caught_exception);
199  HPX_TEST_EQ(data_type::instance_count.load(), std::size_t(0));
200  HPX_TEST_LTE(throw_after_, data_type::max_instance_count.load());
201 
202  std::free(p);
203 }

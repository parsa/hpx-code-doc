118  template <typename B>
119  HPX_HOST_DEVICE
120  void execute(B part_begin, std::size_t part_steps,
121  std::size_t part_index)
122  {
123  auto pack = typename hpx::util::detail::make_index_pack<
124  sizeof...(Ts)>::type();
125  detail::init_iteration(args_, pack, part_index);
126 
127  while (part_steps != 0)
128  {
129  detail::invoke_iteration(args_, pack, f_, part_begin);
130 
131  detail::next_iteration(args_, pack);
132 
133  // NVCC seems to have a bug with std::min...
134  std::size_t chunk =
135  S(part_steps) < stride_ ? part_steps : stride_;
136 
137  // modifies 'chunk'
138  part_begin = parallel::v1::detail::next(
139  part_begin, part_steps, chunk);
140  part_steps -= chunk;
141  }
142  }

161  template <typename ExPolicy, typename InIter, typename Size,
162  typename S, typename F, typename... Args>
163  HPX_HOST_DEVICE
164  static hpx::util::unused_type
165  sequential(ExPolicy policy, InIter first, Size size, S stride,
166  F && f, Args &&... args)
167  {
168  int const init_sequencer[] = {
169  0, (args.init_iteration(0), 0)...
170  };
171  (void)init_sequencer;
172 
173  std::size_t count = size;
174  while (count != 0)
175  {
176  hpx::util::invoke(f, first, args.iteration_value()...);
177 
178  int const next_sequencer[] = {
179  0, (args.next_iteration(), 0)...
180  };
181  (void)next_sequencer;
182 
183  // NVCC seems to have a bug with std::min...
184  std::size_t chunk = (S(count) < stride) ? count : stride;
185 
186  // modifies 'chunk'
187  first = parallel::v1::detail::next(first, count, chunk);
188  count -= chunk;
189  }
190 
191  // make sure live-out variables are properly set on return
192  int const exit_sequencer[] = {
193  0, (args.exit_iteration(size), 0)...
194  };
195  (void)exit_sequencer;
196 
197  return hpx::util::unused_type();
198  }

235  template <typename ExPolicy, typename B, typename E, typename S,
236  std::size_t... Is, typename... Args>
237  typename util::detail::algorithm_result<ExPolicy>::type
238  for_loop(ExPolicy && policy, B first, E last, S stride,
239  hpx::util::detail::pack_c<std::size_t, Is...>, Args &&... args)
240  {
241  // stride shall not be zero
242  HPX_ASSERT(stride != 0);
243 
244  // stride should be negative only if E is an integral type or at
245  // least a bidirectional iterator
246  if (stride < 0)
247  {
248  HPX_ASSERT(std::is_integral<E>::value ||
249  hpx::traits::is_bidirectional_iterator<E>::value);
250  }
251 
252  // the for_loop should be executed sequentially if the
253  // execution policy enforces sequential execution or if the
254  // loop boundaries are integral types
255 #if defined(HPX_HAVE_ALGORITHM_INPUT_ITERATOR_SUPPORT)
256  static_assert(
257  (std::is_integral<B>::value ||
258  hpx::traits::is_input_iterator<B>::value),
259  "Requires at least input iterator or integral loop boundaries.");
260 
261  typedef std::integral_constant<bool,
262  execution::is_sequenced_execution_policy<ExPolicy>::value ||
263  (!std::is_integral<B>::value &&
264  !hpx::traits::is_forward_iterator<B>::value)
265  > is_seq;
266 #else
267  static_assert(
268  (std::is_integral<B>::value ||
269  hpx::traits::is_forward_iterator<B>::value),
270  "Requires at least forward iterator or integral loop boundaries.");
271 
272  typedef std::integral_constant<bool,
273  execution::is_sequenced_execution_policy<ExPolicy>::value ||
274  !std::is_integral<B>::value
275  > is_seq;
276 #endif
277 
278  std::size_t size = parallel::v1::detail::distance(first, last);
279  auto && t = hpx::util::forward_as_tuple(std::forward<Args>(args)...);
280 
281  return for_loop_algo().call(
282  std::forward<ExPolicy>(policy), is_seq(), first, size, stride,
283  hpx::util::get<sizeof...(Args)-1>(t), hpx::util::get<Is>(t)...);

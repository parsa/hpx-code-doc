410  struct when_some : std::enable_shared_from_this<when_some<Sequence> > //-V690
411  {
412  typedef lcos::local::spinlock mutex_type;
413 
414  public:
415  void on_future_ready(std::size_t idx, threads::thread_id_type const& id)
416  {
417  std::size_t const new_count = count_.fetch_add(1) + 1;
418  if (new_count <= needed_count_)
419  {
420  {
421  std::lock_guard<mutex_type> l(this->mtx_);
422  lazy_values_.indices.push_back(idx);
423  }
424  if (new_count == needed_count_) {
425  if (id != threads::get_self_id()) {
426  threads::set_thread_state(id, threads::pending);
427  } else {
428  goal_reached_on_calling_thread_ = true;
429  }
430  }
431  }
432  }

449  when_some_result<Sequence> operator()()
450  {
451  // set callback functions to executed when future is ready
452  set_on_completed_callback(*this);
453 
454  // if all of the requested futures are already set, our
455  // callback above has already been called often enough, otherwise
456  // we suspend ourselves

465  HPX_ASSERT(count_.load(std::memory_order_seq_cst) >= needed_count_);

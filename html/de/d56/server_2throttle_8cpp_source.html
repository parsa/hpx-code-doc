40  void throttle::suspend(std::size_t shepherd)
41  {
42  // If the current thread is not the requested one, re-schedule a new
43  // PX thread in order to retry.
44  std::size_t thread_num = hpx::get_worker_thread_num();
45  if (thread_num != shepherd) {
46  register_suspend_thread(shepherd);
47  return;
48  }
49 
50  std::lock_guard<mutex_type> l(mtx_);
51 
52  if (shepherd >= blocked_os_threads_.size()) {
53  HPX_THROW_EXCEPTION(hpx::bad_parameter, "throttle::suspend",
54  "invalid thread number");
55  }
56 
57  bool is_suspended = blocked_os_threads_[shepherd];
58  if (!is_suspended) {
59  blocked_os_threads_[shepherd] = true;
60  register_thread(shepherd);
61  }
62  }

88  // do the requested throttling
89  void throttle::throttle_controller(std::size_t shepherd)
90  {
91  std::unique_lock<mutex_type> l(mtx_);
92  if (!blocked_os_threads_[shepherd])
93  return; // nothing more to do
94 
95  {
96  // put this shepherd thread to sleep for 100ms
97  hpx::util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
98 
99  hpx::compat::this_thread::sleep_for(std::chrono::milliseconds(100));
100  }
101 
102  // if this thread still needs to be suspended, re-schedule this routine
103  // which will give the thread manager some cycles to tend to the high
104  // priority tasks which might have arrived
105  if (blocked_os_threads_[shepherd])
106  register_thread(shepherd);
107  }

111  {

117  description.c_str(),

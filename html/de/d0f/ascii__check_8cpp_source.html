25  // Legal characters for a source file are defined in section 2.2 of the standard
26  // I have added '@', '^', and '`' to the "legal" chars because they are commonly
27  // used in comments, and they are strictly ASCII.
28  struct non_ascii : public std::unary_function<char, bool> {
29  public:
30  non_ascii () {}
31  ~non_ascii () {}
32  bool operator () ( char c ) const
33  {
34  if ( c == ' ' ) return false;
35  if ( c >= 'a' && c <= 'z' ) return false;
36  if ( c >= 'A' && c <= 'Z' ) return false;
37  if ( c >= '0' && c <= '9' ) return false;
38  // Horizontal/Vertical tab, newline, and form feed
39  if ( c == '\t' || c == '\n'
40  || c == '\r' || c == '\v' || c == '\f' ) return false;
41  return gPunct.find ( c ) == string::npos;
42  }
43  };

57 // Given a position in the file, extract and return the line
58  std::string find_line ( const std::string &contents,
59  std::string::const_iterator iter_pos )
60  {
61  std::size_t pos = iter_pos - contents.begin ();
62 
63  // Search backwards for a CR or LR
64  std::size_t start_pos = contents.find_last_of ( kCRLF, pos );
65  std::string::const_iterator line_start =
66  contents.begin () + ( start_pos == std::string::npos ? 0 : start_pos + 1 );
67 
68 
69  // Search forwards for a CR or LF
70  std::size_t end_pos = contents.find_first_of ( kCRLF, pos + 1 );
71  std::string::const_iterator line_end;
72  if ( end_pos == std::string::npos )
73  line_end = contents.end ();
74  else
75  line_end = contents.begin () + end_pos - 1;
76 
77  return std::string ( line_start, line_end );
78  }

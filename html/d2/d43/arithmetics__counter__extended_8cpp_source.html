108  template <typename Statistic>
109  hpx::performance_counters::counter_value
110  arithmetics_counter_extended<Statistic>::get_counter_value(bool reset)
111  {
112  std::vector<counter_value> base_values =
113  counters_.get_counter_values(hpx::launch::sync);
114 
115  // apply arithmetic Statistic
116  typedef boost::accumulators::accumulator_set<
117  double, boost::accumulators::stats<Statistic>
118  > accumulator_type;
119 
120  accumulator_type accum;
121  for (counter_value const& base_value : base_values)
122  {
123  accum(base_value.get_value<double>());
124  }
125  double value = detail::statistic_get_value<Statistic>::call(accum);
126 
127  if (base_values[0].scale_inverse_ && base_values[0].scaling_ != 1.0) //-V550
128  {
129  base_values[0].value_ =
130  static_cast<std::int64_t>(value * base_values[0].scaling_);
131  }
132  else {
133  base_values[0].value_ =
134  static_cast<std::int64_t>(value / base_values[0].scaling_);
135  }
136 
137  base_values[0].time_ = static_cast<std::int64_t>(hpx::get_system_uptime());
138  base_values[0].count_ = counters_.get_invocation_count();
139 
140  return base_values[0];
141  }

244  naming::gid_type arithmetics_counter_extended_creator(
245  counter_info const& info, error_code& ec)
246  {
247  switch (info.type_) {
248  case counter_aggregating:
249  {
250  counter_path_elements paths;
251  get_counter_path_elements(info.fullname_, paths, ec);
252  if (ec) return naming::invalid_gid;
253 
254  if (!paths.parameters_.empty()) {
255  // try to interpret the additional parameter as a list of
256  // two performance counter names
257  std::vector<std::string> names;
258  boost::split(names, paths.parameters_, boost::is_any_of(","));
259 
260  if (names.empty())
261  {
262  HPX_THROWS_IF(ec, bad_parameter,
263  "arithmetics_counter_extended_creator",
264  "the parameter specification for an arithmetic counter "
265  "has to expand to at least one counter name: " +
266  paths.parameters_);
267  return naming::invalid_gid;
268  }
269 
270  for (std::string const& name : names)
271  {
272  counter_path_elements paths;
273  if (status_valid_data != get_counter_path_elements(
274  name, paths, ec) || ec)
275  {
276  HPX_THROWS_IF(ec, bad_parameter,
277  "arithmetics_counter_extended_creator",
278  "the given (expanded) counter name is not "
279  "a validly formed performance counter name: " +
280  name);
281  return naming::invalid_gid;
282  }
283  }
284 
285  return create_arithmetics_counter_extended(info, names, ec);
286  }
287  else {
288  HPX_THROWS_IF(ec, bad_parameter,
289  "arithmetics_counter_extended_creator",
290  "the parameter specification for an arithmetic counter "
291  "has to be a comma separated list of performance "
292  "counter names, none is given: " +
293  remove_counter_prefix(info.fullname_));
294  }
295  }
296  break;
297 
298  default:
299  HPX_THROWS_IF(ec, bad_parameter,
300  "arithmetics_counter_extended_creator",
301  "invalid counter type requested");
302  break;
303  }
304  return naming::invalid_gid;
305  }

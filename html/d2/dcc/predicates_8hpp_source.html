162  template <typename Iter, typename Stride>
163  HPX_HOST_DEVICE
164  HPX_FORCEINLINE static
165  Iter call(Iter iter, std::size_t max_count, Stride& offset)
166  {
167  // anything less than a bidirectional iterator does not support
168  // negative offsets
169  HPX_ASSERT(!std::is_signed<Stride>::value || !is_negative(offset));
170 
171  // NVCC seems to have a bug with std::min...
172  Stride count =
173  Stride(max_count < std::size_t(offset) ? max_count : offset);
174 
175  // advance through the end or max number of elements
176  for (; count != 0; (void)++iter, --count)
177  ;
178 
179  offset -= count;
180  return iter;
181  }
182  };

53  // Creation function for per-action parcel data counters
54  naming::gid_type per_action_data_counter_creator(counter_info const& info,
55  hpx::parcelset::detail::per_action_data_counter_registry& registry,
56  hpx::util::function_nonser<
57  std::int64_t(std::string const&, bool)
58  > const& counter_func,
59  error_code& ec)
60  {
61  switch (info.type_) {
62  case counter_raw:
63  {
64  counter_path_elements paths;
65  get_counter_path_elements(info.fullname_, paths, ec);
66  if (ec) return naming::invalid_gid;
67 
68  if (paths.parentinstance_is_basename_) {
69  HPX_THROWS_IF(ec, bad_parameter,
70  "per_action_data_counter_creator",
71  "invalid action invocation counter name (instance name "
72  "must not be a valid base counter name)");
73  return naming::invalid_gid;
74  }
75 
76  if (paths.parameters_.empty()) {
77  // if no parameters (action name) is given assume that this
78  // counter should report the overall value for all actions
79  using util::placeholders::_1;
80  auto const& f =
81  util::bind(counter_func, paths.parameters_, _1);
82  return performance_counters::locality_raw_counter_creator(
83  info, f,ec);
84  }
85 
86  // ask registry
87  hpx::util::function_nonser<std::int64_t(bool)> f =
88  registry.get_counter(paths.parameters_, counter_func);
89 
90  return detail::create_raw_counter(info, std::move(f), ec);
91  }
92  break;
93 
94  default:
95  HPX_THROWS_IF(ec, bad_parameter,
96  "per_action_data_counter_creator",
97  "invalid counter type requested");
98  return naming::invalid_gid;
99  }

652  future<std::vector<T> >
653  get_values(std::vector<size_type> const & pos_vec) const
654  {
655  // check if position vector is empty
656  // the following code needs at least one element.
657  if (pos_vec.empty())
658  return make_ready_future(std::vector<T>());
659 
660  // current partition index of the block
661  size_type part_cur = get_partition(pos_vec[0]);
662 
663  // iterator to the begin of current block
664  typename std::vector<size_type>::const_iterator part_begin =
665  pos_vec.begin();
666 
667  // vector holding futures of the values for all blocks
668  std::vector<future<std::vector<T> > > part_values_future;
669  for (typename std::vector<size_type>::const_iterator it =
670  pos_vec.begin(); it != pos_vec.end(); ++it)
671  {
672  // get the partition of the current position
673  size_type part = get_partition(*it);
674 
675  // if the partition of the current position is the same
676  // as the rest of the current block go to next position
677  if (part == part_cur)
678  continue;
679 
680  // if the partition of the current position is NOT the same
681  // as the positions before the block ends here
682  else
683  {
684  // this is the end of a block containing indexes ('pos')
685  // of the same partition ('part').
686  // get async values for this block
687  part_values_future.push_back(get_values(part_cur,
688  get_local_indices(std::vector<size_type>(part_begin, it))));
689 
690  // reset block variables to start a new one from here
691  part_cur = part;
692  part_begin = it;
693  }
694  }
695 
696  // the end of the vector is also an end of a block
697  // get async values for this block
698  part_values_future.push_back(get_values(part_cur,
699  get_local_indices(std::vector<size_type>(
700  part_begin, pos_vec.end()))));
701 
702  // This helper function unwraps the vectors from each partition
703  // and merge them to one vector
704  auto merge_func =
705  [&pos_vec](std::vector<future<std::vector<T> > > && part_values_f)
706  -> std::vector<T>
707  {
708  std::vector<T> values;
709  values.reserve(pos_vec.size());
710 
711  for (future<std::vector<T> >& part_f: part_values_f)
712  {
713  std::vector<T> part_values = part_f.get();
714  std::move(part_values.begin(), part_values.end(),
715  std::back_inserter(values));
716  }
717  return values;
718  };
719 
720  // when all values are here merge them to one vector
721  // and return a future to this vector
722  return dataflow(launch::async, merge_func,
723  std::move(part_values_future));
724  }

1032  future<void>
1033  set_values(std::vector<size_type> const& pos, std::vector<T> const& val)
1034  {
1035  HPX_ASSERT(pos.size() == val.size());
1036 
1037  // check if position vector is empty
1038  // the following code needs at least one element.
1039  if (pos.empty())
1040  return make_ready_future();
1041 
1042  // partition index of the current block
1043  size_type part_cur = get_partition(pos[0]);
1044 
1045  // iterator to the begin of current block
1046  typename std::vector<size_type>::const_iterator pos_block_begin =
1047  pos.begin();
1048  typename std::vector<T>::const_iterator val_block_begin = val.begin();
1049 
1050  // vector holding futures of the state for all blocks
1051  std::vector<future<void> > part_futures;
1052 
1053  // going through the position vector
1054  typename std::vector<size_type>::const_iterator pos_it = pos.begin();
1055  typename std::vector<T>::const_iterator val_it = val.begin();
1056  for (; pos_it != pos.end(); ++pos_it, ++val_it)
1057  {
1058  // get the partition of the current position
1059  size_type part = get_partition(*pos_it);
1060 
1061  // if the partition of the current position is the same
1062  // as the rest of the current block go to next position
1063  if (part == part_cur)
1064  continue;
1065 
1066  // if the partition of the current position is NOT the same
1067  // as the positions before the block ends here
1068  else
1069  {
1070  // this is the end of a block containing indexes ('pos')
1071  // of the same partition ('part').
1072  // set asynchronous values for this block
1073  part_futures.push_back(set_values(part_cur,
1074  get_local_indices(std::vector<size_type>(
1075  pos_block_begin, pos_it)),
1076  std::vector<T>(val_block_begin, val_it)));
1077 
1078  // reset block variables to start a new one from here
1079  part_cur = part;
1080  pos_block_begin = pos_it;
1081  val_block_begin = val_it;
1082  }
1083  }
1084 
1085  // the end of the vector is also an end of a block
1086  // get asynchronous values for this block
1087  part_futures.push_back(set_values(part_cur,
1088  get_local_indices(std::vector<size_type>(
1089  pos_block_begin, pos.end())),
1090  std::vector<T>(val_block_begin, val.end())));
1091 
1092  return when_all(part_futures);
1093  }

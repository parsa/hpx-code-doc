104 void run_benchmark(boost::program_options::variables_map & vm)
105 {
106  // use the first remote locality to bounce messages, if possible
107  hpx::id_type here = hpx::find_here();
108 
109  hpx::id_type there = here;
110  std::vector<hpx::id_type> localities = hpx::find_remote_localities();
111  if (!localities.empty())
112  there = localities[0];
113 
114  // perform actual measurements
115  for (std::size_t size = 1; size <= MAX_MSG_SIZE; size *= 2)
116  {
117  std::size_t loop = LOOP_SMALL;
118  std::size_t skip = SKIP_SMALL;
119  std::size_t window_size = WINDOW_SIZE_SMALL;
120 
121  if (size > LARGE_MESSAGE_SIZE) {
122  loop = LOOP_LARGE;
123  skip = SKIP_LARGE;
124  window_size = WINDOW_SIZE_LARGE;
125  }
126 
127  hpx::util::high_resolution_timer t;
128 
129  for(std::size_t i = 0; i < loop + skip; ++i)
130  {
131  if (i == skip) // don't measure during warm-up phase
132  t.restart();
133 
134  hpx::future<void> recv =
135  hpx::async(recv_async, there, size, window_size);
136 
137  send_async(there, size, window_size);
138 
139  recv.wait();
140  }
141 
142  double bw = (size / 1e6 * loop * window_size * 2) / t.elapsed();
143 
144  hpx::cout << std::left << std::setw(10) << size << bw << std::endl;
145  }
146 }

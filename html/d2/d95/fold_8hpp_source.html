363  Result operator()(
364  hpx::future<std::vector<hpx::future<Result> > > r) const
365  {
366  std::vector<hpx::future<Result> > fres = std::move(r.get());
367  HPX_ASSERT(!fres.empty());
368 
369  // we're at the beginning of the folding chain, incorporate the
370  // initial value
371  if (fres.size() == 1)
372  return fold_op_(init_, fres[0].get());
373 
374  // in the middle of the folding chain we simply apply the folding
375  // operation to the two values we received.
376  HPX_ASSERT(fres.size() == 2);
377  return fold_op_(fres[0].get(), fres[1].get());
378  }

381  Result const& init_;

407  fold_futures.reserve(2);

430  , vs...
431  )

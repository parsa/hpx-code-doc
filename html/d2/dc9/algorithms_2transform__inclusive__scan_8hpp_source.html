45  template <typename InIter, typename OutIter, typename Conv, typename T,
46  typename Op>
47  OutIter sequential_transform_inclusive_scan(InIter first, InIter last,
48  OutIter dest, Conv && conv, T init, Op && op)
49  {
50  for (; first != last; (void) ++first, ++dest)
51  {
52  init = hpx::util::invoke(op, init, hpx::util::invoke(conv, *first));
53  *dest = init;
54  }
55  return dest;

91  template <typename ExPolicy, typename FwdIter1, typename Conv,
92  typename T, typename Op>
93  static typename util::detail::algorithm_result<
94  ExPolicy, FwdIter2
95  >::type
96  parallel(ExPolicy && policy, FwdIter1 first, FwdIter1 last,
97  FwdIter2 dest, Conv && conv, T && init, Op && op)
98  {
99  typedef util::detail::algorithm_result<ExPolicy, FwdIter2> result;
100  typedef hpx::util::zip_iterator<FwdIter1, FwdIter2> zip_iterator;
101  typedef typename std::iterator_traits<FwdIter1>::difference_type
102  difference_type;
103 
104  if (first == last)
105  return result::get(std::move(dest));
106 
107  difference_type count = std::distance(first, last);
108 
109  FwdIter2 final_dest = dest;
110  std::advance(final_dest, count);
111 
112  // The overall scan algorithm is performed by executing 2
113  // subsequent parallel steps. The first calculates the scan
114  // results for each partition and the second produces the
115  // overall result
116 
117  using hpx::util::get;
118  using hpx::util::make_zip_iterator;
119 
120  auto f3 =
121  [op, policy](
122  zip_iterator part_begin, std::size_t part_size,
123  hpx::shared_future<T> curr, hpx::shared_future<T> next
124  ) -> void
125  {
126  HPX_UNUSED(policy);
127 
128  next.get(); // rethrow exceptions
129 
130  T val = curr.get();
131  FwdIter2 dst = get<1>(part_begin.get_iterator_tuple());
132 
133  util::loop_n<ExPolicy>(
134  dst, part_size,
135  [&op, &val](FwdIter2 it) -> void
136  {
137  *it = hpx::util::invoke(op, *it, val);
138  });
139  };
140 
141  return util::scan_partitioner<ExPolicy, FwdIter2, T>::call(
142  std::forward<ExPolicy>(policy),
143  make_zip_iterator(first, dest), count, init,
144  // step 1 performs first part of scan algorithm
145  [op, conv](
146  zip_iterator part_begin, std::size_t part_size
147  ) -> T
148  {
149  T part_init =
150  hpx::util::invoke(conv, get<0>(*part_begin));
151  get<1>(*part_begin++) = part_init;
152 
153  auto iters = part_begin.get_iterator_tuple();
154  return sequential_transform_inclusive_scan_n(
155  get<0>(iters),
156  part_size-1,
157  get<1>(iters),
158  conv, part_init, op);
159  },
160  // step 2 propagates the partition results from left
161  // to right
162  hpx::util::unwrapping(op),
163  // step 3 runs final accumulation on each partition
164  std::move(f3),
165  // step 4 use this return value
166  [final_dest](std::vector<hpx::shared_future<T> > &&,
167  std::vector<hpx::future<void> > &&) -> FwdIter2
168  {
169  return final_dest;
170  });
171  }
172  };

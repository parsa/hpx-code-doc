51  // This version has a callback to be invoked for each future when it
52  // gets ready.

54  struct wait_each

56  protected:
57  void on_future_ready_(threads::thread_id_type const& id)
58  {
59  std::size_t oldcount = ready_count_.fetch_add(1);
60  HPX_ASSERT(oldcount < lazy_values_.size());
61 
62  if (oldcount + 1 == lazy_values_.size())
63  {
64  // reactivate waiting thread only if it's not us
65  if (id != threads::get_self_id())
66  threads::set_thread_state(id, threads::pending);
67  else
68  goal_reached_on_calling_thread_ = true;
69  }
70  }

151  std::vector<Future> operator()()
152  {
153  ready_count_.store(0);
154  goal_reached_on_calling_thread_ = false;
155 
156  // set callback functions to executed when future is ready
157  std::size_t size = lazy_values_.size();
158  threads::thread_id_type id = threads::get_self_id();
159  for (std::size_t i = 0; i != size; ++i)
160  {
161  typedef
162  typename traits::detail::shared_state_ptr_for<Future>::type
163  shared_state_ptr;
164  shared_state_ptr current =
165  traits::detail::get_shared_state(lazy_values_[i]);
166 
167  current->execute_deferred();
168  current->set_on_completed(
169  util::bind(&wait_each::on_future_ready, this, i, id));
170  }
171 
172  // If all of the requested futures are already set then our
173  // callback above has already been called, otherwise we suspend
174  // ourselves.
175  if (!goal_reached_on_calling_thread_)
176  {
177  // wait for all of the futures to return to become ready
178  this_thread::suspend(threads::suspended,
179  "hpx::lcos::detail::wait_each::operator()");
180  }
181 
182  // all futures should be ready
183  HPX_ASSERT(ready_count_ == size);
184 
185  return std::move(lazy_values_);
186  }

190  typename std::remove_reference<F>::type f_;

226  // invoked as soon as a value becomes available, it will not wait for all

229  inline std::size_t
230  wait(std::vector<Future>& lazy_values, F && f,
231  std::int32_t suspend_for = 10)
232  {

49  bool interval_timer::start(bool evaluate_)
50  {
51  std::unique_lock<mutex_type> l(mtx_);
52  if (is_terminated_)
53  return false;
54 
55  if (!is_started_) {
56  if (first_start_) {
57  first_start_ = false;
58 
59  util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
60  if (pre_shutdown_)
61  {
62  register_pre_shutdown_function(
63  util::bind(&interval_timer::terminate,
64  this->shared_from_this()));
65  }
66  else
67  {
68  register_shutdown_function(
69  util::bind(&interval_timer::terminate,
70  this->shared_from_this()));
71  }
72  }
73 
74  is_stopped_ = false;
75 
76  if (evaluate_) {
77  l.unlock();
78  evaluate(threads::wait_signaled);
79  }
80  else {
81  schedule_thread(l);
82  }
83 
84  return true;
85  }
86  return false;

179  threads::thread_result_type interval_timer::evaluate(
180  threads::thread_state_ex_enum statex)
181  {
182  try {
183  std::unique_lock<mutex_type> l(mtx_);
184 
185  if (is_stopped_ || is_terminated_ ||
186  statex == threads::wait_abort || 0 == microsecs_)
187  {
188  // object has been finalized, exit
189  return threads::thread_result_type(threads::terminated, nullptr);
190  }
191 
192  if (id_ != nullptr && id_ != threads::get_self_id())
193  {
194  // obsolete timer thread
195  return threads::thread_result_type(threads::terminated, nullptr);
196  }
197 
198  id_ = nullptr;
199  is_started_ = false;
200 
201  bool result = false;
202 
203  {
204  util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
205  result = f_(); // invoke the supplied function
206  }
207 
208  // some other thread might already have started the timer
209  if (nullptr == id_ && result) {
210  HPX_ASSERT(!is_started_);
211  schedule_thread(l); // wait and repeat
212  }
213 
214  if (!result)
215  is_terminated_ = true;
216  }
217  catch (hpx::exception const& e){
218  // the lock above might throw yield_aborted
219  if (e.get_error() != yield_aborted)
220  throw;
221  }
222 
223  // do not re-schedule this thread
224  return threads::thread_result_type(threads::terminated, nullptr);
225  }

227  // schedule a high priority task after a given time interval
228  void interval_timer::schedule_thread(std::unique_lock<mutex_type> & l)
229  {
230  HPX_ASSERT(l.owns_lock());
231 
232  using namespace hpx::threads;
233 
234  error_code ec;
235 
236  // create a new suspended thread
237  threads::thread_id_type id;
238  {
239  // FIXME: registering threads might lead to thread suspension since
240  // the allocators use hpx::lcos::local::spinlock. Unlocking the
241  // lock here would be the right thing but leads to crashes and hangs
242  // at shutdown.
243  //util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
244  id = hpx::applier::register_thread_plain(
245  util::bind(&interval_timer::evaluate,
246  this->shared_from_this(), util::placeholders::_1),
247  description_.c_str(), threads::suspended, true,
248  threads::thread_priority_boost, std::size_t(-1),
249  threads::thread_stacksize_default, ec);
250  }
251 
252  if (ec) {
253  is_terminated_ = true;
254  is_started_ = false;
255  return;
256  }
257 
258  // schedule this thread to be run after the given amount of seconds
259  threads::set_thread_state(id,
260  std::chrono::microseconds(microsecs_),
261  threads::pending, threads::wait_signaled,
262  threads::thread_priority_boost, ec);
263 
264  if (ec) {
265  is_terminated_ = true;
266  is_started_ = false;
267 
268  // abort the newly created thread
269  threads::set_thread_state(id, threads::pending, threads::wait_abort,
270  threads::thread_priority_boost, ec);
271 
272  return;
273  }
274 
275  id_ = id;
276  is_started_ = true;
277  }

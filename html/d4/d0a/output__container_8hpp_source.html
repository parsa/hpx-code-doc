231  void save_binary(void const* address, std::size_t count) // override
232  {
233  HPX_ASSERT(count != 0);
234 
235  // make sure there is a current serialization_chunk descriptor
236  // available
237  if (chunker_.get_chunk_type() == chunk_type_pointer ||
238  chunker_.get_chunk_size() != 0)
239  {
240  // add a new serialization_chunk,
241  // the chunk size will be set at the end
242  chunker_.push_back(create_index_chunk(current_, 0));
243  }
244 
245  std::size_t new_current = current_ + count;
246  if (access_traits::size(cont_) < new_current)
247  access_traits::resize(cont_, count);
248 
249  access_traits::write(cont_, count, current_, address);
250 
251  current_ = new_current;
252  }
253 
254  std::size_t save_binary_chunk(void const* address, std::size_t count) // override
255  {
256  if (count < HPX_ZERO_COPY_SERIALIZATION_THRESHOLD)
257  {
258  // fall back to serialization_chunk-less archive
259  this->output_container::save_binary(address, count);
260  // the container has grown by count bytes
261  return count;
262  }
263  else {
264  HPX_ASSERT(
265  chunker_.get_chunk_type() == chunk_type_index ||
266  chunker_.get_chunk_size() != 0);
267 
268  // complement current serialization_chunk by setting its length
269  if (chunker_.get_chunk_type() == chunk_type_index)
270  {
271  HPX_ASSERT(chunker_.get_chunk_size() == 0);
272 
273  chunker_.set_chunk_size(
274  current_ - chunker_.get_chunk_data_index());
275  }
276 
277  // add a new serialization_chunk referring to the external
278  // buffer
279  chunker_.push_back(create_pointer_chunk(address, count));
280  // the container did not grow
281  return 0;
282  }
283  }

295  typedef traits::serialization_access_data<Container> access_traits;

298  filtered_output_container(Container& cont,
299  std::vector<serialization_chunk>* chunks = nullptr)

301  start_compressing_at_(0), filter_(nullptr)

307  void flush()
308  {
309  std::size_t written = 0;
310 
311  if (access_traits::size(this->cont_) < this->current_)
312  access_traits::resize(this->cont_, this->current_);
313 
314  this->current_ = start_compressing_at_;
315 
316  do {
317  bool flushed = access_traits::flush(
318  filter_, this->cont_, this->current_,
319  access_traits::size(this->cont_)-this->current_, written);
320 
321  this->current_ += written;
322  if (flushed)
323  break;
324 
325  // resize container
326  std::size_t size = access_traits::size(this->cont_);
327  access_traits::resize(this->cont_, 2 * size);
328 
329  } while (true);
330 
331  // truncate container
332  access_traits::resize(this->cont_, this->current_);
333  }
334 
335  void set_filter(binary_filter* filter) // override
336  {
337  HPX_ASSERT(nullptr == filter_ && filter != nullptr);
338  filter_ = filter;
339  start_compressing_at_ = this->current_;
340 
341  this->base_type::set_filter(nullptr);
342  }
343 
344  void save_binary(void const* address, std::size_t count) // override
345  {
346  HPX_ASSERT(count != 0);
347 
348  // during construction the filter may not have been set yet
349  if (filter_ != nullptr)
350  filter_->save(address, count);
351  this->current_ += count;
352  }

357  {
358  // fall back to serialization_chunk-less archive
359  HPX_ASSERT(count != 0);
360  filter_->save(address, count);
361  this->current_ += count;
362  return count;

364  else {
365  return this->base_type::save_binary_chunk(address, count);
366  }
367  }

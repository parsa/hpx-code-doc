38 int hpx_main()
39 {
40  {
41  // Now it is possible to instantiate a new counter instance based on
42  // the naming scheme "/test{locality#0/total}/data". Try invoking this
43  // example using the command line option:
44  //
45  // --hpx:print-counter=/test{locality#0/total}/data
46  //
47  // This will print something like:
48  //
49  // test{locality#0/total}/data,1,1.005240[s],1
50  //
51  // where the first value is a sequence counter of the invocation of the
52  // counter. The next is the timestamp marking the point in time since
53  // application startup at which this counter has been queried. The last
54  // value printed is the counter value retrieved at the listed point in
55  // time.
56 
57  // By invoking this example with the command line options:
58  //
59  // --hpx:print-counter=/test{locality#0/total}/data
60  // --hpx:print-counter-interval=100
61  //
62  // the counter will be queried periodically and the output will look
63  // like:
64  //
65  // test{locality#0/total}/data,1,0.001937[s],1
66  // test{locality#0/total}/data,2,0.109625[s],2
67  // test{locality#0/total}/data,3,0.217192[s],3
68  // test{locality#0/total}/data,4,0.323497[s],4
69  // test{locality#0/total}/data,5,0.430867[s],5
70  // test{locality#0/total}/data,6,0.536965[s],6
71  // test{locality#0/total}/data,7,0.643422[s],7
72  // test{locality#0/total}/data,8,0.750788[s],8
73  // test{locality#0/total}/data,9,0.857031[s],9
74  // test{locality#0/total}/data,10,0.963330[s],10
75  // test{locality#0/total}/data,11,1.015063[s],11
76  //
77  // which shows that the counter has been queried roughly every 100
78  // milliseconds, as specified. Additionally it will be queried during
79  // application shutdown, as before.
80 
81  hpx::this_thread::suspend(1000); // wait for one second
82  }
83  return hpx::finalize();
84 }

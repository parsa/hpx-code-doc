120  naming::resolver_client& agas_client)
121  {
122  typedef typename Component::type_holder type_holder;
123  if (component_invalid == components::get_component_type<type_holder>())
124  {
125  // First call to get_component_type, ask AGAS for a unique id.
126  if (isenabled_) {
127  component_type const ctype =
128  agas_client.register_factory(locality, get_component_name());

130  if (component_invalid == ctype) {

135  }
136  components::set_component_type<type_holder>(ctype);
137  }

143  }
144  }

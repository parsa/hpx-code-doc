81  void parse_proc_io(proc_io& pio)
82  {
83  boost::format fmt(PROC_IO_PATH);
84  pid_t pid = getpid();
85  std::string fn = boost::str(fmt % pid);
86  std::ifstream ins(fn);
87 
88  if (!ins.is_open())
89  HPX_THROW_EXCEPTION(hpx::no_success,
90  "hpx::performance_counters::io::parse_proc_io",
91  boost::str(boost::format("failed to open " PROC_IO_PATH) % pid));
92 
93  typedef boost::spirit::basic_istream_iterator<char> iterator;
94  iterator it(ins), end;
95  proc_io_parser<iterator> p;
96 
97  if (!qi::phrase_parse(it, end, p, ascii::space, pio))
98  HPX_THROW_EXCEPTION(hpx::no_success,
99  "hpx::performance_counters::io::parse_proc_io",
100  boost::str(boost::format("failed to parse " PROC_IO_PATH) % pid));
101  }

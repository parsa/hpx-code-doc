100 void test_parcel_serialization(hpx::parcelset::parcel outp,
101  int out_archive_flags, bool zero_copy)
102 {
103  // serialize data
104  std::vector<hpx::serialization::serialization_chunk> out_chunks;
105  std::size_t arg_size = get_archive_size(outp, out_archive_flags,
106  zero_copy ? &out_chunks : nullptr);
107  std::vector<char> out_buffer;
108 
109  out_buffer.resize(arg_size + HPX_PARCEL_SERIALIZATION_OVERHEAD);
110 
111  {
112  // create an output archive and serialize the parcel
113  hpx::serialization::output_archive archive(
114  out_buffer, out_archive_flags,
115  zero_copy ? &out_chunks : nullptr);
116  archive << outp;
117 
118  arg_size = archive.bytes_written();
119  }
120 
121  out_buffer.resize(arg_size);
122 
123  // deserialize data
124  hpx::parcelset::parcel inp;
125 
126  {
127  // create an input archive and deserialize the parcel
128  hpx::serialization::input_archive archive(
129  out_buffer, arg_size, &out_chunks);
130 
131  archive >> inp;
132  }
133 
134  // make sure the parcel has been de-serialized properly
135  HPX_TEST_EQ(outp.source_id(), inp.source_id());

143  HPX_TEST_EQ(outact->get_action_name(), inact->get_action_name());
144  HPX_TEST_EQ(int(outact->get_action_type()), int(inact->get_action_type()));

40 double wait_tasks(std::size_t num_samples, std::size_t num_tasks,
41  std::size_t num_chunks, std::size_t delay)
42 {
43  std::size_t num_chunk_tasks = ((num_tasks + num_chunks) / num_chunks) - 1;
44  std::size_t last_num_chunk_tasks = num_tasks - (num_chunks - 1) * num_chunk_tasks;
45 
46  double result = 0;
47 
48  for (std::size_t k = 0; k != num_samples; ++k)
49  {
50  std::vector<std::vector<hpx::future<void> > > chunks;
51  chunks.reserve(num_chunks);
52  for (std::size_t c = 0; c != num_chunks - 1; ++c)
53  {
54  chunks.push_back(create_tasks(num_chunk_tasks, delay));
55  }
56  chunks.push_back(create_tasks(last_num_chunk_tasks, delay));
57 
58  std::vector<hpx::future<void> > chunk_results;
59  chunk_results.reserve(num_chunks);
60 
61  // wait of tasks in chunks
62  hpx::util::high_resolution_timer t;
63  if (num_chunks == 1)
64  {
65  hpx::wait_all(chunks[0]);
66  }
67  else
68  {
69  for (std::size_t c = 0; c != num_chunks; ++c)
70  {
71  chunk_results.push_back(
72  hpx::async([&chunks, c]() { hpx::wait_all(chunks[c]); })
73  );
74  }
75  hpx::wait_all(chunk_results);
76  }
77  result += t.elapsed();
78  }
79 
80  return result / num_samples;
81 }

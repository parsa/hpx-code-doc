150  void interpolator::fill_partitions(std::string const& datafilename,
151  std::string symbolic_name_base, std::size_t num_instances)
152  {
153  auto locs = hpx::find_all_localities();
154  hpx::future<std::vector<partition3d> > result =
155  hpx::new_<partition3d[]>(hpx::default_layout(locs), num_instances);
156 
157  // Read required data from file.
158  {
159  std::lock_guard<mutex_type> l(mtx_);
160  num_values_[dimension::ye] = extract_data_range(datafilename,
161  "ye", minval_[dimension::ye], maxval_[dimension::ye],
162  delta_[dimension::ye]);
163  num_values_[dimension::temp] = extract_data_range(datafilename,
164  "logtemp", minval_[dimension::temp], maxval_[dimension::temp],
165  delta_[dimension::temp]);
166  num_values_[dimension::rho] = extract_data_range(datafilename,
167  "logrho", minval_[dimension::rho], maxval_[dimension::rho],
168  delta_[dimension::rho]);
169  }
170 
171  // Initialize all attached partition objects.
172  num_partitions_per_dim_ = static_cast<std::size_t>(
173  std::cbrt(double(locs.size())));
174 
175  std::size_t partition_size_x =
176  num_values_[dimension::ye] / num_partitions_per_dim_;
177  std::size_t last_partition_size_x = num_values_[dimension::ye] -
178  partition_size_x * (num_partitions_per_dim_-1);
179 
180  std::size_t partition_size_y =
181  num_values_[dimension::temp] / num_partitions_per_dim_;
182  std::size_t last_partition_size_y = num_values_[dimension::temp] -
183  partition_size_y * (num_partitions_per_dim_-1);
184 
185  std::size_t partition_size_z =
186  num_values_[dimension::rho] / num_partitions_per_dim_;
187  std::size_t last_partition_size_z = num_values_[dimension::rho] -
188  partition_size_z * (num_partitions_per_dim_-1);
189 
190  dimension dim_x(num_values_[dimension::ye]);
191  dimension dim_y(num_values_[dimension::temp]);
192  dimension dim_z(num_values_[dimension::rho]);
193 
194  std::vector<hpx::future<void> > lazy_sync;
195  lazy_sync.reserve(2 * num_instances);
196 
197  for (std::size_t x = 0; x != num_partitions_per_dim_; ++x)
198  {
199  dim_x.offset_ = partition_size_x * x;
200  if (x == num_partitions_per_dim_-1)
201  dim_x.count_ = last_partition_size_x;
202  else
203  dim_x.count_ = partition_size_x;
204 
205  for (std::size_t y = 0; y != num_partitions_per_dim_; ++y)
206  {
207  dim_y.offset_ = partition_size_y * y;
208  if (y == num_partitions_per_dim_-1)
209  dim_y.count_ = last_partition_size_y;
210  else
211  dim_y.count_ = partition_size_y;
212 
213  for (std::size_t z = 0; z != num_partitions_per_dim_; ++z)
214  {
215  dim_z.offset_ = partition_size_z * z;
216  if (z == num_partitions_per_dim_-1)
217  dim_z.count_ = last_partition_size_z;
218  else
219  dim_z.count_ = partition_size_z;
220 
221  std::size_t index = x + (y + z * num_partitions_per_dim_) *
222  num_partitions_per_dim_;
223  HPX_ASSERT(index < partitions_.size());
224 
225  lazy_sync.push_back(partitions_[index].init_async(
226  datafilename, dim_x, dim_y, dim_z));
227  }
228  }
229  }
230 
231  if (symbolic_name_base[symbolic_name_base.size() - 1] != '/')
232  symbolic_name_base += "/";
233 
234  // Register symbolic names of all involved components.
235  partitions_ = result.get();
236 
237  std::size_t i = 0;
238  for (partition3d & p : partitions_)
239  {
240  lazy_sync.push_back(p.register_as(
241  symbolic_name_base + std::to_string(i++)));
242  }
243 
244  // Wait for initialization to finish.
245  hpx::wait_all(lazy_sync);
246  }

263  std::size_t
264  interpolator::get_partition_index(std::size_t d, double value) const
265  {
266  std::size_t partition_size = num_values_[d] / num_partitions_per_dim_;
267  std::size_t partition_index = static_cast<std::size_t>(
268  (value - minval_[d]) / (delta_[d] * partition_size));
269  if (partition_index == num_partitions_per_dim_)
270  --partition_index;
271  HPX_ASSERT(partition_index < num_partitions_per_dim_);
272  return partition_index;
273  }

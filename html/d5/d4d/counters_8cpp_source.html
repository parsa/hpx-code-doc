350  template <typename Iterator>
351  struct path_parser : qi::grammar<Iterator, path_elements()>
352  {
353  path_parser()
354  : path_parser::base_type(start)
355  {
356  start = -qi::lit(counter_prefix)
357  >> '/' >> +~qi::char_("/{#@") >> -instance
358  >> -('/' >> +~qi::char_("#}@")) >> -('@' >> +qi::char_);
359  instance =
360  '{' >> parent >> -('/' >> child) >> -('/' >> subchild) >> '}'
361  ;
362  parent =
363  &qi::lit('/') >> qi::raw[start] >> qi::attr(-1) >> qi::attr(true)
364  // base counter
365  | +~qi::char_("#/}")
366  >> ( '#' >> raw_uint // counter parent-instance name
367  | -qi::string("#*") // counter parent-instance skeleton name
368  )
369  >> qi::attr(false)
370  ;
371  child =
372  +~qi::char_("#/}")
373  >> ( qi::char_('#') >> +~qi::char_("/}") // counter instance name
374  | -qi::string("#*") // counter instance skeleton name
375  )
376  >> qi::attr(false)
377  ;
378  subchild =
379  +~qi::char_("#}")
380  >> ( '#' >> raw_uint // counter (sub-)instance name
381  | -qi::string("#*") // counter (sub-)instance skeleton name
382  )
383  >> qi::attr(false)
384  ;
385  raw_uint = qi::raw[qi::uint_];
386  }
387 
388  qi::rule<Iterator, path_elements()> start;
389  qi::rule<Iterator, instance_elements()> instance;
390  qi::rule<Iterator, instance_name()> parent;
391  qi::rule<Iterator, instance_name()> child;
392  qi::rule<Iterator, instance_name()> subchild;
393  qi::rule<Iterator, std::string()> raw_uint;
394  };

402  counter_status get_counter_path_elements(std::string const& name,
403  counter_path_elements& path, error_code& ec)
404  {
405  path_elements elements;
406  path_parser<std::string::const_iterator> p;
407 
408  std::string::const_iterator begin = name.begin();
409  if (!qi::parse(begin, name.end(), p, elements) || begin != name.end())
410  {
411  HPX_THROWS_IF(ec, bad_parameter, "get_counter_path_elements",
412  boost::str(boost::format("invalid counter name format: %s") % name));
413  return status_invalid_data;
414  }
415 
416  path.objectname_ = elements.object_;
417  path.countername_ = elements.counter_;
418 
419  path.parentinstancename_ = elements.instance_.parent_.name_;
420  path.parentinstance_is_basename_ = elements.instance_.parent_.basename_;
421  path.parentinstanceindex_ = -1;
422 
423  path.instancename_ = elements.instance_.child_.name_;
424  path.instanceindex_ = -1;
425 
426  path.subinstancename_ = elements.instance_.subchild_.name_;
427  path.subinstanceindex_ = -1;
428 
429  path.parameters_ = elements.parameters_;
430 
431  if (!path.parentinstance_is_basename_)
432  {
433  if (elements.instance_.parent_.index_ == "#*")
434  {
435  path.parentinstancename_ += "#*";
436  }
437  else if (!elements.instance_.parent_.index_.empty())
438  {
439  path.parentinstanceindex_ =
440  hpx::util::safe_lexical_cast<std::int64_t>(
441  elements.instance_.parent_.index_);
442  }
443 
444  if (elements.instance_.child_.index_ == "#*")
445  {
446  path.instancename_ += "#*";
447  }
448  else if (!elements.instance_.child_.index_.empty())
449  {
450  if (elements.instance_.child_.index_[0] == '#')
451  elements.instance_.child_.index_.erase(0, 1);
452 
453  if (path.instancename_ == "pool")
454  {
455  path.instanceindex_ = hpx::resource::get_pool_index(
456  elements.instance_.child_.index_);
457  }
458  else
459  {
460  path.instanceindex_ =
461  hpx::util::safe_lexical_cast<std::int64_t>(
462  elements.instance_.child_.index_, std::int64_t(-2));
463  if (path.instanceindex_ == std::int64_t(-2))
464  {
465  HPX_THROWS_IF(ec, bad_parameter,
466  "get_counter_path_elements",
467  boost::str(boost::format(
468  "invalid counter name format: %s") % name
469  ));
470  return status_invalid_data;
471  }
472  }
473  }
474 
475  if (elements.instance_.subchild_.index_ == "#*")
476  {
477  path.subinstancename_ += "#*";
478  }
479  else if (!elements.instance_.subchild_.index_.empty())
480  {
481  path.subinstanceindex_ =
482  hpx::util::safe_lexical_cast<std::uint64_t>(
483  elements.instance_.subchild_.index_);
484  }
485  }
486 
487  if (&ec != &throws)
488  ec = make_success_code();
489 
490  return status_valid_data;
491  }

559  counter_status complement_counter_info(counter_info& info, error_code& ec)
560  {
561  counter_path_elements p;
562 
563  counter_status status = get_counter_path_elements(info.fullname_, p, ec);
564  if (!status_is_valid(status)) return status;
565 
566  if (p.parentinstancename_.empty()) {
567  p.parentinstancename_ = "locality";
568  p.parentinstanceindex_ = static_cast<std::int64_t>(get_locality_id());
569  if (p.instancename_.empty())
570  {
571  p.instancename_ = "total";
572  p.instanceindex_ = -1;
573  }
574  if (p.subinstancename_.empty())
575  {
576  p.subinstanceindex_ = -1;
577  }
578  }
579 
580  // fill with complete counter type info
581  std::string type_name;
582  get_counter_type_name(p, type_name, ec);
583  if (!status_is_valid(status)) return status;
584 
585  get_counter_type(type_name, info, ec);
586  if (!status_is_valid(status)) return status;
587 
588  // last, set full counter name
589  return get_counter_name(p, info.fullname_, ec);
590  }

1168  lcos::future<naming::id_type> get_counter_async(
1169  counter_info const& info, error_code& ec)
1170  {
1171  typedef lcos::future<naming::id_type> result_type;
1172 
1173  // complement counter info data
1174  counter_info complemented_info = info;
1175  complement_counter_info(complemented_info, ec);

1177 
1178  ensure_counter_prefix(complemented_info.fullname_);
1179  // pre-pend prefix, if necessary
1180

1182  naming::id_type id = agas::resolve_name(launch::sync,
1183  complemented_info.fullname_, ec);

1186  // figure out target locality
1187  counter_path_elements p;
1188  get_counter_path_elements(complemented_info.fullname_, p, ec);

1195  get_counter_path_elements(p.parentinstancename_, p, ec);
1196  if (ec) return result_type();

1198

1200  ( p.parentinstanceindex_ < 0 ||
1201  p.parentinstanceindex_ >=
1202  static_cast<std::int32_t>(
1203  get_num_localities(hpx::launch::sync))
1204  )
1205  )
1206  {

1208  "attempt to create counter on non-existing locality");

1210  }

1213  // create the new performance counter
1214  using namespace components::stubs;

1218  naming::get_id_from_locality_id(
1219  static_cast<std::uint32_t>(p.parentinstanceindex_)),
1220  complemented_info);
1221  }

1223  f = runtime_support::create_performance_counter_async(
1224  find_here(), complemented_info);
1225  }

1229  hpx::util::placeholders::_1,

1240  }

1244  }

35 std::uint64_t fibonacci(std::uint64_t n)
36 {
37  if (n < 2)
38  return n;
39 
40  // We restrict ourselves to execute the Fibonacci function locally.
41  hpx::naming::id_type const locality_id = hpx::find_here();
42 
43  // Invoking the Fibonacci algorithm twice is inefficient.
44  // However, we intentionally demonstrate it this way to create some
45  // heavy workload.
46 
47  fibonacci_action fib;
48  hpx::future<std::uint64_t> n1 =
49  hpx::async(fib, locality_id, n - 1);
50  hpx::future<std::uint64_t> n2 =
51  hpx::async(fib, locality_id, n - 2);
52 
53  return n1.get() + n2.get(); // wait for the Futures to return their values
54 }

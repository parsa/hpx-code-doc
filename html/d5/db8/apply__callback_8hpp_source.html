151  template <typename Action, typename Continuation, typename Callback, typename ...Ts>
152  inline bool
153  apply_p_cb(Continuation && c, naming::address&& addr,
154  naming::id_type const& gid, threads::thread_priority priority,
155  Callback && cb, Ts&&... vs)
156  {
157  if (!traits::action_is_target_valid<Action>::call(gid)) {
158  HPX_THROW_EXCEPTION(bad_parameter, "apply_p_cb",
159  boost::str(boost::format(
160  "the target (destination) does not match the action type (%s)"
161  ) % hpx::actions::detail::get_action_name<Action>()));
162  return false;
163  }
164 
165  // Determine whether the gid is local or remote
166  if (addr.locality_ == hpx::get_locality()) {
167  // apply locally
168  bool result =
169  applier::detail::apply_l_p<Action>(std::forward<Continuation>(c), gid,
170  std::move(addr), priority, std::forward<Ts>(vs)...);
171 
172  // invoke callback
173  cb(boost::system::error_code(), parcelset::parcel());
174  return result;
175  }
176 
177  // apply remotely
178  return applier::detail::apply_r_p_cb<Action>(std::move(addr),
179  std::forward<Continuation>(c), gid,
180  priority, std::forward<Callback>(cb), std::forward<Ts>(vs)...);
181  }

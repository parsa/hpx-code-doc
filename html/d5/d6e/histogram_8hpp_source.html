49  // The histogram histogram estimator returns a histogram of the sample
50  // distribution. The positions and sizes of the bins are determined using a
51  // specifiable number of cached samples (cache_size). The range between the
52  // minimum and the maximum of the cached samples is subdivided into a
53  // specifiable number of bins (num_bins) of same size. Additionally,
54  // an under- and an overflow bin is added to capture future under- and
55  // overflow samples. Once the bins are determined, the cached samples and
56  // all subsequent samples are added to the correct bins. At the end, a
57  // range of std::pair is return, where each pair contains the position of
58  // the bin (lower bound) and the samples count (normalized with the
59  // total number of samples).
60  template <typename Sample>
61  struct histogram_impl : boost::accumulators::accumulator_base
62  {
63  typedef typename boost::numeric::functional::fdiv_base<
64  Sample, std::size_t
65  >::result_type float_type;
66  typedef std::vector<std::pair<float_type, float_type> > histogram_type;
67  typedef std::vector<float_type> array_type;
68 
69  // for boost::result_of
70  typedef util::iterator_range<
71  typename histogram_type::iterator
72  > result_type;
73 
74 // conversion from 'const __int64' to 'const double', possible loss of data
75 #if defined(HPX_MSVC_WARNING_PRAGMA)
76 #pragma warning(push)
77 #pragma warning(disable: 4244)
78 #endif
79  template <typename Args>
80  histogram_impl(Args const& args)
81  : num_bins_(args[histogram_num_bins]),
82  minimum_(args[histogram_min_range]),
83  maximum_(args[histogram_max_range]),
84  bin_size_ (boost::numeric::average(
85  args[histogram_max_range] - args[histogram_min_range],
86  args[histogram_num_bins])
87  ),
88  samples_in_bin_(std::size_t(args[histogram_num_bins] + 2), 0.),
89  bin_positions_(args[histogram_num_bins] + 2),
90  histogram_(
91  std::size_t(args[histogram_num_bins] + 2),
92  std::make_pair(0, 1)
93  ),
94  is_dirty_(true)
95  {
96  // determine bin positions (their lower bounds)
97  for (std::size_t i = 0; i < this->num_bins_ + 2; ++i)
98  {
99  this->bin_positions_[i] = minimum_ + (i - 1.0) * bin_size_;
100  }
101  }
102 #if defined(HPX_MSVC_WARNING_PRAGMA)
103 #pragma warning(pop)
104 #endif

106  template <typename Args>
107  void operator()(Args const &args)
108  {
109  //std::size_t cnt = count(args);
110  {
111  if (args[boost::accumulators::sample] <
112  this->bin_positions_[1])
113  {
114  ++(this->samples_in_bin_[0]);
115  }
116  else if (args[boost::accumulators::sample] >=
117  this->bin_positions_[this->num_bins_ + 1])
118  {
119  ++(this->samples_in_bin_[this->num_bins_ + 1]);
120  }
121  else
122  {
123  typename array_type::iterator it = std::upper_bound(
124  this->bin_positions_.begin()
125  , this->bin_positions_.end()
126  , args[boost::accumulators::sample]
127  );
128 
129  std::size_t d =
130  std::distance(this->bin_positions_.begin(), it);
131  ++(this->samples_in_bin_[d - 1]);
132  }
133  }
134  }
135 
136  template <typename Args>
137  result_type result(Args const &args) const
138  {
139  {
140  // creates a vector of std::pair where each pair i holds
141  // the values bin_positions[i] (x-axis of histogram) and
142  // samples_in_bin[i] / cnt (y-axis of histogram).
143 
144  for (std::size_t i = 0; i < this->num_bins_ + 2; ++i)
145  {
146  this->histogram_[i] = std::make_pair(
147  this->bin_positions_[i],
148  boost::numeric::average(
149  this->samples_in_bin_[i],
150  boost::accumulators::count(args)));
151  }
152  }
153 
154  // returns a range of pairs
155  return util::make_iterator_range(this->histogram_);
156  }

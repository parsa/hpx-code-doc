451  naming::gid_type retrieve_agas_counter(std::string const& name,
452  naming::id_type const& agas_id, error_code& ec)
453  {
454  naming::gid_type id;
455 
456  // get action code from counter type
457  agas::namespace_action_code service_code =
458  agas::detail::retrieve_action_service_code(name, ec);
459  if (agas::invalid_request == service_code) return id;
460 
461  switch (service_code) {
462  case agas::component_ns_statistics_counter:
463  {
464  agas::server::component_namespace::statistics_counter_action
465  action;
466  return action(agas_id, name).get_gid();
467  }
468  case agas::locality_ns_statistics_counter:
469  {
470  agas::server::locality_namespace::statistics_counter_action
471  action;
472  return action(agas_id, name).get_gid();
473  }
474  case agas::symbol_ns_statistics_counter:
475  {
476  agas::server::symbol_namespace::statistics_counter_action
477  action;
478  return action(agas_id, name).get_gid();
479  }
480  case agas::primary_ns_statistics_counter:
481  {
482  agas::server::primary_namespace::statistics_counter_action
483  action;
484  return action(agas_id, name).get_gid();
485  }
486  default:
487  HPX_THROWS_IF(ec, bad_parameter, "retrieve_statistics_counter",
488  "unknown counter agas counter name: " + name);
489  break;
490  }
491 
492  return id;

502  naming::gid_type agas_raw_counter_creator(
503  counter_info const& info, error_code& ec, char const* const service_name)
504  {
505  // verify the validity of the counter instance name
506  counter_path_elements paths;
507  get_counter_path_elements(info.fullname_, paths, ec);
508  if (ec) return naming::invalid_gid;
509 
510  if (paths.objectname_ != "agas") {
511  HPX_THROWS_IF(ec, bad_parameter, "agas_raw_counter_creator",
512  "unknown performance counter (unrelated to AGAS)");
513  return naming::invalid_gid;
514  }
515  if (paths.parentinstance_is_basename_) {
516  HPX_THROWS_IF(ec, bad_parameter, "agas_raw_counter_creator",
517  "invalid counter instance parent name: " +
518  paths.parentinstancename_);
519  return naming::invalid_gid;
520  }
521 
522  // counter instance name: <agas_instance_name>/total
523  // for instance: locality#0/total
524  if (paths.instancename_ == "total" && paths.instanceindex_ == -1)
525  {
526  // find the referenced AGAS instance and dispatch the request there
527  std::string service(agas::service_name);
528  service += paths.parentinstancename_;
529 
530  if (-1 == paths.parentinstanceindex_) {
531  HPX_THROWS_IF(ec, bad_parameter, "agas_raw_counter_creator",
532  "invalid parent instance index: -1");
533  return naming::invalid_gid;
534  }
535  service += "#";
536  service += std::to_string(paths.parentinstanceindex_);
537 
538  service += "/";
539  service += service_name;
540 
541  naming::id_type id = agas::resolve_name(launch::sync, service, ec);
542  if (id == naming::invalid_id) {
543  HPX_THROWS_IF(ec, not_implemented,
544  "agas_raw_counter_creator",
545  "invalid counter name: " +
546  remove_counter_prefix(info.fullname_));
547  return naming::invalid_gid;
548  }
549 
550  return detail::retrieve_agas_counter(info.fullname_, id, ec);
551  }
552 
553  HPX_THROWS_IF(ec, not_implemented, "agas_raw_counter_creator",
554  "invalid counter type name: " + paths.instancename_);
555  return naming::invalid_gid;
556  }

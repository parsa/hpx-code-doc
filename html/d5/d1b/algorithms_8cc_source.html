413 void gemm(bool transa, bool transb,
414  double alpha, const matrix_t& a, const matrix_t& b,
415  double beta, matrix_t& c)
416 {
417  if (alpha == 0.0) {
418  scal(beta, c);
419  return;
420  }
421  if (!transb) {
422  if (!transa) {
423  // c = alpha a b + beta c
424  assert(b.NI == a.NJ);
425  assert(c.NI == a.NI);
426  assert(c.NJ == b.NJ);
427  for (std::ptrdiff_t j=0; j<c.NJ; ++j) {
428  if (beta == 0.0) {
429  for (std::ptrdiff_t i=0; i<c.NI; ++i) {
430  c(i,j) = 0.0;
431  }
432  } else {
433  for (std::ptrdiff_t i=0; i<c.NI; ++i) {
434  c(i,j) *= beta;
435  }
436  }
437  for (std::ptrdiff_t k=0; k<b.NI; ++k) {
438  if (b(k,j) != 0.0) {
439  double tmp = alpha * b(k,j);
440  for (std::ptrdiff_t i=0; i<c.NI; ++i) {
441  c(i,j) += tmp * a(i,k);
442  }
443  }
444  }
445  }
446  } else {
447  // c = alpha a^T b + beta c
448  assert(b.NI == a.NI);
449  assert(c.NI == a.NJ);
450  assert(c.NJ == b.NJ);
451  for (std::ptrdiff_t j=0; j<c.NJ; ++j) {
452  for (std::ptrdiff_t i=0; i<c.NI; ++i) {
453  double tmp = 0.0;
454  for (std::ptrdiff_t k=0; k<b.NI; ++k) {
455  tmp += a(k,i) * b(k,j);
456  }
457  if (beta == 0.0) {
458  c(i,j) = alpha * tmp;
459  } else {
460  c(i,j) = alpha * tmp + beta * c(i,j);
461  }
462  }
463  }
464  }
465  } else {
466  if (!transa) {
467  // c = alpha a b^T + beta c
468  assert(b.NJ == a.NJ);
469  assert(c.NI == a.NI);
470  assert(c.NJ == b.NI);
471  for (std::ptrdiff_t j=0; j<c.NJ; ++j) {
472  if (beta == 0.0) {
473  for (std::ptrdiff_t i=0; i<c.NI; ++i) {
474  c(i,j) = 0.0;
475  }
476  } else if (beta != 1.0) {
477  for (std::ptrdiff_t i=0; i<c.NI; ++i) {
478  c(i,j) *= beta;
479  }
480  }
481  for (std::ptrdiff_t k=0; k<b.NJ; ++k) {
482  if (b(j,k) != 0.0) {
483  double tmp = alpha * b(j,k);
484  for (std::ptrdiff_t i=0; i<c.NI; ++i) {
485  c(i,j) += tmp * a(i,k);
486  }
487  }
488  }
489  }
490  } else {
491  // c = alpha a^T b^T + beta c
492  assert(b.NJ == a.NI);
493  assert(c.NI == a.NJ);
494  assert(c.NJ == b.NI);
495  for (std::ptrdiff_t j=0; j<c.NJ; ++j) {

502  c(i,j) = alpha * tmp;
503  } else {
504  c(i,j) = alpha * tmp + beta * c(i,j);

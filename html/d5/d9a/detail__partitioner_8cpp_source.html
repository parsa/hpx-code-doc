85  // mechanism for adding resources
86  // num threads = number of threads desired on a PU. defaults to 1.
87  // note: if num_threads > 1 => oversubscription
88  void init_pool_data::add_resource(
89  std::size_t pu_index, bool exclusive, std::size_t num_threads)
90  {
91  if (pu_index >= hpx::threads::hardware_concurrency())
92  {
93  throw_invalid_argument("init_pool_data::add_resource",
94  "init_pool_data::add_resource: processing unit index "
95  "out of bounds. The total available number of "
96  "processing units on this machine is " +
97  std::to_string(hpx::threads::hardware_concurrency()));
98  }
99 
100  // Increment thread_num count (for pool-count and global count)
101  num_threads_ += num_threads;
102  num_threads_overall += num_threads;
103 
104  // Add pu mask to internal data structure
105  threads::mask_type pu_mask = threads::mask_type();
106  threads::resize(pu_mask, threads::hardware_concurrency());
107  threads::set(pu_mask, pu_index);
108 
109  // Add one mask for each OS-thread
110  for (std::size_t i = 0; i != num_threads; i++)
111  {
112  assigned_pus_.push_back(pu_mask);
113  assigned_pu_nums_.push_back(
114  util::make_tuple(pu_index, exclusive, false)
115  );
116  }
117  }

241  void partitioner::fill_topology_vectors()
242  {
243  std::size_t pid = 0;
244  std::size_t num_numa_nodes = topology_.get_number_of_numa_nodes();
245  if (num_numa_nodes == 0)
246  num_numa_nodes = topology_.get_number_of_sockets();
247  numa_domains_.reserve(num_numa_nodes);
248 
249  // loop on the numa-domains
250  for (std::size_t i = 0; i != num_numa_nodes; ++i)
251  {
252  numa_domains_.emplace_back(i); // add a numa domain
253  numa_domain &nd = numa_domains_.back(); // numa-domain just added
254 
255  std::size_t numa_node_cores =
256  topology_.get_number_of_numa_node_cores(i);
257  nd.cores_.reserve(numa_node_cores);
258 
259  bool numa_domain_contains_exposed_cores = false;
260 
261  // loop on the cores
262  for (std::size_t j = 0; j != numa_node_cores; ++j)
263  {
264  nd.cores_.emplace_back(j, &nd);
265  core &c = nd.cores_.back();
266 
267  std::size_t core_pus = topology_.get_number_of_core_pus(j);
268  c.pus_.reserve(core_pus);
269 
270  bool core_contains_exposed_pus = false;
271 
272  // loop on the processing units
273  for (std::size_t k = 0; k != core_pus; ++k)
274  {
275  if (pu_exposed(pid))
276  {
277  c.pus_.emplace_back(pid, &c,
278  affinity_data_.get_thread_occupancy(pid));
279  pu &p = c.pus_.back();
280 
281  if (p.thread_occupancy_ == 0)
282  {
283  throw_runtime_error(
284  "partitioner::fill_topology_vectors",
285  "PU #" + std::to_string(pid) +
286  " has thread occupancy 0");
287  }
288  core_contains_exposed_pus = true;
289  }
290  ++pid;
291  }
292 
293  if (core_contains_exposed_pus)
294  {
295  numa_domain_contains_exposed_cores = true;
296  }
297  else
298  {
299  nd.cores_.pop_back();
300  }
301  }
302 
303  if (!numa_domain_contains_exposed_cores)
304  {
305  numa_domains_.pop_back();
306  }
307  }
308  }

310  std::size_t partitioner::assign_cores(std::size_t first_core)
311  {
312  std::lock_guard<mutex_type> l(mtx_);
313 
314  // adjust first_core, if needed
315  if (first_core_ != first_core)
316  {
317  std::size_t offset = first_core;
318  std::size_t num_pus_core = topology_.get_number_of_core_pus(offset);
319 
320  if (first_core_ != std::size_t(-1))
321  {
322  offset -= first_core_;
323  }
324 
325  if (offset != 0)
326  {
327  offset *= num_pus_core;
328  for (auto& d : initial_thread_pools_)
329  {
330  d.assign_first_core(offset);
331  }
332  }
333  first_core_ = first_core;
334  reconfigure_affinities_locked();
335  }
336 
337  // should have been initialized by now
338  HPX_ASSERT(cores_needed_ != std::size_t(-1));
339  return cores_needed_;
340  }

342  // This function is called in hpx_init, before the instantiation of the
343  // runtime It takes care of configuring some internal parameters of the
344  // resource partitioner related to the pools
345  // -1 assigns all free resources to the default pool
346  // -2 checks whether there are empty pools
347  void partitioner::setup_pools()
348  {
349  // Assign all free resources to the default pool
350  bool first = true;
351  for (hpx::resource::numa_domain &d : numa_domains_)
352  {
353  for (hpx::resource::core &c : d.cores_)
354  {
355  for (hpx::resource::pu &p : c.pus_)
356  {
357  if (p.thread_occupancy_count_ == 0)
358  {
359  // The default pool resources are assigned non-
360  // exclusively if dynamic pools are enabled.
361  // Also, by default, the first PU is always exclusive
362  // (to avoid deadlocks).
363  add_resource(p, "default",
364  first || !(mode_ & mode_allow_dynamic_pools));
365  first = false;
366  }
367  }
368  }
369  }
370 
371  std::lock_guard<mutex_type> l(mtx_);
372 
373  // @TODO allow empty pools
374  if (get_pool_data("default").num_threads_ == 0)
375  {
376  throw_runtime_error("partitioner::setup_pools",
377  "Default pool has no threads assigned. Please rerun with "
378  "--hpx:threads=X and check the pool thread assignment");
379  }
380 
381  // Check whether any of the pools defined up to now are empty
382  if (check_empty_pools())
383  {
384  throw_runtime_error("partitioner::setup_pools",
385  "Pools empty of resources are not allowed. Please re-run this "
386  "application with allow-empty-pool-policy (not implemented "
387  "yet)");
388  }
390  }

392  // This function is called in hpx_init, before the instantiation of the runtime

395  void partitioner::setup_schedulers()
396  {
397  // select the default scheduler
398  scheduling_policy default_scheduler;
399 
400  if (0 == std::string("local").find(cfg_.queuing_))
401  {
402  default_scheduler = scheduling_policy::local;
403  }
404  else if (0 == std::string("local-priority-fifo").find(cfg_.queuing_))
405  {
406  default_scheduler = scheduling_policy::local_priority_fifo;
407  }
408  else if (0 == std::string("local-priority-lifo").find(cfg_.queuing_))
409  {
410  default_scheduler = scheduling_policy::local_priority_lifo;
411  }
412  else if (0 == std::string("static").find(cfg_.queuing_))
413  {
414  default_scheduler = scheduling_policy::static_;
415  }
416  else if (0 == std::string("static-priority").find(cfg_.queuing_))
417  {
418  default_scheduler = scheduling_policy::static_priority;
419  }
420  else if (0 == std::string("abp-priority").find(cfg_.queuing_))
421  {
422  default_scheduler = scheduling_policy::abp_priority;
423  }
424  else if (0 == std::string("hierarchy").find(cfg_.queuing_))
425  {
426  default_scheduler = scheduling_policy::hierarchy;
427  }
428  else if (0 == std::string("periodic-priority").find(cfg_.queuing_))
429  {
430  default_scheduler = scheduling_policy::periodic_priority;
431  }
432  else if (0 == std::string("throttle").find(cfg_.queuing_))
433  {
434  default_scheduler = scheduling_policy::throttle;
435  }
436  else
437  {
438  throw hpx::detail::command_line_error(
439  "Bad value for command line option --hpx:queuing");
440  }

442  // set this scheduler on the pools that do not have a specified scheduler yet
443  std::lock_guard<mutex_type> l(mtx_);
444  std::size_t npools = initial_thread_pools_.size();
445  for (std::size_t i = 0; i != npools; ++i)
446  {

450  }
451  }

454  // This function is called in hpx_init, before the instantiation of the
455  // runtime. It takes care of configuring some internal parameters of the
456  // resource partitioner related to the affinity bindings
457  //
458  // If we use the resource partitioner, OS-thread numbering gets slightly
459  // complicated: The affinity_masks_ data member of affinity_data considers
460  // OS-threads to be numbered in order of occupation of the consecutive
461  // processing units, while the thread manager will consider them to be
462  // ordered according to their assignment to pools (first all threads
463  // belonging to the default pool, then all threads belonging to the first
464  // pool created, etc.) and instantiate them according to this system.
465  // We need to re-write affinity_data_ with the masks in the correct order
466  // at this stage.
467  void partitioner::reconfigure_affinities()
468  {
469  std::lock_guard<mutex_type> l(mtx_);
470  reconfigure_affinities_locked();
471  }

473  void partitioner::reconfigure_affinities_locked()
474  {
475  std::vector<std::size_t> new_pu_nums;
476  std::vector<threads::mask_type> new_affinity_masks;
477 
478  new_pu_nums.reserve(initial_thread_pools_.size());
479  new_affinity_masks.reserve(initial_thread_pools_.size());
480 
481  {
482  for (auto &itp : initial_thread_pools_)
483  {
484  for (auto const& mask : itp.assigned_pus_)
485  {
486  new_affinity_masks.push_back(mask);
487  }
488  for (auto const& pu_num : itp.assigned_pu_nums_)
489  {
490  new_pu_nums.push_back(util::get<0>(pu_num));
491  }
492  }
493  }
494 
495  affinity_data_.set_num_threads(new_pu_nums.size());
496  affinity_data_.set_pu_nums(std::move(new_pu_nums));
497  affinity_data_.set_affinity_masks(std::move(new_affinity_masks));
498  affinity_data_.init_cached_pu_nums(new_pu_nums.size());
499  }

616  void partitioner::add_resource(
617  pu const& p, std::string const& pool_name, bool exclusive,
618  std::size_t num_threads)
619  {
620  if (get_runtime_ptr() != nullptr)
621  {
622  HPX_THROW_EXCEPTION(invalid_status,
623  "partitioner::add_resource",
624  "this function must be called before the runtime system has "
625  "been started");
626  }
627 
628  std::lock_guard<mutex_type> l(mtx_);
629 
630  if (!exclusive && !(mode_ & mode_allow_dynamic_pools))
631  {
632  throw std::invalid_argument(
633  "partitioner::add_resource: dynamic pools have not been "
634  "enabled for this partitioner");
635  }
636 
637  if (mode_ & mode_allow_oversubscription)
638  {
639  // increment occupancy counter
640  get_pool_data(pool_name).add_resource(
641  p.id_, exclusive, num_threads);
642  ++p.thread_occupancy_count_;

644  }
645 
646  // check occupancy counter and increment it
647  if (p.thread_occupancy_count_ == 0)
648  {
649  get_pool_data(pool_name).add_resource(
650  p.id_, exclusive, num_threads);
651  ++p.thread_occupancy_count_;
652 
653  // Make sure the total number of requested threads does not exceed
654  // the number of threads requested on the command line
655  if (detail::init_pool_data::num_threads_overall > cfg_.num_threads_)
656  {
658  /* if (rp-policy == allow_empty_default_pool
659  && detail::init_pool_data::num_threads_overall == cfg_.num_threads_) {
660  // then it's all fine
661  } else {*/
662  throw std::runtime_error(
663  "partitioner::add_resource: " "Creation of " +
664  std::to_string(detail::init_pool_data::num_threads_overall) +
665  " threads requested by the resource partitioner, but "
666  "only " +
667  std::to_string(cfg_.num_threads_) +
668  " provided on the command-line.");
669  // }
670  }
671  }

673  {
674  throw std::runtime_error(
675  "partitioner::add_resource: " "PU #" + std::to_string(p.id_) +

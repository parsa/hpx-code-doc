257  std::size_t pu_num = rp.get_pu_num(global_id);

260  {
261  global_id = local_to_global_thread_index(local_id);
262  pu_num = rp.get_pu_num(global_id);
263  if (pred(numa, topo.get_numa_node_number(pu_num)))
264  {
265  result.push_back(local_id);
266  }
267  }

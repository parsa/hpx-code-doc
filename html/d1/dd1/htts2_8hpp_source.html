31 template <typename BaseClock>
32 struct clocksource
33 {
34  typedef BaseClock base_clock;
35  typedef typename base_clock::rep rep;
36  typedef typename std::nano period;
37  typedef std::chrono::duration<rep, period> duration;
38 
39  static_assert(base_clock::is_steady == true,
40  "base_clock is not steady");
41 
42  // Returns: current time in nanoseconds.
43  static rep now()
44  {
45  duration d = std::chrono::duration_cast<duration>(
46  base_clock::now().time_since_epoch());
47  rep t = d.count();
48  BOOST_ASSERT(t >= 0);
49  return t;
50  }

52  // Returns: uncertainty of the base_clock in nanoseconds.
53  static double clock_uncertainty()
54  {
55  // For steady clocks, we use instrument uncertainty, ie:
56  // instrument_uncertainty = instrument_least_count/2
57  return 1.0/2.0;

61 // Performs approximately 'expected_' nanoseconds of artificial work.
62 // Returns: nanoseconds of work performed.
63 template <typename BaseClock>
64 typename clocksource<BaseClock>::rep
65 payload(typename clocksource<BaseClock>::rep expected)
66 {
67  typedef typename clocksource<BaseClock>::rep rep;
68 
69  rep const start = clocksource<BaseClock>::now();
70 
71  while (true)
72  {
73  rep const measured = clocksource<BaseClock>::now() - start;
74 
75  if (measured >= expected)
76  return measured;
77  }
78 }

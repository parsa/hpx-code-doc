136  {
137  // empty parcels are always valid
138 #if defined(HPX_HAVE_PARCEL_PROFILING)

140  return true;
141 #endif
142 
143  // verify target destination

155  if (!components::types_are_compatible(type, data_.addr_.type_))
156  {
157  return false;
158  }
159  }
160 
161  return true;
162  }

369  std::pair<naming::address_type, naming::component_type>
370  parcel::determine_lva()
371  {
372  naming::resolver_client& client = hpx::naming::get_agas_client();
373  int comptype = action_->get_component_type();
374 
375  // decode the local virtual address of the parcel
376  naming::address::address_type lva = data_.addr_.address_;
377  // by convention, a zero address references either the local
378  // runtime support component or one of the AGAS components
379  if (0 == lva)
380  {
381  switch(comptype)
382  {
383  case components::component_runtime_support:
384  lva = hpx::applier::get_applier()
385  .get_runtime_support_raw_gid()
386  .get_lsb();
387  break;
388 
389  case components::component_agas_primary_namespace:
390  lva = client.get_primary_ns_lva();
391  break;
392 
393  case components::component_agas_symbol_namespace:
394  lva = client.get_symbol_ns_lva();
395  break;
396 
397  case components::component_plain_function:
398  break;
399 
400  default:
401  HPX_ASSERT(false);
402  }
403  }
404  else if (comptype == components::component_memory)
405  {
406  HPX_ASSERT(naming::refers_to_virtual_memory(data_.dest_));
407  lva = hpx::applier::get_applier().get_memory_raw_gid().get_lsb();
408  }
409 
410  // make sure the component_type of the action matches the
411  // component type in the destination address
412  if (HPX_UNLIKELY(!components::types_are_compatible(
413  data_.addr_.type_, comptype)))
414  {
415  std::ostringstream strm;
416  strm << " types are not compatible: destination_type("
417  << data_.addr_.type_ << ") action_type(" << comptype
418  << ") parcel (" << *this << ")";
419  HPX_THROW_EXCEPTION(bad_component_type,
420  "applier::schedule_action",
421  strm.str());
422  }
423 
424  return std::make_pair(lva, comptype);
425  }

427  bool parcel::load_schedule(serialization::input_archive & ar,
428  std::size_t num_thread, bool& deferred_schedule)
429  {
430  load_data(ar);
431 
432  // make sure this parcel destination matches the proper locality
433  HPX_ASSERT(destination_locality() == data_.addr_.locality_);
434 
435  std::pair<naming::address_type, naming::component_type> p = determine_lva();
436 
437  // make sure the target has not been migrated away
438  auto r = action_->was_object_migrated(data_.dest_, p.first);
439  if (r.first)
440  {
441  // If the object was migrated, just load the action and return.
442  action_->load(ar);
443  return true;
444  }
445 
446  // continuation support, this is handled in the transfer action
447  action_->load_schedule(ar, std::move(data_.dest_), p.first, p.second,
448  num_thread, deferred_schedule);
449 
450 #if HPX_HAVE_ITTNOTIFY != 0 && !defined(HPX_HAVE_APEX)
451  static util::itt::event parcel_recv("recv_parcel");
452  util::itt::event_tick(parcel_recv);
453 #endif
454 
455 #if defined(HPX_HAVE_APEX) && defined(HPX_HAVE_PARCEL_PROFILING)
456  // tell APEX about the received parcel
457  apex::recv(data_.parcel_id_.get_lsb(), size_,
458  naming::get_locality_id_from_gid(data_.source_id_),
459  reinterpret_cast<std::uint64_t>(action_->get_parent_thread_id()));
460 #endif
461 
462  return false;
463  }

465  void parcel::schedule_action(std::size_t num_thread)
466  {
467  // make sure this parcel destination matches the proper locality
468  HPX_ASSERT(destination_locality() == data_.addr_.locality_);
469 
470  std::pair<naming::address_type, naming::component_type> p = determine_lva();
471 
472  // make sure the target has not been migrated away
473  auto r = action_->was_object_migrated(data_.dest_, p.first);
474  if (r.first)
475  {
476  // If the object was migrated, just route.
477  naming::resolver_client& client = hpx::naming::get_agas_client();
478  client.route(
479  std::move(*this),
480  &detail::parcel_route_handler,
481  threads::thread_priority_normal);
482  return;
483  }
484 
485  // dispatch action, register work item either with or without
486  // continuation support, this is handled in the transfer action
487  action_->schedule_thread(std::move(data_.dest_), p.first, p.second,
488  num_thread);
489  }

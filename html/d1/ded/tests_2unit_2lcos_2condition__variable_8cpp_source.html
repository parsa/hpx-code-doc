465 void condition_test_waits(condition_test_data* data)
466 {
467  std::unique_lock<hpx::lcos::local::mutex> lock(data->mutex);
468  HPX_TEST(lock ? true : false);
469 
470  // Test wait.
471  while (data->notified != 1)
472  data->condition.wait(lock);
473  HPX_TEST(lock ? true : false);
474  HPX_TEST_EQ(data->notified, 1);
475  data->awoken++;
476  data->condition.notify_one();
477 
478  // Test predicate wait.
479  data->condition.wait(lock, cond_predicate(data->notified, 2));
480  HPX_TEST(lock ? true : false);
481  HPX_TEST_EQ(data->notified, 2);
482  data->awoken++;
483  data->condition.notify_one();
484 
485  // Test wait_until.
486  std::chrono::system_clock::time_point xt =
487  std::chrono::system_clock::now()
488  + std::chrono::milliseconds(10);
489  while (data->notified != 3)
490  data->condition.wait_until(lock, xt);
491  HPX_TEST(lock ? true : false);
492  HPX_TEST_EQ(data->notified, 3);
493  data->awoken++;
494  data->condition.notify_one();
495 
496  // Test predicate wait_until.
497  xt = std::chrono::system_clock::now()
498  + std::chrono::milliseconds(10);
499  cond_predicate pred(data->notified, 4);
500  HPX_TEST(data->condition.wait_until(lock, xt, pred));
501  HPX_TEST(lock ? true : false);
502  HPX_TEST(pred());
503  HPX_TEST_EQ(data->notified, 4);
504  data->awoken++;
505  data->condition.notify_one();
506 
507  // Test predicate wait_for
508  cond_predicate pred_rel(data->notified, 5);
509  HPX_TEST(data->condition.wait_for(lock, std::chrono::milliseconds(10), pred_rel));
510  HPX_TEST(lock ? true : false);
511  HPX_TEST(pred_rel());
512  HPX_TEST_EQ(data->notified, 5);
513  data->awoken++;
514  data->condition.notify_one();
515 }

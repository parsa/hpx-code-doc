107  bool pool_timer::start(util::steady_duration const& time_duration,
108  bool evaluate_)
109  {
110  std::unique_lock<mutex_type> l(mtx_);
111  if (is_terminated_)
112  return false;
113 
114  if (!is_started_) {
115  is_stopped_ = false;
116  is_started_ = true;
117 
118  if (first_start_) {
119  first_start_ = false;
120 
121  util::unlock_guard<std::unique_lock<mutex_type> > ul(l);
122  if (pre_shutdown_)
123  {
124  register_pre_shutdown_function(
125  util::bind(&pool_timer::terminate,
126  this->shared_from_this()));
127  }
128  else
129  {
130  register_shutdown_function(
131  util::bind(&pool_timer::terminate,
132  this->shared_from_this()));
133  }
134  }
135 
136  HPX_ASSERT(timer_ != nullptr);
137  timer_->expires_from_now(time_duration.value());
138  timer_->async_wait(util::bind(&pool_timer::timer_handler,
139  this->shared_from_this(), util::placeholders::_1));
140 
141  return true;
142  }
143  return false;

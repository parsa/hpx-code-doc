417  template <typename Statistic>
418  hpx::performance_counters::counter_value
419  statistics_counter<Statistic>::get_counter_value(bool reset)
420  {
421  std::lock_guard<mutex_type> l(mtx_);
422 
423  hpx::performance_counters::counter_value value;
424 
425  prev_value_.value_ = static_cast<std::int64_t>(value_->get_value());
426  prev_value_.status_ = status_new_data;
427  prev_value_.time_ = static_cast<std::int64_t>(hpx::get_system_uptime());
428  prev_value_.count_ = ++invocation_count_;
429  has_prev_value_ = true;
430 
431  value = prev_value_; // return value
432 
433  if (reset || value_->need_reset())
434  {
435  value_.reset(new detail::counter_type_from_statistic<Statistic>(
436  parameter2_)); // reset accumulator
437  value_->add_value(static_cast<double>(prev_value_.value_));
438  // start off with last base value
439  }
440 
441  return value;
442  }

444  template <typename Statistic>
445  bool statistics_counter<Statistic>::evaluate()
446  {
447  // gather current base value
448  counter_value base_value;
449  if (!evaluate_base_counter(base_value))
450  return false;
451 
452  // simply average the measured base counter values since it got queried
453  // for the last time
454  counter_value value;
455  if (base_value.scaling_ != prev_value_.scaling_ ||
456  base_value.scale_inverse_ != prev_value_.scale_inverse_)
457  {
458  // not supported right now
459  HPX_THROW_EXCEPTION(not_implemented,
460  "statistics_counter<Statistic>::evaluate",
461  "base counter should keep scaling constant over time");
462  return false;
463  }
464  else {
465  // accumulate new value
466  std::lock_guard<mutex_type> l(mtx_);
467  value_->add_value(static_cast<double>(base_value.value_));
468  }
469  return true;
470  }

472  template <typename Statistic>
473  bool statistics_counter<Statistic>::ensure_base_counter()
474  {
475  // lock here to avoid checking out multiple reference counted GIDs
476  // from AGAS. This
477  std::unique_lock<mutex_type> l(mtx_);
478 
479  if (!base_counter_id_) {
480  // get or create the base counter
481  error_code ec(lightweight);
482  hpx::id_type base_counter_id;
483  {
484  // We need to unlock the lock here since get_counter might suspend
485  util::unlock_guard<std::unique_lock<mutex_type> > unlock(l);
486  base_counter_id = get_counter(base_counter_name_, ec);
487  }
488 
489  // After reacquiring the lock, we need to check again if base_counter_id_
490  // hasn't been set yet
491  if (!base_counter_id_)
492  {
493  base_counter_id_ = base_counter_id;
494  }
495  else
496  {
497  // If it was set already by a different thread, return true.
498  return true;
499  }
500 
501  if (HPX_UNLIKELY(ec || !base_counter_id_))
502  {
503  // base counter could not be retrieved
504  HPX_THROW_EXCEPTION(bad_parameter,
505  "statistics_counter<Statistic>::evaluate_base_counter",
506  boost::str(boost::format(
507  "could not get or create performance counter: '%s'") %
508  base_counter_name_)
509  );
510  return false;
511  }
512  }
513 
514  return true;
515  }

517  template <typename Statistic>
518  bool statistics_counter<Statistic>::evaluate_base_counter(
519  counter_value& value)
520  {
521  // query the actual value
522  if (!base_counter_id_ && !ensure_base_counter())
523  return false;
524 
525  value = stubs::performance_counter::get_value(
526  launch::sync, base_counter_id_, reset_base_counter_);
527 
528  if (!has_prev_value_)
529  {
530  has_prev_value_ = true;
531  prev_value_ = value;
532  }
533 
534  return true;
535  }

538  // Start and stop this counter. We dispatch the calls to the base counter
539  // and control our own interval_timer.
540  template <typename Statistic>
541  bool statistics_counter<Statistic>::start()
542  {
543  if (!timer_.is_started()) {
544  // start base counter
545  if (!base_counter_id_ && !ensure_base_counter())
546  return false;
547 
548  bool result = stubs::performance_counter::start(
549  launch::sync, base_counter_id_);
550  if (result) {
551  // acquire the current value of the base counter
552  counter_value base_value;
553  if (evaluate_base_counter(base_value))
554  {
555  std::lock_guard<mutex_type> l(mtx_);
556  value_->add_value(static_cast<double>(base_value.value_));
557  prev_value_ = base_value;
558  }
559 
560  // start timer
561  timer_.start();
562  }
563  else {
564  // start timer even if base counter does not support being
565  // start/stop operations
566  timer_.start(true);
567  }
568  return result;
569  }
570  return false;
571  }
572 
573  template <typename Statistic>
574  bool statistics_counter<Statistic>::stop()
575  {
576  if (timer_.is_started()) {
577  timer_.stop();
578 
579  if (!base_counter_id_ && !ensure_base_counter())
580  return false;
581  return stubs::performance_counter::stop(
582  launch::sync, base_counter_id_);
583  }
584  return false;
585  }

587  template <typename Statistic>
588  void statistics_counter<Statistic>::reset_counter_value()
589  {
590  std::lock_guard<mutex_type> l(mtx_);
591 
592  // reset accumulator
593  value_.reset(new detail::counter_type_from_statistic<Statistic>(

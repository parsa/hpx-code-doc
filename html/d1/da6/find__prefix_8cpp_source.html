124  std::string get_executable_filename(char const* argv0)
125  {
126  std::string r;
127 
128 #if defined(HPX_WINDOWS)
129  HPX_UNUSED(argv0);
130 
131  char exe_path[MAX_PATH + 1] = { '\0' };
132  if (!GetModuleFileNameA(nullptr, exe_path, sizeof(exe_path)))
133  {
134  HPX_THROW_EXCEPTION(hpx::dynamic_link_failure,
135  "get_executable_filename",
136  "unable to find executable filename");
137  }
138  r = exe_path;
139 
140 #elif defined(__linux) || defined(linux) || defined(__linux__)
141  char buf[PATH_MAX + 1];
142  ssize_t length = ::readlink("/proc/self/exe", buf, sizeof(buf));
143 
144  if (length != -1)
145  {
146  buf[length] = '\0';
147  r = buf;
148  return r;
149  }
150 
151  std::string argv0_(argv0);
152 
153  // REVIEW: Should we resolve symlinks at any point here?
154  if (argv0_.length() > 0)
155  {
156  // Check for an absolute path.
157  if (argv0_[0] == '/')
158  return argv0_;
159 
160  // Check for a relative path.
161  if (argv0_.find('/') != std::string::npos)
162  {
163  // Get the current working directory.
164 
165  // NOTE: getcwd does give you a null terminated string,
166  // while readlink (above) does not.
167  if (::getcwd(buf, PATH_MAX))
168  {
169  r = buf;
170  r += '/';
171  r += argv0_;
172  return r;
173  }
174  }
175 
176  // Search PATH
177  char const* epath = ::getenv("PATH");
178  if (epath)
179  {
180  std::vector<std::string> path_dirs;
181 
182  boost::algorithm::split(path_dirs, epath,
183  boost::algorithm::is_any_of(":"),
184  boost::algorithm::token_compress_on);
185 
186  for (std::uint64_t i = 0; i < path_dirs.size(); ++i)
187  {
188  r = path_dirs[i];
189  r += '/';
190  r += argv0_;
191 
192  // Can't use Boost.Filesystem as it doesn't let me access
193  // st_uid and st_gid.
194  struct stat s;
195 
196  // Make sure the file is executable and shares our
197  // effective uid and gid.
198  // NOTE: If someone was using an HPX application that was
199  // seteuid'd to root, this may fail.
200  if (0 == ::stat(r.c_str(), &s))
201  if ((s.st_uid == ::geteuid()) && (s.st_mode & S_IXUSR)
202  && (s.st_gid == ::getegid()) && (s.st_mode & S_IXGRP)
203  && (s.st_mode & S_IXOTH))
204  return r;
205  }
206  }
207  }
208 
209  HPX_THROW_EXCEPTION(hpx::dynamic_link_failure,
210  "get_executable_filename",
211  "unable to find executable filename");
212 
213 #elif defined(__APPLE__)
214  HPX_UNUSED(argv0);
215 
216  char exe_path[PATH_MAX + 1];
217  std::uint32_t len = sizeof(exe_path) / sizeof(exe_path[0]);
218 
219  if (0 != _NSGetExecutablePath(exe_path, &len))
220  {
221  HPX_THROW_EXCEPTION(hpx::dynamic_link_failure,
222  "get_executable_filename",
223  "unable to find executable filename");
224  }
225  exe_path[len-1] = '\0';
226  r = exe_path;
227 
228 #elif defined(__FreeBSD__)
229  HPX_UNUSED(argv0);
230 
231  int mib[] = { CTL_KERN, KERN_PROC, KERN_PROC_PATHNAME, -1 };
232  size_t cb = 0;
233  sysctl(mib, 4, nullptr, &cb, nullptr, 0);
234  if (cb)
235  {
236  std::vector<char> buf(cb);
237  sysctl(mib, 4, &buf[0], &cb, nullptr, 0);
238  r = &buf[0];
239  }
240 
241 #else
242 # error Unsupported platform
243 #endif
244 
245  return r;
246  }

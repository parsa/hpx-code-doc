449 void test_inclusive_scan_validate(ExPolicy p, std::vector<int> &a, std::vector<int> &b)
450 {
451  using namespace hpx::parallel;
452  typedef std::vector<int>::iterator Iter;
453 
454  // test 1, fill array with numbers counting from 0, then run scan algorithm
455  a.clear();
456  std::copy(boost::counting_iterator<int>(0),
457  boost::counting_iterator<int>(ARRAY_SIZE), std::back_inserter(a));
458  b.resize(a.size());

463  // counting from zero,

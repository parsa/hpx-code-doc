26 int fractal_pixel_value(float x0, float y0, int max_iteration)
27 {
28  float x = 0, y = 0;
29  int iteration = 0;
30  while (x*x + y*y < 2*2 && iteration < max_iteration)
31  {
32  float xtemp = x*x - y*y + x0;
33  y = 2*x*y + y0;
34  x = xtemp;
35 
36  ++iteration;
37  }
38  return iteration;
39 }

283  struct loop<detail::prefetching_iterator<Itr, Ts...> >
284  {
285  typedef detail::prefetching_iterator<Itr, Ts...> iterator_type;
286  typedef typename iterator_type::base_iterator type;
287  typedef typename hpx::util::detail::make_index_pack<
288  sizeof...(Ts)
289  >::type index_pack_type;
290 
291  template <typename End, typename F>
292  static iterator_type
293  call(iterator_type it, End end, F && f)
294  {
295  for (; it != end; ++it)
296  {
297  Itr base = it.base();
298  std::size_t j = it.index();
299 
300  std::size_t last = (std::min)(it.index() + it.chunk_size(),
301  it.range_size());
302 
303  for (; j != last; (void) ++j, ++base)
304  f(base);
305 
306  if (j != it.range_size())
307  prefetch_containers(it.ranges(), index_pack_type(), j);
308  }
309  return it;
310  }

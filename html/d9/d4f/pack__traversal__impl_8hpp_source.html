548  template <typename M, typename T>
549  auto remap_container(
550  container_category_tag<false, false>, M&& mapper, T&& container)
551  -> decltype(
552  rebind_container<mapped_type_from_t<T, M>>(container))
553  {
554  static_assert(has_push_back<typename std::decay<T>::type,
555  element_of_t<T>>::value,
556  "Can only remap containers, that provide a push_back "
557  "method!");
558 
559  // Create the new container, which is capable of holding
560  // the remappped types.
561  auto remapped =
562  rebind_container<mapped_type_from_t<T, M>>(container);
563 
564  // We try to reserve the original size from the source
565  // container to the destination container.
566  reserve_if_possible(
567  is_reservable<decltype(remapped)>{}, remapped, container);
568 
569  // Perform the actual value remapping from the source to
570  // the destination.
571  // We could have used std::transform for this, however,
572  // I didn't want to pull a whole header for it in.
573  for (auto&& val :
574  container_accessor_of(std::forward<T>(container)))
575  {
576  remapped.push_back(spreading::unpack(std::forward<M>(
577  mapper)(std::forward<decltype(val)>(val))));
578  }
579 
580  return remapped; // RVO
581  }

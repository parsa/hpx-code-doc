362  {
363  std::uint32_t this_locality = get_locality_id();
364  std::vector<future<void> > ptrs;
365 
366  std::size_t l = 0;

383  )

386  ++l;
387  }

431  // Perform a deep copy from the given unordered_map
432  void copy_from(unordered_map const& rhs)
433  {
434  typedef typename partitions_vector_type::const_iterator
435  const_iterator;
436 
437  std::vector<future<id_type> > objs;
438  const_iterator end = rhs.partitions_.end();
439  for (const_iterator it = rhs.partitions_.begin(); it != end; ++it)
440  {
441  typedef partition_unordered_map_server component_type;
442  objs.push_back(hpx::components::copy<component_type>(
443  it->partition_.get()));
444  }
445  wait_all(objs);
446 
447  std::uint32_t this_locality = get_locality_id();
448  std::vector<future<void> > ptrs;
449 
450  partitions_vector_type partitions;
451  partitions.reserve(rhs.partitions_.size());
452  for (std::size_t i = 0; i != rhs.partitions_.size(); ++i)
453  {
454  std::uint32_t locality = rhs.partitions_[i].locality_id_;
455 
456  partitions.push_back(partition_data(objs[i].get(), locality));
457 
458  if (locality == this_locality)
459  {
460  using util::placeholders::_1;
461  ptrs.push_back(get_ptr<partition_unordered_map_server>(
462  partitions[i].partition_.get()).then(
463  util::bind(&unordered_map::get_ptr_helper,
464  i, std::ref(partitions), _1)));
465  }
466  }
467 
468  wait_all(ptrs);
469 
470  std::swap(partitions_, partitions);
471  }

105  inline threads::detail::scheduled_executor_base*
106  get_service_executor(service_executor_type t,
107  char const* name_suffix = "")
108  {
109  switch(t)
110  {
111  case service_executor_type::io_thread_pool:
112  return new detail::service_executor("io-pool");
113 
114  case service_executor_type::parcel_thread_pool:
115  {
116  char const* suffix = *name_suffix ? name_suffix : "-tcp";
117  return new detail::service_executor("parcel-pool", suffix);
118  }
119 
120  case service_executor_type::timer_thread_pool:
121  return new detail::service_executor("timer-pool");
122 
123  case service_executor_type::main_thread:
124  return new detail::service_executor("main-pool");
125 
126  default:
127  break;
128  }
129 
130  HPX_THROW_EXCEPTION(bad_parameter,
131  "hpx::threads::detail::get_service_executor",
132  "unknown pool executor type");

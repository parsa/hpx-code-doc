124  // -------------------------------------------------------------------
125  // callable that actually computes the state using the stencil iterator
126  // -------------------------------------------------------------------
127  template<
128  typename Transformer, typename StencilIterType,
129  typename KeyStateIterType, typename Compare
130  >
131  struct reduce_stencil_generate
132  {
133  typedef typename Transformer::template result<
134  Transformer(StencilIterType)
135  >::element_type element_type;
136  typedef typename Transformer::template result<
137  Transformer(StencilIterType)
138  >::type tuple_type;
139  typedef typename std::iterator_traits<
140  KeyStateIterType
141  >::reference KeyStateType;
142 
143  reduce_stencil_generate() { }
144 
145  void operator()(const tuple_type &value, KeyStateType &kiter,
146  const Compare &comp) const
147  {
148  // resolves to a tuple of values for *(it-1), *it, *(it+1)
149 
150  element_type left = hpx::util::get<0>(value);
151  element_type mid = hpx::util::get<1>(value);
152  element_type right = hpx::util::get<2>(value);
153 
154  // we need to determine which of three states this
155  // index is. It can be:
156  // 1. Middle of a set of equivalent keys.
157  // 2. Start of a set of equivalent keys.
158  // 3. End of a set of equivalent keys.
159  // 4. Both the start and end of a set of keys
160 
161  {
162  const bool leftMatches(comp(left, mid));
163  const bool rightMatches(comp(mid, right));
164  kiter = reduce_key_series_states(!leftMatches, !rightMatches);
165  }
166  }
167  };

281  // we need to determine based on the keys what is the keystate for
282  // each key. The states are start, middle, end of a series and the special
283  // state start and end of the sequence
284  std::vector<reduce_key_series_states> key_state;
285  typedef std::vector<reduce_key_series_states>::iterator keystate_iter_type;
286  typedef detail::reduce_stencil_iterator<RanIter, reduce_stencil_transformer>
287  reducebykey_iter;
288  typedef typename std::iterator_traits<RanIter>::reference element_type;
289  typedef typename zip_iterator<reducebykey_iter, keystate_iter_type>
290  ::reference zip_ref;
291  //
292  const std::uint64_t number_of_keys = std::distance(key_first, key_last);
293  //
294  key_state.assign(number_of_keys, reduce_key_series_states());
295  {
296  reduce_stencil_transformer r_s_t;
297  reducebykey_iter reduce_begin =
298  make_reduce_stencil_iterator(key_first, r_s_t);
299  reducebykey_iter reduce_end =
300  make_reduce_stencil_iterator(key_last, r_s_t);
301 
302  // FIXME: handle cases number_of_keys == 0 and
303  // number_of_keys == 1
304 
305  if (number_of_keys == 2) {
306  // for two entries, one is a start, the other an end,
307  // if they are different, then they are both start/end
308  element_type left = *key_first;
309  element_type right = *std::next(key_first);
310  key_state[0] = reduce_key_series_states(true, !comp(left, right));
311  key_state[1] = reduce_key_series_states(!comp(left, right), true);
312  } else {
313  // do the first and last elements by hand to simplify the iterator
314  // traversal as there is no prev/next for first/last
315  element_type elem0 = *key_first;
316  element_type elem1 = *std::next(key_first);
317  key_state[0] = reduce_key_series_states(true, elem0 != elem1);
318  // middle elements
319  reduce_stencil_generate<
320  reduce_stencil_transformer, RanIter,
321  keystate_iter_type, Compare
322  > kernel;
323  hpx::parallel::for_each(sync_policy,
324  make_zip_iterator(reduce_begin + 1, key_state.begin() + 1),
325  make_zip_iterator(reduce_end - 1, key_state.end() - 1),
326  [&kernel, &comp](zip_ref ref)
327  {
328  kernel(get<0>(ref), get<1>(ref), comp);
329  });
330  // Last element
331  element_type elemN = *std::prev(key_last);
332  element_type elemn = *std::prev(std::prev(key_last));
333  key_state.back() = reduce_key_series_states(elemN != elemn, true);
334  }
335  }
336  {
337  typedef zip_iterator<
338  RanIter2, std::vector<reduce_key_series_states>::iterator
339  > zip_iterator_in;
340  typedef typename zip_iterator_in::value_type zip_type_in;
341 
342  typedef zip_iterator<
343  FwdIter2, std::vector<reduce_key_series_states>::iterator
344  > zip_iterator_vout;
345  typedef typename zip_iterator_vout::value_type zip_type_out;
346 
347  typedef typename std::iterator_traits<RanIter2>::value_type value_type;
348 
349  zip_iterator_in states_begin = make_zip_iterator(values_first,
350  hpx::util::begin(key_state));
351  zip_iterator_in states_end = make_zip_iterator(
352  values_first + number_of_keys, hpx::util::end(key_state));
353  zip_iterator_vout states_out_begin = make_zip_iterator(values_output,
354  hpx::util::begin(key_state));
355  //
356 
357  zip_type_in initial;
358  //
359  typedef hpx::util::tuple<value_type, reduce_key_series_states>
360  lambda_type;
361  hpx::parallel::inclusive_scan(sync_policy, states_begin,
362  states_end, states_out_begin,
363  // B is the current entry, A is the one passed in from 'previous'
364  [&func](zip_type_in a, zip_type_in b)->lambda_type
365  {
366  value_type a_val = get<0>(a);
367  reduce_key_series_states a_state = get<1>(a);
368  value_type b_val = get<0>(b);
369  reduce_key_series_states b_state = get<1>(b);
370  debug_reduce_by_key(
371  "{ " << a_val << "+" << b_val << " },\t" << a_state <<
372  b_state);
373  // if carrying a start flag, then copy - don't add
374  if (b_state.start) {
375  debug_reduce_by_key(" = " << b_val << std::endl);
376  return make_tuple(
377  b_val,
378  reduce_key_series_states(
379  a_state.start || b_state.start, b_state.end));
380  }
381  // normal add of previous + this
382  else {
383  debug_reduce_by_key(
384  " = " << func(a_val, b_val) << std::endl);
385  value_type temp = func(a_val, b_val);
386  return make_tuple(
387  temp,
388  reduce_key_series_states(
389  a_state.start || b_state.start, b_state.end));
390  }
391  }, initial);
392 
393  // now copy the values and keys for each element that
394  // is marked by an 'END' state to the final output
395  typedef typename hpx::util::zip_iterator<
396  RanIter, FwdIter2, std::vector<reduce_key_series_states>::iterator
397  >::reference zip2_ref;
398 
399  return make_pair_result(
400  hpx::parallel::copy_if(sync_policy,
401  make_zip_iterator(key_first, values_output,
402  hpx::util::begin(key_state)),
403  make_zip_iterator(key_last, values_output + number_of_keys,
404  hpx::util::end(key_state)),
405  make_zip_iterator(keys_output, values_output,
406  hpx::util::begin(key_state)),
407  // copies to dest only when 'end' state is true
408  [](zip2_ref it)
409  {
410  return get<2>(it).end;
411  }),
412  keys_output, values_output);
413  }
414  }
